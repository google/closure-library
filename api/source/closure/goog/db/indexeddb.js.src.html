33/**
34 * Creates an IDBDatabase wrapper object. The database object has methods for
35 * setting the version to change the structure of the database and for creating
36 * transactions to get or modify the stored records. Should not be created
37 * directly, call {@link goog.db.openDatabase} to set up the connection.
38 *
39 * @param {!IDBDatabase} db Underlying IndexedDB database object.
40 * @constructor
41 * @extends {goog.events.EventTarget}
42 * @final
43 */
44goog.db.IndexedDb = function(db) {
45 goog.db.IndexedDb.base(this, 'constructor');
46
47 /**
48 * Underlying IndexedDB database object.
49 *
50 * @type {!IDBDatabase}
51 * @private
52 */
53 this.db_ = db;
54
55 /**
56 * Internal event handler that listens to IDBDatabase events.
57 * @type {!goog.events.EventHandler<!goog.db.IndexedDb>}
58 * @private
59 */
60 this.eventHandler_ = new goog.events.EventHandler(this);
61
62 this.eventHandler_.listen(
63 this.db_,
64 goog.db.IndexedDb.EventType.ABORT,
65 goog.bind(
66 this.dispatchEvent,
67 this,
68 goog.db.IndexedDb.EventType.ABORT));
69 this.eventHandler_.listen(
70 this.db_,
71 goog.db.IndexedDb.EventType.ERROR,
72 this.dispatchError_);
73 this.eventHandler_.listen(
74 this.db_,
75 goog.db.IndexedDb.EventType.VERSION_CHANGE,
76 this.dispatchVersionChange_);
77 this.eventHandler_.listen(
78 this.db_,
79 goog.db.IndexedDb.EventType.CLOSE,
80 goog.bind(
81 this.dispatchEvent,
82 this,
83 goog.db.IndexedDb.EventType.CLOSE));
84};
85goog.inherits(goog.db.IndexedDb, goog.events.EventTarget);

168/**
169 * Creates an object store in this database. Can only be called inside a
170 * {@link goog.db.UpgradeNeededCallback} or the callback for the Deferred
171 * returned from #setVersion.
172 *
173 * @param {string} name Name for the new object store.
174 * @param {Object=} opt_params Options object. The available options are:
175 * keyPath, which is a string and determines what object attribute

211/**
212 * Updates the version of the database and returns a Deferred transaction.
213 * The database's structure can be changed inside this Deferred's callback, but
214 * nowhere else. This means adding or deleting object stores, and adding or
215 * deleting indexes. The version change will not succeed unless there are no
216 * other connections active for this database anywhere. A new database
217 * connection should be opened after the version change is finished to pick
218 * up changes.
219 *
220 * This is deprecated, and only supported on Chrome prior to version 25. New
221 * applications should use the version parameter to {@link goog.db.openDatabase}
222 * instead.
223 *
224 * @param {string} version The new version of the database.
225 * @return {!goog.async.Deferred} The deferred transaction for changing the
226 * version.
227 */
228goog.db.IndexedDb.prototype.setVersion = function(version) {
229 var self = this;
230 var d = new goog.async.Deferred();
231 var request = this.db_.setVersion(version);
232 request.onsuccess = function(ev) {
233 // the transaction is in the result field (the transaction field is null
234 // for version change requests)
235 d.callback(new goog.db.Transaction(ev.target.result, self));
236 };
237 request.onerror = function(ev) {
238 // If a version change is blocked, onerror and onblocked may both fire.
239 // Check d.hasFired() to avoid an AlreadyCalledError.
240 if (!d.hasFired()) {
241 d.errback(goog.db.Error.fromRequest(ev.target, 'setting version'));
242 }
243 };
244 request.onblocked = function(ev) {
245 // If a version change is blocked, onerror and onblocked may both fire.
246 // Check d.hasFired() to avoid an AlreadyCalledError.
247 if (!d.hasFired()) {
248 d.errback(new goog.db.Error.VersionChangeBlockedError());
249 }
250 };
251 return d;
252};

255/**
256 * Creates a new transaction.
257 *
258 * @param {!Array<string>} storeNames A list of strings that contains the
259 * transaction's scope, the object stores that this transaction can operate
260 * on.
261 * @param {goog.db.Transaction.TransactionMode=} opt_mode The mode of the
262 * transaction. If not present, the default is READ_ONLY. For VERSION_CHANGE
263 * transactions call {@link goog.db.IndexedDB#setVersion} instead.
264 * @return {!goog.db.Transaction} The wrapper for the newly created transaction.
265 * @throws {goog.db.Error} If there's a problem creating the transaction.
266 */
267goog.db.IndexedDb.prototype.createTransaction = function(storeNames, opt_mode) {
268 try {
269 // IndexedDB on Chrome 22+ requires that opt_mode not be passed rather than
270 // be explicitly passed as undefined.
271 var transaction = opt_mode ?
272 this.db_.transaction(storeNames, opt_mode) :
273 this.db_.transaction(storeNames);
274 return new goog.db.Transaction(transaction, this);
275 } catch (ex) {
276 throw goog.db.Error.fromException(ex, 'creating transaction');
277 }
278};

288/**
289 * Event types fired by a database.
290 *
291 * @enum {string} The event types for the web socket.
292 */
293goog.db.IndexedDb.EventType = {
294
295 /**
296 * Fired when a transaction is aborted and the event bubbles to its database.
297 */
298 ABORT: 'abort',
299
300 /**
301 * Fired when the database connection is forcibly closed by the browser,
302 * without an explicit call to IDBDatabase#close. This behavior is not in the
303 * spec yet but will be added since it is necessary, see
304 * https://www.w3.org/Bugs/Public/show_bug.cgi?id=22540.
305 */
306 CLOSE: 'close',
307
308 /**
309 * Fired when a transaction has an error.
310 */
311 ERROR: 'error',
312
313 /**
314 * Fired when someone (possibly in another window) is attempting to modify the
315 * structure of the database. Since a change can only be made when there are
316 * no active database connections, this usually means that the database should
317 * be closed so that the other client can make its changes.
318 */
319 VERSION_CHANGE: 'versionchange'
320};
