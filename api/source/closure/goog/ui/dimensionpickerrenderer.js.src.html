16 * @fileoverview The default renderer for a goog.dom.DimensionPicker. A
17 * dimension picker allows the user to visually select a row and column count.
18 * It looks like a palette but in order to minimize DOM load it is rendered.

21 * @author robbyw@google.com (Robby Walker)

37/**
38 * Default renderer for {@link goog.ui.DimensionPicker}s. Renders the
39 * palette as two divs, one with the un-highlighted background, and one with the
40 * highlighted background.
41 *
42 * @constructor
43 * @extends {goog.ui.ControlRenderer}
44 */
45goog.ui.DimensionPickerRenderer = function() {
46 goog.ui.ControlRenderer.call(this);
47
48 /** @private {goog.a11y.aria.Announcer} */
49 this.announcer_ = new goog.a11y.aria.Announcer();
50};
51goog.inherits(goog.ui.DimensionPickerRenderer, goog.ui.ControlRenderer);
52goog.addSingletonGetter(goog.ui.DimensionPickerRenderer);

145/**
146 * Scales various elements in order to update the palette's size.
147 * @param {goog.ui.DimensionPicker} palette The palette object.
148 * @param {Element} element The element to set the style of.
149 */
150goog.ui.DimensionPickerRenderer.prototype.updateSize =
151 function(palette, element) {
152 var size = palette.getSize();
153
154 element.style.width = size.width + 'em';
155
156 var underlyingDiv = this.getUnderlyingDiv_(element);
157 underlyingDiv.style.width = size.width + 'em';
158 underlyingDiv.style.height = size.height + 'em';
159
160 if (palette.isRightToLeft()) {
161 this.adjustParentDirection_(palette, element);
162 }
163};

166/**
167 * Adds the appropriate content elements to the given outer DIV.
168 * @param {goog.ui.DimensionPicker} palette The palette object.
169 * @param {Element} element The element to decorate.
170 * @private
171 */
172goog.ui.DimensionPickerRenderer.prototype.addElementContents_ = function(
173 palette, element) {
174 // First we create a single div containing three stacked divs. The bottom div
175 // catches mouse events. We can't use document level mouse move detection as
176 // we could lose events to iframes. This is especially important in Firefox 2
177 // in which TrogEdit creates iframes. The middle div uses a css tiled
178 // background image to represent deselected tiles. The top div uses a
179 // different css tiled background image to represent selected tiles.
180 var mouseCatcherDiv = palette.getDomHelper().createDom(goog.dom.TagName.DIV,
181 goog.getCssName(this.getCssClass(), 'mousecatcher'));
182 var unhighlightedDiv = palette.getDomHelper().createDom(goog.dom.TagName.DIV,
183 {
184 'class': goog.getCssName(this.getCssClass(), 'unhighlighted'),
185 'style': 'width:100%;height:100%'
186 });
187 var highlightedDiv = palette.getDomHelper().createDom(goog.dom.TagName.DIV,
188 goog.getCssName(this.getCssClass(), 'highlighted'));
189 element.appendChild(
190 palette.getDomHelper().createDom(goog.dom.TagName.DIV,
191 {'style': 'width:100%;height:100%'},
192 mouseCatcherDiv, unhighlightedDiv, highlightedDiv));
193
194 // Lastly we add a div to store the text version of the current state.
195 element.appendChild(palette.getDomHelper().createDom(goog.dom.TagName.DIV,
196 goog.getCssName(this.getCssClass(), 'status')));
197};

200/**
201 * Creates a div and adds the appropriate contents to it.
202 * @param {goog.ui.Control} control Picker to render.
203 * @return {!Element} Root element for the palette.
204 * @override
205 */
206goog.ui.DimensionPickerRenderer.prototype.createDom = function(control) {
207 var palette = /** @type {goog.ui.DimensionPicker} */ (control);
208 var classNames = this.getClassNames(palette);
209 // Hide the element from screen readers so they don't announce "1 of 1" for
210 // the perceived number of items in the palette.
211 var element = palette.getDomHelper().createDom(goog.dom.TagName.DIV, {
212 'class': classNames ? classNames.join(' ') : '',
213 'aria-hidden': 'true'
214 });
215 this.addElementContents_(palette, element);
216 this.updateSize(palette, element);
217 return element;
218};

221/**
222 * Initializes the control's DOM when the control enters the document. Called
223 * from {@link goog.ui.Control#enterDocument}.
224 * @param {goog.ui.Control} control Palette whose DOM is to be
225 * initialized as it enters the document.
226 * @override

229 control) {
230 var palette = /** @type {goog.ui.DimensionPicker} */ (control);
231 goog.ui.DimensionPickerRenderer.superClass_.initializeDom.call(this, palette);

234 var highlightedSize = palette.getValue();
235 this.setHighlightedSize(palette,
236 highlightedSize.width, highlightedSize.height);

278/**
279 * Sets the highlighted size. Does nothing if the palette hasn't been rendered.
280 * @param {goog.ui.DimensionPicker} palette The table size palette.
281 * @param {number} columns The number of columns to highlight.
282 * @param {number} rows The number of rows to highlight.
283 */
284goog.ui.DimensionPickerRenderer.prototype.setHighlightedSize = function(
285 palette, columns, rows) {
286 var element = palette.getElement();
287 // Can't update anything if DimensionPicker hasn't been rendered.
288 if (!element) {
289 return;
290 }
291
292 // Style the highlight div.
293 var style = this.getHighlightDiv_(element).style;
294 style.width = columns + 'em';
295 style.height = rows + 'em';
296
297 // Explicitly set style.right so the element grows to the left when increase
298 // in width.
299 if (palette.isRightToLeft()) {
300 style.right = '0';
301 }
302
303 /**
304 * @desc The dimension of the columns and rows currently selected in the
305 * dimension picker, as text that can be spoken by a screen reader.
306 */
307 var MSG_DIMENSION_PICKER_HIGHLIGHTED_DIMENSIONS = goog.getMsg(
308 '{$numCols} by {$numRows}',
309 {'numCols': columns, 'numRows': rows});
310 this.announcer_.say(MSG_DIMENSION_PICKER_HIGHLIGHTED_DIMENSIONS,
311 goog.a11y.aria.LivePriority.ASSERTIVE);
312
313 // Update the size text.
314 goog.dom.setTextContent(this.getStatusDiv_(element),
315 goog.i18n.bidi.enforceLtrInText(columns + ' x ' + rows));
316};

319/**
320 * Position the mouse catcher such that it receives mouse events past the
321 * selectedsize up to the maximum size. Takes care to not introduce scrollbars.
322 * Should be called on enter document and when the window changes size.
323 * @param {goog.ui.DimensionPicker} palette The table size palette.
324 */
325goog.ui.DimensionPickerRenderer.prototype.positionMouseCatcher = function(
326 palette) {
327 var mouseCatcher = this.getMouseCatcher_(palette.getElement());
328 var doc = goog.dom.getOwnerDocument(mouseCatcher);
329 var body = doc.body;
330
331 var position = goog.style.getRelativePosition(mouseCatcher, body);
332
333 // Hide the mouse catcher so it doesn't affect the body's scroll size.
334 mouseCatcher.style.display = 'none';
335
336 // Compute the maximum size the catcher can be without introducing scrolling.
337 var xAvailableEm = (palette.isRightToLeft() && position.x > 0) ?
338 Math.floor(position.x / 18) :
339 Math.floor((body.scrollWidth - position.x) / 18);
340
341 // Computing available height is more complicated - we need to check the
342 // window's inner height.
343 var height;
344 if (goog.userAgent.IE) {
345 // Offset 20px to make up for scrollbar size.
346 height = goog.style.getClientViewportElement(body).scrollHeight - 20;
347 } else {
348 var win = goog.dom.getWindow(doc);
349 // Offset 20px to make up for scrollbar size.
350 height = Math.max(win.innerHeight, body.scrollHeight) - 20;
351 }
352 var yAvailableEm = Math.floor((height - position.y) / 18);
353
354 // Resize and display the mouse catcher.
355 mouseCatcher.style.width = Math.min(palette.maxColumns, xAvailableEm) + 'em';
356 mouseCatcher.style.height = Math.min(palette.maxRows, yAvailableEm) + 'em';
357 mouseCatcher.style.display = '';
358
359 // Explicitly set style.right so the mouse catcher is positioned on the left
360 // side instead of right.
361 if (palette.isRightToLeft()) {
362 mouseCatcher.style.right = '0';
363 }
364};

378/**
379 * This function adjusts the positioning from 'left' and 'top' to 'right' and
380 * 'top' as appropriate for RTL control. This is so when the dimensionpicker
381 * grow in width, the containing element grow to the left instead of right.
382 * This won't be necessary if goog.ui.SubMenu rendering code would position RTL
383 * control with 'right' and 'top'.
384 * @private
385 *
386 * @param {goog.ui.DimensionPicker} palette The palette object.
387 * @param {Element} element The palette's element.
388 */
389goog.ui.DimensionPickerRenderer.prototype.adjustParentDirection_ =
390 function(palette, element) {
391 var parent = palette.getParent();
392 if (parent) {
393 var parentElement = parent.getElement();
394
395 // Anchors the containing element to the right so it grows to the left
396 // when it increase in width.
397 var right = goog.style.getStyle(parentElement, 'right');
398 if (right == '') {
399 var parentPos = goog.style.getPosition(parentElement);
400 var parentSize = goog.style.getSize(parentElement);
401 if (parentSize.width != 0 && parentPos.x != 0) {
402 var visibleRect = goog.style.getBounds(
403 goog.style.getClientViewportElement());
404 var visibleWidth = visibleRect.width;
405 right = visibleWidth - parentPos.x - parentSize.width;
406 goog.style.setStyle(parentElement, 'right', right + 'px');
407 }
408 }
409
410 // When a table is inserted, the containing elemet's position is
411 // recalculated the next time it shows, set left back to '' to prevent
412 // extra white space on the left.
413 var left = goog.style.getStyle(parentElement, 'left');
414 if (left != '') {
415 goog.style.setStyle(parentElement, 'left', '');
416 }
417 } else {
418 goog.style.setStyle(element, 'right', '0px');
419 }
