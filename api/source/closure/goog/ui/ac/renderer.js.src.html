49/**
50 * Class for rendering the results of an auto-complete in a drop down list.
51 *
52 * @constructor
53 * @param {Element=} opt_parentNode optional reference to the parent element
54 * that will hold the autocomplete elements. goog.dom.getDocument().body
55 * will be used if this is null.
56 * @param {?({renderRow}|{render})=} opt_customRenderer Custom full renderer to
57 * render each row. Should be something with a renderRow or render method.
58 * @param {boolean=} opt_rightAlign Determines if the autocomplete will always
59 * be right aligned. False by default.
60 * @param {boolean=} opt_useStandardHighlighting Determines if standard
61 * highlighting should be applied to each row of data. Standard highlighting
62 * bolds every matching substring for a given token in each row. True by
63 * default.
64 * @extends {goog.events.EventTarget}

436/**
437 * Hide the object.
438 */
439goog.ui.ac.Renderer.prototype.dismiss = function() {
440 if (this.target_) {
441 goog.a11y.aria.setActiveDescendant(this.target_, null);
442 }
443 if (this.visible_) {
444 this.visible_ = false;
445
446 // Clear ARIA popup role for the target input box.
447 if (this.target_) {
448 goog.a11y.aria.setState(this.target_,
449 goog.a11y.aria.State.HASPOPUP,
450 false);
451 }
452
453 if (this.menuFadeDuration_ > 0) {
454 goog.dispose(this.animation_);
455 this.animation_ = new goog.fx.dom.FadeOutAndHide(this.element_,
456 this.menuFadeDuration_);
457 this.animation_.play();
458 } else {
459 goog.style.setElementShown(this.element_, false);
460 }
461 }
462};

464
465/**
466 * Show the object.
467 */
468goog.ui.ac.Renderer.prototype.show = function() {
469 if (!this.visible_) {
470 this.visible_ = true;
471
472 // Set ARIA roles and states for the target input box.
473 if (this.target_) {
474 goog.a11y.aria.setRole(this.target_,
475 goog.a11y.aria.Role.COMBOBOX);
476 goog.a11y.aria.setState(this.target_,
477 goog.a11y.aria.State.AUTOCOMPLETE,
478 'list');
479 goog.a11y.aria.setState(this.target_,
480 goog.a11y.aria.State.HASPOPUP,
481 true);
482 }
483
484 if (this.menuFadeDuration_ > 0) {
485 goog.dispose(this.animation_);
486 this.animation_ = new goog.fx.dom.FadeInAndShow(this.element_,
487 this.menuFadeDuration_);
488 this.animation_.play();
489 } else {
490 goog.style.setElementShown(this.element_, true);
491 }
492 }

504/**
505 * Sets the 'active' class of the nth item.
506 * @param {number} index Index of the item to highlight.
507 */
508goog.ui.ac.Renderer.prototype.hiliteRow = function(index) {
509 var row = index >= 0 && index < this.rows_.length ?
510 this.rows_[index] : undefined;
511 var rowDiv = index >= 0 && index < this.rowDivs_.length ?
512 this.rowDivs_[index] : undefined;
513
514 var evtObj = /** @lends {goog.events.Event.prototype} */ ({
515 type: goog.ui.ac.AutoComplete.EventType.ROW_HILITE,
516 rowNode: rowDiv,
517 row: row ? row.data : null
518 });
519 if (this.dispatchEvent(evtObj)) {
520 this.hiliteNone();
521 this.hilitedRow_ = index;
522 if (rowDiv) {
523 goog.dom.classlist.addAll(rowDiv, [this.activeClassName,
524 this.legacyActiveClassName_]);
525 if (this.target_) {
526 goog.a11y.aria.setActiveDescendant(this.target_, rowDiv);
527 }
528 goog.style.scrollIntoContainerView(rowDiv, this.element_);
529 }
530 }
531};
532
533
534/**
535 * Removes the 'active' class from the currently selected row.
536 */
537goog.ui.ac.Renderer.prototype.hiliteNone = function() {
538 if (this.hilitedRow_ >= 0) {
539 goog.dom.classlist.removeAll(
540 goog.asserts.assert(this.rowDivs_[this.hilitedRow_]),
541 [this.activeClassName, this.legacyActiveClassName_]);
542 }
543};
544
545
546/**
547 * Sets the 'active' class of the item with a given id.
548 * @param {number} id Id of the row to hilight. If id is -1 then no rows get
549 * hilited.
550 */
551goog.ui.ac.Renderer.prototype.hiliteId = function(id) {
552 if (id == -1) {
553 this.hiliteRow(-1);
554 } else {
555 for (var i = 0; i < this.rows_.length; i++) {
556 if (this.rows_[i].id == id) {
557 this.hiliteRow(i);
558 return;
559 }
560 }
561 }

579/**
580 * If the main HTML element hasn't been made yet, creates it and appends it
581 * to the parent.
582 * @private
583 */
584goog.ui.ac.Renderer.prototype.maybeCreateElement_ = function() {
585 if (!this.element_) {
586 // Make element and add it to the parent
587 var el = this.dom_.createDom(goog.dom.TagName.DIV, {style: 'display:none'});
588 if (this.showScrollbarsIfTooLarge_) {
589 // Make sure that the dropdown will get scrollbars if it isn't large
590 // enough to show all rows.
591 el.style.overflowY = 'auto';
592 }
593 this.element_ = el;
594 this.setMenuClasses_(el);
595 goog.a11y.aria.setRole(el, goog.a11y.aria.Role.LISTBOX);
596
597 el.id = goog.ui.IdGenerator.getInstance().getNextUniqueId();
598
599 this.dom_.appendChild(this.parent_, el);
600
601 // Add this object as an event handler
602 goog.events.listen(el, goog.events.EventType.CLICK,
603 this.handleClick_, false, this);
604 goog.events.listen(el, goog.events.EventType.MOUSEDOWN,
605 this.handleMouseDown_, false, this);
606 goog.events.listen(el, goog.events.EventType.MOUSEOVER,
607 this.handleMouseOver_, false, this);
608 }
609};

612/**
613 * Redraw (or draw if this is the first call) the rendered auto-complete drop
614 * down.
615 */
616goog.ui.ac.Renderer.prototype.redraw = function() {
617 // Create the element if it doesn't yet exist
618 this.maybeCreateElement_();
619
620 // For top aligned with target (= bottom aligned element),
621 // we need to hide and then add elements while hidden to prevent
622 // visible repositioning
623 if (this.topAlign_) {
624 this.element_.style.visibility = 'hidden';
625 }
626
627 if (this.widthProvider_) {
628 var width = this.widthProvider_.clientWidth + 'px';
629 this.element_.style.minWidth = width;
630 }
631
632 // Remove the current child nodes
633 this.rowDivs_.length = 0;
634 this.dom_.removeChildren(this.element_);
635
636 // Generate the new rows (use forEach so we can change rows_ from an
637 // array to a different datastructure if required)
638 if (this.customRenderer_ && this.customRenderer_.render) {
639 this.customRenderer_.render(this, this.element_, this.rows_, this.token_);
640 } else {
641 var curRow = null;
642 goog.array.forEach(this.rows_, function(row) {
643 row = this.renderRowHtml(row, this.token_);
644 if (this.topAlign_) {
645 // Aligned with top of target = best match at bottom
646 this.element_.insertBefore(row, curRow);
647 } else {
648 this.dom_.appendChild(this.element_, row);
649 }
650 curRow = row;
651 }, this);
652 }
653
654 // Don't show empty result sets
655 if (this.rows_.length == 0) {
656 this.dismiss();
657 return;
658 } else {
659 this.show();
660 }
661
662 this.reposition();
663
664 // Make the autocompleter unselectable, so that it
665 // doesn't steal focus from the input field when clicked.
666 goog.style.setUnselectable(this.element_, true);
667};

685/**
686 * Repositions the auto complete popup relative to the location node, if it
687 * exists and the auto position has been set.
688 */
689goog.ui.ac.Renderer.prototype.reposition = function() {
690 if (this.target_ && this.reposition_) {
691 var anchorElement = this.anchorElement_ || this.target_;
692 var anchorCorner = this.getAnchorCorner();
693
694 var overflowMode = goog.positioning.Overflow.ADJUST_X_EXCEPT_OFFSCREEN;
695 if (this.showScrollbarsIfTooLarge_) {
696 // positionAtAnchor will set the height of this.element_ when it runs
697 // (because of RESIZE_HEIGHT), and it will never increase it relative to
698 // its current value when it runs again. But if the user scrolls their
699 // page, then we might actually want a bigger height when the dropdown is
700 // displayed next time. So we clear the height before calling
701 // positionAtAnchor, so it is free to set the height as large as it
702 // chooses.
703 this.element_.style.height = '';
704 overflowMode |= goog.positioning.Overflow.RESIZE_HEIGHT;
705 }
706
707 goog.positioning.positionAtAnchor(
708 anchorElement, anchorCorner,
709 this.element_, goog.positioning.flipCornerVertical(anchorCorner),
710 null, null, overflowMode);
711
712 if (this.topAlign_) {
713 // This flickers, but is better than the alternative of positioning
714 // in the wrong place and then moving.
715 this.element_.style.visibility = 'visible';
716 }
717 }
718};

791/**
792 * Goes through a node and all of its child nodes, replacing HTML text that
793 * matches a token with <b>token</b>.
794 * The replacement will happen on the first match or all matches depending on
795 * this.highlightAllTokens_ value.
796 *
797 * @param {Node} node Node to match.
798 * @param {string|Array<string>} tokenOrArray Token to match or array of tokens
799 * to match. By default, only the first match will be highlighted. If
800 * highlightAllTokens is set, then all tokens appearing at the start of a
801 * word, in whatever order and however many times, will be highlighted.
802 * @private
803 */
804goog.ui.ac.Renderer.prototype.startHiliteMatchingText_ =
805 function(node, tokenOrArray) {

811/**
812 * @param {Node} node Node to match.
813 * @param {string|Array<string>} tokenOrArray Token to match or array of tokens
814 * to match.
815 * @private
816 */
817goog.ui.ac.Renderer.prototype.hiliteMatchingText_ =
818 function(node, tokenOrArray) {
819 if (!this.highlightAllTokens_ && this.wasHighlightedAtLeastOnce_) {
820 return;
821 }
822
823 if (node.nodeType == goog.dom.NodeType.TEXT) {
824 var rest = null;
825 if (goog.isArray(tokenOrArray) &&
826 tokenOrArray.length > 1 &&
827 !this.highlightAllTokens_) {
828 rest = goog.array.slice(tokenOrArray, 1);
829 }
830
831 var token = this.getTokenRegExp_(tokenOrArray);
832 if (token.length == 0) return;
833
834 var text = node.nodeValue;
835
836 // Create a regular expression to match a token at the beginning of a line
837 // or preceeded by non-alpha-numeric characters. Note: token could have |
838 // operators in it, so we need to parenthesise it before adding \b to it.
839 // or preceeded by non-alpha-numeric characters
840 //
841 // NOTE(user): When using word matches, this used to have
842 // a (^|\\W+) clause where it now has \\b but it caused various
843 // browsers to hang on really long strings. The (^|\\W+) matcher was also
844 // unnecessary, because \b already checks that the character before the
845 // is a non-word character, and ^ matches the start of the line or following
846 // a line terminator character, which is also \W. The regexp also used to
847 // have a capturing match before the \\b, which would capture the
848 // non-highlighted content, but that caused the regexp matching to run much
849 // slower than the current version.
850 var re = this.matchWordBoundary_ ?
851 new RegExp('\\b(?:' + token + ')', 'gi') :
852 new RegExp(token, 'gi');
853 var textNodes = [];
854 var lastIndex = 0;
855
856 // Find all matches
857 // Note: text.split(re) has inconsistencies between IE and FF, so
858 // manually recreated the logic
859 var match = re.exec(text);
860 var numMatches = 0;
861 while (match) {
862 numMatches++;
863 textNodes.push(text.substring(lastIndex, match.index));
864 textNodes.push(text.substring(match.index, re.lastIndex));
865 lastIndex = re.lastIndex;
866 match = re.exec(text);
867 }
868 textNodes.push(text.substring(lastIndex));
869
870 // Replace the tokens with bolded text. Each pair of textNodes
871 // (starting at index idx) includes a node of text before the bolded
872 // token, and a node (at idx + 1) consisting of what should be
873 // enclosed in bold tags.
874 if (textNodes.length > 1) {
875 var maxNumToBold = !this.highlightAllTokens_ ? 1 : numMatches;
876 for (var i = 0; i < maxNumToBold; i++) {
877 var idx = 2 * i;
878
879 node.nodeValue = textNodes[idx];
880 var boldTag = this.dom_.createElement(goog.dom.TagName.B);
881 boldTag.className = this.highlightedClassName;
882 this.dom_.appendChild(boldTag,
883 this.dom_.createTextNode(textNodes[idx + 1]));
884 boldTag = node.parentNode.insertBefore(boldTag, node.nextSibling);
885 node.parentNode.insertBefore(this.dom_.createTextNode(''),
886 boldTag.nextSibling);
887 node = boldTag.nextSibling;
888 }
889
890 // Append the remaining text nodes to the end.
891 var remainingTextNodes = goog.array.slice(textNodes, maxNumToBold * 2);
892 node.nodeValue = remainingTextNodes.join('');
893
894 this.wasHighlightedAtLeastOnce_ = true;
895 } else if (rest) {
896 this.hiliteMatchingText_(node, rest);
897 }
898 } else {
899 var child = node.firstChild;
900 while (child) {
901 var nextChild = child.nextSibling;
902 this.hiliteMatchingText_(child, tokenOrArray);
903 child = nextChild;
904 }
905 }
906};

909/**
910 * Transforms a token into a string ready to be put into the regular expression
911 * in hiliteMatchingText_.
912 * @param {string|Array<string>} tokenOrArray The token or array to get the
913 * regex string from.
914 * @return {string} The regex-ready token.
915 * @private
916 */
917goog.ui.ac.Renderer.prototype.getTokenRegExp_ = function(tokenOrArray) {
918 var token = '';
919
920 if (!tokenOrArray) {
921 return token;
922 }
923
924 if (goog.isArray(tokenOrArray)) {
925 // Remove invalid tokens from the array, which may leave us with nothing.
926 tokenOrArray = goog.array.filter(tokenOrArray, function(str) {
927 return !goog.string.isEmptyOrWhitespace(goog.string.makeSafe(str));
928 });
929 }
930
931 // If highlighting all tokens, join them with '|' so the regular expression
932 // will match on any of them.
933 if (this.highlightAllTokens_) {
934 if (goog.isArray(tokenOrArray)) {
935 var tokenArray = goog.array.map(tokenOrArray, goog.string.regExpEscape);
936 token = tokenArray.join('|');
937 } else {
938 // Remove excess whitespace from the string so bars will separate valid
939 // tokens in the regular expression.
940 token = goog.string.collapseWhitespace(tokenOrArray);
941
942 token = goog.string.regExpEscape(token);
943 token = token.replace(/ /g, '|');
944 }
945 } else {
946 // Not highlighting all matching tokens. If tokenOrArray is a string, use
947 // that as the token. If it is an array, use the first element in the
948 // array.
949 // TODO(user): why is this this way?. We should match against all
950 // tokens in the array, but only accept the first match.
951 if (goog.isArray(tokenOrArray)) {
952 token = tokenOrArray.length > 0 ?
953 goog.string.regExpEscape(tokenOrArray[0]) : '';
954 } else {
955 // For the single-match string token, we refuse to match anything if
956 // the string begins with a non-word character, as matches by definition
957 // can only occur at the start of a word. (This also handles the
958 // goog.string.isEmptyOrWhitespace(goog.string.makeSafe(tokenOrArray))
959 // case.)
960 if (!/^\W/.test(tokenOrArray)) {
961 token = goog.string.regExpEscape(tokenOrArray);
962 }
963 }
964 }
965
966 return token;
967};
