40/**
41 * Creates a new drag and drop detector.
42 * @param {string=} opt_filePath The URL of the page to use for the detector.
43 * It should contain the same contents as dragdropdetector_target.html in
44 * the demos directory.
45 * @constructor
46 * @extends {goog.events.EventTarget}
47 * @final
48 */
49goog.ui.DragDropDetector = function(opt_filePath) {
50 goog.ui.DragDropDetector.base(this, 'constructor');
51
52 var iframe = goog.dom.createDom(goog.dom.TagName.IFRAME, {
53 'frameborder': 0
54 });
55 // In Firefox, we do all drop detection with an IFRAME. In IE, we only use
56 // the IFRAME to capture copied, non-linked images. (When we don't need it,
57 // we put a text INPUT before it and push it off screen.)
58 iframe.className = goog.userAgent.IE ?
59 goog.getCssName(
60 goog.ui.DragDropDetector.BASE_CSS_NAME_, 'ie-editable-iframe') :
61 goog.getCssName(
62 goog.ui.DragDropDetector.BASE_CSS_NAME_, 'w3c-editable-iframe');
63 iframe.src = opt_filePath || goog.ui.DragDropDetector.DEFAULT_FILE_PATH_;
64
65 this.element_ = /** @type {!HTMLIFrameElement} */ (iframe);
66
67 this.handler_ = new goog.events.EventHandler(this);
68 this.handler_.listen(iframe, goog.events.EventType.LOAD, this.initIframe_);
69
70 if (goog.userAgent.IE) {
71 // In IE, we have to bounce between an INPUT for catching links and an
72 // IFRAME for catching images.
73 this.textInput_ = goog.dom.createDom(goog.dom.TagName.INPUT, {
74 'type': goog.dom.InputType.TEXT,
75 'className': goog.getCssName(
76 goog.ui.DragDropDetector.BASE_CSS_NAME_, 'ie-input')
77 });
78
79 this.root_ = goog.dom.createDom(goog.dom.TagName.DIV,
80 goog.getCssName(goog.ui.DragDropDetector.BASE_CSS_NAME_, 'ie-div'),
81 this.textInput_, iframe);
82 } else {
83 this.root_ = iframe;
84 }
85
86 document.body.appendChild(this.root_);
87};
88goog.inherits(goog.ui.DragDropDetector, goog.events.EventTarget);

254/**
255 * Initialize the iframe after it has loaded.
256 * @private
257 */
258goog.ui.DragDropDetector.prototype.initIframe_ = function() {
259 // Set up a holder for position data.
260 this.mousePosition_ = new goog.math.Coordinate(
261 goog.ui.DragDropDetector.INIT_POSITION,
262 goog.ui.DragDropDetector.INIT_POSITION);
263
264 // Set up pointers to the important parts of the IFrame.
265 this.window_ = this.element_.contentWindow;
266 this.document_ = this.window_.document;
267 this.body_ = this.document_.body;
268
269 if (goog.userAgent.GECKO) {
270 this.document_.designMode = 'on';
271 } else if (!goog.userAgent.IE) {
272 // Bug 1667110
273 // In IE, we only set the IFrame body as content-editable when we bring it
274 // into view at the top of the page. Otherwise it may take focus when the
275 // page is loaded, scrolling the user far offscreen.
276 // Note that this isn't easily unit-testable, since it depends on a
277 // browser-specific behavior with content-editable areas.
278 this.body_.contentEditable = true;
279 }
280
281 this.handler_.listen(document.body, goog.events.EventType.DRAGENTER,
282 this.coverScreen_);
283
284 if (goog.userAgent.IE) {
285 // IE only events.
286 // Set up events on the IFrame.
287 this.handler_.
288 listen(this.body_,
289 [goog.events.EventType.DRAGENTER, goog.events.EventType.DRAGOVER],
290 goog.ui.DragDropDetector.enforceCopyEffect_).
291 listen(this.body_, goog.events.EventType.MOUSEOUT,
292 this.switchToInput_).
293 listen(this.body_, goog.events.EventType.DRAGLEAVE,
294 this.uncoverScreen_).
295 listen(this.body_, goog.ui.DragDropDetector.DROP_EVENT_TYPE_,
296 function(e) {
297 this.trackMouse_(e);
298
299 // The drop event occurs before the content is added to the
300 // iframe. We setTimeout so that handleNodeInserted_ is called
301 // after the content is in the document.
302 goog.global.setTimeout(
303 goog.bind(this.handleNodeInserted_, this, e), 0);
304 return true;
305 }).
306
307 // Set up events on the DIV.
308 listen(this.root_,
309 [goog.events.EventType.DRAGENTER, goog.events.EventType.DRAGOVER],
310 this.handleNewDrag_).
311 listen(this.root_,
312 [
313 goog.events.EventType.MOUSEMOVE,
314 goog.events.EventType.KEYPRESS
315 ], this.uncoverScreen_).
316
317 // Set up events on the text INPUT.
318 listen(this.textInput_, goog.events.EventType.DRAGOVER,
319 goog.events.Event.preventDefault).
320 listen(this.textInput_, goog.ui.DragDropDetector.DROP_EVENT_TYPE_,
321 this.handleInputDrop_);
322 } else {
323 // W3C events.
324 this.handler_.
325 listen(this.body_, goog.ui.DragDropDetector.DROP_EVENT_TYPE_,
326 function(e) {
327 this.trackMouse_(e);
328 this.uncoverScreen_();
329 }).
330 listen(this.body_,
331 [goog.events.EventType.MOUSEMOVE, goog.events.EventType.KEYPRESS],
332 this.uncoverScreen_).
333
334 // Detect content insertion.
335 listen(this.document_, 'DOMNodeInserted',
336 this.handleNodeInserted_);
337 }
338};

341/**
342 * Enforce that anything dragged over the IFRAME is copied in to it, rather
343 * than making it navigate to a different URL.
344 * @param {goog.events.BrowserEvent} e The event to enforce copying on.
345 * @private
346 */
347goog.ui.DragDropDetector.enforceCopyEffect_ = function(e) {
348 var event = e.getBrowserEvent();
349 // This function is only called on IE.
350 if (event.dataTransfer.dropEffect.toLowerCase() != 'copy') {
351 event.dataTransfer.dropEffect = 'copy';
352 }

356/**
357 * Cover the screen with the iframe.
358 * @param {goog.events.BrowserEvent} e The event that caused this function call.
359 * @private
360 */
361goog.ui.DragDropDetector.prototype.coverScreen_ = function(e) {
362 // Don't do anything if the drop effect is 'none' and we are in IE.
363 // It is set to 'none' in cases like dragging text inside a text area.
364 if (goog.userAgent.IE &&
365 e.getBrowserEvent().dataTransfer.dropEffect == 'none') {
366 return;
367 }
368
369 if (!this.isCoveringScreen_) {
370 this.isCoveringScreen_ = true;
371 if (goog.userAgent.IE) {
372 goog.style.setStyle(this.root_, 'top', '0');
373 this.body_.contentEditable = true;
374 this.switchToInput_(e);
375 } else {
376 goog.style.setStyle(this.root_, 'height', '5000px');
377 }
378 }
379};

382/**
383 * Uncover the screen.
384 * @private
385 */
386goog.ui.DragDropDetector.prototype.uncoverScreen_ = function() {
387 if (this.isCoveringScreen_) {
388 this.isCoveringScreen_ = false;
389 if (goog.userAgent.IE) {
390 this.body_.contentEditable = false;
391 goog.style.setStyle(this.root_, 'top', '-5000px');
392 } else {
393 goog.style.setStyle(this.root_, 'height', '10px');
394 }
395 }

426/**
427 * Handle a new drag event.
428 * @param {goog.events.BrowserEvent} e The event object.
429 * @return {boolean|undefined} Returns false in IE to cancel the event.
430 * @private
431 */
432goog.ui.DragDropDetector.prototype.handleNewDrag_ = function(e) {
433 var event = e.getBrowserEvent();
434
435 // This is only called on IE.
436 if (event.dataTransfer.dropEffect == 'link') {
437 this.switchToInput_(e);
438 e.preventDefault();
439 return false;
440 }
441
442 // Things that aren't links can be placed in the contentEditable iframe.
443 this.switchToIframe_(e);
444
445 // No need to return true since for events return true is the same as no
446 // return.
447};

450/**
451 * Handle mouse tracking.
452 * @param {goog.events.BrowserEvent} e The event object.
453 * @private
454 */
455goog.ui.DragDropDetector.prototype.trackMouse_ = function(e) {
456 this.mousePosition_.x = e.clientX;
457 this.mousePosition_.y = e.clientY;
458
459 // Check if the event is coming from within the iframe.
460 if (goog.dom.getOwnerDocument(/** @type {Node} */ (e.target)) != document) {
461 var iframePosition = goog.style.getClientPosition(this.element_);
462 this.mousePosition_.x += iframePosition.x;
463 this.mousePosition_.y += iframePosition.y;
464 }

482/**
483 * Clear the contents of the iframe.
484 * @private
485 */
486goog.ui.DragDropDetector.prototype.clearContents_ = function() {
487 if (goog.userAgent.WEBKIT) {
488 // Since this is called on a mutation event for the nodes we are going to
489 // clear, calling this right away crashes some versions of WebKit. Wait
490 // until the events are finished.
491 goog.global.setTimeout(goog.bind(function() {
492 goog.dom.setTextContent(this, '');
493 }, this.body_), 0);
494 } else {
495 this.document_.execCommand('selectAll', false, null);
496 this.document_.execCommand('delete', false, null);
497 this.document_.execCommand('selectAll', false, null);
498 }
499};

502/**
503 * Event handler called when the content of the iframe changes.
504 * @param {goog.events.BrowserEvent} e The event that caused this function call.
505 * @private
506 */
507goog.ui.DragDropDetector.prototype.handleNodeInserted_ = function(e) {
508 var uri;
509
510 if (this.body_.innerHTML.indexOf('<') == -1) {
511 // If the document contains no tags (i.e. is just text), try it out.
512 uri = goog.string.trim(goog.dom.getTextContent(this.body_));
513
514 // See if it looks kind of like a url.
515 if (!uri.match(goog.ui.DragDropDetector.URL_LIKE_REGEX_)) {
516 uri = null;
517 }
518 }
519
520 if (!uri) {
521 var imgs = this.body_.getElementsByTagName(goog.dom.TagName.IMG);
522 if (imgs && imgs.length) {
523 // TODO(robbyw): Grab all the images, instead of just the first.
524 var img = imgs[0];
525 uri = img.src;
526 }
527 }
528
529 if (uri) {
530 var specialCases = goog.ui.DragDropDetector.SPECIAL_CASE_URLS_;
531 var len = specialCases.length;
532 for (var i = 0; i < len; i++) {
533 var specialCase = specialCases[i];
534 if (uri.match(specialCase.regex)) {
535 alert(specialCase.message);
536 break;
537 }
538 }
539
540 // If no special cases matched, add the image.
541 if (i == len) {
542 this.dispatchEvent(
543 new goog.ui.DragDropDetector.ImageDropEvent(
544 uri, this.mousePosition_));
545 return;
546 }
547 }
548
549 var links = this.body_.getElementsByTagName(goog.dom.TagName.A);
550 if (links) {
551 for (i = 0, len = links.length; i < len; i++) {
552 this.dispatchEvent(
553 new goog.ui.DragDropDetector.LinkDropEvent(links[i].href));
554 }
555 }
556
557 this.clearContents_();
558 this.uncoverScreen_();
559};
