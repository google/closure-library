15/**
16 * @fileoverview Tree-like drilldown components for HTML tables.
17 *
18 * This component supports expanding and collapsing groups of rows in
19 * HTML tables. The behavior is like typical Tree widgets, but tables
20 * need special support to enable the tree behaviors.
21 *
22 * Any row or rows in an HTML table can be DrilldownRows. The root
23 * DrilldownRow nodes are always visible in the table, but the rest show
24 * or hide as input events expand and collapse their ancestors.
25 *
26 * Programming them: Top-level DrilldownRows are made by decorating
27 * a TR element. Children are made with addChild or addChildAt, and
28 * are entered into the document by the render() method.
29 *
30 * A DrilldownRow can have any number of children. If it has no children
31 * it can be loaded, not loaded, or with a load in progress.
32 * Top-level DrilldownRows are always displayed (though setting
33 * style.display on a containing DOM node could make one be not
34 * visible to the user). A DrilldownRow can be expanded, or not. A
35 * DrilldownRow displays if all of its ancestors are expanded.
36 *
37 * Set up event handlers and style each row for the application in an
38 * enterDocument method.
39 *
40 * Children normally render into the document lazily, at the first
41 * moment when all ancestors are expanded.
42 *
43 * @see ../demos/drilldownrow.html
44 */

51// The render method is redefined in this class because when addChildAt renders
52// the new child it assumes that the child's DOM node will be a child
53// of the parent component's DOM node, but all DOM nodes of DrilldownRows
54// in the same tree of DrilldownRows are siblings to each other.
55//
56// Arguments (or lack of arguments) to the render methods in Component
57// all determine the place of the new DOM node in the DOM tree, but
58// the place of a new DrilldownRow in the DOM needs to be determined by
59// its position in the tree of DrilldownRows.

74/**
75 * Builds a DrilldownRow component, which can overlay a tree
76 * structure onto sections of an HTML table.
77 *
78 * @param {!goog.ui.DrilldownRow.DrilldownRowProperties=} opt_properties
79 * Optional properties.
80 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
81 * @constructor
82 * @extends {goog.ui.Component}
83 * @final
84 */
85goog.ui.DrilldownRow = function(opt_properties, opt_domHelper) {
86 goog.ui.Component.call(this, opt_domHelper);
87 var properties = opt_properties || {};
88
89 // Initialize instance variables.
90
91 var html;
92 if (!goog.isDefAndNotNull(properties.html)) {
93 html = goog.html.SafeHtml.EMPTY;
94 } else {
95 goog.asserts.assert(properties.html instanceof goog.html.SafeHtml);
96 html = properties.html;
97 }
98 /**
99 * String of HTML to initialize the DOM structure for the table row.
100 * Should have the form '<tr attr="etc">Row contents here</tr>'.
101 * @type {!goog.html.SafeHtml}
102 * @private
103 */
104 this.html_ = html;
105
106 /**
107 * Controls whether this component's children will show when it shows.
108 * @type {boolean}
109 * @private
110 */
111 this.expanded_ = typeof properties.expanded != 'undefined' ?
112 properties.expanded : true;
113
114 /**
115 * If this component's DOM element is created from a string of
116 * HTML, this is the function to call when it is entered into the DOM tree.
117 * @type {Function} args are DrilldownRow and goog.events.EventHandler
118 * of the DrilldownRow.
119 * @private
120 */
121 this.decoratorFn_ = properties.decorator || goog.ui.DrilldownRow.decorate;
122
123 /**
124 * Is the DrilldownRow to be displayed? If it is rendered, this mirrors
125 * the style.display of the DrilldownRow's row.
126 * @type {boolean}
127 * @private
128 */
129 this.displayed_ = true;
130};
131goog.inherits(goog.ui.DrilldownRow, goog.ui.Component);

165/**
166 * Builds a DrilldownRow component. This function exists for
167 * backwards-compatibility only and uses goog.html.legacyconversions. For all
168 * new code use the goog.ui.DrilldownRow() constructor.
169 * @param {goog.ui.DrilldownRow.DrilldownRowPropertiesUnsafe=} opt_properties
170 * Optional properties.
171 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
172 * @return {!goog.ui.DrilldownRow}

174 *
175 */
176goog.ui.DrilldownRow.unsafeCreate = function(opt_properties, opt_domHelper) {
177 var properties = {};
178 if (opt_properties) {
179 properties.loaded = opt_properties.loaded;
180 properties.expanded = opt_properties.expanded;
181 if (goog.isString(opt_properties.html)) {

183 opt_properties.html);
184 }
185 properties.decorator = opt_properties.decorator;
186 }
187 return new goog.ui.DrilldownRow(properties, opt_domHelper);

276/**
277 * Rendering of DrilldownRow's is on need, do not call this directly
278 * from application code.
279 *
280 * Rendering a DrilldownRow places it according to its position in its
281 * tree of DrilldownRows. DrilldownRows cannot be placed any other
282 * way so this method does not use any arguments. This does not call
283 * the base class method and does not modify any of this
284 * DrilldownRow's children.
285 * @override
286 */
287goog.ui.DrilldownRow.prototype.render = function() {
288 if (arguments.length) {
289 throw Error('A DrilldownRow cannot be placed under a specific parent.');
290 } else {
291 var parent = this.getParent();
292 if (!parent.isInDocument()) {
293 throw Error('Cannot render child of un-rendered parent');
294 }
295 // The new child's TR node needs to go just after the last TR
296 // of the part of the parent's subtree that is to the left
297 // of this. The subtree includes the parent.
298 goog.asserts.assertInstanceof(parent, goog.ui.DrilldownRow);
299 var previous = parent.previousRenderedChild_(this);
300 var row;
301 if (previous) {
302 goog.asserts.assertInstanceof(previous, goog.ui.DrilldownRow);
303 row = previous.lastRenderedLeaf_().getElement();
304 } else {
305 row = parent.getElement();
306 }
307 row = /** @type {Element} */ (row.nextSibling);
308 // Render the child row component into the document.
309 if (row) {
310 this.renderBefore(row);
311 } else {
312 // Render at the end of the parent of this DrilldownRow's
313 // DOM element.
314 var tbody = /** @type {Element} */ (parent.getElement().parentNode);
315 goog.ui.DrilldownRow.superClass_.render.call(this, tbody);
316 }
317 }
318};

351/**
352 * Sets the expanded state of this DrilldownRow: makes all children
353 * displayable or not displayable corresponding to the expanded state.
354 *
355 * @param {boolean} expanded whether this should be expanded or not.
356 */
357goog.ui.DrilldownRow.prototype.setExpanded = function(expanded) {
358 if (expanded != this.expanded_) {
359 this.expanded_ = expanded;
360 var elem = this.getElement();
361 goog.asserts.assert(elem);
362 goog.dom.classlist.toggle(elem,
363 goog.getCssName('goog-drilldown-expanded'));
364 goog.dom.classlist.toggle(elem,
365 goog.getCssName('goog-drilldown-collapsed'));
366 if (this.isVisible_()) {
367 this.forEachChild(function(child) {
368 child.setDisplayable_(expanded);
369 });
370 }
371 }
372};

388/**
389 * This static function is a default decorator that adds HTML at the
390 * beginning of the first cell to display indentation and an expander
391 * image; sets up a click handler on the toggler; initializes a class
392 * for the row: either goog-drilldown-expanded or
393 * goog-drilldown-collapsed, depending on the initial state of the
394 * DrilldownRow; and sets up a click event handler on the toggler
395 * element.
396 *
397 * This creates a DIV with class=toggle. Your application can set up
398 * CSS style rules something like this:
399 *
400 * tr.goog-drilldown-expanded .toggle {
401 * background-image: url('minus.png');
402 * }
403 *
404 * tr.goog-drilldown-collapsed .toggle {
405 * background-image: url('plus.png');
406 * }
407 *
408 * These background images show whether the DrilldownRow is expanded.
409 *
410 * @param {goog.ui.DrilldownRow} selfObj DrilldownRow to be decorated.
411 */
412goog.ui.DrilldownRow.decorate = function(selfObj) {
413 var depth = selfObj.getDepth();
414 var row = selfObj.getElement();
415 goog.asserts.assert(row);
416 if (!row.cells) {
417 throw Error('No cells');
418 }
419 var cell = row.cells[0];
420 var html = '<div style="float: left; width: ' + depth +
421 'em;"><div class=toggle style="width: 1em; float: right;">' +
422 '&nbsp;</div></div>';
423 var fragment = selfObj.getDomHelper().htmlToDocumentFragment(html);
424 cell.insertBefore(fragment, cell.firstChild);
425 goog.dom.classlist.add(row, selfObj.isExpanded() ?
426 goog.getCssName('goog-drilldown-expanded') :
427 goog.getCssName('goog-drilldown-collapsed'));
428 // Default mouse event handling:
429 var toggler = fragment.getElementsByTagName(goog.dom.TagName.DIV)[0];
430 var key = selfObj.getHandler().listen(toggler, 'click', function(event) {
431 selfObj.setExpanded(!selfObj.isExpanded());
432 });
433};

441/**
442 * Turn display of a DrilldownRow on or off. If the DrilldownRow has not
443 * yet been rendered, this renders it. This propagates the effect
444 * of the change recursively as needed -- children displaying iff the
445 * parent is displayed and expanded.
446 *
447 * @param {boolean} display state, true iff display is desired.
448 * @private
449 */
450goog.ui.DrilldownRow.prototype.setDisplayable_ = function(display) {
451 if (display && !this.isInDocument()) {
452 this.render();
453 }
454 if (this.displayed_ == display) {
455 return;
456 }
457 this.displayed_ = display;
458 if (this.isInDocument()) {
459 this.getElement().style.display = display ? '' : 'none';
460 }
461 var selfObj = this;
462 this.forEachChild(function(child) {
463 child.setDisplayable_(display && selfObj.expanded_);
464 });
465};

468/**
469 * True iff this and all its DrilldownRow parents are displayable. The
470 * value is an approximation to actual visibility, since it does not
471 * look at whether DOM nodes containing the top-level component have
472 * display: none, visibility: hidden or are otherwise not displayable.
473 * So this visibility is relative to the top-level component.
474 *
475 * @return {boolean} visibility of this relative to its top-level drilldown.
476 * @private
477 */
478goog.ui.DrilldownRow.prototype.isVisible_ = function() {
479 for (var component = this;
480 component instanceof goog.ui.DrilldownRow;
481 component = component.getParent()) {
482 if (!component.displayed_)
483 return false;
484 }
485 return true;
486};

489/**
490 * Create and return a TR element from HTML that looks like
491 * "<tr> ... </tr>".
492 *
493 * @param {!goog.html.SafeHtml} html for one row.
494 * @param {Document} doc object to hold the Element.
495 * @return {Element} table row node created from the HTML.
496 * @private
497 */
498goog.ui.DrilldownRow.createRowNode_ = function(html, doc) {
499 // Note: this may be slow.
500 var tableHtml = goog.html.SafeHtml.create(goog.dom.TagName.TABLE, {}, html);
501 var div = doc.createElement(goog.dom.TagName.DIV);
502 goog.dom.safe.setInnerHtml(div, tableHtml);
503 return div.firstChild.rows[0];

507/**
508 * Get the recursively rightmost child that is in the document.
509 *
510 * @return {goog.ui.DrilldownRow} rightmost child currently entered in
511 * the document, potentially this DrilldownRow. If this is in the
512 * document, result is non-null.
513 * @private
514 */
515goog.ui.DrilldownRow.prototype.lastRenderedLeaf_ = function() {
516 var leaf = null;
517 for (var node = this;
518 node && node.isInDocument();
519 // Node will become undefined if parent has no children.
520 node = node.getChildAt(node.getChildCount() - 1)) {
521 leaf = node;
522 }
523 return /** @type {goog.ui.DrilldownRow} */ (leaf);
524};

527/**
528 * Search this node's direct children for the last one that is in the
529 * document and is before the given child.
530 * @param {goog.ui.DrilldownRow} child The child to stop the search at.
531 * @return {goog.ui.Component?} The last child component before the given child
532 * that is in the document.
533 * @private
534 */
535goog.ui.DrilldownRow.prototype.previousRenderedChild_ = function(child) {
536 for (var i = this.getChildCount() - 1; i >= 0; i--) {
537 if (this.getChildAt(i) == child) {
538 for (var j = i - 1; j >= 0; j--) {
539 var prev = this.getChildAt(j);
540 if (prev.isInDocument()) {
541 return prev;
542 }
543 }
544 }
545 }
546 return null;
