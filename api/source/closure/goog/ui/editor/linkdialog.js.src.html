316/** @override */
317goog.ui.editor.LinkDialog.prototype.show = function() {
318 goog.ui.editor.LinkDialog.base(this, 'show');
319
320
321 this.selectAppropriateTab_(this.textToDisplayInput_.value,
322 this.getTargetUrl_());
323 this.syncOkButton_();
324
325 if (this.showOpenLinkInNewWindow_) {
326 if (!this.targetLink_.isNew()) {
327 // If link is not new, checkbox should reflect current target.
328 this.isOpenLinkInNewWindowChecked_ =
329 this.targetLink_.getAnchor().target == '_blank';
330 }
331 this.openInNewWindowCheckbox_.checked = this.isOpenLinkInNewWindowChecked_;
332 }
333
334 if (this.showRelNoFollow_) {
335 this.relNoFollowCheckbox_.checked =
336 goog.ui.editor.LinkDialog.hasNoFollow(this.targetLink_.getAnchor().rel);
337 }
338};

384/**
385 * Checks if {@code str} contains {@code "nofollow"} as a separate word.
386 * @param {string} str String to be tested. This is usually {@code rel}
387 * attribute of an {@code HTMLAnchorElement} object.
388 * @return {boolean} {@code true} if {@code str} contains {@code nofollow}.
389 */
390goog.ui.editor.LinkDialog.hasNoFollow = function(str) {
391 return goog.ui.editor.LinkDialog.NO_FOLLOW_REGEX_.test(str);
392};
393
394
395/**
396 * Removes {@code "nofollow"} from {@code rel} if it's present as a separate
397 * word.
398 * @param {string} rel Input string. This is usually {@code rel} attribute of
399 * an {@code HTMLAnchorElement} object.
400 * @return {string} {@code rel} with any {@code "nofollow"} removed.
401 */
402goog.ui.editor.LinkDialog.removeNoFollow = function(rel) {
403 return rel.replace(goog.ui.editor.LinkDialog.NO_FOLLOW_REGEX_, '');
404};

450/**
451 * Creates contents of this dialog.
452 * @return {!Element} Contents of the dialog as a DOM element.
453 * @private
454 */
455goog.ui.editor.LinkDialog.prototype.createDialogContent_ = function() {
456 this.textToDisplayDiv_ = /** @type {!HTMLDivElement} */(
457 this.buildTextToDisplayDiv_());
458 var content = this.dom.createDom(goog.dom.TagName.DIV, null,
459 this.textToDisplayDiv_);
460
461 this.tabPane_ = new goog.ui.editor.TabPane(this.dom,
462 goog.ui.editor.messages.MSG_LINK_TO);
463 this.registerDisposable(this.tabPane_);
464 this.tabPane_.addTab(goog.ui.editor.LinkDialog.Id_.ON_WEB_TAB,
465 goog.ui.editor.messages.MSG_ON_THE_WEB,
466 goog.ui.editor.messages.MSG_ON_THE_WEB_TIP,
467 goog.ui.editor.LinkDialog.BUTTON_GROUP_,
468 this.buildTabOnTheWeb_());
469 this.tabPane_.addTab(goog.ui.editor.LinkDialog.Id_.EMAIL_ADDRESS_TAB,
470 goog.ui.editor.messages.MSG_EMAIL_ADDRESS,
471 goog.ui.editor.messages.MSG_EMAIL_ADDRESS_TIP,
472 goog.ui.editor.LinkDialog.BUTTON_GROUP_,
473 this.buildTabEmailAddress_());
474 this.tabPane_.render(content);
475
476 this.eventHandler_.listen(this.tabPane_, goog.ui.Component.EventType.SELECT,
477 this.onChangeTab_);
478
479 if (this.showOpenLinkInNewWindow_) {
480 content.appendChild(this.buildOpenInNewWindowDiv_());
481 }
482 if (this.showRelNoFollow_) {
483 content.appendChild(this.buildRelNoFollowDiv_());
484 }
485
486 return content;
487};

490/**
491 * Builds and returns the text to display section of the edit link dialog.
492 * @return {!Element} A div element to be appended into the dialog div.
493 * @private
494 */
495goog.ui.editor.LinkDialog.prototype.buildTextToDisplayDiv_ = function() {
496 var table = this.dom.createTable(1, 2);
497 table.cellSpacing = '0';
498 table.cellPadding = '0';
499 table.style.fontSize = '10pt';
500 // Build the text to display input.
501 var textToDisplayDiv = this.dom.createDom(goog.dom.TagName.DIV);
502 var html = goog.html.SafeHtml.create('span', {
503 'style': {
504 'position': 'relative',
505 'bottom': '2px',
506 'padding-right': '1px',
507 'white-space': 'nowrap'
508 },
509 id: goog.ui.editor.LinkDialog.Id_.TEXT_TO_DISPLAY_LABEL
510 }, [goog.ui.editor.messages.MSG_TEXT_TO_DISPLAY, goog.string.Unicode.NBSP]);
511 goog.dom.safe.setInnerHtml(table.rows[0].cells[0], html);
512 this.textToDisplayInput_ = /** @type {!HTMLInputElement} */(
513 this.dom.createDom(goog.dom.TagName.INPUT,
514 {id: goog.ui.editor.LinkDialog.Id_.TEXT_TO_DISPLAY}));
515 var textInput = this.textToDisplayInput_;
516 // 98% prevents scroll bars in standards mode.
517 // TODO(robbyw): Is this necessary for quirks mode?
518 goog.style.setStyle(textInput, 'width', '98%');
519 goog.style.setStyle(table.rows[0].cells[1], 'width', '100%');
520 goog.dom.appendChild(table.rows[0].cells[1], textInput);
521
522 goog.a11y.aria.setState(/** @type {!Element} */ (textInput),
523 goog.a11y.aria.State.LABELLEDBY,
524 goog.ui.editor.LinkDialog.Id_.TEXT_TO_DISPLAY_LABEL);
525 textInput.value = this.targetLink_.getCurrentText();
526
527 this.textInputHandler_ = new goog.events.InputHandler(textInput);
528 this.registerDisposable(this.textInputHandler_);
529 this.eventHandler_.listen(this.textInputHandler_,
530 goog.events.InputHandler.EventType.INPUT, this.onTextToDisplayEdit_);
531
532 goog.dom.appendChild(textToDisplayDiv, table);
533 return textToDisplayDiv;
534};

579/**
580* Builds and returns the div containing the tab "On the web".
581* @return {!Element} The div element containing the tab.
582* @private
583*/
584goog.ui.editor.LinkDialog.prototype.buildTabOnTheWeb_ = function() {
585 var onTheWebDiv = this.dom.createElement(goog.dom.TagName.DIV);
586
587 var headingDiv = this.dom.createDom(goog.dom.TagName.DIV, {},
588 this.dom.createDom(goog.dom.TagName.B, {},
589 goog.ui.editor.messages.MSG_WHAT_URL));
590 var urlInput = this.dom.createDom(goog.dom.TagName.INPUT,
591 {
592 id: goog.ui.editor.LinkDialog.Id_.ON_WEB_INPUT,
593 className: goog.ui.editor.LinkDialog.TARGET_INPUT_CLASSNAME_
594 });
595 goog.a11y.aria.setState(urlInput,
596 goog.a11y.aria.State.LABELLEDBY,
597 goog.ui.editor.LinkDialog.Id_.ON_WEB_TAB);
598 // IE throws on unknown values for type, but IE10+ supports type=url
599 if (!goog.userAgent.IE || goog.userAgent.isVersionOrHigher('10')) {
600 // On browsers that support Web Forms 2.0, allow autocompletion of URLs.
601 urlInput.type = goog.dom.InputType.URL;
602 }
603
604 if (goog.editor.BrowserFeature.NEEDS_99_WIDTH_IN_STANDARDS_MODE &&
605 goog.editor.node.isStandardsMode(urlInput)) {
606 urlInput.style.width = '99%';
607 }
608
609 var inputDiv = this.dom.createDom(goog.dom.TagName.DIV, null, urlInput);
610
611 this.urlInputHandler_ = new goog.events.InputHandler(urlInput);
612 this.registerDisposable(this.urlInputHandler_);
613 this.eventHandler_.listen(this.urlInputHandler_,
614 goog.events.InputHandler.EventType.INPUT,
615 this.onUrlOrEmailInputChange_);
616
617 var testLink = new goog.ui.Button(goog.ui.editor.messages.MSG_TEST_THIS_LINK,
618 goog.ui.LinkButtonRenderer.getInstance(),
619 this.dom);
620 testLink.render(inputDiv);
621 testLink.getElement().style.marginTop = '1em';
622 this.eventHandler_.listen(testLink,
623 goog.ui.Component.EventType.ACTION,
624 this.onWebTestLink_);
625
626 // Build the "On the web" explanation text div.
627 var explanationDiv = this.dom.createDom(goog.dom.TagName.DIV,
628 goog.ui.editor.LinkDialog.EXPLANATION_TEXT_CLASSNAME_);
629 goog.dom.safe.setInnerHtml(explanationDiv,
630 goog.ui.editor.messages.getTrLinkExplanationSafeHtml());
631 onTheWebDiv.appendChild(headingDiv);
632 onTheWebDiv.appendChild(inputDiv);
633 onTheWebDiv.appendChild(explanationDiv);
634
635 return onTheWebDiv;
636};

692/**
693 * Returns the url that the target points to.
694 * @return {string} The url that the target points to.
695 * @private
696 */
697goog.ui.editor.LinkDialog.prototype.getTargetUrl_ = function() {
698 // Get the href-attribute through getAttribute() rather than the href property
699 // because Google-Toolbar on Firefox with "Send with Gmail" turned on
700 // modifies the href-property of 'mailto:' links but leaves the attribute
701 // untouched.
702 return this.targetLink_.getAnchor().getAttribute('href') || '';
703};

707 * Selects the correct tab based on the URL, and fills in its inputs.
708 * For new links, it suggests a url based on the link text.
709 * @param {string} text The inner text of the link.
710 * @param {string} url The href for the link.
711 * @private
712 */
713goog.ui.editor.LinkDialog.prototype.selectAppropriateTab_ = function(
714 text, url) {
715 if (this.isNewLink_()) {
716 // Newly created non-empty link: try to infer URL from the link text.
717 this.guessUrlAndSelectTab_(text);
718 } else if (goog.editor.Link.isMailto(url)) {
719 // The link is for an email.
720 this.tabPane_.setSelectedTabId(
721 goog.ui.editor.LinkDialog.Id_.EMAIL_ADDRESS_TAB);
722 this.dom.getElement(goog.ui.editor.LinkDialog.Id_.EMAIL_ADDRESS_INPUT)
723 .value = url.substring(url.indexOf(':') + 1);
724 this.setAutogenFlagFromCurInput_();
725 } else {
726 // No specific tab was appropriate, default to on the web tab.
727 this.tabPane_.setSelectedTabId(goog.ui.editor.LinkDialog.Id_.ON_WEB_TAB);
728 this.dom.getElement(goog.ui.editor.LinkDialog.Id_.ON_WEB_INPUT)
729 .value = this.isNewLink_() ? 'http://' : url;
730 this.setAutogenFlagFromCurInput_();
731 }
732};

735/**
736 * Select a url/tab based on the link's text. This function is simply
737 * the isNewLink_() == true case of selectAppropriateTab_().
738 * @param {string} text The inner text of the link.
739 * @private
740 */
741goog.ui.editor.LinkDialog.prototype.guessUrlAndSelectTab_ = function(text) {
742 if (goog.editor.Link.isLikelyEmailAddress(text)) {
743 // The text is for an email address.
744 this.tabPane_.setSelectedTabId(
745 goog.ui.editor.LinkDialog.Id_.EMAIL_ADDRESS_TAB);
746 this.dom.getElement(goog.ui.editor.LinkDialog.Id_.EMAIL_ADDRESS_INPUT)
747 .value = text;
748 this.setAutogenFlag_(true);
749 // TODO(user): Why disable right after enabling? What bug are we
750 // working around?
751 this.disableAutogenFlag_(true);
752 } else if (goog.editor.Link.isLikelyUrl(text)) {
753 // The text is for a web URL.
754 this.tabPane_.setSelectedTabId(goog.ui.editor.LinkDialog.Id_.ON_WEB_TAB);
755 this.dom.getElement(goog.ui.editor.LinkDialog.Id_.ON_WEB_INPUT)
756 .value = text;
757 this.setAutogenFlag_(true);
758 this.disableAutogenFlag_(true);
759 } else {
760 // No meaning could be deduced from text, choose a default tab.
761 if (!this.targetLink_.getCurrentText()) {
762 this.setAutogenFlag_(true);
763 }
764 this.tabPane_.setSelectedTabId(goog.ui.editor.LinkDialog.Id_.ON_WEB_TAB);
765 }
766};
