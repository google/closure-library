42/**
43 * Base class for modal popup UI components. This can also be used as
44 * a standalone component to render a modal popup with an empty div.
45 *
46 * WARNING: goog.ui.ModalPopup is only guaranteed to work when it is rendered
47 * directly in the 'body' element.
48 *
49 * The Html structure of the modal popup is:
50 * <pre>
51 * Element Function Class-name, goog-modalpopup = default
52 * ----------------------------------------------------------------------------
53 * - iframe Iframe mask goog-modalpopup-bg
54 * - div Background mask goog-modalpopup-bg
55 * - div Modal popup area goog-modalpopup
56 * - span Tab catcher
57 * </pre>
58 * @constructor
59 * @param {boolean=} opt_useIframeMask Work around windowed controls z-index
60 * issue by using an iframe instead of a div for bg element.
61 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper; see {@link
62 * goog.ui.Component} for semantics.
63 * @extends {goog.ui.Component}
64 */
65goog.ui.ModalPopup = function(opt_useIframeMask, opt_domHelper) {
66 goog.ui.ModalPopup.base(this, 'constructor', opt_domHelper);
67
68 /**
69 * Whether the modal popup should use an iframe as the background
70 * element to work around z-order issues.
71 * @type {boolean}
72 * @private
73 */
74 this.useIframeMask_ = !!opt_useIframeMask;
75
76 /**
77 * The element that had focus before the popup was displayed.
78 * @type {Element}
79 * @private
80 */
81 this.lastFocus_ = null;
82};
83goog.inherits(goog.ui.ModalPopup, goog.ui.Component);
84goog.tagUnsealableClass(goog.ui.ModalPopup);

227/**
228 * Creates and disposes of the DOM for background mask elements.
229 * @private
230 */
231goog.ui.ModalPopup.prototype.manageBackgroundDom_ = function() {
232 if (this.useIframeMask_ && !this.bgIframeEl_) {
233 // IE renders the iframe on top of the select elements while still
234 // respecting the z-index of the other elements on the page. See
235 // http://support.microsoft.com/kb/177378 for more information.
236 // Flash and other controls behave in similar ways for other browsers
237 this.bgIframeEl_ = goog.dom.iframe.createBlank(this.getDomHelper());
238 this.bgIframeEl_.className = goog.getCssName(this.getCssClass(), 'bg');
239 goog.style.setElementShown(this.bgIframeEl_, false);
240 goog.style.setOpacity(this.bgIframeEl_, 0);
241 }
242
243 // Create the backgound mask, initialize its opacity, and make sure it's
244 // hidden.
245 if (!this.bgEl_) {
246 this.bgEl_ = this.getDomHelper().createDom(
247 goog.dom.TagName.DIV, goog.getCssName(this.getCssClass(), 'bg'));
248 goog.style.setElementShown(this.bgEl_, false);
249 }
250};

269/**
270 * Allow a shift-tab from the top of the modal popup to the last tabbable
271 * element by moving focus to the tab catcher. This should be called after
272 * catching a wrapping shift-tab event and before allowing it to propagate, so
273 * that focus will land on the last tabbable element before the tab catcher.
274 * @protected
275 */
276goog.ui.ModalPopup.prototype.setupBackwardTabWrap = function() {
277 this.backwardTabWrapInProgress_ = true;
278 try {
279 this.tabCatcherElement_.focus();
280 } catch (e) {
281 // Swallow this. IE can throw an error if the element can not be focused.
282 }
283 // Reset the flag on a timer in case anything goes wrong with the followup
284 // event.
285 goog.Timer.callOnce(this.resetBackwardTabWrap_, 0, this);
286};

372/**
373 * Sets the visibility of the modal popup box and focus to the popup.
374 * @param {boolean} visible Whether the modal popup should be visible.
375 */
376goog.ui.ModalPopup.prototype.setVisible = function(visible) {
377 goog.asserts.assert(
378 this.isInDocument(), 'ModalPopup must be rendered first.');
379
380 if (visible == this.visible_) {
381 return;
382 }
383
384 if (this.popupShowTransition_) this.popupShowTransition_.stop();
385 if (this.bgShowTransition_) this.bgShowTransition_.stop();
386 if (this.popupHideTransition_) this.popupHideTransition_.stop();
387 if (this.bgHideTransition_) this.bgHideTransition_.stop();
388
389 if (this.isInDocument()) {
390 this.setA11YDetectBackground(visible);
391 }
392 if (visible) {
393 this.show_();
394 } else {
395 this.hide_();
396 }
397};

400/**
401 * Sets aria-hidden on the rest of the page to restrict screen reader focus.
402 * Top-level elements with an explicit aria-hidden state are not altered.
403 * @param {boolean} hide Whether to hide or show the rest of the page.
404 * @protected
405 */
406goog.ui.ModalPopup.prototype.setA11YDetectBackground = function(hide) {
407 if (hide) {
408 if (!this.hiddenElements_) {
409 this.hiddenElements_ = [];
410 }
411 var dom = this.getDomHelper();
412 var topLevelChildren = dom.getChildren(dom.getDocument().body);
413 for (var i = 0; i < topLevelChildren.length; i++) {
414 var child = topLevelChildren[i];
415 if (child != this.getElementStrict() &&
416 !goog.a11y.aria.getState(child, goog.a11y.aria.State.HIDDEN)) {
417 goog.a11y.aria.setState(child, goog.a11y.aria.State.HIDDEN, true);
418 this.hiddenElements_.push(child);
419 }
420 }
421 } else if (this.hiddenElements_) {
422 for (var i = 0; i < this.hiddenElements_.length; i++) {
423 goog.a11y.aria.removeState(
424 this.hiddenElements_[i], goog.a11y.aria.State.HIDDEN);
425 }
426 this.hiddenElements_ = null;
427 }

451/**
452 * Shows the popup.
453 * @private
454 */
455goog.ui.ModalPopup.prototype.show_ = function() {
456 if (!this.dispatchEvent(goog.ui.PopupBase.EventType.BEFORE_SHOW)) {
457 return;
458 }
459
460 try {
461 this.lastFocus_ = this.getDomHelper().getDocument().activeElement;
462 } catch (e) {
463 // Focus-related actions often throw exceptions.
464 // Sample past issue: https://bugzilla.mozilla.org/show_bug.cgi?id=656283
465 }
466 this.resizeBackground_();
467 this.reposition();
468
469 // Listen for keyboard and resize events while the modal popup is visible.
470 this.getHandler().listen(
471 this.getDomHelper().getWindow(), goog.events.EventType.RESIZE,
472 this.resizeBackground_);
473
474 this.showPopupElement_(true);
475 this.focus();
476 this.visible_ = true;
477
478 if (this.popupShowTransition_ && this.bgShowTransition_) {
479 goog.events.listenOnce(
480 /** @type {!goog.events.EventTarget} */ (this.popupShowTransition_),
481 goog.fx.Transition.EventType.END, this.onShow, false, this);
482 this.bgShowTransition_.play();
483 this.popupShowTransition_.play();
484 } else {
485 this.onShow();
486 }
487};

490/**
491 * Hides the popup.
492 * @private
493 */
494goog.ui.ModalPopup.prototype.hide_ = function() {
495 if (!this.dispatchEvent(goog.ui.PopupBase.EventType.BEFORE_HIDE)) {
496 return;
497 }
498
499 // Stop listening for keyboard and resize events while the modal
500 // popup is hidden.
501 this.getHandler().unlisten(
502 this.getDomHelper().getWindow(), goog.events.EventType.RESIZE,
503 this.resizeBackground_);
504
505 // Set visibility to hidden even if there is a transition. This
506 // reduces complexity in subclasses who may want to override
507 // setVisible (such as goog.ui.Dialog).
508 this.visible_ = false;
509
510 if (this.popupHideTransition_ && this.bgHideTransition_) {
511 goog.events.listenOnce(
512 /** @type {!goog.events.EventTarget} */ (this.popupHideTransition_),
513 goog.fx.Transition.EventType.END, this.onHide, false, this);
514 this.bgHideTransition_.play();
515 // The transition whose END event you are listening to must be played last
516 // to prevent errors when disposing on hide event, which occur on browsers
517 // that do not support CSS3 transitions.
518 this.popupHideTransition_.play();
519 } else {
520 this.onHide();
521 }
522
523 this.returnFocus_();
524};

527/**
528 * Attempts to return the focus back to the element that had it before the popup
529 * was opened.
530 * @private
531 */
532goog.ui.ModalPopup.prototype.returnFocus_ = function() {
533 try {
534 var dom = this.getDomHelper();
535 var body = dom.getDocument().body;
536 var active = dom.getDocument().activeElement || body;
537 if (!this.lastFocus_ || this.lastFocus_ == body) {
538 this.lastFocus_ = null;
539 return;
540 }
541 // We only want to move the focus if we actually have it, i.e.:
542 // - if we immediately hid the popup the focus should have moved to the
543 // body element
544 // - if there is a hiding transition in progress the focus would still be
545 // within the dialog and it is safe to move it if the current focused
546 // element is a child of the dialog
547 if (active == body || dom.contains(this.getElement(), active)) {
548 this.lastFocus_.focus();
549 }
550 } catch (e) {
551 // Swallow this. IE can throw an error if the element can not be focused.
552 }
553 // Explicitly want to null this out even if there was an error focusing to
554 // avoid bleed over between dialog invocations.
555 this.lastFocus_ = null;
556};

613/**
614 * Make the background element the size of the document.
615 *
616 * NOTE(user): We must hide the background element before measuring the
617 * document, otherwise the size of the background will stop the document from
618 * shrinking to fit a smaller window. This does cause a slight flicker in Linux
619 * browsers, but should not be a common scenario.
620 * @private
621 */
622goog.ui.ModalPopup.prototype.resizeBackground_ = function() {
623 if (this.bgIframeEl_) {
624 goog.style.setElementShown(this.bgIframeEl_, false);
625 }
626 if (this.bgEl_) {
627 goog.style.setElementShown(this.bgEl_, false);
628 }
629
630 var doc = this.getDomHelper().getDocument();
631 var win = goog.dom.getWindow(doc) || window;
632
633 // Take the max of document height and view height, in case the document does
634 // not fill the viewport. Read from both the body element and the html element
635 // to account for browser differences in treatment of absolutely-positioned
636 // content.
637 var viewSize = goog.dom.getViewportSize(win);
638 var w = Math.max(viewSize.width,
639 Math.max(doc.body.scrollWidth, doc.documentElement.scrollWidth));
640 var h = Math.max(viewSize.height,
641 Math.max(doc.body.scrollHeight, doc.documentElement.scrollHeight));
642
643 if (this.bgIframeEl_) {
644 goog.style.setElementShown(this.bgIframeEl_, true);
645 goog.style.setSize(this.bgIframeEl_, w, h);
646 }
647 if (this.bgEl_) {
648 goog.style.setElementShown(this.bgEl_, true);
649 goog.style.setSize(this.bgEl_, w, h);
650 }
651};

654/**
655 * Centers the modal popup in the viewport, taking scrolling into account.
656 */
657goog.ui.ModalPopup.prototype.reposition = function() {
658 // TODO(chrishenry): Make this use goog.positioning as in goog.ui.PopupBase?
659
660 // Get the current viewport to obtain the scroll offset.
661 var doc = this.getDomHelper().getDocument();
662 var win = goog.dom.getWindow(doc) || window;
663 if (goog.style.getComputedPosition(this.getElement()) == 'fixed') {
664 var x = 0;
665 var y = 0;
666 } else {
667 var scroll = this.getDomHelper().getDocumentScroll();
668 var x = scroll.x;
669 var y = scroll.y;
670 }
671
672 var popupSize = goog.style.getSize(this.getElement());
673 var viewSize = goog.dom.getViewportSize(win);
674
675 // Make sure left and top are non-negatives.
676 var left = Math.max(x + viewSize.width / 2 - popupSize.width / 2, 0);
677 var top = Math.max(y + viewSize.height / 2 - popupSize.height / 2, 0);
678 goog.style.setPosition(this.getElement(), left, top);
679
680 // We place the tab catcher at the same position as the dialog to
681 // prevent IE from scrolling when users try to tab out of the dialog.
682 goog.style.setPosition(this.tabCatcherElement_, left, top);
683};

686/**
687 * Handles focus events. Makes sure that if the user tabs past the
688 * elements in the modal popup, the focus wraps back to the beginning, and that
689 * if the user shift-tabs past the front of the modal popup, focus wraps around
690 * to the end.
691 * @param {goog.events.BrowserEvent} e Browser's event object.
692 * @protected
693 */
694goog.ui.ModalPopup.prototype.onFocus = function(e) {
695 if (this.backwardTabWrapInProgress_) {
696 this.resetBackwardTabWrap_();
697 } else if (e.target == this.tabCatcherElement_) {
698 goog.Timer.callOnce(this.focusElement_, 0, this);
699 }
