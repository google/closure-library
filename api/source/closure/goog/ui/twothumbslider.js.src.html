15/**
16 * @fileoverview Twothumbslider is a slider that allows to select a subrange
17 * within a range by dragging two thumbs. The selected sub-range is exposed
18 * through getValue() and getExtent().
19 *
20 * To decorate, the twothumbslider should be bound to an element with the class
21 * name 'goog-twothumbslider-[vertical / horizontal]' containing children with
22 * the classname 'goog-twothumbslider-value-thumb' and
23 * 'goog-twothumbslider-extent-thumb', respectively.
24 *
25 * Decorate Example:
26 * <div id="twothumbslider" class="goog-twothumbslider-horizontal">
27 * <div class="goog-twothumbslider-value-thumb">
28 * <div class="goog-twothumbslider-extent-thumb">
29 * </div>
30 * <script>
31 *
32 * var slider = new goog.ui.TwoThumbSlider;
33 * slider.decorate(document.getElementById('twothumbslider'));
34 *
35 * TODO(user): add a11y once we know what this element is
36 *
37 * @see ../demos/twothumbslider.html
38 */
39

50/**
51 * This creates a TwoThumbSlider object.
52 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
53 * @constructor
54 * @extends {goog.ui.SliderBase}
55 */
56goog.ui.TwoThumbSlider = function(opt_domHelper) {
57 goog.ui.SliderBase.call(this, opt_domHelper);
58 this.rangeModel.setValue(this.getMinimum());
59 this.rangeModel.setExtent(this.getMaximum() - this.getMinimum());
60};
61goog.inherits(goog.ui.TwoThumbSlider, goog.ui.SliderBase);
62goog.tagUnsealableClass(goog.ui.TwoThumbSlider);

110/**
111 * This creates a thumb element with the specified CSS class name.
112 * @param {string} cs CSS class name of the thumb to be created.
113 * @return {!HTMLDivElement} The created thumb element.
114 * @private
115 */
116goog.ui.TwoThumbSlider.prototype.createThumb_ = function(cs) {
117 var thumb = this.getDomHelper().createDom(goog.dom.TagName.DIV, cs);
118 goog.a11y.aria.setRole(thumb, goog.a11y.aria.Role.BUTTON);
119 return /** @type {!HTMLDivElement} */ (thumb);
120};

123/**
124 * Creates the thumb members for a twothumbslider. If the
125 * element contains a child with a class name 'goog-twothumbslider-value-thumb'
126 * (or 'goog-twothumbslider-extent-thumb', respectively), then that will be used
127 * as the valueThumb (or as the extentThumb, respectively). If the element
128 * contains a child with a class name 'goog-twothumbslider-rangehighlight',
129 * then that will be used as the range highlight.
130 * @override
131 */
132goog.ui.TwoThumbSlider.prototype.createThumbs = function() {
133 // find range highlight and thumbs
134 var valueThumb = goog.dom.getElementsByTagNameAndClass(
135 null, goog.ui.TwoThumbSlider.VALUE_THUMB_CSS_CLASS, this.getElement())[0];
136 var extentThumb = goog.dom.getElementsByTagNameAndClass(null,
137 goog.ui.TwoThumbSlider.EXTENT_THUMB_CSS_CLASS, this.getElement())[0];
138 var rangeHighlight = goog.dom.getElementsByTagNameAndClass(null,
139 goog.ui.TwoThumbSlider.RANGE_HIGHLIGHT_CSS_CLASS, this.getElement())[0];
140 if (!valueThumb) {
141 valueThumb =
142 this.createThumb_(goog.ui.TwoThumbSlider.VALUE_THUMB_CSS_CLASS);
143 this.getElement().appendChild(valueThumb);
144 }
145 if (!extentThumb) {
146 extentThumb =
147 this.createThumb_(goog.ui.TwoThumbSlider.EXTENT_THUMB_CSS_CLASS);
148 this.getElement().appendChild(extentThumb);
149 }
150 if (!rangeHighlight) {
151 rangeHighlight = this.getDomHelper().createDom(goog.dom.TagName.DIV,
152 goog.ui.TwoThumbSlider.RANGE_HIGHLIGHT_CSS_CLASS);
153 // Insert highlight before value thumb so that it renders under the thumbs.
154 this.getDomHelper().insertSiblingBefore(rangeHighlight, valueThumb);
155 }
156 this.valueThumb = valueThumb;
157 this.extentThumb = extentThumb;
158 this.rangeHighlight = rangeHighlight;
159};
