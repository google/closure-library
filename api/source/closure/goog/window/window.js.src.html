49/**
50 * Opens a new window.
51 *
52 * @param {string|Object} linkRef A string or an object that supports toString,
53 * for example goog.Uri. If this is an object with a 'href' attribute, such
54 * as HTMLAnchorElement, it will be used instead.
55 *
56 * @param {Object=} opt_options supports the following options:
57 * 'target': (string) target (window name). If null, linkRef.target will
58 * be used.
59 * 'width': (number) window width.
60 * 'height': (number) window height.
61 * 'top': (number) distance from top of screen
62 * 'left': (number) distance from left of screen
63 * 'toolbar': (boolean) show toolbar
64 * 'scrollbars': (boolean) show scrollbars
65 * 'location': (boolean) show location
66 * 'statusbar': (boolean) show statusbar
67 * 'menubar': (boolean) show menubar
68 * 'resizable': (boolean) resizable
69 * 'noreferrer': (boolean) whether to attempt to remove the referrer header
70 * from the request headers. Does this by opening a blank window that
71 * then redirects to the target url, so users may see some flickering.
72 *
73 * @param {Window=} opt_parentWin Parent window that should be used to open the
74 * new window.
75 *
76 * @return {Window} Returns the window object that was opened. This returns
77 * null if a popup blocker prevented the window from being
78 * opened. In case when a new window is opened in a different
79 * browser sandbox (such as iOS standalone mode), the returned
80 * object is a emulated Window object that functions as if
81 * a cross-origin window has been opened.
82 */
83goog.window.open = function(linkRef, opt_options, opt_parentWin) {
84 if (!opt_options) {
85 opt_options = {};
86 }
87 var parentWin = opt_parentWin || window;
88
89 // HTMLAnchorElement has a toString() method with the same behavior as
90 // goog.Uri in all browsers except for Safari, which returns
91 // '[object HTMLAnchorElement]'. We check for the href first, then
92 // assume that it's a goog.Uri or String otherwise.
93 var href = typeof linkRef.href != 'undefined' ? linkRef.href :
94 String(linkRef);
95 var target = opt_options.target || linkRef.target;
96
97 var sb = [];
98 for (var option in opt_options) {
99 switch (option) {
100 case 'width':
101 case 'height':
102 case 'top':
103 case 'left':
104 sb.push(option + '=' + opt_options[option]);
105 break;
106 case 'target':
107 case 'noreferrer':
108 break;
109 default:
110 sb.push(option + '=' + (opt_options[option] ? 1 : 0));
111 }
112 }
113 var optionString = sb.join(',');
114
115 var newWin;
116 if (goog.labs.userAgent.platform.isIos() &&
117 parentWin.navigator && parentWin.navigator['standalone'] &&
118 target && target != '_self') {
119 // iOS in standalone mode disregards "target" in window.open and always
120 // opens new URL in the same window. The workout around is to create an "A"
121 // element and send a click event to it.
122 // Notice that the "A" tag does NOT have to be added to the DOM.
123 var a = parentWin.document.createElement(goog.dom.TagName.A);
124 a.setAttribute('href', href);
125 a.setAttribute('target', target);
126 if (opt_options['noreferrer']) {
127 a.setAttribute('rel', 'noreferrer');
128 }
129 var click = document.createEvent('MouseEvent');
130 click.initMouseEvent('click',
131 true, // canBubble
132 true, // cancelable
133 parentWin,
134 1); // detail = mousebutton
135 a.dispatchEvent(click);
136 // New window is not available in this case. Instead, a fake Window object
137 // is returned. In particular, it will have window.document undefined. In
138 // general, it will appear to most of clients as a Window for a different
139 // origin. Since iOS standalone web apps are run in their own sandbox, this
140 // is the most appropriate return value.
141 newWin = /** @type {!Window} */ ({});
142 } else if (opt_options['noreferrer']) {
143 // Use a meta-refresh to stop the referrer from being included in the
144 // request headers.
145 newWin = parentWin.open('', target, optionString);
146 if (newWin) {
147 if (goog.userAgent.IE) {
148 // IE has problems parsing the content attribute if the url contains
149 // a semicolon. We can fix this by adding quotes around the url, but
150 // then we can't parse quotes in the URL correctly. We take a
151 // best-effort approach.
152 //
153 // If the URL has semicolons, wrap it in single quotes to protect
154 // the semicolons.
155 // If the URL has semicolons and single quotes, url-encode the single
156 // quotes as well.
157 //
158 // This is imperfect. Notice that both ' and ; are reserved characters
159 // in URIs, so this could do the wrong thing, but at least it will
160 // do the wrong thing in only rare cases.
161 // ugh.
162 if (href.indexOf(';') != -1) {
163 href = "'" + href.replace(/'/g, '%27') + "'";
164 }
165 }
166 newWin.opener = null;
167 href = goog.string.htmlEscape(href);
168 newWin.document.write('<META HTTP-EQUIV="refresh" content="0; url=' +
169 href + '">');
170 newWin.document.close();
171 }
172 } else {
173 newWin = parentWin.open(href, target, optionString);
174 }
175 // newWin is null if a popup blocker prevented the window open.
176 return newWin;
177};

180/**
181 * Opens a new window without any real content in it.
182 *
183 * This can be used to get around popup blockers if you need to open a window
184 * in response to a user event, but need to do asynchronous work to determine
185 * the URL to open, and then set the URL later.
186 *
187 * Example usage:
188 *
189 * var newWin = goog.window.openBlank('Loading...');
190 * setTimeout(
191 * function() {
192 * newWin.location.href = 'http://www.google.com';
193 * }, 100);
194 *
195 * @param {string=} opt_message String to show in the new window. This string
196 * will be HTML-escaped to avoid XSS issues.
197 * @param {Object=} opt_options Options to open window with.
198 * {@see goog.window.open for exact option semantics}.
199 * @param {Window=} opt_parentWin Parent window that should be used to open the
200 * new window.
201 * @return {Window} Returns the window object that was opened. This returns
202 * null if a popup blocker prevented the window from being
203 * opened.
204 */
205goog.window.openBlank = function(opt_message, opt_options, opt_parentWin) {
206
207 // Open up a window with the loading message and nothing else.
208 // This will be interpreted as HTML content type with a missing doctype
209 // and html/body tags, but is otherwise acceptable.
210 var loadingMessage = opt_message ? goog.string.htmlEscape(opt_message) : '';
211 return /** @type {Window} */ (goog.window.open(
212 'javascript:"' + encodeURI(loadingMessage) + '"',
213 opt_options, opt_parentWin));
214};

217/**
218 * Raise a help popup window, defaulting to "Google standard" size and name.
219 *
220 * (If your project is using GXPs, consider using {@link PopUpLink.gxp}.)
221 *
222 * @param {string|Object} linkRef if this is a string, it will be used as the
223 * URL of the popped window; otherwise it's assumed to be an HTMLAnchorElement
224 * (or some other object with "target" and "href" properties).
225 *
226 * @param {Object=} opt_options Options to open window with.
227 * {@see goog.window.open for exact option semantics}
228 * Additional wrinkles to the options:
229 * - if 'target' field is null, linkRef.target will be used. If *that's*
230 * null, the default is "google_popup".
231 * - if 'width' field is not specified, the default is 690.
232 * - if 'height' field is not specified, the default is 500.
233 *
234 * @return {boolean} true if the window was not popped up, false if it was.
235 */
236goog.window.popup = function(linkRef, opt_options) {
237 if (!opt_options) {
238 opt_options = {};
239 }
240
241 // set default properties
242 opt_options['target'] = opt_options['target'] ||
243 linkRef['target'] || goog.window.DEFAULT_POPUP_TARGET;
244 opt_options['width'] = opt_options['width'] ||
245 goog.window.DEFAULT_POPUP_WIDTH;
246 opt_options['height'] = opt_options['height'] ||
247 goog.window.DEFAULT_POPUP_HEIGHT;
248
249 var newWin = goog.window.open(linkRef, opt_options);
250 if (!newWin) {
251 return true;
252 }
253 newWin.focus();
254
255 return false;
256};
