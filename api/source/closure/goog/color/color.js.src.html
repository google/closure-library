57/**
58 * Parses a color out of a string.
59 * @param {string} str Color in some format.
60 * @return {{hex: string, type: string}} 'hex' is a string containing a hex
61 * representation of the color, 'type' is a string containing the type
62 * of color format passed in ('hex', 'rgb', 'named').
63 */
64goog.color.parse = function(str) {
65 var result = {};
66 str = String(str);
67
68 var maybeHex = goog.color.prependHashIfNecessaryHelper(str);
69 if (goog.color.isValidHexColor_(maybeHex)) {
70 result.hex = goog.color.normalizeHex(maybeHex);
71 result.type = 'hex';
72 return result;
73 } else {
74 var rgb = goog.color.isValidRgbColor_(str);
75 if (rgb.length) {
76 result.hex = goog.color.rgbArrayToHex(rgb);
77 result.type = 'rgb';
78 return result;
79 } else if (goog.color.names) {
80 var hex = goog.color.names[str.toLowerCase()];
81 if (hex) {
82 result.hex = hex;
83 result.type = 'named';
84 return result;
85 }
86 }
87 }
88 throw Error(str + ' is not a valid color string');
89};

92/**
93 * Determines if the given string can be parsed as a color.
94 * {@see goog.color.parse}.
95 * @param {string} str Potential color string.
96 * @return {boolean} True if str is in a format that can be parsed to a color.
97 */
98goog.color.isValidColor = function(str) {
99 var maybeHex = goog.color.prependHashIfNecessaryHelper(str);
100 return !!(goog.color.isValidHexColor_(maybeHex) ||
101 goog.color.isValidRgbColor_(str).length ||
102 goog.color.names && goog.color.names[str.toLowerCase()]);

141/**
142 * Normalize an hex representation of a color
143 * @param {string} hexColor an hex color string.
144 * @return {string} hex color in the format '#rrggbb' with all lowercase
145 * literals.
146 */
147goog.color.normalizeHex = function(hexColor) {
148 if (!goog.color.isValidHexColor_(hexColor)) {
149 throw Error("'" + hexColor + "' is not a valid hex color");
150 }
151 if (hexColor.length == 4) { // of the form #RGB
152 hexColor = hexColor.replace(goog.color.hexTripletRe_, '#$1$1$2$2$3$3');
153 }
154 return hexColor.toLowerCase();
155};

173/**
174 * Converts a color from RGB to hex representation.
175 * @param {number} r Amount of red, int between 0 and 255.
176 * @param {number} g Amount of green, int between 0 and 255.
177 * @param {number} b Amount of blue, int between 0 and 255.
178 * @return {string} hex representation of the color.
179 */
180goog.color.rgbToHex = function(r, g, b) {
181 r = Number(r);
182 g = Number(g);
183 b = Number(b);
184 if (isNaN(r) || r < 0 || r > 255 ||
185 isNaN(g) || g < 0 || g > 255 ||
186 isNaN(b) || b < 0 || b > 255) {
187 throw Error('"(' + r + ',' + g + ',' + b + '") is not a valid RGB color');
188 }
189 var hexR = goog.color.prependZeroIfNecessaryHelper(r.toString(16));
190 var hexG = goog.color.prependZeroIfNecessaryHelper(g.toString(16));
191 var hexB = goog.color.prependZeroIfNecessaryHelper(b.toString(16));
192 return '#' + hexR + hexG + hexB;
193};

206/**
207 * Converts a color from RGB color space to HSL color space.
208 * Modified from {@link http://en.wikipedia.org/wiki/HLS_color_space}.
209 * @param {number} r Value of red, in [0, 255].
210 * @param {number} g Value of green, in [0, 255].
211 * @param {number} b Value of blue, in [0, 255].
212 * @return {!goog.color.Hsl} hsl representation of the color.
213 */
214goog.color.rgbToHsl = function(r, g, b) {
215 // First must normalize r, g, b to be between 0 and 1.
216 var normR = r / 255;
217 var normG = g / 255;
218 var normB = b / 255;
219 var max = Math.max(normR, normG, normB);
220 var min = Math.min(normR, normG, normB);
221 var h = 0;
222 var s = 0;
223
224 // Luminosity is the average of the max and min rgb color intensities.
225 var l = 0.5 * (max + min);
226
227 // The hue and saturation are dependent on which color intensity is the max.
228 // If max and min are equal, the color is gray and h and s should be 0.
229 if (max != min) {
230 if (max == normR) {
231 h = 60 * (normG - normB) / (max - min);
232 } else if (max == normG) {
233 h = 60 * (normB - normR) / (max - min) + 120;
234 } else if (max == normB) {
235 h = 60 * (normR - normG) / (max - min) + 240;
236 }
237
238 if (0 < l && l <= 0.5) {
239 s = (max - min) / (2 * l);
240 } else {
241 s = (max - min) / (2 - 2 * l);
242 }
243 }
244
245 // Make sure the hue falls between 0 and 360.
246 return [Math.round(h + 360) % 360, s, l];
247};

260/**
261 * Helper for hslToRgb.
262 * @param {number} v1 Helper variable 1.
263 * @param {number} v2 Helper variable 2.
264 * @param {number} vH Helper variable 3.
265 * @return {number} Appropriate RGB value, given the above.
266 * @private
267 */
268goog.color.hueToRgb_ = function(v1, v2, vH) {
269 if (vH < 0) {
270 vH += 1;
271 } else if (vH > 1) {
272 vH -= 1;
273 }
274 if ((6 * vH) < 1) {
275 return (v1 + (v2 - v1) * 6 * vH);
276 } else if (2 * vH < 1) {
277 return v2;
278 } else if (3 * vH < 2) {
279 return (v1 + (v2 - v1) * ((2 / 3) - vH) * 6);
280 }
281 return v1;
282};

285/**
286 * Converts a color from HSL color space to RGB color space.
287 * Modified from {@link http://www.easyrgb.com/math.html}
288 * @param {number} h Hue, in [0, 360].
289 * @param {number} s Saturation, in [0, 1].
290 * @param {number} l Luminosity, in [0, 1].
291 * @return {!goog.color.Rgb} rgb representation of the color.
292 */
293goog.color.hslToRgb = function(h, s, l) {
294 var r = 0;
295 var g = 0;
296 var b = 0;
297 var normH = h / 360; // normalize h to fall in [0, 1]
298
299 if (s == 0) {
300 r = g = b = l * 255;
301 } else {
302 var temp1 = 0;
303 var temp2 = 0;
304 if (l < 0.5) {
305 temp2 = l * (1 + s);
306 } else {
307 temp2 = l + s - (s * l);
308 }
309 temp1 = 2 * l - temp2;
310 r = 255 * goog.color.hueToRgb_(temp1, temp2, normH + (1 / 3));
311 g = 255 * goog.color.hueToRgb_(temp1, temp2, normH);
312 b = 255 * goog.color.hueToRgb_(temp1, temp2, normH - (1 / 3));
313 }
314
315 return [Math.round(r), Math.round(g), Math.round(b)];
316};

380/**
381 * Checks if a string is a valid rgb color. We expect strings of the format
382 * '(r, g, b)', or 'rgb(r, g, b)', where each color component is an int in
383 * [0, 255].
384 * @param {string} str String to check.
385 * @return {!goog.color.Rgb} the rgb representation of the color if it is
386 * a valid color, or the empty array otherwise.
387 * @private
388 */
389goog.color.isValidRgbColor_ = function(str) {
390 // Each component is separate (rather than using a repeater) so we can
391 // capture the match. Also, we explicitly set each component to be either 0,
392 // or start with a non-zero, to prevent octal numbers from slipping through.
393 var regExpResultArray = str.match(goog.color.rgbColorRe_);
394 if (regExpResultArray) {
395 var r = Number(regExpResultArray[1]);
396 var g = Number(regExpResultArray[2]);
397 var b = Number(regExpResultArray[3]);
398 if (r >= 0 && r <= 255 &&
399 g >= 0 && g <= 255 &&
400 b >= 0 && b <= 255) {
401 return [r, g, b];
402 }
403 }
404 return [];
405};

444/**
445 * Converts an HSV triplet to an RGB array. V is brightness because b is
446 * reserved for blue in RGB.
447 * @param {number} h Hue value in [0, 360].
448 * @param {number} s Saturation value in [0, 1].
449 * @param {number} brightness brightness in [0, 255].
450 * @return {!goog.color.Rgb} rgb representation of the color.
451 */
452goog.color.hsvToRgb = function(h, s, brightness) {
453 var red = 0;
454 var green = 0;
455 var blue = 0;
456 if (s == 0) {
457 red = brightness;
458 green = brightness;
459 blue = brightness;
460 } else {
461 var sextant = Math.floor(h / 60);
462 var remainder = (h / 60) - sextant;
463 var val1 = brightness * (1 - s);
464 var val2 = brightness * (1 - (s * remainder));
465 var val3 = brightness * (1 - (s * (1 - remainder)));
466 switch (sextant) {
467 case 1:
468 red = val2;
469 green = brightness;
470 blue = val1;
471 break;
472 case 2:
473 red = val1;
474 green = brightness;
475 blue = val3;
476 break;
477 case 3:
478 red = val1;
479 green = val2;
480 blue = brightness;
481 break;
482 case 4:
483 red = val3;
484 green = val1;
485 blue = brightness;
486 break;
487 case 5:
488 red = brightness;
489 green = val1;
490 blue = val2;
491 break;
492 case 6:
493 case 0:
494 red = brightness;
495 green = val3;
496 blue = val1;
497 break;
498 }
499 }
500
501 return [Math.floor(red), Math.floor(green), Math.floor(blue)];
502};

505/**
506 * Converts from RGB values to an array of HSV values.
507 * @param {number} red Red value in [0, 255].
508 * @param {number} green Green value in [0, 255].
509 * @param {number} blue Blue value in [0, 255].
510 * @return {!goog.color.Hsv} hsv representation of the color.
511 */
512goog.color.rgbToHsv = function(red, green, blue) {
513
514 var max = Math.max(Math.max(red, green), blue);
515 var min = Math.min(Math.min(red, green), blue);
516 var hue;
517 var saturation;
518 var value = max;
519 if (min == max) {
520 hue = 0;
521 saturation = 0;
522 } else {
523 var delta = (max - min);
524 saturation = delta / max;
525
526 if (red == max) {
527 hue = (green - blue) / delta;
528 } else if (green == max) {
529 hue = 2 + ((blue - red) / delta);
530 } else {
531 hue = 4 + ((red - green) / delta);
532 }
533 hue *= 60;
534 if (hue < 0) {
535 hue += 360;
536 }
537 if (hue > 360) {
538 hue -= 360;
539 }
540 }
541
542 return [hue, saturation, value];
543};

631/**
632 * Calculates the Euclidean distance between two color vectors on an HSL sphere.
633 * A demo of the sphere can be found at:
634 * http://en.wikipedia.org/wiki/HSL_color_space
635 * In short, a vector for color (H, S, L) in this system can be expressed as
636 * (S*L'*cos(2*PI*H), S*L'*sin(2*PI*H), L), where L' = abs(L - 0.5), and we
637 * simply calculate the 1-2 distance using these coordinates
638 * @param {goog.color.Hsl} hsl1 First color in hsl representation.
639 * @param {goog.color.Hsl} hsl2 Second color in hsl representation.
640 * @return {number} Distance between the two colors, in the range [0, 1].
641 */
642goog.color.hslDistance = function(hsl1, hsl2) {
643 var sl1, sl2;
644 if (hsl1[2] <= 0.5) {
645 sl1 = hsl1[1] * hsl1[2];
646 } else {
647 sl1 = hsl1[1] * (1.0 - hsl1[2]);
648 }
649
650 if (hsl2[2] <= 0.5) {
651 sl2 = hsl2[1] * hsl2[2];
652 } else {
653 sl2 = hsl2[1] * (1.0 - hsl2[2]);
654 }
655
656 var h1 = hsl1[0] / 360.0;
657 var h2 = hsl2[0] / 360.0;
658 var dh = (h1 - h2) * 2.0 * Math.PI;
659 return (hsl1[2] - hsl2[2]) * (hsl1[2] - hsl2[2]) +
660 sl1 * sl1 + sl2 * sl2 - 2 * sl1 * sl2 * Math.cos(dh);
661};

664/**
665 * Blend two colors together, using the specified factor to indicate the weight
666 * given to the first color
667 * @param {goog.color.Rgb} rgb1 First color represented in rgb.
668 * @param {goog.color.Rgb} rgb2 Second color represented in rgb.
669 * @param {number} factor The weight to be given to rgb1 over rgb2. Values
670 * should be in the range [0, 1]. If less than 0, factor will be set to 0.
671 * If greater than 1, factor will be set to 1.
672 * @return {!goog.color.Rgb} Combined color represented in rgb.
673 */
674goog.color.blend = function(rgb1, rgb2, factor) {
675 factor = goog.math.clamp(factor, 0, 1);
676
677 return [
678 Math.round(factor * rgb1[0] + (1.0 - factor) * rgb2[0]),
679 Math.round(factor * rgb1[1] + (1.0 - factor) * rgb2[1]),
680 Math.round(factor * rgb1[2] + (1.0 - factor) * rgb2[2])
681 ];
682};

713/**
714 * Find the "best" (highest-contrast) of the suggested colors for the prime
715 * color. Uses W3C formula for judging readability and visual accessibility:
716 * http://www.w3.org/TR/AERT#color-contrast
717 * @param {goog.color.Rgb} prime Color represented as a rgb array.
718 * @param {Array<goog.color.Rgb>} suggestions Array of colors,
719 * each representing a rgb array.
720 * @return {!goog.color.Rgb} Highest-contrast color represented by an array..
721 */
722goog.color.highContrast = function(prime, suggestions) {
723 var suggestionsWithDiff = [];
724 for (var i = 0; i < suggestions.length; i++) {
725 suggestionsWithDiff.push({
726 color: suggestions[i],
727 diff: goog.color.yiqBrightnessDiff_(suggestions[i], prime) +
728 goog.color.colorDiff_(suggestions[i], prime)
729 });
730 }
731 suggestionsWithDiff.sort(function(a, b) {
732 return b.diff - a.diff;
733 });
734 return suggestionsWithDiff[0].color;
735};
