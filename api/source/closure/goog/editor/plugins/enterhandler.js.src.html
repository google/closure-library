42/**
43 * Plugin to handle enter keys. This does all the crazy to normalize (as much as
44 * is reasonable) what happens when you hit enter. This also handles the
45 * special casing of hitting enter in a blockquote.
46 *
47 * In IE, Webkit, and Opera, the resulting HTML uses one DIV tag per line. In
48 * Firefox, the resulting HTML uses BR tags at the end of each line.
49 *
50 * @constructor
51 * @extends {goog.editor.Plugin}
52 */
53goog.editor.plugins.EnterHandler = function() {
54 goog.editor.Plugin.call(this);
55};
56goog.inherits(goog.editor.plugins.EnterHandler, goog.editor.Plugin);

135/**
136 * Fix paragraphs to be the correct type of node.
137 * @param {goog.events.Event} e The <enter> key event.
138 * @param {boolean} split Whether we already split up a blockquote by
139 * manually inserting elements.
140 * @protected
141 */
142goog.editor.plugins.EnterHandler.prototype.processParagraphTagsInternal =
143 function(e, split) {
144 // Force IE to turn the node we are leaving into a DIV. If we do turn
145 // it into a DIV, the node IE creates in response to ENTER will also be
146 // a DIV. If we don't, it will be a P. We handle that case
147 // in handleKeyUpIE_
148 if (goog.userAgent.IE || goog.userAgent.OPERA) {
149 this.ensureBlockIeOpera(goog.dom.TagName.DIV);
150 } else if (!split && goog.userAgent.WEBKIT) {
151 // WebKit duplicates a blockquote when the user hits enter. Let's cancel
152 // this and insert a BR instead, to make it more consistent with the other
153 // browsers.
154 var range = this.getFieldObject().getRange();
155 if (!range || !goog.editor.plugins.EnterHandler.isDirectlyInBlockquote(
156 range.getContainerElement())) {
157 return;
158 }
159
160 var dh = this.getFieldDomHelper();
161 var br = dh.createElement(goog.dom.TagName.BR);
162 range.insertNode(br, true);
163
164 // If the BR is at the end of a block element, Safari still thinks there is
165 // only one line instead of two, so we need to add another BR in that case.
166 if (goog.editor.node.isBlockTag(br.parentNode) &&
167 !goog.editor.node.skipEmptyTextNodes(br.nextSibling)) {
168 goog.dom.insertSiblingBefore(
169 dh.createElement(goog.dom.TagName.BR), br);
170 }
171
172 goog.editor.range.placeCursorNextTo(br, false);
173 e.preventDefault();
174 }
175};

178/**
179 * Determines whether the lowest containing block node is a blockquote.
180 * @param {Node} n The node.
181 * @return {boolean} Whether the deepest block ancestor of n is a blockquote.
182 */
183goog.editor.plugins.EnterHandler.isDirectlyInBlockquote = function(n) {
184 for (var current = n; current; current = current.parentNode) {
185 if (goog.editor.node.isBlockTag(current)) {
186 return current.tagName == goog.dom.TagName.BLOCKQUOTE;
187 }
188 }
189
190 return false;

204/**
205 * Deletes the element at the cursor if it is a BR node, and if it does, calls
206 * e.preventDefault to stop the browser from deleting. Only necessary in Gecko
207 * as a workaround for mozilla bug 205350 where deleting a BR that is followed
208 * by a block element doesn't work (the BR gets immediately replaced). We also
209 * need to account for an ill-formed cursor which occurs from us trying to
210 * stop the browser from deleting.
211 *
212 * @param {goog.events.Event} e The DELETE keypress event.
213 * @protected
214 */
215goog.editor.plugins.EnterHandler.prototype.deleteBrGecko = function(e) {
216 var range = this.getFieldObject().getRange();
217 if (range.isCollapsed()) {
218 var container = range.getEndNode();
219 if (container.nodeType == goog.dom.NodeType.ELEMENT) {
220 var nextNode = container.childNodes[range.getEndOffset()];
221 if (nextNode && nextNode.tagName == goog.dom.TagName.BR) {
222 // We want to retrieve the first non-whitespace previous sibling
223 // as we could have added an empty text node below and want to
224 // properly handle deleting a sequence of BR's.
225 var previousSibling = goog.editor.node.getPreviousSibling(nextNode);
226 var nextSibling = nextNode.nextSibling;
227
228 container.removeChild(nextNode);
229 e.preventDefault();
230
231 // When we delete a BR followed by a block level element, the cursor
232 // has a line-height which spans the height of the block level element.
233 // e.g. If we delete a BR followed by a UL, the resulting HTML will
234 // appear to the end user like:-
235 //
236 // | * one
237 // | * two
238 // | * three
239 //
240 // There are a couple of cases that we have to account for in order to
241 // properly conform to what the user expects when DELETE is pressed.
242 //
243 // 1. If the BR has a previous sibling and the previous sibling is
244 // not a block level element or a BR, we place the cursor at the
245 // end of that.
246 // 2. If the BR doesn't have a previous sibling or the previous sibling
247 // is a block level element or a BR, we place the cursor at the
248 // beginning of the leftmost leaf of its next sibling.
249 if (nextSibling && goog.editor.node.isBlockTag(nextSibling)) {
250 if (previousSibling &&
251 !(previousSibling.tagName == goog.dom.TagName.BR ||
252 goog.editor.node.isBlockTag(previousSibling))) {
253 goog.dom.Range.createCaret(
254 previousSibling,
255 goog.editor.node.getLength(previousSibling)).select();
256 } else {
257 var leftMostLeaf = goog.editor.node.getLeftMostLeaf(nextSibling);
258 goog.dom.Range.createCaret(leftMostLeaf, 0).select();
259 }
260 }
261 }
262 }
263 }
264};

267/** @override */
268goog.editor.plugins.EnterHandler.prototype.handleKeyPress = function(e) {
269 // If a dialog doesn't have selectable field, Gecko grabs the event and
270 // performs actions in editor window. This solves that problem and allows
271 // the event to be passed on to proper handlers.
272 if (goog.userAgent.GECKO && this.getFieldObject().inModalMode()) {
273 return false;
274 }
275
276 // Firefox will allow the first node in an iframe to be deleted
277 // on a backspace. Disallow it if the node is empty.
278 if (e.keyCode == goog.events.KeyCodes.BACKSPACE) {
279 this.handleBackspaceInternal(e, this.getFieldObject().getRange());
280
281 } else if (e.keyCode == goog.events.KeyCodes.ENTER) {
282 if (goog.userAgent.GECKO) {
283 if (!e.shiftKey) {
284 // Behave similarly to IE's content editable return carriage:
285 // If the shift key is down or specified by the application, insert a
286 // BR, otherwise split paragraphs
287 this.handleEnterGecko_(e);
288 }
289 } else {
290 // In Gecko-based browsers, this is handled in the handleEnterGecko_
291 // method.
292 this.getFieldObject().dispatchBeforeChange();
293 var cursorPosition = this.deleteCursorSelection_();
294
295 var split = !!this.getFieldObject().execCommand(
296 goog.editor.plugins.Blockquote.SPLIT_COMMAND, cursorPosition);
297 if (split) {
298 // TODO(user): I think we probably don't need to stopPropagation here
299 e.preventDefault();
300 e.stopPropagation();
301 }
302
303 this.releasePositionObject_(cursorPosition);
304
305 if (goog.userAgent.WEBKIT) {
306 this.handleEnterWebkitInternal(e);
307 }
308
309 this.processParagraphTagsInternal(e, split);
310 this.getFieldObject().dispatchChange();
311 }
312
313 } else if (goog.userAgent.GECKO && e.keyCode == goog.events.KeyCodes.DELETE) {
314 this.handleDeleteGecko(e);
315 }
316
317 return false;
318};

347/**
348 * Handles an enter keypress event on fields in Gecko.
349 * @param {goog.events.BrowserEvent} e The key event.
350 * @private
351 */
352goog.editor.plugins.EnterHandler.prototype.handleEnterGecko_ = function(e) {
353 // Retrieve whether the selection is collapsed before we delete it.
354 var range = this.getFieldObject().getRange();
355 var wasCollapsed = !range || range.isCollapsed();
356 var cursorPosition = this.deleteCursorSelection_();
357
358 var handled = this.getFieldObject().execCommand(
359 goog.editor.plugins.Blockquote.SPLIT_COMMAND, cursorPosition);
360 if (handled) {
361 // TODO(user): I think we probably don't need to stopPropagation here
362 e.preventDefault();
363 e.stopPropagation();
364 }
365
366 this.releasePositionObject_(cursorPosition);
367 if (!handled) {
368 this.handleEnterAtCursorGeckoInternal(e, wasCollapsed, range);
369 }
370};

423/**
424 * Ensures all text in IE and Opera to be in the given tag in order to control
425 * Enter spacing. Call this when Enter is pressed if desired.
426 *
427 * We want to make sure the user is always inside of a block (or other nodes
428 * listed in goog.editor.plugins.EnterHandler.IGNORE_ENSURE_BLOCK_NODES_). We
429 * listen to keypress to force nodes that the user is leaving to turn into
430 * blocks, but we also need to listen to keyup to force nodes that the user is
431 * entering to turn into blocks.
432 * Example: html is: "<h2>foo[cursor]</h2>", and the user hits enter. We
433 * don't want to format the h2, but we do want to format the P that is
434 * created on enter. The P node is not available until keyup.
435 * @param {goog.dom.TagName} tag The tag name to convert to.
436 * @param {boolean=} opt_keyUp Whether the function is being called on key up.
437 * When called on key up, the cursor is in the newly created node, so the
438 * semantics for when to change it to a block are different. Specifically,
439 * if the resulting node contains only a BR, it is converted to <tag>.
440 * @protected
441 */
442goog.editor.plugins.EnterHandler.prototype.ensureBlockIeOpera = function(tag,
443 opt_keyUp) {
444 var range = this.getFieldObject().getRange();
445 var container = range.getContainer();
446 var field = this.getFieldObject().getElement();
447
448 var paragraph;
449 while (container && container != field) {
450 // We don't need to ensure a block if we are already in the same block, or
451 // in another block level node that we don't want to change the format of
452 // (unless we're handling keyUp and that block node just contains a BR).
453 var nodeName = container.nodeName;
454 // Due to @bug 2455389, the call to isBrElem needs to be inlined in the if
455 // instead of done before and saved in a variable, so that it can be
456 // short-circuited and avoid a weird IE edge case.
457 if (nodeName == tag ||
458 (goog.editor.plugins.EnterHandler.
459 DO_NOT_ENSURE_BLOCK_NODES_[nodeName] && !(opt_keyUp &&
460 goog.editor.plugins.EnterHandler.isBrElem(container)))) {
461 // Opera can create a <p> inside of a <div> in some situations,
462 // such as when breaking out of a list that is contained in a <div>.
463 if (goog.userAgent.OPERA && paragraph) {
464 if (nodeName == tag &&
465 paragraph == container.lastChild &&
466 goog.editor.node.isEmpty(paragraph)) {
467 goog.dom.insertSiblingAfter(paragraph, container);
468 goog.dom.Range.createFromNodeContents(paragraph).select();
469 }
470 break;
471 }
472 return;
473 }
474 if (goog.userAgent.OPERA && opt_keyUp && nodeName == goog.dom.TagName.P &&
475 nodeName != tag) {
476 paragraph = container;
477 }
478
479 container = container.parentNode;
480 }
481
482
483 if (goog.userAgent.IE && !goog.userAgent.isVersionOrHigher(9)) {
484 // IE (before IE9) has a bug where if the cursor is directly before a block
485 // node (e.g., the content is "foo[cursor]<blockquote>bar</blockquote>"),
486 // the FormatBlock command actually formats the "bar" instead of the "foo".
487 // This is just wrong. To work-around this, we want to move the
488 // selection back one character, and then restore it to its prior position.
489 // NOTE: We use the following "range math" to detect this situation because
490 // using Closure ranges here triggers a bug in IE that causes a crash.
491 // parent2 != parent3 ensures moving the cursor forward one character
492 // crosses at least 1 element boundary, and therefore tests if the cursor is
493 // at such a boundary. The second check, parent3 != range.parentElement()
494 // weeds out some cases where the elements are siblings instead of cousins.
495 var needsHelp = false;
496 range = range.getBrowserRangeObject();
497 var range2 = range.duplicate();
498 range2.moveEnd('character', 1);
499 // In whitebox mode, when the cursor is at the end of the field, trying to
500 // move the end of the range will do nothing, and hence the range's text
501 // will be empty. In this case, the cursor clearly isn't sitting just
502 // before a block node, since it isn't before anything.
503 if (range2.text.length) {
504 var parent2 = range2.parentElement();
505
506 var range3 = range2.duplicate();
507 range3.collapse(false);
508 var parent3 = range3.parentElement();
509
510 if ((needsHelp = parent2 != parent3 &&
511 parent3 != range.parentElement())) {
512 range.move('character', -1);
513 range.select();
514 }
515 }
516 }
517
518 this.getFieldObject().getEditableDomHelper().getDocument().execCommand(
519 'FormatBlock', false, '<' + tag + '>');
520
521 if (needsHelp) {
522 range.move('character', 1);
523 range.select();
524 }
525};

574/**
575 * Delete the selection at the current cursor position, then returns the node
576 * at the current position.
577 * @return {!goog.editor.range.Point} The current cursor position. Note that
578 * unlike simulateEnterIE_, this should not be removed from the DOM.
579 * @private
580 */
581goog.editor.plugins.EnterHandler.prototype.deleteCursorSelectionW3C_ =
582 function() {
583 var range = this.getFieldObject().getRange();
584
585 // Delete the current selection if it's is non-collapsed.
586 // Although this is redundant in FF, it's necessary for Safari
587 if (!range.isCollapsed()) {
588 var shouldDelete = true;
589 // Opera selects the <br> in an empty block if there is no text node
590 // preceding it. To preserve inline formatting when pressing [enter] inside
591 // an empty block, don't delete the selection if it only selects a <br> at
592 // the end of the block.
593 // TODO(user): Move this into goog.dom.Range. It should detect this state
594 // when creating a range from the window selection and fix it in the created
595 // range.
596 if (goog.userAgent.OPERA) {
597 var startNode = range.getStartNode();
598 var startOffset = range.getStartOffset();
599 if (startNode == range.getEndNode() &&
600 // This weeds out cases where startNode is a text node.
601 startNode.lastChild &&
602 startNode.lastChild.tagName == goog.dom.TagName.BR &&
603 // If this check is true, then endOffset is implied to be
604 // startOffset + 1, because the selection is not collapsed and
605 // it starts and ends within the same element.
606 startOffset == startNode.childNodes.length - 1) {
607 shouldDelete = false;
608 }
609 }
610 if (shouldDelete) {
611 goog.editor.plugins.EnterHandler.deleteW3cRange_(range);
612 }
613 }
614
615 return goog.editor.range.getDeepEndPoint(range, true);
616};

619/**
620 * Deletes the contents of the selection from the DOM.
621 * @param {goog.dom.AbstractRange} range The range to remove contents from.
622 * @return {goog.dom.AbstractRange} The resulting range. Used for testing.
623 * @private
624 */
625goog.editor.plugins.EnterHandler.deleteW3cRange_ = function(range) {
626 if (range && !range.isCollapsed()) {
627 var reselect = true;
628 var baseNode = range.getContainerElement();
629 var nodeOffset = new goog.dom.NodeOffset(range.getStartNode(), baseNode);
630 var rangeOffset = range.getStartOffset();
631
632 // Whether the selection crosses no container boundaries.
633 var isInOneContainer =
634 goog.editor.plugins.EnterHandler.isInOneContainerW3c_(range);
635
636 // Whether the selection ends in a container it doesn't fully select.
637 var isPartialEnd = !isInOneContainer &&
638 goog.editor.plugins.EnterHandler.isPartialEndW3c_(range);
639
640 // Remove The range contents, and ensure the correct content stays selected.
641 range.removeContents();
642 var node = nodeOffset.findTargetNode(baseNode);
643 if (node) {
644 range = goog.dom.Range.createCaret(node, rangeOffset);
645 } else {
646 // This occurs when the node that would have been referenced has now been
647 // deleted and there are no other nodes in the baseNode. Thus need to
648 // set the caret to the end of the base node.
649 range =
650 goog.dom.Range.createCaret(baseNode, baseNode.childNodes.length);
651 reselect = false;
652 }
653 range.select();
654
655 // If we just deleted everything from the container, add an nbsp
656 // to the container, and leave the cursor inside of it
657 if (isInOneContainer) {
658 var container = goog.editor.style.getContainer(range.getStartNode());
659 if (goog.editor.node.isEmpty(container, true)) {
660 var html = '&nbsp;';
661 if (goog.userAgent.OPERA &&
662 container.tagName == goog.dom.TagName.LI) {
663 // Don't break Opera's native break-out-of-lists behavior.
664 html = '<br>';
665 }
666 goog.editor.node.replaceInnerHtml(container, html);
667 goog.editor.range.selectNodeStart(container.firstChild);
668 reselect = false;
669 }
670 }
671
672 if (isPartialEnd) {
673 /*
674 This code handles the following, where | is the cursor:
675 <div>a|b</div><div>c|d</div>
676 After removeContents, the remaining HTML is
677 <div>a</div><div>d</div>
678 which means the line break between the two divs remains. This block
679 moves children of the second div in to the first div to get the correct
680 result:
681 <div>ad</div>
682

685 */

688 if (rangeStart && redundantContainer) {

707/**
708 * Checks whether the whole range is in a single block-level element.
709 * @param {goog.dom.AbstractRange} range The range to check.
710 * @return {boolean} Whether the whole range is in a single block-level element.

713goog.editor.plugins.EnterHandler.isInOneContainerW3c_ = function(range) {
714 // Find the block element containing the start of the selection.
715 var startContainer = range.getStartNode();
716 if (goog.editor.style.isContainer(startContainer)) {
717 startContainer = startContainer.childNodes[range.getStartOffset()] ||
718 startContainer;
719 }
720 startContainer = goog.editor.style.getContainer(startContainer);
721
722 // Find the block element containing the end of the selection.

724 if (goog.editor.style.isContainer(endContainer)) {
725 endContainer = endContainer.childNodes[range.getEndOffset()] ||
726 endContainer;
727 }
728 endContainer = goog.editor.style.getContainer(endContainer);
729

736 * Checks whether the end of the range is not at the end of a block-level
737 * element.
738 * @param {goog.dom.AbstractRange} range The range to check.
739 * @return {boolean} Whether the end of the range is not at the end of a
740 * block-level element.
741 * @private
742 */
743goog.editor.plugins.EnterHandler.isPartialEndW3c_ = function(range) {
744 var endContainer = range.getEndNode();
745 var endOffset = range.getEndOffset();
746 var node = endContainer;
747 if (goog.editor.style.isContainer(node)) {
748 var child = node.childNodes[endOffset];
749 // Child is null when end offset is >= length, which indicates the entire
750 // container is selected. Otherwise, we also know the entire container
751 // is selected if the selection ends at a new container.
752 if (!child ||
753 child.nodeType == goog.dom.NodeType.ELEMENT &&
754 goog.editor.style.isContainer(child)) {
755 return false;
756 }
757 }
758
759 var container = goog.editor.style.getContainer(node);
760 while (container != node) {
761 if (goog.editor.node.getNextSibling(node)) {
762 return true;
763 }
764 node = node.parentNode;
765 }
766
767 return endOffset != goog.editor.node.getLength(endContainer);
