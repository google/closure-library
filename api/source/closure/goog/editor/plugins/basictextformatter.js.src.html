151/**
152 * Execute a user-initiated command.
153 * @param {string} command Command to execute.
154 * @param {...*} var_args For color commands, this
155 * should be the hex color (with the #). For FORMAT_BLOCK, this should be
156 * the goog.editor.plugins.BasicTextFormatter.BLOCK_COMMAND.
157 * It will be unused for other commands.
158 * @return {Object|undefined} The result of the command.
159 * @override
160 */
161goog.editor.plugins.BasicTextFormatter.prototype.execCommandInternal = function(
162 command, var_args) {
163 var preserveDir, styleWithCss, needsFormatBlockDiv, hasDummySelection;
164 var result;
165 var opt_arg = arguments[1];
166
167 switch (command) {
168 case goog.editor.plugins.BasicTextFormatter.COMMAND.BACKGROUND_COLOR:
169 // Don't bother for no color selected, color picker is resetting itself.
170 if (!goog.isNull(opt_arg)) {
171 if (goog.editor.BrowserFeature.EATS_EMPTY_BACKGROUND_COLOR) {
172 this.applyBgColorManually_(opt_arg);
173 } else if (goog.userAgent.OPERA) {
174 // backColor will color the block level element instead of
175 // the selected span of text in Opera.
176 this.execCommandHelper_('hiliteColor', opt_arg);
177 } else {
178 this.execCommandHelper_(command, opt_arg);
179 }
180 }
181 break;
182
183 case goog.editor.plugins.BasicTextFormatter.COMMAND.LINK:
184 result = this.toggleLink_(opt_arg);
185 break;
186
187 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_CENTER:
188 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_FULL:
189 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_RIGHT:
190 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_LEFT:
191 this.justify_(command);
192 break;
193
194 default:
195 if (goog.userAgent.IE &&
196 command ==
197 goog.editor.plugins.BasicTextFormatter.COMMAND.FORMAT_BLOCK &&
198 opt_arg) {
199 // IE requires that the argument be in the form of an opening
200 // tag, like <h1>, including angle brackets. WebKit will accept
201 // the arguemnt with or without brackets, and Firefox pre-3 supports
202 // only a fixed subset of tags with brackets, and prefers without.
203 // So we only add them IE only.
204 opt_arg = '<' + opt_arg + '>';
205 }
206
207 if (command ==
208 goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_COLOR &&
209 goog.isNull(opt_arg)) {
210 // If we don't have a color, then FONT_COLOR is a no-op.
211 break;
212 }
213
214 switch (command) {
215 case goog.editor.plugins.BasicTextFormatter.COMMAND.INDENT:
216 case goog.editor.plugins.BasicTextFormatter.COMMAND.OUTDENT:
217 if (goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS) {
218 if (goog.userAgent.GECKO) {
219 styleWithCss = true;
220 }
221 if (goog.userAgent.OPERA) {
222 if (command ==
223 goog.editor.plugins.BasicTextFormatter.COMMAND.OUTDENT) {
224 // styleWithCSS actually sets negative margins on <blockquote>
225 // to outdent them. If the command is enabled without
226 // styleWithCSS flipped on, then the caret is in a blockquote so
227 // styleWithCSS must not be used. But if the command is not
228 // enabled, styleWithCSS should be used so that elements such as
229 // a <div> with a margin-left style can still be outdented.
230 // (Opera bug: CORE-21118)
231 styleWithCss =
232 !this.getDocument_().queryCommandEnabled('outdent');
233 } else {
234 // Always use styleWithCSS for indenting. Otherwise, Opera will
235 // make separate <blockquote>s around *each* indented line,
236 // which adds big default <blockquote> margins between each
237 // indented line.
238 styleWithCss = true;
239 }
240 }
241 }
242 // Fall through.
243
244 case goog.editor.plugins.BasicTextFormatter.COMMAND.ORDERED_LIST:
245 case goog.editor.plugins.BasicTextFormatter.COMMAND.UNORDERED_LIST:
246 if (goog.editor.BrowserFeature.LEAVES_P_WHEN_REMOVING_LISTS &&
247 this.queryCommandStateInternal_(this.getDocument_(),
248 command)) {
249 // IE leaves behind P tags when unapplying lists.
250 // If we're not in P-mode, then we want divs
251 // So, unlistify, then convert the Ps into divs.
252 needsFormatBlockDiv = this.getFieldObject().queryCommandValue(
253 goog.editor.Command.DEFAULT_TAG) != goog.dom.TagName.P;
254 } else if (!goog.editor.BrowserFeature.CAN_LISTIFY_BR) {
255 // IE doesn't convert BRed line breaks into separate list items.
256 // So convert the BRs to divs, then do the listify.
257 this.convertBreaksToDivs_();
258 }
259
260 // This fix only works in Gecko.
261 if (goog.userAgent.GECKO &&
262 goog.editor.BrowserFeature.FORGETS_FORMATTING_WHEN_LISTIFYING &&
263 !this.queryCommandValue(command)) {
264 hasDummySelection |= this.beforeInsertListGecko_();
265 }
266 // Fall through to preserveDir block
267
268 case goog.editor.plugins.BasicTextFormatter.COMMAND.FORMAT_BLOCK:
269 // Both FF & IE may lose directionality info. Save/restore it.
270 // TODO(user): Does Safari also need this?
271 // TODO (gmark, jparent): This isn't ideal because it uses a string
272 // literal, so if the plugin name changes, it would break. We need a
273 // better solution. See also other places in code that use
274 // this.getPluginByClassId('Bidi').
275 preserveDir = !!this.getFieldObject().getPluginByClassId('Bidi');
276 break;
277
278 case goog.editor.plugins.BasicTextFormatter.COMMAND.SUBSCRIPT:
279 case goog.editor.plugins.BasicTextFormatter.COMMAND.SUPERSCRIPT:
280 if (goog.editor.BrowserFeature.NESTS_SUBSCRIPT_SUPERSCRIPT) {
281 // This browser nests subscript and superscript when both are
282 // applied, instead of canceling out the first when applying the
283 // second.
284 this.applySubscriptSuperscriptWorkarounds_(command);
285 }
286 break;
287
288 case goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE:
289 case goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD:
290 case goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC:
291 // If we are applying the formatting, then we want to have
292 // styleWithCSS false so that we generate html tags (like <b>). If we
293 // are unformatting something, we want to have styleWithCSS true so
294 // that we can unformat both html tags and inline styling.
295 // TODO(user): What about WebKit and Opera?
296 styleWithCss = goog.userAgent.GECKO &&
297 goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
298 this.queryCommandValue(command);
299 break;
300
301 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_COLOR:
302 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_FACE:
303 // It is very expensive in FF (order of magnitude difference) to use
304 // font tags instead of styled spans. Whenever possible,
305 // force FF to use spans.
306 // Font size is very expensive too, but FF always uses font tags,
307 // regardless of which styleWithCSS value you use.
308 styleWithCss = goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
309 goog.userAgent.GECKO;
310 }
311
312 /**
313 * Cases where we just use the default execCommand (in addition
314 * to the above fall-throughs)
315 * goog.editor.plugins.BasicTextFormatter.COMMAND.STRIKE_THROUGH:
316 * goog.editor.plugins.BasicTextFormatter.COMMAND.HORIZONTAL_RULE:
317 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUBSCRIPT:
318 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUPERSCRIPT:
319 * goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE:
320 * goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD:
321 * goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC:
322 * goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_SIZE:
323 * goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_FACE:
324 */
325 this.execCommandHelper_(command, opt_arg, preserveDir, !!styleWithCss);
326
327 if (hasDummySelection) {
328 this.getDocument_().execCommand('Delete', false, true);
329 }
330
331 if (needsFormatBlockDiv) {
332 this.getDocument_().execCommand('FormatBlock', false, '<div>');
333 }
334 }
335 // FF loses focus, so we have to set the focus back to the document or the
336 // user can't type after selecting from menu. In IE, focus is set correctly
337 // and resetting it here messes it up.
338 if (goog.userAgent.GECKO && !this.getFieldObject().inModalMode()) {
339 this.focusField_();
340 }
341 return result;
342};

354/**
355 * Gets the command value.
356 * @param {string} command The command value to get.
357 * @return {string|boolean|null} The current value of the command in the given
358 * selection. NOTE: This return type list is not documented in MSDN or MDC
359 * and has been constructed from experience. Please update it
360 * if necessary.
361 * @override
362 */
363goog.editor.plugins.BasicTextFormatter.prototype.queryCommandValue = function(
364 command) {
365 var styleWithCss;
366 switch (command) {
367 case goog.editor.plugins.BasicTextFormatter.COMMAND.LINK:
368 return this.isNodeInState_(goog.dom.TagName.A);
369
370 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_CENTER:
371 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_FULL:
372 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_RIGHT:
373 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_LEFT:
374 return this.isJustification_(command);
375
376 case goog.editor.plugins.BasicTextFormatter.COMMAND.FORMAT_BLOCK:
377 // TODO(nicksantos): See if we can use queryCommandValue here.
378 return goog.editor.plugins.BasicTextFormatter.getSelectionBlockState_(
379 this.getFieldObject().getRange());
380
381 case goog.editor.plugins.BasicTextFormatter.COMMAND.INDENT:
382 case goog.editor.plugins.BasicTextFormatter.COMMAND.OUTDENT:
383 case goog.editor.plugins.BasicTextFormatter.COMMAND.HORIZONTAL_RULE:
384 // TODO: See if there are reasonable results to return for
385 // these commands.
386 return false;
387
388 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_SIZE:
389 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_FACE:
390 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_COLOR:
391 case goog.editor.plugins.BasicTextFormatter.COMMAND.BACKGROUND_COLOR:
392 // We use queryCommandValue here since we don't just want to know if a
393 // color/fontface/fontsize is applied, we want to know WHICH one it is.
394 return this.queryCommandValueInternal_(this.getDocument_(), command,
395 goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
396 goog.userAgent.GECKO);
397
398 case goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE:
399 case goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD:
400 case goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC:
401 styleWithCss = goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
402 goog.userAgent.GECKO;
403
404 default:
405 /**
406 * goog.editor.plugins.BasicTextFormatter.COMMAND.STRIKE_THROUGH
407 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUBSCRIPT
408 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUPERSCRIPT
409 * goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE
410 * goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD
411 * goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC
412 * goog.editor.plugins.BasicTextFormatter.COMMAND.ORDERED_LIST
413 * goog.editor.plugins.BasicTextFormatter.COMMAND.UNORDERED_LIST
414 */
415 // This only works for commands that use the default execCommand
416 return this.queryCommandStateInternal_(this.getDocument_(), command,
417 styleWithCss);
418 }

422/**
423 * @override
424 */
425goog.editor.plugins.BasicTextFormatter.prototype.prepareContentsHtml =
426 function(html) {
427 // If the browser collapses empty nodes and the field has only a script
428 // tag in it, then it will collapse this node. Which will mean the user
429 // can't click into it to edit it.
430 if (goog.editor.BrowserFeature.COLLAPSES_EMPTY_NODES &&
431 html.match(/^\s*<script/i)) {
432 html = '&nbsp;' + html;
433 }
434
435 if (goog.editor.BrowserFeature.CONVERT_TO_B_AND_I_TAGS) {
436 // Some browsers (FF) can't undo strong/em in some cases, but can undo b/i!
437 html = html.replace(/<(\/?)strong([^\w])/gi, '<$1b$2');
438 html = html.replace(/<(\/?)em([^\w])/gi, '<$1i$2');
439 }
440
441 return html;
442};

474/**
475 * @override
476 */
477goog.editor.plugins.BasicTextFormatter.prototype.cleanContentsHtml =
478 function(html) {
479 if (goog.editor.BrowserFeature.MOVES_STYLE_TO_HEAD) {
480 // Safari creates a new <head> element for <style> tags, so prepend their
481 // contents to the output.
482 var heads = this.getFieldObject().getEditableDomHelper().
483 getElementsByTagNameAndClass(goog.dom.TagName.HEAD);
484 var stylesHtmlArr = [];
485
486 // i starts at 1 so we don't copy in the original, legitimate <head>.
487 var numHeads = heads.length;
488 for (var i = 1; i < numHeads; ++i) {
489 var styles = heads[i].getElementsByTagName(goog.dom.TagName.STYLE);
490 var numStyles = styles.length;
491 for (var j = 0; j < numStyles; ++j) {
492 stylesHtmlArr.push(styles[j].outerHTML);
493 }
494 }
495 return stylesHtmlArr.join('') + html;
496 }
497
498 return html;

541/**
542 * Regular expression to match BRs in HTML. Saves the BRs' attributes in $1 for
543 * use with replace(). In non-IE browsers, does not match BRs adjacent to an
544 * opening or closing DIV or P tag, since nonrendered BR elements can occur at
545 * the end of block level containers in those browsers' editors.
546 * @type {RegExp}
547 * @private
548 */
549goog.editor.plugins.BasicTextFormatter.BR_REGEXP_ =
550 goog.userAgent.IE ? /<br([^\/>]*)\/?>/gi :
551 /<br([^\/>]*)\/?>(?!<\/(div|p)>)/gi;
552
553
554/**
555 * Convert BRs in the selection to divs.
556 * This is only intended to be used in IE and Opera.
557 * @return {boolean} Whether any BR's were converted.
558 * @private
559 */
560goog.editor.plugins.BasicTextFormatter.prototype.convertBreaksToDivs_ =
561 function() {
562 if (!goog.userAgent.IE && !goog.userAgent.OPERA) {
563 // This function is only supported on IE and Opera.
564 return false;
565 }
566 var range = this.getRange_();
567 var parent = range.getContainerElement();
568 var doc = this.getDocument_();
569
570 goog.editor.plugins.BasicTextFormatter.BR_REGEXP_.lastIndex = 0;
571 // Only mess with the HTML/selection if it contains a BR.
572 if (goog.editor.plugins.BasicTextFormatter.BR_REGEXP_.test(
573 parent.innerHTML)) {
574 // Insert temporary markers to remember the selection.
575 var savedRange = range.saveUsingCarets();
576
577 if (parent.tagName == goog.dom.TagName.P) {
578 // Can't append paragraphs to paragraph tags. Throws an exception in IE.
579 goog.editor.plugins.BasicTextFormatter.convertParagraphToDiv_(
580 parent, true);
581 } else {
582 // Used to do:
583 // IE: <div>foo<br>bar</div> --> <div>foo<p id="temp_br">bar</div>
584 // Opera: <div>foo<br>bar</div> --> <div>foo<p class="temp_br">bar</div>
585 // To fix bug 1939883, now does for both:
586 // <div>foo<br>bar</div> --> <div>foo<p trtempbr="temp_br">bar</div>
587 // TODO(user): Confirm if there's any way to skip this
588 // intermediate step of converting br's to p's before converting those to
589 // div's. The reason may be hidden in CLs 5332866 and 8530601.
590 var attribute = 'trtempbr';
591 var value = 'temp_br';
592 var newHtml = parent.innerHTML.replace(
593 goog.editor.plugins.BasicTextFormatter.BR_REGEXP_,
594 '<p$1 ' + attribute + '="' + value + '">');
595 goog.editor.node.replaceInnerHtml(parent, newHtml);
596
597 var paragraphs =
598 goog.array.toArray(parent.getElementsByTagName(goog.dom.TagName.P));
599 goog.iter.forEach(paragraphs, function(paragraph) {
600 if (paragraph.getAttribute(attribute) == value) {
601 paragraph.removeAttribute(attribute);
602 if (goog.string.isBreakingWhitespace(
603 goog.dom.getTextContent(paragraph))) {
604 // Prevent the empty blocks from collapsing.
605 // A <BR> is preferable because it doesn't result in any text being
606 // added to the "blank" line. In IE, however, it is possible to
607 // place the caret after the <br>, which effectively creates a
608 // visible line break. Because of this, we have to resort to using a
609 // &nbsp; in IE.
610 var child = goog.userAgent.IE ?
611 doc.createTextNode(goog.string.Unicode.NBSP) :
612 doc.createElement(goog.dom.TagName.BR);
613 paragraph.appendChild(child);
614 }
615 goog.editor.plugins.BasicTextFormatter.convertParagraphToDiv_(
616 paragraph);
617 }
618 });
619 }
620
621 // Select the previously selected text so we only listify
622 // the selected portion and maintain the user's selection.
623 savedRange.restore();
624 return true;
625 }
626
627 return false;
628};

631/**
632 * Convert the given paragraph to being a div. This clobbers the
633 * passed-in node!
634 * This is only intended to be used in IE and Opera.
635 * @param {Node} paragraph Paragragh to convert to a div.
636 * @param {boolean=} opt_convertBrs If true, also convert BRs to divs.
637 * @private
638 */
639goog.editor.plugins.BasicTextFormatter.convertParagraphToDiv_ =
640 function(paragraph, opt_convertBrs) {
641 if (!goog.userAgent.IE && !goog.userAgent.OPERA) {
642 // This function is only supported on IE and Opera.
643 return;
644 }
645 var outerHTML = paragraph.outerHTML.replace(/<(\/?)p/gi, '<$1div');
646 if (opt_convertBrs) {
647 // IE fills in the closing div tag if it's missing!
648 outerHTML = outerHTML.replace(
649 goog.editor.plugins.BasicTextFormatter.BR_REGEXP_,
650 '</div><div$1>');
651 }
652 if (goog.userAgent.OPERA && !/<\/div>$/i.test(outerHTML)) {
653 // Opera doesn't automatically add the closing tag, so add it if needed.
654 outerHTML += '</div>';
655 }
656 paragraph.outerHTML = outerHTML;

660/**
661 * If this is a goog.editor.plugins.BasicTextFormatter.COMMAND,
662 * convert it to something that we can pass into execCommand,
663 * queryCommandState, etc.
664 *
665 * TODO(user): Consider doing away with the + and converter completely.
666 *
667 * @param {goog.editor.plugins.BasicTextFormatter.COMMAND|string}
668 * command A command key.
