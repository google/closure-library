142/**
143 * Removes formatting from the current selection. Removes basic formatting
144 * (B/I/U) using the browser's execCommand. Then extracts the html from the
145 * selection to convert, calls either a client's specified removeFormattingFunc
146 * callback or trogedit's general built-in removeFormattingWorker_,
147 * and then replaces the current selection with the converted text.
148 * @private
149 */
150goog.editor.plugins.RemoveFormatting.prototype.removeFormatting_ = function() {
151 var range = this.getFieldObject().getRange();
152 if (range.isCollapsed()) {
153 return;
154 }
155
156 // Get the html to format and send it off for formatting. Built in
157 // removeFormat only strips some inline elements and some inline CSS styles
158 var convFunc = this.optRemoveFormattingFunc_ ||
159 goog.bind(this.removeFormattingWorker_, this);
160 this.convertSelectedHtmlText_(convFunc);
161
162 // Do the execCommand last as it needs block elements removed to work
163 // properly on background/fontColor in FF. There are, unfortunately, still
164 // cases where background/fontColor are not removed here.
165 var doc = this.getFieldDomHelper().getDocument();
166 doc.execCommand('RemoveFormat', false, undefined);
167
168 if (goog.editor.BrowserFeature.ADDS_NBSPS_IN_REMOVE_FORMAT) {
169 // WebKit converts spaces to non-breaking spaces when doing a RemoveFormat.
170 // See: https://bugs.webkit.org/show_bug.cgi?id=14062
171 this.convertSelectedHtmlText_(function(text) {
172 // This loses anything that might have legitimately been a non-breaking
173 // space, but that's better than the alternative of only having non-
174 // breaking spaces.
175 // Old versions of WebKit (Safari 3, Chrome 1) incorrectly match /u00A0
176 // and newer versions properly match &nbsp;.
177 var nbspRegExp =
178 goog.userAgent.isVersionOrHigher('528') ? /&nbsp;/g : /\u00A0/g;
179 return text.replace(nbspRegExp, ' ');
180 });
181 }
182};

185/**
186 * Finds the nearest ancestor of the node that is a table.
187 * @param {Node} nodeToCheck Node to search from.
188 * @return {Node} The table, or null if one was not found.
189 * @private
190 */
191goog.editor.plugins.RemoveFormatting.prototype.getTableAncestor_ = function(
192 nodeToCheck) {
193 var fieldElement = this.getFieldObject().getElement();
194 while (nodeToCheck && nodeToCheck != fieldElement) {
195 if (nodeToCheck.tagName == goog.dom.TagName.TABLE) {
196 return nodeToCheck;
197 }
198 nodeToCheck = nodeToCheck.parentNode;
199 }
200 return null;
201};

204/**
205 * Replaces the contents of the selection with html. Does its best to maintain
206 * the original selection. Also does its best to result in a valid DOM.
207 *
208 * TODO(user): See if there's any way to make this work on Ranges, and then
209 * move it into goog.editor.range. The Firefox implementation uses execCommand
210 * on the document, so must work on the actual selection.
211 *
212 * @param {string} html The html string to insert into the range.
213 * @private
214 */
215goog.editor.plugins.RemoveFormatting.prototype.pasteHtml_ = function(html) {
216 var range = this.getFieldObject().getRange();
217
218 var dh = this.getFieldDomHelper();
219 // Use markers to set the extent of the selection so that we can reselect it
220 // afterwards. This works better than builtin range manipulation in FF and IE
221 // because their implementations are so self-inconsistent and buggy.
222 var startSpanId = goog.string.createUniqueString();
223 var endSpanId = goog.string.createUniqueString();
224 html = '<span id="' + startSpanId + '"></span>' + html +
225 '<span id="' + endSpanId + '"></span>';
226 var dummyNodeId = goog.string.createUniqueString();
227 var dummySpanText = '<span id="' + dummyNodeId + '"></span>';
228
229 if (goog.editor.BrowserFeature.HAS_IE_RANGES) {
230 // IE's selection often doesn't include the outermost tags.
231 // We want to use pasteHTML to replace the range contents with the newly
232 // unformatted text, so we have to check to make sure we aren't just
233 // pasting into some stray tags. To do this, we first clear out the
234 // contents of the range and then delete all empty nodes parenting the now
235 // empty range. This way, the pasted contents are never re-embedded into
236 // formated nodes. Pasting purely empty html does not work, since IE moves
237 // the selection inside the next node, so we insert a dummy span.
238 var textRange = range.getTextRange(0).getBrowserRangeObject();
239 textRange.pasteHTML(dummySpanText);
240 var parent;
241 while ((parent = textRange.parentElement()) &&
242 goog.editor.node.isEmpty(parent) &&
243 !goog.editor.node.isEditableContainer(parent)) {
244 var tag = parent.nodeName;
245 // We can't remove these table tags as it will invalidate the table dom.
246 if (tag == goog.dom.TagName.TD ||
247 tag == goog.dom.TagName.TR ||
248 tag == goog.dom.TagName.TH) {
249 break;
250 }
251
252 goog.dom.removeNode(parent);
253 }
254 textRange.pasteHTML(html);
255 var dummySpan = dh.getElement(dummyNodeId);
256 // If we entered the while loop above, the node has already been removed
257 // since it was a child of parent and parent was removed.
258 if (dummySpan) {
259 goog.dom.removeNode(dummySpan);
260 }
261 } else if (goog.editor.BrowserFeature.HAS_W3C_RANGES) {
262 // insertHtml and range.insertNode don't merge blocks correctly.
263 // (e.g. if your selection spans two paragraphs)
264 dh.getDocument().execCommand('insertImage', false, dummyNodeId);
265 var dummyImageNodePattern = new RegExp('<[^<]*' + dummyNodeId + '[^>]*>');
266 var parent = this.getFieldObject().getRange().getContainerElement();
267 if (parent.nodeType == goog.dom.NodeType.TEXT) {
268 // Opera sometimes returns a text node here.
269 // TODO(user): perhaps we should modify getParentContainer?
270 parent = parent.parentNode;
271 }
272
273 // We have to search up the DOM because in some cases, notably when
274 // selecting li's within a list, execCommand('insertImage') actually splits
275 // tags in such a way that parent that used to contain the selection does
276 // not contain inserted image.
277 while (!dummyImageNodePattern.test(parent.innerHTML)) {
278 parent = parent.parentNode;
279 }
280
281 // Like the IE case above, sometimes the selection does not include the
282 // outermost tags. For Gecko, we have already expanded the range so that
283 // it does, so we can just replace the dummy image with the final html.
284 // For WebKit, we use the same approach as we do with IE - we
285 // inject a dummy span where we will eventually place the contents, and
286 // remove parentNodes of the span while they are empty.
287
288 if (goog.userAgent.GECKO) {
289 goog.editor.node.replaceInnerHtml(parent,
290 parent.innerHTML.replace(dummyImageNodePattern, html));
291 } else {
292 goog.editor.node.replaceInnerHtml(parent,
293 parent.innerHTML.replace(dummyImageNodePattern, dummySpanText));
294 var dummySpan = dh.getElement(dummyNodeId);
295 parent = dummySpan;
296 while ((parent = dummySpan.parentNode) &&
297 goog.editor.node.isEmpty(parent) &&
298 !goog.editor.node.isEditableContainer(parent)) {
299 var tag = parent.nodeName;
300 // We can't remove these table tags as it will invalidate the table dom.
301 if (tag == goog.dom.TagName.TD ||
302 tag == goog.dom.TagName.TR ||
303 tag == goog.dom.TagName.TH) {
304 break;
305 }
306
307 // We can't just remove parent since dummySpan is inside it, and we need
308 // to keep dummy span around for the replacement. So we move the
309 // dummySpan up as we go.
310 goog.dom.insertSiblingAfter(dummySpan, parent);
311 goog.dom.removeNode(parent);
312 }
313 goog.editor.node.replaceInnerHtml(parent,
314 parent.innerHTML.replace(new RegExp(dummySpanText, 'i'), html));
315 }
316 }
317
318 var startSpan = dh.getElement(startSpanId);
319 var endSpan = dh.getElement(endSpanId);
320 goog.dom.Range.createFromNodes(startSpan, 0, endSpan,
321 endSpan.childNodes.length).select();
322 goog.dom.removeNode(startSpan);
323 goog.dom.removeNode(endSpan);
324};

327/**
328 * Gets the html inside the selection to send off for further processing.
329 *
330 * TODO(user): Make this general so that it can be moved into
331 * goog.editor.range. The main reason it can't be moved is becuase we need to
332 * get the range before we do the execCommand and continue to operate on that
333 * same range (reasons are documented above).
334 *
335 * @param {goog.dom.AbstractRange} range The selection.
336 * @return {string} The html string to format.
337 * @private
338 */
339goog.editor.plugins.RemoveFormatting.prototype.getHtmlText_ = function(range) {
340 var div = this.getFieldDomHelper().createDom(goog.dom.TagName.DIV);
341 var textRange = range.getBrowserRangeObject();
342
343 if (goog.editor.BrowserFeature.HAS_W3C_RANGES) {
344 // Get the text to convert.
345 div.appendChild(textRange.cloneContents());
346 } else if (goog.editor.BrowserFeature.HAS_IE_RANGES) {
347 // Trim the whitespace on the ends of the range, so that it the container
348 // will be the container of only the text content that we are changing.
349 // This gets around issues in IE where the spaces are included in the
350 // selection, but ignored sometimes by execCommand, and left orphaned.
351 var rngText = range.getText();
352
353 // BRs get reported as \r\n, but only count as one character for moves.
354 // Adjust the string so our move counter is correct.
355 rngText = rngText.replace(/\r\n/g, '\r');
356
357 var rngTextLength = rngText.length;
358 var left = rngTextLength - goog.string.trimLeft(rngText).length;
359 var right = rngTextLength - goog.string.trimRight(rngText).length;
360
361 textRange.moveStart('character', left);
362 textRange.moveEnd('character', -right);
363
364 var htmlText = textRange.htmlText;
365 // Check if in pretag and fix up formatting so that new lines are preserved.
366 if (textRange.queryCommandValue('formatBlock') == 'Formatted') {
367 htmlText = goog.string.newLineToBr(textRange.htmlText);
368 }
369 div.innerHTML = htmlText;
370 }
371
372 // Get the innerHTML of the node instead of just returning the text above
373 // so that its properly html escaped.
374 return div.innerHTML;
375};

378/**
379 * Move the range so that it doesn't include any partially selected tables.
380 * @param {goog.dom.AbstractRange} range The range to adjust.
381 * @param {Node} startInTable Table node that the range starts in.
382 * @param {Node} endInTable Table node that the range ends in.
383 * @return {!goog.dom.SavedCaretRange} Range to use to restore the
384 * selection after we run our custom remove formatting.
385 * @private
386 */
387goog.editor.plugins.RemoveFormatting.prototype.adjustRangeForTables_ =
388 function(range, startInTable, endInTable) {
389 // Create placeholders for the current selection so we can restore it
390 // later.
391 var savedCaretRange = goog.editor.range.saveUsingNormalizedCarets(range);
392
393 var startNode = range.getStartNode();
394 var startOffset = range.getStartOffset();
395 var endNode = range.getEndNode();
396 var endOffset = range.getEndOffset();
397 var dh = this.getFieldDomHelper();
398
399 // Move start after the table.
400 if (startInTable) {
401 var textNode = dh.createTextNode('');
402 goog.dom.insertSiblingAfter(textNode, startInTable);
403 startNode = textNode;
404 startOffset = 0;
405 }
406 // Move end before the table.
407 if (endInTable) {
408 var textNode = dh.createTextNode('');
409 goog.dom.insertSiblingBefore(textNode, endInTable);
410 endNode = textNode;
411 endOffset = 0;
412 }
413
414 goog.dom.Range.createFromNodes(startNode, startOffset,
415 endNode, endOffset).select();
416
417 return savedCaretRange;
418};

421/**
422 * Remove a caret from the dom and hide it in a safe place, so it can
423 * be restored later via restoreCaretsFromCave.
424 * @param {goog.dom.SavedCaretRange} caretRange The caret range to
425 * get the carets from.
426 * @param {boolean} isStart Whether this is the start or end caret.
427 * @private
428 */
429goog.editor.plugins.RemoveFormatting.prototype.putCaretInCave_ = function(
430 caretRange, isStart) {
431 var cavedCaret = goog.dom.removeNode(caretRange.getCaret(isStart));
432 if (isStart) {
433 this.startCaretInCave_ = cavedCaret;
434 } else {
435 this.endCaretInCave_ = cavedCaret;
436 }
437};

440/**
441 * Restore carets that were hidden away by adding them back into the dom.
442 * Note: this does not restore to the original dom location, as that
443 * will likely have been modified with remove formatting. The only
444 * guarentees here are that start will still be before end, and that
445 * they will be in the editable region. This should only be used when
446 * you don't actually intend to USE the caret again.
447 * @private
448 */
449goog.editor.plugins.RemoveFormatting.prototype.restoreCaretsFromCave_ =
450 function() {
451 // To keep start before end, we put the end caret at the bottom of the field
452 // and the start caret at the start of the field.
453 var field = this.getFieldObject().getElement();
454 if (this.startCaretInCave_) {
455 field.insertBefore(this.startCaretInCave_, field.firstChild);
456 this.startCaretInCave_ = null;
457 }
458 if (this.endCaretInCave_) {
459 field.appendChild(this.endCaretInCave_);
460 this.endCaretInCave_ = null;
461 }
462};

465/**
466 * Gets the html inside the current selection, passes it through the given
467 * conversion function, and puts it back into the selection.
468 *
469 * @param {function(string): string} convertFunc A conversion function that
470 * transforms an html string to new html string.
471 * @private
472 */
473goog.editor.plugins.RemoveFormatting.prototype.convertSelectedHtmlText_ =
474 function(convertFunc) {
475 var range = this.getFieldObject().getRange();
476
477 // For multiple ranges, it is really hard to do our custom remove formatting
478 // without invalidating other ranges. So instead of always losing the
479 // content, this solution at least lets the browser do its own remove
480 // formatting which works correctly most of the time.
481 if (range.getTextRangeCount() > 1) {
482 return;
483 }
484
485 if (goog.userAgent.GECKO) {
486 // Determine if we need to handle tables, since they are special cases.
487 // If the selection is entirely within a table, there is no extra
488 // formatting removal we can do. If a table is fully selected, we will
489 // just blow it away. If a table is only partially selected, we can
490 // perform custom remove formatting only on the non table parts, since we
491 // we can't just remove the parts and paste back into it (eg. we can't
492 // inject html where a TR used to be).
493 // If the selection contains the table and more, this is automatically
494 // handled, but if just the table is selected, it can be tricky to figure
495 // this case out, because of the numerous ways selections can be formed -
496 // ex. if a table has a single tr with a single td with a single text node
497 // in it, and the selection is (textNode: 0), (textNode: nextNode.length)
498 // then the entire table is selected, even though the start and end aren't
499 // the table itself. We are truly inside a table if the expanded endpoints
500 // are still inside the table.
501
502 // Expand the selection to include any outermost tags that weren't included
503 // in the selection, but have the same visible selection. Stop expanding
504 // if we reach the top level field.
505 var expandedRange = goog.editor.range.expand(range,
506 this.getFieldObject().getElement());
507
508 var startInTable = this.getTableAncestor_(expandedRange.getStartNode());
509 var endInTable = this.getTableAncestor_(expandedRange.getEndNode());
510
511 if (startInTable || endInTable) {
512 if (startInTable == endInTable) {
513 // We are fully contained in the same table, there is no extra
514 // remove formatting that we can do, just return and run browser
515 // formatting only.
516 return;
517 }
518
519 // Adjust the range to not contain any partially selected tables, since
520 // we don't want to run our custom remove formatting on them.
521 var savedCaretRange = this.adjustRangeForTables_(range,
522 startInTable, endInTable);
523
524 // Hack alert!!
525 // If start is not in a table, then the saved caret will get sent out
526 // for uber remove formatting, and it will get blown away. This is
527 // fine, except that we need to be able to re-create a range from the
528 // savedCaretRange later on. So, we just remove it from the dom, and
529 // put it back later so we can create a range later (not exactly in the
530 // same spot, but don't worry we don't actually try to use it later)
531 // and then it will be removed when we dispose the range.
532 if (!startInTable) {
533 this.putCaretInCave_(savedCaretRange, true);
534 }
535 if (!endInTable) {
536 this.putCaretInCave_(savedCaretRange, false);
537 }
538
539 // Re-fetch the range, and re-expand it, since we just modified it.
540 range = this.getFieldObject().getRange();
541 expandedRange = goog.editor.range.expand(range,
542 this.getFieldObject().getElement());
543 }
544
545 expandedRange.select();
546 range = expandedRange;
547 }
548
549 // Convert the selected text to the format-less version, paste back into
550 // the selection.
551 var text = this.getHtmlText_(range);
552 this.pasteHtml_(convertFunc(text));
553
554 if (goog.userAgent.GECKO && savedCaretRange) {
555 // If we moved the selection, move it back so the user can't tell we did
556 // anything crazy and so the browser removeFormat that we call next
557 // will operate on the entire originally selected range.
558 range = this.getFieldObject().getRange();
559 this.restoreCaretsFromCave_();
560 var realSavedCaretRange = savedCaretRange.toAbstractRange();
561 var startRange = startInTable ? realSavedCaretRange : range;
562 var endRange = endInTable ? realSavedCaretRange : range;
563 var restoredRange =
564 goog.editor.plugins.RemoveFormatting.createRangeDelimitedByRanges_(
565 startRange, endRange);
566 restoredRange.select();
567 savedCaretRange.dispose();
568 }
569};

572/**
573 * Does a best-effort attempt at clobbering all formatting that the
574 * browser's execCommand couldn't clobber without being totally inefficient.
575 * Attempts to convert visual line breaks to BRs. Leaves anchors that contain an
576 * href and images.
577 * Adapted from Gmail's MessageUtil's htmlToPlainText. http://go/messageutil.js
578 * @param {string} html The original html of the message.
579 * @return {string} The unformatted html, which is just text, br's, anchors and
580 * images.
581 * @private
582 */
583goog.editor.plugins.RemoveFormatting.prototype.removeFormattingWorker_ =
584 function(html) {
585 var el = goog.dom.createElement(goog.dom.TagName.DIV);
586 el.innerHTML = html;
587
588 // Put everything into a string buffer to avoid lots of expensive string
589 // concatenation along the way.
590 var sb = [];
591 var stack = [el.childNodes, 0];
592
593 // Keep separate stacks for places where we need to keep track of
594 // how deeply embedded we are. These are analogous to the general stack.
595 var preTagStack = [];
596 var preTagLevel = 0; // Length of the prestack.
597 var tableStack = [];
598 var tableLevel = 0;
599
600 // sp = stack pointer, pointing to the stack array.
601 // decrement by 2 since the stack alternates node lists and
602 // processed node counts
603 for (var sp = 0; sp >= 0; sp -= 2) {
604 // Check if we should pop the table level.
605 var changedLevel = false;
606 while (tableLevel > 0 && sp <= tableStack[tableLevel - 1]) {
607 tableLevel--;
608 changedLevel = true;
609 }
610 if (changedLevel) {
611 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
612 }
613
614
615 // Check if we should pop the <pre>/<xmp> level.
616 changedLevel = false;
617 while (preTagLevel > 0 && sp <= preTagStack[preTagLevel - 1]) {
618 preTagLevel--;
619 changedLevel = true;
620 }
621 if (changedLevel) {
622 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
623 }
624
625 // The list of of nodes to process at the current stack level.
626 var nodeList = stack[sp];
627 // The number of nodes processed so far, stored in the stack immediately
628 // following the node list for that stack level.
629 var numNodesProcessed = stack[sp + 1];
630
631 while (numNodesProcessed < nodeList.length) {
632 var node = nodeList[numNodesProcessed++];
633 var nodeName = node.nodeName;
634
635 var formatted = this.getValueForNode(node);
636 if (goog.isDefAndNotNull(formatted)) {
637 sb.push(formatted);
638 continue;
639 }
640
641 // TODO(user): Handle case 'EMBED' and case 'OBJECT'.
642 switch (nodeName) {
643 case '#text':
644 // Note that IE does not preserve whitespace in the dom
645 // values, even in a pre tag, so this is useless for IE.
646 var nodeValue = preTagLevel > 0 ?
647 node.nodeValue :
648 goog.string.stripNewlines(node.nodeValue);
649 nodeValue = goog.string.htmlEscape(nodeValue);
650 sb.push(nodeValue);
651 continue;
652
653 case goog.dom.TagName.P:
654 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
655 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
656 break; // break (not continue) so that child nodes are processed.
657
658 case goog.dom.TagName.BR:
659 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
660 continue;
661
662 case goog.dom.TagName.TABLE:
663 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
664 tableStack[tableLevel++] = sp;
665 break;
666
667 case goog.dom.TagName.PRE:
668 case 'XMP':
669 // This doesn't fully handle xmp, since
670 // it doesn't actually ignore tags within the xmp tag.
671 preTagStack[preTagLevel++] = sp;
672 break;
673
674 case goog.dom.TagName.STYLE:
675 case goog.dom.TagName.SCRIPT:
676 case goog.dom.TagName.SELECT:
677 continue;
678
679 case goog.dom.TagName.A:
680 if (node.href && node.href != '') {
681 sb.push("<a href='");
682 sb.push(node.href);
683 sb.push("'>");
684 sb.push(this.removeFormattingWorker_(node.innerHTML));
685 sb.push('</a>');
686 continue; // Children taken care of.
687 } else {
688 break; // Take care of the children.
689 }
690
691 case goog.dom.TagName.IMG:
692 sb.push("<img src='");
693 sb.push(node.src);
694 sb.push("'");
695 // border=0 is a common way to not show a blue border around an image
696 // that is wrapped by a link. If we remove that, the blue border will
697 // show up, which to the user looks like adding format, not removing.
698 if (node.border == '0') {
699 sb.push(" border='0'");
700 }
701 sb.push('>');
702 continue;
703
704 case goog.dom.TagName.TD:
705 // Don't add a space for the first TD, we only want spaces to
706 // separate td's.
707 if (node.previousSibling) {
708 sb.push(' ');
709 }
710 break;
711
712 case goog.dom.TagName.TR:
713 // Don't add a newline for the first TR.
714 if (node.previousSibling) {
715 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
716 }
717 break;
718
719 case goog.dom.TagName.DIV:
720 var parent = node.parentNode;
721 if (parent.firstChild == node &&
722 goog.editor.plugins.RemoveFormatting.BLOCK_RE_.test(
723 parent.tagName)) {
724 // If a DIV is the first child of another element that itself is a
725 // block element, the DIV does not add a new line.
726 break;
727 }
728 // Otherwise, the DIV does add a new line. Fall through.
729
730 default:
731 if (goog.editor.plugins.RemoveFormatting.BLOCK_RE_.test(nodeName)) {
732 goog.editor.plugins.RemoveFormatting.appendNewline_(sb);
733 }
734 }
735
736 // Recurse down the node.
737 var children = node.childNodes;
738 if (children.length > 0) {
739 // Push the current state on the stack.
740 stack[sp++] = nodeList;
741 stack[sp++] = numNodesProcessed;
742
743 // Iterate through the children nodes.
744 nodeList = children;
745 numNodesProcessed = 0;
746 }
747 }
748 }
749
750 // Replace &nbsp; with white space.
751 return goog.string.normalizeSpaces(sb.join(''));
752};
