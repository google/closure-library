312/**
313 * Handles keyboard shortcuts. Preferred to using handleKey* as it will use
314 * the proper event based on browser and will be more performant. If
315 * handleKeyPress/handleKeyDown returns true, this will not be called. If the
316 * plugin handles the shortcut, it is responsible for dispatching appropriate
317 * events (change, selection change at the time of this comment). If the plugin
318 * calls execCommand on the editable field, then execCommand already takes care
319 * of dispatching events.
320 * NOTE: For performance reasons this is only called when any key is pressed

325 * @param {!goog.events.BrowserEvent} e The browser event.

327 * @param {boolean} isModifierPressed Whether the ctrl/meta key was pressed or
328 * not.

336/**
337 * Handles execCommand. This default implementation handles dispatching
338 * BEFORECHANGE, CHANGE, and SELECTIONCHANGE events, and calls
339 * execCommandInternal to perform the actual command. Plugins that want to
340 * do their own event dispatching should override execCommand, otherwise
341 * it is preferred to only override execCommandInternal.
342 *
343 * This version of execCommand will only work for single field plugins.
344 * Multi-field plugins must override execCommand.
345 *
346 * @param {string} command The command to execute.
347 * @param {...*} var_args Any additional parameters needed to
348 * execute the command.
349 * @return {*} The result of the execCommand, if any.
350 */
351goog.editor.Plugin.prototype.execCommand = function(command, var_args) {
352 // TODO(user): Replace all uses of isSilentCommand with plugins that just
353 // override this base execCommand method.
354 var silent = this.isSilentCommand(command);
355 if (!silent) {
356 // Stop listening to mutation events in Firefox while text formatting
357 // is happening. This prevents us from trying to size the field in the
358 // middle of an execCommand, catching the field in a strange intermediary
359 // state where both replacement nodes and original nodes are appended to
360 // the dom. Note that change events get turned back on by
361 // fieldObj.dispatchChange.
362 if (goog.userAgent.GECKO) {
363 this.getFieldObject().stopChangeEvents(true, true);
364 }
365
366 this.getFieldObject().dispatchBeforeChange();
367 }
368
369 try {
370 var result = this.execCommandInternal.apply(this, arguments);
371 } finally {
372 // If the above execCommandInternal call throws an exception, we still need
373 // to turn change events back on (see http://b/issue?id=1471355).
374 // NOTE: If if you add to or change the methods called in this finally
375 // block, please add them as expected calls to the unit test function
376 // testExecCommandException().
377 if (!silent) {
378 // dispatchChange includes a call to startChangeEvents, which unwinds the
379 // call to stopChangeEvents made before the try block.
380 this.getFieldObject().dispatchChange();
381 this.getFieldObject().dispatchSelectionChangeEvent();
382 }
383 }
384
385 return result;
386};
