15/**
16 * @fileoverview Class for managing requests via iFrames. Supports a number of
17 * methods of transfer.
18 *
19 * Gets and Posts can be performed and the resultant page read in as text,
20 * JSON, or from the HTML DOM.
21 *
22 * Using an iframe causes the throbber to spin, this is good for providing
23 * feedback to the user that an action has occurred.
24 *
25 * Requests do not affect the history stack, see goog.History if you require
26 * this behavior.
27 *
28 * The responseText and responseJson methods assume the response is plain,
29 * text. You can access the Iframe's DOM through responseXml if you need
30 * access to the raw HTML.
31 *
32 * Tested:
33 * + FF2.0 (Win Linux)
34 * + IE6, IE7
35 * + Opera 9.1,
36 * + Chrome
37 * - Opera 8.5 fails because of no textContent and buggy innerText support
38 *
39 * NOTE: Safari doesn't fire the onload handler when loading plain text files
40 *
41 * This has been tested with Drip in IE to ensure memory usage is as constant
42 * as possible. When making making thousands of requests, memory usage stays
43 * constant for a while but then starts increasing (<500k for 2000
44 * requests) -- this hasn't yet been tracked down yet, though it is cleared up
45 * after a refresh.
46 *
47 *
48 * BACKGROUND FILE UPLOAD:
49 * By posting an arbitrary form through an IframeIo object, it is possible to
50 * implement background file uploads. Here's how to do it:
51 *
52 * - Create a form:
53 * <pre>
54 * &lt;form id="form" enctype="multipart/form-data" method="POST"&gt;
55 * &lt;input name="userfile" type="file" /&gt;
56 * &lt;/form&gt;
57 * </pre>
58 *
59 * - Have the user click the file input
60 * - Create an IframeIo instance
61 * <pre>
62 * var io = new goog.net.IframeIo;
63 * goog.events.listen(io, goog.net.EventType.COMPLETE,
64 * function() { alert('Sent'); });
65 * io.sendFromForm(document.getElementById('form'));
66 * </pre>
67 *
68 *
69 * INCREMENTAL LOADING:
70 * Gmail sends down multiple script blocks which get executed as they are
71 * received by the client. This allows incremental rendering of the thread
72 * list and conversations.
73 *
74 * This requires collaboration with the server that is sending the requested
75 * page back. To set incremental loading up, you should:
76 *
77 * A) In the application code there should be an externed reference to
78 * <code>handleIncrementalData()</code>. e.g.
79 * goog.exportSymbol('GG_iframeFn', goog.net.IframeIo.handleIncrementalData);
80 *
81 * B) The response page should them call this method directly, an example
82 * response would look something like this:
83 * <pre>
84 * &lt;html&gt;
85 * &lt;head&gt;
86 * &lt;meta content="text/html;charset=UTF-8" http-equiv="content-type"&gt;
87 * &lt;/head&gt;
88 * &lt;body&gt;
89 * &lt;script&gt;
90 * D = top.P ? function(d) { top.GG_iframeFn(window, d) } : function() {};
91 * &lt;/script&gt;
92 *
93 * &lt;script&gt;D([1, 2, 3, 4, 5]);&lt;/script&gt;
94 * &lt;script&gt;D([6, 7, 8, 9, 10]);&lt;/script&gt;
95 * &lt;script&gt;D([11, 12, 13, 14, 15]);&lt;/script&gt;
96 * &lt;/body&gt;
97 * &lt;/html&gt;
98 * </pre>
99 *
100 * Your application should then listen, on the IframeIo instance, to the event
101 * goog.net.EventType.INCREMENTAL_DATA. The event object contains a
102 * 'data' member which is the content from the D() calls above.
103 *
104 * NOTE: There can be problems if you save a reference to the data object in IE.
105 * If you save an array, and the iframe is dispose, then the array looses its
106 * prototype and thus array methods like .join(). You can get around this by
107 * creating arrays using the parent window's Array constructor, or you can
108 * clone the array.
109 *
110 *
111 * EVENT MODEL:
112 * The various send methods work asynchronously. You can be notified about
113 * the current status of the request (completed, success or error) by
114 * listening for events on the IframeIo object itself. The following events
115 * will be sent:
116 * - goog.net.EventType.COMPLETE: when the request is completed
117 * (either sucessfully or unsuccessfully). You can find out about the result
118 * using the isSuccess() and getLastError
119 * methods.
120 * - goog.net.EventType.SUCCESS</code>: when the request was completed
121 * successfully
122 * - goog.net.EventType.ERROR: when the request failed
123 * - goog.net.EventType.ABORT: when the request has been aborted
124 *
125 * Example:
126 * <pre>
127 * var io = new goog.net.IframeIo();
128 * goog.events.listen(io, goog.net.EventType.COMPLETE,
129 * function() { alert('request complete'); });
130 * io.sendFromForm(...);
131 * </pre>
132 *
133 */

245/**
246 * Static send that creates a short lived instance of IframeIo to send the
247 * request.
248 * @param {goog.Uri|string} uri Uri of the request, it is up the caller to
249 * manage query string params.
250 * @param {Function=} opt_callback Event handler for when request is completed.
251 * @param {string=} opt_method Default is GET, POST uses a form to submit the
252 * request.
253 * @param {boolean=} opt_noCache Append a timestamp to the request to avoid
254 * caching.
255 * @param {Object|goog.structs.Map=} opt_data Map of key-value pairs that
256 * will be posted to the server via the iframe's form.
257 */
258goog.net.IframeIo.send = function(
259 uri, opt_callback, opt_method, opt_noCache, opt_data) {
260
261 var io = new goog.net.IframeIo();
262 goog.events.listen(io, goog.net.EventType.READY, io.dispose, false, io);
263 if (opt_callback) {
264 goog.events.listen(io, goog.net.EventType.COMPLETE, opt_callback);
265 }
266 io.send(uri, opt_method, opt_noCache, opt_data);
267};

291/**
292 * Handles incremental data and routes it to the correct iframeIo instance.
293 * The HTML page requested by the IframeIo instance should contain script blocks
294 * that call an externed reference to this method.
295 * @param {Window} win The window object.
296 * @param {Object} data The data object.
297 */
298goog.net.IframeIo.handleIncrementalData = function(win, data) {
299 // If this is the inner-frame, then we need to use the parent instead.
300 var iframeName = goog.string.endsWith(win.name,
301 goog.net.IframeIo.INNER_FRAME_SUFFIX) ? win.parent.name : win.name;
302
303 var iframeIoName = iframeName.substring(0, iframeName.lastIndexOf('_'));
304 var iframeIo = goog.net.IframeIo.getInstanceByName(iframeIoName);
305 if (iframeIo && iframeName == iframeIo.iframeName_) {
306 iframeIo.handleIncrementalData_(data);
307 } else {
308 var logger = goog.log.getLogger('goog.net.IframeIo');
309 goog.log.info(logger,
310 'Incremental iframe data routed for unknown iframe');
311 }
312};

513/**
514 * Sends a request via an iframe.
515 *
516 * A HTML form is used and submitted to the iframe, this simplifies the
517 * difference between GET and POST requests. The iframe needs to be created and
518 * destroyed for each request otherwise the request will contribute to the
519 * history stack.
520 *
521 * sendFromForm does some clever trickery (thanks jlim) in non-IE browsers to
522 * stop a history entry being added for POST requests.
523 *
524 * @param {goog.Uri|string} uri Uri of the request.
525 * @param {string=} opt_method Default is GET, POST uses a form to submit the
526 * request.
527 * @param {boolean=} opt_noCache Append a timestamp to the request to avoid
528 * caching.
529 * @param {Object|goog.structs.Map=} opt_data Map of key-value pairs.
530 */
531goog.net.IframeIo.prototype.send = function(
532 uri, opt_method, opt_noCache, opt_data) {
533
534 if (this.active_) {
535 throw Error('[goog.net.IframeIo] Unable to send, already active.');
536 }
537
538 var uriObj = new goog.Uri(uri);
539 this.lastUri_ = uriObj;
540 var method = opt_method ? opt_method.toUpperCase() : 'GET';
541
542 if (opt_noCache) {
543 uriObj.makeUnique();
544 }
545
546 goog.log.info(this.logger_,
547 'Sending iframe request: ' + uriObj + ' [' + method + ']');
548
549 // Build a form for this request
550 this.form_ = goog.net.IframeIo.getForm_();
551
552 if (method == 'GET') {
553 // For GET requests, we assume that the caller didn't want the queryparams
554 // already specified in the URI to be clobbered by the form, so we add the
555 // params here.
556 goog.net.IframeIo.addFormInputs_(this.form_, uriObj.getQueryData());
557 }
558
559 if (opt_data) {
560 // Create form fields for each of the data values
561 goog.net.IframeIo.addFormInputs_(this.form_, opt_data);
562 }
563
564 // Set the URI that the form will be posted
565 this.form_.action = uriObj.toString();
566 this.form_.method = method;
567
568 this.sendFormInternal_();
569 this.clearForm_();
570};

573/**
574 * Sends the data stored in an existing form to the server. The HTTP method
575 * should be specified on the form, the action can also be specified but can
576 * be overridden by the optional URI param.
577 *
578 * This can be used in conjunction will a file-upload input to upload a file in
579 * the background without affecting history.
580 *
581 * Example form:
582 * <pre>
583 * &lt;form action="/server/" enctype="multipart/form-data" method="POST"&gt;
584 * &lt;input name="userfile" type="file"&gt;
585 * &lt;/form&gt;
586 * </pre>
587 *
588 * @param {HTMLFormElement} form Form element used to send the request to the
589 * server.
590 * @param {string=} opt_uri Uri to set for the destination of the request, by
591 * default the uri will come from the form.
592 * @param {boolean=} opt_noCache Append a timestamp to the request to avoid
593 * caching.
594 */
595goog.net.IframeIo.prototype.sendFromForm = function(form, opt_uri,
596 opt_noCache) {
597 if (this.active_) {
598 throw Error('[goog.net.IframeIo] Unable to send, already active.');
599 }
600
601 var uri = new goog.Uri(opt_uri || form.action);
602 if (opt_noCache) {
603 uri.makeUnique();
604 }
605
606 goog.log.info(this.logger_, 'Sending iframe request from form: ' + uri);
607
608 this.lastUri_ = uri;
609 this.form_ = form;
610 this.form_.action = uri.toString();
611 this.sendFormInternal_();
612};

615/**
616 * Abort the current Iframe request
617 * @param {goog.net.ErrorCode=} opt_failureCode Optional error code to use -
618 * defaults to ABORT.
619 */
620goog.net.IframeIo.prototype.abort = function(opt_failureCode) {
621 if (this.active_) {
622 goog.log.info(this.logger_, 'Request aborted');
623 var requestIframe = this.getRequestIframe();
624 goog.asserts.assert(requestIframe);
625 goog.events.removeAll(requestIframe);
626 this.complete_ = false;
627 this.active_ = false;
628 this.success_ = false;
629 this.lastErrorCode_ = opt_failureCode || goog.net.ErrorCode.ABORT;
630
631 this.dispatchEvent(goog.net.EventType.ABORT);
632
633 this.makeReady_();
634 }
635};

638/** @override */
639goog.net.IframeIo.prototype.disposeInternal = function() {
640 goog.log.fine(this.logger_, 'Disposing iframeIo instance');
641
642 // If there is an active request, abort it
643 if (this.active_) {
644 goog.log.fine(this.logger_, 'Aborting active request');
645 this.abort();
646 }
647
648 // Call super-classes implementation (remove listeners)
649 goog.net.IframeIo.superClass_.disposeInternal.call(this);
650
651 // Add the current iframe to the list of iframes for disposal.
652 if (this.iframe_) {
653 this.scheduleIframeDisposal_();
654 }
655
656 // Disposes of the form
657 this.disposeForm_();
658
659 // Nullify anything that might cause problems and clear state
660 delete this.errorChecker_;
661 this.form_ = null;
662 this.lastCustomError_ = this.lastContent_ = this.lastContentHtml_ = null;
663 this.lastUri_ = null;
664 this.lastErrorCode_ = goog.net.ErrorCode.NO_ERROR;
665
666 delete goog.net.IframeIo.instances_[this.name_];
667};

817/**
818 * Submits the internal form to the iframe.
819 * @private
820 */
821goog.net.IframeIo.prototype.sendFormInternal_ = function() {
822 this.active_ = true;
823 this.complete_ = false;
824 this.lastErrorCode_ = goog.net.ErrorCode.NO_ERROR;
825
826 // Make Iframe
827 this.createIframe_();
828
829 if (goog.net.IframeIo.useIeReadyStateCodePath_()) {
830 // In IE<11 we simply create the frame, wait until it is ready, then post
831 // the form to the iframe and wait for the readystate to change to
832 // 'complete'
833
834 // Set the target to the iframe's name
835 this.form_.target = this.iframeName_ || '';
836 this.appendIframe_();
837 if (!this.ignoreResponse_) {
838 goog.events.listen(this.iframe_, goog.events.EventType.READYSTATECHANGE,
839 this.onIeReadyStateChange_, false, this);
840 }
841
842 /** @preserveTry */
843 try {
844 this.errorHandled_ = false;
845 this.form_.submit();
846 } catch (e) {
847 // If submit threw an exception then it probably means the page that the
848 // code is running on the local file system and the form's action was
849 // pointing to a file that doesn't exist, causing the browser to fire an
850 // exception. IE also throws an exception when it is working offline and
851 // the URL is not available.
852
853 if (!this.ignoreResponse_) {
854 goog.events.unlisten(
855 this.iframe_,
856 goog.events.EventType.READYSTATECHANGE,
857 this.onIeReadyStateChange_,
858 false,
859 this);
860 }
861
862 this.handleError_(goog.net.ErrorCode.ACCESS_DENIED);
863 }
864
865 } else {
866 // For all other browsers we do some trickery to ensure that there is no
867 // entry on the history stack. Thanks go to jlim for the prototype for this
868
869 goog.log.fine(this.logger_, 'Setting up iframes and cloning form');
870
871 this.appendIframe_();
872
873 var innerFrameName = this.iframeName_ +
874 goog.net.IframeIo.INNER_FRAME_SUFFIX;
875
876 // Open and document.write another iframe into the iframe
877 var doc = goog.dom.getFrameContentDocument(this.iframe_);
878 var html;
879 if (document.baseURI) {
880 // On Safari 4 and 5 the new iframe doesn't inherit the current baseURI.
881 html = goog.net.IframeIo.createIframeHtmlWithBaseUri_(innerFrameName);
882 } else {
883 html = goog.net.IframeIo.createIframeHtml_(innerFrameName);
884 }
885 if (goog.userAgent.OPERA) {
886 // Opera adds a history entry when document.write is used.
887 // Change the innerHTML of the page instead.
888 goog.dom.safe.setInnerHtml(doc.documentElement, html);
889 } else {
890 goog.dom.safe.documentWrite(doc, html);
891 }
892
893 // Listen for the iframe's load
894 if (!this.ignoreResponse_) {
895 goog.events.listen(doc.getElementById(innerFrameName),
896 goog.events.EventType.LOAD, this.onIframeLoaded_, false, this);
897 }
898
899 // Fix text areas, since importNode won't clone changes to the value
900 var textareas = this.form_.getElementsByTagName(goog.dom.TagName.TEXTAREA);
901 for (var i = 0, n = textareas.length; i < n; i++) {
902 // The childnodes represent the initial child nodes for the text area
903 // appending a text node essentially resets the initial value ready for
904 // it to be clones - while maintaining HTML escaping.
905 var value = textareas[i].value;
906 if (goog.dom.getRawTextContent(textareas[i]) != value) {
907 goog.dom.setTextContent(textareas[i], value);
908 textareas[i].value = value;
909 }
910 }
911
912 // Append a cloned form to the iframe
913 var clone = doc.importNode(this.form_, true);
914 clone.target = innerFrameName;
915 // Work around crbug.com/66987
916 clone.action = this.form_.action;
917 doc.body.appendChild(clone);
918
919 // Fix select boxes, importNode won't override the default value
920 var selects = this.form_.getElementsByTagName(goog.dom.TagName.SELECT);
921 var clones = clone.getElementsByTagName(goog.dom.TagName.SELECT);
922 for (var i = 0, n = selects.length; i < n; i++) {
923 var selectsOptions = selects[i].getElementsByTagName(
924 goog.dom.TagName.OPTION);
925 var clonesOptions = clones[i].getElementsByTagName(
926 goog.dom.TagName.OPTION);
927 for (var j = 0, m = selectsOptions.length; j < m; j++) {
928 clonesOptions[j].selected = selectsOptions[j].selected;
929 }
930 }
931
932 // IE and some versions of Firefox (1.5 - 1.5.07?) fail to clone the value
933 // attribute for <input type="file"> nodes, which results in an empty
934 // upload if the clone is submitted. Check, and if the clone failed, submit
935 // using the original form instead.
936 var inputs = this.form_.getElementsByTagName(goog.dom.TagName.INPUT);
937 var inputClones = clone.getElementsByTagName(goog.dom.TagName.INPUT);
938 for (var i = 0, n = inputs.length; i < n; i++) {
939 if (inputs[i].type == goog.dom.InputType.FILE) {
940 if (inputs[i].value != inputClones[i].value) {
941 goog.log.fine(this.logger_,
942 'File input value not cloned properly. Will ' +
943 'submit using original form.');
944 this.form_.target = innerFrameName;
945 clone = this.form_;
946 break;
947 }
948 }
949 }
950
951 goog.log.fine(this.logger_, 'Submitting form');
952
953 /** @preserveTry */
954 try {
955 this.errorHandled_ = false;
956 clone.submit();
957 doc.close();
958
959 if (goog.userAgent.GECKO) {
960 // This tests if firefox silently fails, this can happen, for example,
961 // when the server resets the connection because of a large file upload
962 this.firefoxSilentErrorTimeout_ =
963 goog.Timer.callOnce(this.testForFirefoxSilentError_, 250, this);
964 }
965
966 } catch (e) {
967 // If submit threw an exception then it probably means the page that the
968 // code is running on the local file system and the form's action was
969 // pointing to a file that doesn't exist, causing the browser to fire an
970 // exception.
971
972 goog.log.error(this.logger_,
973 'Error when submitting form: ' + goog.debug.exposeException(e));
974
975 if (!this.ignoreResponse_) {
976 goog.events.unlisten(doc.getElementById(innerFrameName),
977 goog.events.EventType.LOAD, this.onIframeLoaded_, false, this);
978 }
979
980 doc.close();
981
982 this.handleError_(goog.net.ErrorCode.FILE_NOT_FOUND);
983 }
984 }

1123 this.lastErrorCode_ = goog.net.ErrorCode.NO_ERROR;
1124 this.dispatchEvent(goog.net.EventType.COMPLETE);

1233/**
1234 * Schedules an iframe for disposal, async. We can't remove the iframes in the
1235 * same execution context as the response, otherwise some versions of Firefox
1236 * will not detect that the response has correctly finished and the loading bar
1237 * will stay active forever.
1238 * @private
1239 */
1240goog.net.IframeIo.prototype.scheduleIframeDisposal_ = function() {
1241 var iframe = this.iframe_;

1243 // There shouldn't be a case where the iframe is null and we get to this
1244 // stage, but the error reports in http://b/909448 indicate it is possible.
1245 if (iframe) {
1246 // NOTE(user): Stops Internet Explorer leaking the iframe object. This
1247 // shouldn't be needed, since the events have all been removed, which
1248 // should in theory clean up references. Oh well...
1249 iframe.onreadystatechange = null;
1250 iframe.onload = null;
1251 iframe.onerror = null;

1253 this.iframesForDisposal_.push(iframe);
1254 }

1265 // affect any other actions that may have been triggered by the request.
1266 this.iframeDisposalTimer_ = goog.Timer.callOnce(
1267 this.disposeIframes_, goog.net.IframeIo.IFRAME_DISPOSE_DELAY_MS, this);
