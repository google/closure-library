100/**
101 * Parses an IP Address in a string.
102 * If the string is malformed, the function will simply return null
103 * instead of raising an exception.
104 *
105 * @param {string} address The IP Address.
106 * @see {goog.net.Ipv4Address}
107 * @see {goog.net.Ipv6Address}
108 * @return {goog.net.IpAddress} The IP Address or null.
109 */
110goog.net.IpAddress.fromString = function(address) {
111 try {
112 if (address.indexOf(':') != -1) {
113 return new goog.net.Ipv6Address(address);
114 }
115
116 return new goog.net.Ipv4Address(address);
117 } catch (e) {
118 // Both constructors raise exception if the address is malformed (ie.
119 // invalid). The user of this function should not care about catching
120 // the exception, espcially if it's used to validate an user input.
121 return null;
122 }
123};

127 * Tries to parse a string represented as a host portion of an URI.
128 * See RFC 3986 for more details on IPv6 addresses inside URI.
129 * If the string is malformed, the function will simply return null
130 * instead of raising an exception.
131 *
132 * @param {string} address A RFC 3986 encoded IP address.
133 * @see {goog.net.Ipv4Address}
134 * @see {goog.net.Ipv6Address}
135 * @return {goog.net.IpAddress} The IP Address.
136 */
137goog.net.IpAddress.fromUriString = function(address) {
138 try {
139 if (goog.string.startsWith(address, '[') &&
140 goog.string.endsWith(address, ']')) {
141 return new goog.net.Ipv6Address(

143 }
144
145 return new goog.net.Ipv4Address(address);

147 // Both constructors raise exception if the address is malformed (ie.
148 // invalid). The user of this function should not care about catching
149 // the exception, espcially if it's used to validate an user input.
150 return null;
151 }

156/**
157 * Takes a string or a number and returns a IPv4 Address.
158 *
159 * This constructor accepts strings and instance of goog.math.Integer.
160 * If you pass a goog.math.Integer, make sure that its sign is set to positive.
161 * @param {(string|!goog.math.Integer)} address The address to store.
162 * @extends {goog.net.IpAddress}
163 * @constructor
164 * @final
165 */
166goog.net.Ipv4Address = function(address) {
167 var ip = goog.math.Integer.ZERO;
168 if (address instanceof goog.math.Integer) {
169 if (address.getSign() != 0 ||
170 address.lessThan(goog.math.Integer.ZERO) ||
171 address.greaterThan(goog.net.Ipv4Address.MAX_ADDRESS_)) {
172 throw Error('The address does not look like an IPv4.');
173 } else {
174 ip = goog.object.clone(address);
175 }
176 } else {
177 if (!goog.net.Ipv4Address.REGEX_.test(address)) {
178 throw Error(address + ' does not look like an IPv4 address.');
179 }
180
181 var octets = address.split('.');
182 if (octets.length != 4) {
183 throw Error(address + ' does not look like an IPv4 address.');
184 }
185
186 for (var i = 0; i < octets.length; i++) {
187 var parsedOctet = goog.string.toNumber(octets[i]);
188 if (isNaN(parsedOctet) ||
189 parsedOctet < 0 || parsedOctet > 255 ||
190 (octets[i].length != 1 && goog.string.startsWith(octets[i], '0'))) {
191 throw Error('In ' + address + ', octet ' + i + ' is not valid');
192 }
193 var intOctet = goog.math.Integer.fromNumber(parsedOctet);
194 ip = ip.shiftLeft(8).or(intOctet);
195 }
196 }
197 goog.net.Ipv4Address.base(
198 this, 'constructor', /** @type {!goog.math.Integer} */ (ip), 4);
199};
200goog.inherits(goog.net.Ipv4Address, goog.net.IpAddress);

231 * @override
232 */
233goog.net.Ipv4Address.prototype.toString = function() {
234 if (this.ipStr_) {
235 return this.ipStr_;
236 }
237
238 var ip = this.ip_.getBitsUnsigned(0);
239 var octets = [];
240 for (var i = 3; i >= 0; i--) {
241 octets[i] = String((ip & 0xff));
242 ip = ip >>> 8;
243 }
244
245 this.ipStr_ = octets.join('.');
246
247 return this.ipStr_;

260/**
261 * Takes a string or a number and returns an IPv6 Address.
262 *
263 * This constructor accepts strings and instance of goog.math.Integer.
264 * If you pass a goog.math.Integer, make sure that its sign is set to positive.
265 * @param {(string|!goog.math.Integer)} address The address to store.
266 * @constructor
267 * @extends {goog.net.IpAddress}
268 * @final
269 */
270goog.net.Ipv6Address = function(address) {
271 var ip = goog.math.Integer.ZERO;
272 if (address instanceof goog.math.Integer) {
273 if (address.getSign() != 0 ||
274 address.lessThan(goog.math.Integer.ZERO) ||
275 address.greaterThan(goog.net.Ipv6Address.MAX_ADDRESS_)) {
276 throw Error('The address does not look like a valid IPv6.');
277 } else {
278 ip = goog.object.clone(address);
279 }
280 } else {
281 if (!goog.net.Ipv6Address.REGEX_.test(address)) {
282 throw Error(address + ' is not a valid IPv6 address.');
283 }
284
285 var splitColon = address.split(':');
286 if (splitColon[splitColon.length - 1].indexOf('.') != -1) {
287 var newHextets = goog.net.Ipv6Address.dottedQuadtoHextets_(
288 splitColon[splitColon.length - 1]);
289 goog.array.removeAt(splitColon, splitColon.length - 1);
290 goog.array.extend(splitColon, newHextets);
291 address = splitColon.join(':');
292 }
293
294 var splitDoubleColon = address.split('::');
295 if (splitDoubleColon.length > 2 ||
296 (splitDoubleColon.length == 1 && splitColon.length != 8)) {
297 throw Error(address + ' is not a valid IPv6 address.');
298 }
299
300 var ipArr;
301 if (splitDoubleColon.length > 1) {
302 ipArr = goog.net.Ipv6Address.explode_(splitDoubleColon);
303 } else {
304 ipArr = splitColon;
305 }
306
307 if (ipArr.length != 8) {
308 throw Error(address + ' is not a valid IPv6 address');
309 }
310
311 for (var i = 0; i < ipArr.length; i++) {
312 var parsedHextet = goog.math.Integer.fromString(ipArr[i], 16);
313 if (parsedHextet.lessThan(goog.math.Integer.ZERO) ||
314 parsedHextet.greaterThan(goog.net.Ipv6Address.MAX_HEXTET_VALUE_)) {
315 throw Error(ipArr[i] + ' in ' + address + ' is not a valid hextet.');
316 }
317 ip = ip.shiftLeft(16).or(parsedHextet);
318 }
319 }
320 goog.net.Ipv6Address.base(
321 this, 'constructor', /** @type {!goog.math.Integer} */ (ip), 6);
322};
323goog.inherits(goog.net.Ipv6Address, goog.net.IpAddress);

362/**
363 * @override
364 */
365goog.net.Ipv6Address.prototype.toString = function() {
366 if (this.ipStr_) {
367 return this.ipStr_;
368 }
369
370 var outputArr = [];
371 for (var i = 3; i >= 0; i--) {
372 var bits = this.ip_.getBitsUnsigned(i);
373 var firstHextet = bits >>> 16;
374 var secondHextet = bits & 0xffff;
375 outputArr.push(firstHextet.toString(16));
376 outputArr.push(secondHextet.toString(16));
377 }
378
379 outputArr = goog.net.Ipv6Address.compress_(outputArr);
380 this.ipStr_ = outputArr.join(':');
381 return this.ipStr_;
382};

393/**
394 * This method is in charge of expanding/exploding an IPv6 string from its
395 * compressed form.
396 * @private
397 * @param {!Array<string>} address An IPv6 address split around '::'.
398 * @return {!Array<string>} The expanded version of the IPv6.
399 */
400goog.net.Ipv6Address.explode_ = function(address) {
401 var basePart = address[0].split(':');
402 var secondPart = address[1].split(':');
403
404 if (basePart.length == 1 && basePart[0] == '') {
405 basePart = [];
406 }
407 if (secondPart.length == 1 && secondPart[0] == '') {
408 secondPart = [];
409 }
410
411 // Now we fill the gap with 0.
412 var gap = 8 - (basePart.length + secondPart.length);
413
414 if (gap < 1) {
415 return [];
416 }
417
418 goog.array.extend(basePart, goog.array.repeat('0', gap));
419
420 // Now we merge the basePart + gap + secondPart
421 goog.array.extend(basePart, secondPart);
422
423 return basePart;
424};

427/**
428 * This method is in charge of compressing an expanded IPv6 array of hextets.
429 * @private
430 * @param {!Array<string>} hextets The array of hextet.
431 * @return {!Array<string>} The compressed version of this array.
432 */
433goog.net.Ipv6Address.compress_ = function(hextets) {
434 var bestStart = -1;
435 var start = -1;
436 var bestSize = 0;
437 var size = 0;
438 for (var i = 0; i < hextets.length; i++) {
439 if (hextets[i] == '0') {
440 size++;
441 if (start == -1) {
442 start = i;
443 }
444 if (size > bestSize) {
445 bestSize = size;
446 bestStart = start;
447 }
448 } else {
449 start = -1;
450 size = 0;
451 }
452 }
453
454 if (bestSize > 0) {
455 if ((bestStart + bestSize) == hextets.length) {
456 hextets.push('');
457 }
458 hextets.splice(bestStart, bestSize, '');
459
460 if (bestStart == 0) {
461 hextets = [''].concat(hextets);
462 }
463 }
464 return hextets;
465};

468/**
469 * This method will convert an IPv4 to a list of 2 hextets.
470 *
471 * For instance, 1.2.3.4 will be converted to ['0102', '0304'].
472 * @private
473 * @param {string} quads An IPv4 as a string.
474 * @return {!Array<string>} A list of 2 hextets.
475 */
476goog.net.Ipv6Address.dottedQuadtoHextets_ = function(quads) {
477 var ip4 = new goog.net.Ipv4Address(quads).toInteger();
478 var bits = ip4.getBitsUnsigned(0);
479 var hextets = [];
480
481 hextets.push(((bits >>> 16) & 0xffff).toString(16));
482 hextets.push((bits & 0xffff).toString(16));
483
484 return hextets;
485};
