139/**
140 * Create a new range wrapper that selects the area between the given nodes,
141 * accounting for the given offsets. Do not use this method directly - please
142 * use goog.dom.Range.createFrom* instead.
143 * @param {Node} anchorNode The node to start with.
144 * @param {number} anchorOffset The offset within the node to start.
145 * @param {Node} focusNode The node to end with.
146 * @param {number} focusOffset The offset within the node to end.
147 * @return {!goog.dom.TextRange} A range wrapper object.
148 */
149goog.dom.TextRange.createFromNodes = function(anchorNode, anchorOffset,
150 focusNode, focusOffset) {
151 var range = new goog.dom.TextRange();
152 range.isReversed_ = /** @suppress {missingRequire} */ (
153 goog.dom.Range.isReversed(anchorNode, anchorOffset,
154 focusNode, focusOffset));
155
156 // Avoid selecting terminal elements directly
157 if (goog.dom.isElement(anchorNode) && !goog.dom.canHaveChildren(anchorNode)) {
158 var parent = anchorNode.parentNode;
159 anchorOffset = goog.array.indexOf(parent.childNodes, anchorNode);
160 anchorNode = parent;
161 }
162
163 if (goog.dom.isElement(focusNode) && !goog.dom.canHaveChildren(focusNode)) {
164 var parent = focusNode.parentNode;
165 focusOffset = goog.array.indexOf(parent.childNodes, focusNode);
166 focusNode = parent;
167 }
168
169 // Initialize the range as a W3C style range.
170 if (range.isReversed_) {
171 range.startNode_ = focusNode;
172 range.startOffset_ = focusOffset;
173 range.endNode_ = anchorNode;
174 range.endOffset_ = anchorOffset;
175 } else {
176 range.startNode_ = anchorNode;
177 range.startOffset_ = anchorOffset;
178 range.endNode_ = focusNode;
179 range.endOffset_ = focusOffset;
180 }
181
182 return range;
183};

342/** @override */
343goog.dom.TextRange.prototype.containsRange = function(otherRange,
344 opt_allowPartial) {
345 var otherRangeType = otherRange.getType();
346 if (otherRangeType == goog.dom.RangeType.TEXT) {
347 return this.getBrowserRangeWrapper_().containsRange(
348 otherRange.getBrowserRangeWrapper_(), opt_allowPartial);
349 } else if (otherRangeType == goog.dom.RangeType.CONTROL) {
350 var elements = otherRange.getElements();
351 var fn = opt_allowPartial ? goog.array.some : goog.array.every;
352 return fn(elements, function(el) {
353 return this.containsNode(el, opt_allowPartial);
354 }, this);
355 }
356 return false;
357};

360/**
361 * Tests if the given node is in a document.
362 * @param {Node} node The node to check.
363 * @return {boolean} Whether the given node is in the given document.
364 */
365goog.dom.TextRange.isAttachedNode = function(node) {
366 if (goog.userAgent.IE && !goog.userAgent.isDocumentModeOrHigher(9)) {
367 var returnValue = false;
368 /** @preserveTry */
369 try {
370 returnValue = node.parentNode;
371 } catch (e) {
372 // IE sometimes throws Invalid Argument errors when a node is detached.
373 // Note: trying to return a value from the above try block can cause IE
374 // to crash. It is necessary to use the local returnValue
375 }
376 return !!returnValue;
377 } else {
378 return goog.dom.contains(node.ownerDocument.body, node);
379 }
380};

384goog.dom.TextRange.prototype.isRangeInDocument = function() {
385 // Ensure any cached nodes are in the document. IE also allows ranges to
386 // become detached, so we check if the range is still in the document as
387 // well for IE.
388 return (!this.startNode_ ||
389 goog.dom.TextRange.isAttachedNode(this.startNode_)) &&
390 (!this.endNode_ ||

393 this.getBrowserRangeWrapper_().isRangeInDocument());

424/** @override */
425goog.dom.TextRange.prototype.getPastableHtml = function() {
426 // TODO(robbyw): Get any attributes the table or tr has.
427
428 var html = this.getValidHtml();
429
430 if (html.match(/^\s*<td\b/i)) {
431 // Match html starting with a TD.
432 html = '<table><tbody><tr>' + html + '</tr></tbody></table>';
433 } else if (html.match(/^\s*<tr\b/i)) {
434 // Match html starting with a TR.
435 html = '<table><tbody>' + html + '</tbody></table>';
436 } else if (html.match(/^\s*<tbody\b/i)) {
437 // Match html starting with a TBODY.
438 html = '<table>' + html + '</table>';
439 } else if (html.match(/^\s*<li\b/i)) {
440 // Match html starting with an LI.
441 var container = this.getContainer();
442 var tagType = goog.dom.TagName.UL;
443 while (container) {
444 if (container.tagName == goog.dom.TagName.OL) {
445 tagType = goog.dom.TagName.OL;
446 break;
447 } else if (container.tagName == goog.dom.TagName.UL) {
448 break;
449 }
450 container = container.parentNode;
451 }
452 html = goog.string.buildString('<', tagType, '>', html, '</', tagType, '>');
453 }
454
455 return html;
456};

532/** @override */
533goog.dom.TextRange.prototype.collapse = function(toAnchor) {
534 var toStart = this.isReversed() ? !toAnchor : toAnchor;
535
536 if (this.browserRangeWrapper_) {
537 this.browserRangeWrapper_.collapse(toStart);
538 }
539
540 if (toStart) {
541 this.endNode_ = this.startNode_;
542 this.endOffset_ = this.startOffset_;
543 } else {
544 this.startNode_ = this.endNode_;
545 this.startOffset_ = this.endOffset_;
546 }
547
548 // Collapsed ranges can't be reversed
549 this.isReversed_ = false;
550};
