15/**
16 * @fileoverview Provides a 'paste' event detector that works consistently
17 * across different browsers.
18 *
19 * IE5, IE6, IE7, Safari3.0 and FF3.0 all fire 'paste' events on textareas.
20 * FF2 doesn't. This class uses 'paste' events when they are available
21 * and uses heuristics to detect the 'paste' event when they are not available.
22 *
23 * Known issue: will not detect paste events in FF2 if you pasted exactly the
24 * same existing text.
25 * Known issue: Opera + Mac doesn't work properly because of the meta key. We
26 * can probably fix that. TODO(user): {@link KeyboardShortcutHandler} does not
27 * work either very well with opera + mac. fix that.
28 *
29 * @supported IE5, IE6, IE7, Safari3.0, Chrome, FF2.0 (linux) and FF3.0 and
30 * Opera (mac and windows).
31 *
32 * @see ../demos/pastehandler.html
33 */

51/**
52 * A paste event detector. Gets an {@code element} as parameter and fires
53 * {@code goog.events.PasteHandler.EventType.PASTE} events when text is
54 * pasted in the {@code element}. Uses heuristics to detect paste events in FF2.
55 * See more details of the heuristic on {@link #handleEvent_}.
56 *
57 * @param {Element} element The textarea element we are listening on.
58 * @constructor
59 * @extends {goog.events.EventTarget}
60 */
61goog.events.PasteHandler = function(element) {
62 goog.events.EventTarget.call(this);
63
64 /**
65 * The element that you want to listen for paste events on.
66 * @type {Element}
67 * @private
68 */
69 this.element_ = element;
70
71 /**
72 * The last known value of the element. Kept to check if things changed. See
73 * more details on {@link #handleEvent_}.
74 * @type {string}
75 * @private
76 */
77 this.oldValue_ = this.element_.value;
78
79 /**
80 * Handler for events.
81 * @type {goog.events.EventHandler<!goog.events.PasteHandler>}
82 * @private
83 */
84 this.eventHandler_ = new goog.events.EventHandler(this);
85
86 /**
87 * The last time an event occurred on the element. Kept to check whether the
88 * last event was generated by two input events or by multiple fast key events
89 * that got swallowed. See more details on {@link #handleEvent_}.
90 * @type {number}
91 * @private
92 */
93 this.lastTime_ = goog.now();
94
95 if (goog.userAgent.WEBKIT ||
96 goog.userAgent.IE ||
97 goog.userAgent.GECKO && goog.userAgent.isVersionOrHigher('1.9')) {
98 // Most modern browsers support the paste event.
99 this.eventHandler_.listen(element, goog.events.EventType.PASTE,
100 this.dispatch_);
101 } else {
102 // But FF2 and Opera doesn't. we listen for a series of events to try to
103 // find out if a paste occurred. We enumerate and cover all known ways to
104 // paste text on textareas. See more details on {@link #handleEvent_}.
105 var events = [
106 goog.events.EventType.KEYDOWN,
107 goog.events.EventType.BLUR,
108 goog.events.EventType.FOCUS,
109 goog.events.EventType.MOUSEOVER,
110 'input'
111 ];
112 this.eventHandler_.listen(element, events, this.handleEvent_);
113 }
114
115 /**
116 * ConditionalDelay used to poll for changes in the text element once users
117 * paste text. Browsers fire paste events BEFORE the text is actually present
118 * in the element.value property.
119 * @type {goog.async.ConditionalDelay}
120 * @private
121 */
122 this.delay_ = new goog.async.ConditionalDelay(
123 goog.bind(this.checkUpdatedText_, this));
124
125};
126goog.inherits(goog.events.PasteHandler, goog.events.EventTarget);

243/**
244 * Checks whether the element.value property was updated, and if so, dispatches
245 * the event that let clients know that the text is available.
246 * @return {boolean} Whether the polling should stop or not, based on whether
247 * we found a text change or not.
248 * @private
249 */
250goog.events.PasteHandler.prototype.checkUpdatedText_ = function() {
251 if (this.oldValue_ == this.element_.value) {
252 return false;
253 }
254 goog.log.info(this.logger_, 'detected textchange after paste');
255 this.dispatchEvent(goog.events.PasteHandler.EventType.AFTER_PASTE);
256 return true;
257};

260/**
261 * Dispatches the paste event.
262 * @param {goog.events.BrowserEvent} e The underlying browser event.
263 * @private
264 */
265goog.events.PasteHandler.prototype.dispatch_ = function(e) {
266 var event = new goog.events.BrowserEvent(e.getBrowserEvent());
267 event.type = goog.events.PasteHandler.EventType.PASTE;
268 this.dispatchEvent(event);
269
270 // Starts polling for updates in the element.value property so we can tell
271 // when do dispatch the AFTER_PASTE event. (We do an initial check after an
272 // async delay of 0 msec since some browsers update the text right away and
273 // our poller will always wait one period before checking).
274 goog.Timer.callOnce(function() {
275 if (!this.checkUpdatedText_()) {
276 this.delay_.start(
277 goog.events.PasteHandler.PASTE_POLLING_PERIOD_MS_,
278 goog.events.PasteHandler.PASTE_POLLING_TIMEOUT_MS_);
279 }
280 }, 0, this);
281};

284/**
285 * The main event handler which implements a state machine.
286 *
287 * To handle FF2, we enumerate and cover all the known ways a user can paste:
288 *
289 * 1) ctrl+v, shift+insert, cmd+v
290 * 2) right click -> paste
291 * 3) edit menu -> paste
292 * 4) drag and drop
293 * 5) middle click
294 *
295 * (1) is easy and can be detected by listening for key events and finding out
296 * which keys are pressed. (2), (3), (4) and (5) do not generate a key event,
297 * so we need to listen for more than that. (2-5) all generate 'input' events,
298 * but so does key events. So we need to have some sort of 'how did the input
299 * event was generated' history algorithm.
300 *
301 * (2) is an interesting case in Opera on a Mac: since Macs does not have two
302 * buttons, right clicking involves pressing the CTRL key. Even more interesting
303 * is the fact that opera does NOT set the e.ctrlKey bit. Instead, it sets
304 * e.keyCode = 0.
305 * {@link http://www.quirksmode.org/js/keys.html}
306 *
307 * (1) is also an interesting case in Opera on a Mac: Opera is the only browser
308 * covered by this class that can detect the cmd key (FF2 can't apparently). And
309 * it fires e.keyCode = 17, which is the CTRL key code.
310 * {@link http://www.quirksmode.org/js/keys.html}
311 *
312 * NOTE(user, pbarry): There is an interesting thing about (5): on Linux, (5)
313 * pastes the last thing that you highlighted, not the last thing that you
314 * ctrl+c'ed. This code will still generate a {@code PASTE} event though.
315 *
316 * We enumerate all the possible steps a user can take to paste text and we
317 * implemented the transition between the steps in a state machine. The
318 * following is the design of the state machine:
319 *
320 * matching paths:
321 *
322 * (1) happens on INIT -> FOCUSED -> TYPING -> [e.ctrlKey & e.keyCode = 'v']
323 * (2-3) happens on INIT -> FOCUSED -> [input event happened]
324 * (4) happens on INIT -> [mouseover && text changed]
325 *
326 * non matching paths:
327 *
328 * user is typing normally
329 * INIT -> FOCUS -> TYPING -> INPUT -> INIT
330 *
331 * @param {goog.events.BrowserEvent} e The underlying browser event.
332 * @private
333 */
334goog.events.PasteHandler.prototype.handleEvent_ = function(e) {
335 // transition between states happen at each browser event, and depend on the
336 // current state, the event that led to this state, and the event input.
337 switch (this.state_) {
338 case goog.events.PasteHandler.State.INIT: {
339 this.handleUnderInit_(e);
340 break;
341 }
342 case goog.events.PasteHandler.State.FOCUSED: {
343 this.handleUnderFocused_(e);
344 break;
345 }
346 case goog.events.PasteHandler.State.TYPING: {
347 this.handleUnderTyping_(e);
348 break;
349 }
350 default: {
351 goog.log.error(this.logger_, 'invalid ' + this.state_ + ' state');
352 }
353 }
354 this.lastTime_ = goog.now();
355 this.oldValue_ = this.element_.value;
356 goog.log.info(this.logger_, e.type + ' -> ' + this.state_);
357 this.previousEvent_ = e.type;
358};

361/**
362 * {@code goog.events.PasteHandler.EventType.INIT} is the first initial state
363 * the textarea is found. You can only leave this state by setting focus on the
364 * textarea, which is how users will input text. You can also paste things using
365 * drag and drop, which will not generate a {@code goog.events.EventType.FOCUS}
366 * event, but will generate a {@code goog.events.EventType.MOUSEOVER}.
367 *
368 * For browsers that support the 'paste' event, we match it and stay on the same
369 * state.
370 *
371 * @param {goog.events.BrowserEvent} e The underlying browser event.
372 * @private
373 */
374goog.events.PasteHandler.prototype.handleUnderInit_ = function(e) {
375 switch (e.type) {
376 case goog.events.EventType.BLUR: {
377 this.state_ = goog.events.PasteHandler.State.INIT;
378 break;
379 }
380 case goog.events.EventType.FOCUS: {
381 this.state_ = goog.events.PasteHandler.State.FOCUSED;
382 break;
383 }
384 case goog.events.EventType.MOUSEOVER: {
385 this.state_ = goog.events.PasteHandler.State.INIT;
386 if (this.element_.value != this.oldValue_) {
387 goog.log.info(this.logger_, 'paste by dragdrop while on init!');
388 this.dispatch_(e);
389 }
390 break;
391 }
392 default: {
393 goog.log.error(this.logger_,
394 'unexpected event ' + e.type + 'during init');
395 }
396 }
397};

400/**
401 * {@code goog.events.PasteHandler.EventType.FOCUSED} is typically the second
402 * state the textarea will be, which is followed by the {@code INIT} state. On
403 * this state, users can paste in three different ways: edit -> paste,
404 * right click -> paste and drag and drop.
405 *
406 * The latter will generate a {@code goog.events.EventType.MOUSEOVER} event,
407 * which we match by making sure the textarea text changed. The first two will
408 * generate an 'input', which we match by making sure it was NOT generated by a
409 * key event (which also generates an 'input' event).
410 *
411 * Unfortunately, in Firefox, if you type fast, some KEYDOWN events are
412 * swallowed but an INPUT event may still happen. That means we need to
413 * differentiate between two consecutive INPUT events being generated either by
414 * swallowed key events OR by a valid edit -> paste -> edit -> paste action. We
415 * do this by checking a minimum time between the two events. This heuristic
416 * seems to work well, but it is obviously a heuristic :).
417 *
418 * @param {goog.events.BrowserEvent} e The underlying browser event.
419 * @private
420 */
421goog.events.PasteHandler.prototype.handleUnderFocused_ = function(e) {
422 switch (e.type) {
423 case 'input' : {
424 // there are two different events that happen in practice that involves
425 // consecutive 'input' events. we use a heuristic to differentiate
426 // between the one that generates a valid paste action and the one that
427 // doesn't.
428 // @see testTypingReallyFastDispatchesTwoInputEventsBeforeTheKEYDOWNEvent
429 // and
430 // @see testRightClickRightClickAlsoDispatchesTwoConsecutiveInputEvents
431 // Notice that an 'input' event may be also triggered by a 'middle click'
432 // paste event, which is described in
433 // @see testMiddleClickWithoutFocusTriggersPasteEvent
434 var minimumMilisecondsBetweenInputEvents = this.lastTime_ +
435 goog.events.PasteHandler.
436 MANDATORY_MS_BETWEEN_INPUT_EVENTS_TIE_BREAKER;
437 if (goog.now() > minimumMilisecondsBetweenInputEvents ||
438 this.previousEvent_ == goog.events.EventType.FOCUS) {
439 goog.log.info(this.logger_, 'paste by textchange while focused!');
440 this.dispatch_(e);
441 }
442 break;
443 }
444 case goog.events.EventType.BLUR: {
445 this.state_ = goog.events.PasteHandler.State.INIT;
446 break;
447 }
448 case goog.events.EventType.KEYDOWN: {
449 goog.log.info(this.logger_, 'key down ... looking for ctrl+v');
450 // Opera + MAC does not set e.ctrlKey. Instead, it gives me e.keyCode = 0.
451 // http://www.quirksmode.org/js/keys.html
452 if (goog.userAgent.MAC && goog.userAgent.OPERA && e.keyCode == 0 ||
453 goog.userAgent.MAC && goog.userAgent.OPERA && e.keyCode == 17) {
454 break;
455 }
456 this.state_ = goog.events.PasteHandler.State.TYPING;
457 break;
458 }
459 case goog.events.EventType.MOUSEOVER: {
460 if (this.element_.value != this.oldValue_) {
461 goog.log.info(this.logger_, 'paste by dragdrop while focused!');
462 this.dispatch_(e);
463 }
464 break;
465 }
466 default: {
467 goog.log.error(this.logger_,
468 'unexpected event ' + e.type + ' during focused');
469 }
470 }
471};

474/**
475 * {@code goog.events.PasteHandler.EventType.TYPING} is the third state
476 * this class can be. It exists because each KEYPRESS event will ALSO generate
477 * an INPUT event (because the textarea value changes), and we need to
478 * differentiate between an INPUT event generated by a key event and an INPUT
479 * event generated by edit -> paste actions.
480 *
481 * This is the state that we match the ctrl+v pattern.
482 *
483 * @param {goog.events.BrowserEvent} e The underlying browser event.
484 * @private
485 */
486goog.events.PasteHandler.prototype.handleUnderTyping_ = function(e) {
487 switch (e.type) {
488 case 'input' : {
489 this.state_ = goog.events.PasteHandler.State.FOCUSED;
490 break;
491 }
492 case goog.events.EventType.BLUR: {
493 this.state_ = goog.events.PasteHandler.State.INIT;
494 break;
495 }
496 case goog.events.EventType.KEYDOWN: {
497 if (e.ctrlKey && e.keyCode == goog.events.KeyCodes.V ||
498 e.shiftKey && e.keyCode == goog.events.KeyCodes.INSERT ||
499 e.metaKey && e.keyCode == goog.events.KeyCodes.V) {
500 goog.log.info(this.logger_, 'paste by ctrl+v while keypressed!');
501 this.dispatch_(e);
502 }
503 break;
504 }
505 case goog.events.EventType.MOUSEOVER: {
506 if (this.element_.value != this.oldValue_) {
507 goog.log.info(this.logger_, 'paste by dragdrop while keypressed!');
508 this.dispatch_(e);
509 }
510 break;
511 }
512 default: {
513 goog.log.error(this.logger_,
514 'unexpected event ' + e.type + ' during keypressed');
515 }
516 }
