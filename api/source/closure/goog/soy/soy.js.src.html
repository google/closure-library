31/**
32 * @define {boolean} Whether to require all Soy templates to be "strict html".
33 * Soy templates that use strict autoescaping forbid noAutoescape along with
34 * many dangerous directives, and return a runtime type SanitizedContent that
35 * marks them as safe.
36 *
37 * If this flag is enabled, Soy templates will fail to render if a template
38 * returns plain text -- indicating it is a non-strict template.
39 */
40goog.define('goog.soy.REQUIRE_STRICT_AUTOESCAPE', false);

43/**
44 * Renders a Soy template and then set the output string as
45 * the innerHTML of an element. It is recommended to use this helper function
46 * instead of directly setting innerHTML in your hand-written code, so that it
47 * will be easier to audit the code for cross-site scripting vulnerabilities.
48 *
49 * @param {Element} element The element whose content we are rendering into.
50 * @param {null|function(ARG_TYPES, null=, Object<string, *>=):*} template
51 * The Soy template defining the element's content.
52 * @param {ARG_TYPES=} opt_templateData The data for the template.
53 * @param {Object=} opt_injectedData The injected data for the template.
54 * @template ARG_TYPES
55 */
56goog.soy.renderElement = function(element, template, opt_templateData,
57 opt_injectedData) {
58 // Soy template parameter is only nullable for historical reasons.
59 goog.asserts.assert(template, 'Soy template may not be null.');
60 element.innerHTML = goog.soy.ensureTemplateOutputHtml_(template(
61 opt_templateData || goog.soy.defaultTemplateData_, undefined,
62 opt_injectedData));
63};

66/**
67 * Renders a Soy template into a single node or a document
68 * fragment. If the rendered HTML string represents a single node, then that
69 * node is returned (note that this is *not* a fragment, despite them name of
70 * the method). Otherwise a document fragment is returned containing the
71 * rendered nodes.
72 *
73 * @param {null|function(ARG_TYPES, null=, Object<string, *>=):*} template
74 * The Soy template defining the element's content.
75 * @param {ARG_TYPES=} opt_templateData The data for the template.
76 * @param {Object=} opt_injectedData The injected data for the template.
77 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper used to
78 * create DOM nodes; defaults to {@code goog.dom.getDomHelper}.
79 * @return {!Node} The resulting node or document fragment.
80 * @template ARG_TYPES
81 */
82goog.soy.renderAsFragment = function(template, opt_templateData,
83 opt_injectedData, opt_domHelper) {
84 // Soy template parameter is only nullable for historical reasons.
85 goog.asserts.assert(template, 'Soy template may not be null.');
86 var dom = opt_domHelper || goog.dom.getDomHelper();
87 var html = goog.soy.ensureTemplateOutputHtml_(
88 template(opt_templateData || goog.soy.defaultTemplateData_,
89 undefined, opt_injectedData));
90 goog.soy.assertFirstTagValid_(html);
91 return dom.htmlToDocumentFragment(html);
92};

95/**
96 * Renders a Soy template into a single node. If the rendered
97 * HTML string represents a single node, then that node is returned. Otherwise,
98 * a DIV element is returned containing the rendered nodes.
99 *
100 * @param {null|function(ARG_TYPES, null=, Object<string, *>=):*} template
101 * The Soy template defining the element's content.
102 * @param {ARG_TYPES=} opt_templateData The data for the template.
103 * @param {Object=} opt_injectedData The injected data for the template.
104 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper used to
105 * create DOM nodes; defaults to {@code goog.dom.getDomHelper}.
106 * @return {!Element} Rendered template contents, wrapped in a parent DIV
107 * element if necessary.
108 * @template ARG_TYPES
109 */
110goog.soy.renderAsElement = function(template, opt_templateData,
111 opt_injectedData, opt_domHelper) {
112 // Soy template parameter is only nullable for historical reasons.
113 goog.asserts.assert(template, 'Soy template may not be null.');
114 var dom = opt_domHelper || goog.dom.getDomHelper();
115 var wrapper = dom.createElement(goog.dom.TagName.DIV);
116 var html = goog.soy.ensureTemplateOutputHtml_(template(
117 opt_templateData || goog.soy.defaultTemplateData_,
118 undefined, opt_injectedData));
119 goog.soy.assertFirstTagValid_(html);
120 wrapper.innerHTML = html;
121
122 // If the template renders as a single element, return it.
123 if (wrapper.childNodes.length == 1) {
124 var firstChild = wrapper.firstChild;
125 if (firstChild.nodeType == goog.dom.NodeType.ELEMENT) {
126 return /** @type {!Element} */ (firstChild);
127 }
128 }
129
130 // Otherwise, return the wrapper DIV.
131 return wrapper;
132};

135/**
136 * Ensures the result is "safe" to insert as HTML.
137 *
138 * Note if the template has non-strict autoescape, the guarantees here are very
139 * weak. It is recommended applications switch to requiring strict
140 * autoescaping over time by tweaking goog.soy.REQUIRE_STRICT_AUTOESCAPE.
141 *
142 * In the case the argument is a SanitizedContent object, it either must
143 * already be of kind HTML, or if it is kind="text", the output will be HTML
144 * escaped.
145 *
146 * @param {*} templateResult The template result.
147 * @return {string} The assumed-safe HTML output string.
148 * @private
149 */
150goog.soy.ensureTemplateOutputHtml_ = function(templateResult) {
151 // Allow strings as long as strict autoescaping is not mandated. Note we
152 // allow everything that isn't an object, because some non-escaping templates
153 // end up returning non-strings if their only print statement is a
154 // non-escaped argument, plus some unit tests spoof templates.
155 // TODO(gboyer): Track down and fix these cases.
156 if (!goog.soy.REQUIRE_STRICT_AUTOESCAPE && !goog.isObject(templateResult)) {
157 return String(templateResult);
158 }
159
160 // Allow SanitizedContent of kind HTML.
161 if (templateResult instanceof goog.soy.data.SanitizedContent) {
162 templateResult = /** @type {!goog.soy.data.SanitizedContent} */ (
163 templateResult);
164 var ContentKind = goog.soy.data.SanitizedContentKind;
165 if (templateResult.contentKind === ContentKind.HTML) {
166 return goog.asserts.assertString(templateResult.getContent());
167 }
168 if (templateResult.contentKind === ContentKind.TEXT) {
169 // Allow text to be rendered, as long as we escape it. Other content
170 // kinds will fail, since we don't know what to do with them.
171 // TODO(gboyer): Perhaps also include URI in this case.
172 return goog.string.htmlEscape(templateResult.getContent());
173 }
174 }
175
176 goog.asserts.fail('Soy template output is unsafe for use as HTML: ' +
177 templateResult);
178
179 // In production, return a safe string, rather than failing hard.
180 return 'zSoyz';
181};

184/**
185 * Checks that the rendered HTML does not start with an invalid tag that would
186 * likely cause unexpected output from renderAsElement or renderAsFragment.
187 * See {@link http://www.w3.org/TR/html5/semantics.html#semantics} for reference
188 * as to which HTML elements can be parents of each other.
189 * @param {string} html The output of a template.
190 * @private
191 */
192goog.soy.assertFirstTagValid_ = function(html) {
193 if (goog.asserts.ENABLE_ASSERTS) {
194 var matches = html.match(goog.soy.INVALID_TAG_TO_RENDER_);
195 goog.asserts.assert(!matches, 'This template starts with a %s, which ' +
196 'cannot be a child of a <div>, as required by soy internals. ' +
197 'Consider using goog.soy.renderElement instead.\nTemplate output: %s',
198 matches && matches[0], html);
199 }
200};

203/**
204 * A pattern to find templates that cannot be rendered by renderAsElement or
205 * renderAsFragment, as these elements cannot exist as the child of a <div>.
206 * @type {!RegExp}
207 * @private
208 */
209goog.soy.INVALID_TAG_TO_RENDER_ =
210 /^<(body|caption|col|colgroup|head|html|tr|td|tbody|thead|tfoot)>/i;
