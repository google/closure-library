15/**
16 * @fileoverview Datastructure: AvlTree.
17 *
18 *
19 * This file provides the implementation of an AVL-Tree datastructure. The tree
20 * maintains a set of unique values in a sorted order. The values can be
21 * accessed efficiently in their sorted order since the tree enforces an O(logn)
22 * maximum height. See http://en.wikipedia.org/wiki/Avl_tree for more detail.
23 *
24 * The big-O notation for all operations are below:
25 * <pre>
26 * Method big-O
27 * ----------------------------------------------------------------------------
28 * - add O(logn)
29 * - remove O(logn)
30 * - clear O(1)
31 * - contains O(logn)
32 * - indexOf O(logn)
33 * - getCount O(1)
34 * - getMinimum O(1), or O(logn) when optional root is specified
35 * - getMaximum O(1), or O(logn) when optional root is specified
36 * - getHeight O(1)
37 * - getValues O(n)
38 * - inOrderTraverse O(logn + k), where k is number of traversed nodes
39 * - reverseOrderTraverse O(logn + k), where k is number of traversed nodes
40 * </pre>
41 */
42

51/**
52 * Constructs an AVL-Tree, which uses the specified comparator to order its
53 * values. The values can be accessed efficiently in their sorted order since
54 * the tree enforces a O(logn) maximum height.
55 *
56 * @param {Function=} opt_comparator Function used to order the tree's nodes.
57 * @constructor
58 * @implements {goog.structs.Collection<T>}
59 * @final
60 * @template T
61 */
62goog.structs.AvlTree = function(opt_comparator) {
63 this.comparator_ = opt_comparator ||
64 goog.structs.AvlTree.DEFAULT_COMPARATOR_;
65};

68/**
69 * String comparison function used to compare values in the tree. This function
70 * is used by default if no comparator is specified in the tree's constructor.
71 *
72 * @param {T} a The first value.
73 * @param {T} b The second value.
74 * @return {number} -1 if a < b, 1 if a > b, 0 if a = b.
75 * @template T
76 * @private
77 */
78goog.structs.AvlTree.DEFAULT_COMPARATOR_ = function(a, b) {
79 if (String(a) < String(b)) {
80 return -1;
81 } else if (String(a) > String(b)) {
82 return 1;
83 }
84 return 0;
85};

129/**
130 * Inserts a node into the tree with the specified value if the tree does
131 * not already contain a node with the specified value. If the value is
132 * inserted, the tree is balanced to enforce the AVL-Tree height property.
133 *
134 * @param {T} value Value to insert into the tree.
135 * @return {boolean} Whether value was inserted into the tree.
136 * @override
137 */
138goog.structs.AvlTree.prototype.add = function(value) {
139 // If the tree is empty, create a root node with the specified value
140 if (this.root_ == null) {
141 this.root_ = new goog.structs.AvlTree.Node(value);
142 this.minNode_ = this.root_;
143 this.maxNode_ = this.root_;
144 return true;
145 }
146
147 // This will be set to the new node if a new node is added.
148 var newNode = null;
149
150 // Depth traverse the tree and insert the value if we reach a null node
151 this.traverse_(function(node) {
152 var retNode = null;
153 var comparison = this.comparator_(node.value, value);
154 if (comparison > 0) {
155 retNode = node.left;
156 if (node.left == null) {
157 newNode = new goog.structs.AvlTree.Node(value, node);
158 node.left = newNode;
159 if (node == this.minNode_) {
160 this.minNode_ = newNode;
161 }
162 }
163 } else if (comparison < 0) {
164 retNode = node.right;
165 if (node.right == null) {
166 newNode = new goog.structs.AvlTree.Node(value, node);
167 node.right = newNode;
168 if (node == this.maxNode_) {
169 this.maxNode_ = newNode;
170 }
171 }
172 }
173 return retNode; // If null, we'll stop traversing the tree
174 });
175
176 // If a node was added, increment counts and balance tree.
177 if (newNode) {
178 this.traverse_(
179 function(node) {
180 node.count++;
181 return node.parent;
182 },
183 newNode.parent);
184 this.balance_(newNode.parent); // Maintain the AVL-tree balance
185 }
186
187 // Return true if a node was added, false otherwise
188 return !!newNode;
189};

192/**
193 * Removes a node from the tree with the specified value if the tree contains a
194 * node with this value. If a node is removed the tree is balanced to enforce
195 * the AVL-Tree height property. The value of the removed node is returned.
196 *
197 * @param {T} value Value to find and remove from the tree.
198 * @return {T} The value of the removed node or null if the value was not in
199 * the tree.
200 * @override
201 */
202goog.structs.AvlTree.prototype.remove = function(value) {
203 // Assume the value is not removed and set the value when it is removed
204 var retValue = null;
205
206 // Depth traverse the tree and remove the value if we find it
207 this.traverse_(function(node) {
208 var retNode = null;
209 var comparison = this.comparator_(node.value, value);
210 if (comparison > 0) {
211 retNode = node.left;
212 } else if (comparison < 0) {
213 retNode = node.right;
214 } else {
215 retValue = node.value;
216 this.removeNode_(node);
217 }
218 return retNode; // If null, we'll stop traversing the tree
219 });
220
221 // Return the value that was removed, null if the value was not in the tree
222 return retValue;
223};

245 // Assume the value is not in the tree and set this value if it is found

249 this.traverse_(function(node) {
250 var retNode = null;

252 if (comparison > 0) {
253 retNode = node.left;
254 } else if (comparison < 0) {
255 retNode = node.right;
256 } else {
257 isContained = true;
258 }
259 return retNode; // If null, we'll stop traversing the tree
260 });

262 // Return true if the value is contained in the tree, false otherwise
263 return isContained;

267/**
268 * Returns the index (in an in-order traversal) of the node in the tree with
269 * the specified value. For example, the minimum value in the tree will
270 * return an index of 0 and the maximum will return an index of n - 1 (where
271 * n is the number of nodes in the tree). If the value is not found then -1
272 * is returned.
273 *
274 * @param {T} value Value in the tree whose in-order index is returned.
275 * @return {!number} The in-order index of the given value in the
276 * tree or -1 if the value is not found.
277 */
278goog.structs.AvlTree.prototype.indexOf = function(value) {
279 // Assume the value is not in the tree and set this value if it is found
280 var retIndex = -1;
281 var currIndex = 0;
282
283 // Depth traverse the tree and set retIndex if we find the node
284 this.traverse_(function(node) {
285 var comparison = this.comparator_(node.value, value);
286 if (comparison > 0) {
287 // The value is less than this node, so recurse into the left subtree.
288 return node.left;
289 }
290
291 if (node.left) {
292 // The value is greater than all of the nodes in the left subtree.
293 currIndex += node.left.count;
294 }
295
296 if (comparison < 0) {
297 // The value is also greater than this node.
298 currIndex++;
299 // Recurse into the right subtree.
300 return node.right;
301 }
302 // We found the node, so stop traversing the tree.
303 retIndex = currIndex;
304 return null;
305 });
306
307 // Return index if the value is contained in the tree, -1 otherwise
308 return retIndex;
309};

386/**
387 * Performs an in-order traversal of the tree and calls {@code func} with each
388 * traversed node, optionally starting from the smallest node with a value >= to
389 * the specified start value. The traversal ends after traversing the tree's
390 * maximum node or when {@code func} returns a value that evaluates to true.
391 *
392 * @param {Function} func Function to call on each traversed node.
393 * @param {Object=} opt_startValue If specified, traversal will begin on the
394 * node with the smallest value >= opt_startValue.
395 */
396goog.structs.AvlTree.prototype.inOrderTraverse =
397 function(func, opt_startValue) {
398 // If our tree is empty, return immediately
399 if (!this.root_) {
400 return;
401 }
402
403 // Depth traverse the tree to find node to begin in-order traversal from
404 var startNode;
405 if (opt_startValue) {
406 this.traverse_(function(node) {
407 var retNode = null;
408 var comparison = this.comparator_(node.value, opt_startValue);
409 if (comparison > 0) {
410 retNode = node.left;
411 startNode = node;
412 } else if (comparison < 0) {
413 retNode = node.right;
414 } else {
415 startNode = node;
416 }
417 return retNode; // If null, we'll stop traversing the tree
418 });
419 if (!startNode) {
420 return;
421 }
422 } else {
423 startNode = this.getMinNode_();
424 }
425
426 // Traverse the tree and call func on each traversed node's value
427 var node = startNode, prev = startNode.left ? startNode.left : startNode;
428 while (node != null) {
429 if (node.left != null && node.left != prev && node.right != prev) {
430 node = node.left;
431 } else {
432 if (node.right != prev) {
433 if (func(node.value)) {
434 return;
435 }
436 }
437 var temp = node;
438 node = node.right != null && node.right != prev ?
439 node.right :
440 node.parent;
441 prev = temp;
442 }
443 }
444};

447/**
448 * Performs a reverse-order traversal of the tree and calls {@code func} with
449 * each traversed node, optionally starting from the largest node with a value
450 * <= to the specified start value. The traversal ends after traversing the
451 * tree's minimum node or when func returns a value that evaluates to true.
452 *
453 * @param {function(T):?} func Function to call on each traversed node.
454 * @param {Object=} opt_startValue If specified, traversal will begin on the
455 * node with the largest value <= opt_startValue.
456 */
457goog.structs.AvlTree.prototype.reverseOrderTraverse =
458 function(func, opt_startValue) {
459 // If our tree is empty, return immediately
460 if (!this.root_) {
461 return;
462 }
463
464 // Depth traverse the tree to find node to begin reverse-order traversal from
465 var startNode;
466 if (opt_startValue) {
467 this.traverse_(goog.bind(function(node) {
468 var retNode = null;
469 var comparison = this.comparator_(node.value, opt_startValue);
470 if (comparison > 0) {
471 retNode = node.left;
472 } else if (comparison < 0) {
473 retNode = node.right;
474 startNode = node;
475 } else {
476 startNode = node;
477 }
478 return retNode; // If null, we'll stop traversing the tree
479 }, this));
480 if (!startNode) {
481 return;
482 }
483 } else {
484 startNode = this.getMaxNode_();
485 }
486
487 // Traverse the tree and call func on each traversed node's value
488 var node = startNode, prev = startNode.right ? startNode.right : startNode;
489 while (node != null) {
490 if (node.right != null && node.right != prev && node.left != prev) {
491 node = node.right;
492 } else {
493 if (node.left != prev) {
494 if (func(node.value)) {
495 return;
496 }
497 }
498 var temp = node;
499 node = node.left != null && node.left != prev ?
500 node.left :
501 node.parent;
502 prev = temp;
503 }
504 }
505};

508/**
509 * Performs a traversal defined by the supplied {@code traversalFunc}. The first
510 * call to {@code traversalFunc} is passed the root or the optionally specified
511 * startNode. After that, calls {@code traversalFunc} with the node returned
512 * by the previous call to {@code traversalFunc} until {@code traversalFunc}
513 * returns null or the optionally specified endNode. The first call to
514 * traversalFunc is passed the root or the optionally specified startNode.
515 *
516 * @param {function(
517 * this:goog.structs.AvlTree<T>,
518 * !goog.structs.AvlTree.Node):?goog.structs.AvlTree.Node} traversalFunc
519 * Function used to traverse the tree.
520 * @param {goog.structs.AvlTree.Node<T>=} opt_startNode The node at which the
521 * traversal begins.
522 * @param {goog.structs.AvlTree.Node<T>=} opt_endNode The node at which the
523 * traversal ends.
524 * @private
525 */
526goog.structs.AvlTree.prototype.traverse_ =
527 function(traversalFunc, opt_startNode, opt_endNode) {
528 var node = opt_startNode ? opt_startNode : this.root_;
529 var endNode = opt_endNode ? opt_endNode : null;
530 while (node && node != endNode) {
531 node = traversalFunc.call(this, node);
532 }
533};

536/**
537 * Ensures that the specified node and all its ancestors are balanced. If they
538 * are not, performs left and right tree rotations to achieve a balanced
539 * tree. This method assumes that at most 2 rotations are necessary to balance
540 * the tree (which is true for AVL-trees that are balanced after each node is
541 * added or removed).
542 *
543 * @param {goog.structs.AvlTree.Node<T>} node Node to begin balance from.
544 * @private
545 */
546goog.structs.AvlTree.prototype.balance_ = function(node) {
547
548 this.traverse_(function(node) {
549 // Calculate the left and right node's heights
550 var lh = node.left ? node.left.height : 0;
551 var rh = node.right ? node.right.height : 0;
552
553 // Rotate tree rooted at this node if it is not AVL-tree balanced
554 if (lh - rh > 1) {
555 if (node.left.right && (!node.left.left ||
556 node.left.left.height < node.left.right.height)) {
557 this.leftRotate_(node.left);
558 }
559 this.rightRotate_(node);
560 } else if (rh - lh > 1) {
561 if (node.right.left && (!node.right.right ||
562 node.right.right.height < node.right.left.height)) {
563 this.rightRotate_(node.right);
564 }
565 this.leftRotate_(node);
566 }
567
568 // Recalculate the left and right node's heights
569 lh = node.left ? node.left.height : 0;
570 rh = node.right ? node.right.height : 0;
571
572 // Set this node's height
573 node.height = Math.max(lh, rh) + 1;
574
575 // Traverse up tree and balance parent
576 return node.parent;
577 }, node);
578
579};

582/**
583 * Performs a left tree rotation on the specified node.
584 *
585 * @param {goog.structs.AvlTree.Node<T>} node Pivot node to rotate from.
586 * @private
587 */
588goog.structs.AvlTree.prototype.leftRotate_ = function(node) {
589 // Re-assign parent-child references for the parent of the node being removed
590 if (node.isLeftChild()) {
591 node.parent.left = node.right;
592 node.right.parent = node.parent;
593 } else if (node.isRightChild()) {
594 node.parent.right = node.right;
595 node.right.parent = node.parent;
596 } else {
597 this.root_ = node.right;
598 this.root_.parent = null;
599 }
600
601 // Re-assign parent-child references for the child of the node being removed
602 var temp = node.right;
603 node.right = node.right.left;
604 if (node.right != null) node.right.parent = node;
605 temp.left = node;
606 node.parent = temp;
607
608 // Update counts.
609 temp.count = node.count;
610 node.count -= (temp.right ? temp.right.count : 0) + 1;
611};

614/**
615 * Performs a right tree rotation on the specified node.
616 *
617 * @param {goog.structs.AvlTree.Node<T>} node Pivot node to rotate from.
618 * @private
619 */
620goog.structs.AvlTree.prototype.rightRotate_ = function(node) {
621 // Re-assign parent-child references for the parent of the node being removed
622 if (node.isLeftChild()) {
623 node.parent.left = node.left;
624 node.left.parent = node.parent;
625 } else if (node.isRightChild()) {
626 node.parent.right = node.left;
627 node.left.parent = node.parent;
628 } else {
629 this.root_ = node.left;
630 this.root_.parent = null;
631 }
632
633 // Re-assign parent-child references for the child of the node being removed
634 var temp = node.left;
635 node.left = node.left.right;
636 if (node.left != null) node.left.parent = node;
637 temp.right = node;
638 node.parent = temp;
639
640 // Update counts.
641 temp.count = node.count;
642 node.count -= (temp.left ? temp.left.count : 0) + 1;
643};

646/**
647 * Removes the specified node from the tree and ensures the tree still
648 * maintains the AVL-tree balance.
649 *
650 * @param {goog.structs.AvlTree.Node<T>} node The node to be removed.
651 * @private
652 */
653goog.structs.AvlTree.prototype.removeNode_ = function(node) {
654 // Perform normal binary tree node removal, but balance the tree, starting
655 // from where we removed the node
656 if (node.left != null || node.right != null) {
657 var b = null; // Node to begin balance from
658 var r; // Node to replace the node being removed
659 if (node.left != null) {
660 r = this.getMaxNode_(node.left);
661
662 // Update counts.
663 this.traverse_(function(node) {
664 node.count--;
665 return node.parent;
666 }, r);
667
668 if (r != node.left) {
669 r.parent.right = r.left;
670 if (r.left) r.left.parent = r.parent;
671 r.left = node.left;
672 r.left.parent = r;
673 b = r.parent;
674 }
675 r.parent = node.parent;
676 r.right = node.right;
677 if (r.right) r.right.parent = r;
678 if (node == this.maxNode_) this.maxNode_ = r;
679 r.count = node.count;
680 } else {
681 r = this.getMinNode_(node.right);
682
683 // Update counts.
684 this.traverse_(function(node) {
685 node.count--;
686 return node.parent;
687 }, r);
688
689 if (r != node.right) {
690 r.parent.left = r.right;
691 if (r.right) r.right.parent = r.parent;
692 r.right = node.right;
693 r.right.parent = r;
694 b = r.parent;
695 }
696 r.parent = node.parent;
697 r.left = node.left;
698 if (r.left) r.left.parent = r;
699 if (node == this.minNode_) this.minNode_ = r;
700 r.count = node.count;
701 }
702
703 // Update the parent of the node being removed to point to its replace
704 if (node.isLeftChild()) {
705 node.parent.left = r;
706 } else if (node.isRightChild()) {
707 node.parent.right = r;
708 } else {
709 this.root_ = r;
710 }
711
712 // Balance the tree
713 this.balance_(b ? b : r);
714 } else {
715 // Update counts.
716 this.traverse_(function(node) {
717 node.count--;
718 return node.parent;
719 }, node.parent);
720
721 // If the node is a leaf, remove it and balance starting from its parent
722 if (node.isLeftChild()) {
723 this.special = 1;
724 node.parent.left = null;
725 if (node == this.minNode_) this.minNode_ = node.parent;
726 this.balance_(node.parent);
727 } else if (node.isRightChild()) {
728 node.parent.right = null;
729 if (node == this.maxNode_) this.maxNode_ = node.parent;
730 this.balance_(node.parent);
731 } else {
732 this.clear();
733 }
734 }
735};

738/**
739 * Returns the node in the tree that has k nodes before it in an in-order
740 * traversal, optionally rooted at {@code opt_rootNode}.
741 *
742 * @param {number} k The number of nodes before the node to be returned in an
743 * in-order traversal, where 0 <= k < root.count.
744 * @param {goog.structs.AvlTree.Node<T>=} opt_rootNode Optional root node.
745 * @return {goog.structs.AvlTree.Node<T>} The node at the specified index.
746 * @private
747 */
748goog.structs.AvlTree.prototype.getKthNode_ = function(k, opt_rootNode) {
749 var root = opt_rootNode || this.root_;
750 var numNodesInLeftSubtree = root.left ? root.left.count : 0;
751
752 if (k < numNodesInLeftSubtree) {
753 return this.getKthNode_(k, root.left);
754 } else if (k == numNodesInLeftSubtree) {
755 return root;
756 } else {
