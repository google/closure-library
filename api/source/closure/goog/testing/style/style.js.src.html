28/**
29 * Determines whether the bounding rectangles of the given elements intersect.
30 * @param {Element} element The first element.
31 * @param {Element} otherElement The second element.
32 * @return {boolean} Whether the bounding rectangles of the given elements
33 * intersect.
34 */
35goog.testing.style.intersects = function(element, otherElement) {
36 var elementRect = goog.style.getBounds(element);
37 var otherElementRect = goog.style.getBounds(otherElement);
38 return goog.math.Rect.intersects(elementRect, otherElementRect);
39};

42/**
43 * Determines whether the element has visible dimensions, i.e. x > 0 && y > 0.
44 * @param {Element} element The element to check.
45 * @return {boolean} Whether the element has visible dimensions.
46 */
47goog.testing.style.hasVisibleDimensions = function(element) {
48 var elSize = goog.style.getSize(element);
49 var shortest = elSize.getShortest();
50 if (shortest <= 0) {
51 return false;
52 }
53
54 return true;
55};

58/**
59 * Determines whether the CSS style of the element renders it visible.
60 * @param {!Element} element The element to check.
61 * @return {boolean} Whether the CSS style of the element renders it visible.
62 */
63goog.testing.style.isVisible = function(element) {
64 var visibilityStyle =
65 goog.testing.style.getAvailableStyle_(element, 'visibility');
66 var displayStyle =
67 goog.testing.style.getAvailableStyle_(element, 'display');
68
69 return (visibilityStyle != 'hidden' && displayStyle != 'none');
70};

73/**
74 * Test whether the given element is on screen.
75 * @param {!Element} el The element to test.
76 * @return {boolean} Whether the element is on the screen.
77 */
78goog.testing.style.isOnScreen = function(el) {
79 var doc = goog.dom.getDomHelper(el).getDocument();
80 var viewport = goog.style.getVisibleRectForElement(doc.body);
81 var viewportRect = goog.math.Rect.createFromBox(viewport);
82 return goog.dom.contains(doc, el) &&
83 goog.style.getBounds(el).intersects(viewportRect);
84};

87/**
88 * This is essentially goog.style.getStyle_. goog.style.getStyle_ is private
89 * and is not a recommended way for general purpose style extractor. For the
90 * purposes of layout testing, we only use this function for retrieving
91 * 'visiblity' and 'display' style.
92 * @param {!Element} element The element to retrieve the style from.
93 * @param {string} style Style property name.
94 * @return {string} Style value.
95 * @private
96 */
97goog.testing.style.getAvailableStyle_ = function(element, style) {
98 return goog.style.getComputedStyle(element, style) ||
99 goog.style.getCascadedStyle(element, style) ||
100 goog.style.getStyle(element, style);
101};
