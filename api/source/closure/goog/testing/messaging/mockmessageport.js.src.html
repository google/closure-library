26/**
27 * Class for unit-testing code that uses MessagePorts.
28 * @param {*} id An opaque identifier, used because message ports otherwise have
29 * no distinguishing characteristics.
30 * @param {goog.testing.MockControl} mockControl The mock control used to create
31 * the method mock for #postMessage.
32 * @constructor
33 * @extends {goog.events.EventTarget}
34 * @final
35 */
36goog.testing.messaging.MockMessagePort = function(id, mockControl) {
37 goog.testing.messaging.MockMessagePort.base(this, 'constructor');
38
39 /**
40 * An opaque identifier, used because message ports otherwise have no
41 * distinguishing characteristics.
42 * @type {*}
43 */
44 this.id = id;
45
46 /**
47 * Whether or not the port has been started.
48 * @type {boolean}
49 */
50 this.started = false;
51
52 /**
53 * Whether or not the port has been closed.
54 * @type {boolean}
55 */
56 this.closed = false;
57
58 mockControl.createMethodMock(this, 'postMessage');
59};
60goog.inherits(goog.testing.messaging.MockMessagePort, goog.events.EventTarget);
