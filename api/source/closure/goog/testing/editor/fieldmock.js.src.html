31/**
32 * Mock of goog.editor.Field.
33 * @param {Window=} opt_window Window the field would edit. Defaults to
34 * {@code window}.
35 * @param {Window=} opt_appWindow "AppWindow" of the field, which can be
36 * different from {@code opt_window} when mocking a field that uses an
37 * iframe. Defaults to {@code opt_window}.
38 * @param {goog.dom.AbstractRange=} opt_range An object (mock or real) to be
39 * returned by getRange(). If ommitted, a new goog.dom.Range is created
40 * from the window every time getRange() is called.
41 * @constructor
42 * @extends {goog.testing.LooseMock}
43 * @suppress {missingProperties} Mocks do not fit in the type system well.
44 * @final
45 */
46goog.testing.editor.FieldMock =
47 function(opt_window, opt_appWindow, opt_range) {
48 goog.testing.LooseMock.call(this, goog.editor.Field);
49 opt_window = opt_window || window;
50 opt_appWindow = opt_appWindow || opt_window;
51
52 this.getAppWindow();
53 this.$anyTimes();
54 this.$returns(opt_appWindow);
55
56 this.getRange();
57 this.$anyTimes();
58 this.$does(function() {
59 return opt_range || goog.dom.Range.createFromWindow(opt_window);
60 });
61
62 this.getEditableDomHelper();
63 this.$anyTimes();
64 this.$returns(goog.dom.getDomHelper(opt_window.document));
65
66 this.usesIframe();
67 this.$anyTimes();
68
69 this.getBaseZindex();
70 this.$anyTimes();
71 this.$returns(0);
72
73 this.restoreSavedRange(goog.testing.mockmatchers.ignoreArgument);
74 this.$anyTimes();
75 this.$does(function(range) {
76 if (range) {
77 range.restore();
78 }
79 this.focus();
80 });
