33/**
34 * Create a new test controller.
35 * @param {Element} root The root editable element.
36 * @constructor
37 * @extends {goog.Disposable}
38 * @final
39 */
40goog.testing.editor.TestHelper = function(root) {
41 if (!root) {
42 throw Error('Null root');
43 }
44 goog.Disposable.call(this);
45
46 /**
47 * Convenience variable for root DOM element.
48 * @type {!Element}
49 * @private
50 */
51 this.root_ = root;
52
53 /**
54 * The starting HTML of the editable element.
55 * @type {string}
56 * @private
57 */
58 this.savedHtml_ = '';
59};
60goog.inherits(goog.testing.editor.TestHelper, goog.Disposable);

75/**
76 * Make the root element editable. Alse saves its HTML to be restored
77 * in tearDown.
78 */
79goog.testing.editor.TestHelper.prototype.setUpEditableElement = function() {
80 this.savedHtml_ = this.root_.innerHTML;
81 if (goog.editor.BrowserFeature.HAS_CONTENT_EDITABLE) {
82 this.root_.contentEditable = true;
83 } else {
84 this.root_.ownerDocument.designMode = 'on';
85 }
86 this.root_.setAttribute('g_editable', 'true');
87};
88
89
90/**
91 * Reset the element previously initialized, restoring its HTML and making it
92 * non editable.
93 * @suppress {accessControls} Private state of
94 * {@link goog.editor.plugins.AbstractBubblePlugin} is accessed for test
95 * purposes.
96 */
97goog.testing.editor.TestHelper.prototype.tearDownEditableElement = function() {
98 if (goog.editor.BrowserFeature.HAS_CONTENT_EDITABLE) {
99 this.root_.contentEditable = false;
100 } else {
101 this.root_.ownerDocument.designMode = 'off';
102 }
103 goog.dom.removeChildren(this.root_);
104 this.root_.innerHTML = this.savedHtml_;
105 this.root_.removeAttribute('g_editable');
106
107 if (goog.editor.plugins && goog.editor.plugins.AbstractBubblePlugin) {
108 // Remove old bubbles.
109 for (var key in goog.editor.plugins.AbstractBubblePlugin.bubbleMap_) {
110 goog.editor.plugins.AbstractBubblePlugin.bubbleMap_[key].dispose();
111 }
112 // Ensure we get a new bubble for each test.
113 goog.editor.plugins.AbstractBubblePlugin.bubbleMap_ = {};
114 }
115};

118/**
119 * Assert that the html in 'root' is substantially similar to htmlPattern.
120 * This method tests for the same set of styles, and for the same order of
121 * nodes. Breaking whitespace nodes are ignored. Elements can be annotated
122 * with classnames corresponding to keys in goog.userAgent and will be
123 * expected to show up in that user agent and expected not to show up in
124 * others.
125 * @param {string} htmlPattern The pattern to match.
126 */
127goog.testing.editor.TestHelper.prototype.assertHtmlMatches = function(
128 htmlPattern) {
129 goog.testing.dom.assertHtmlContentsMatch(htmlPattern, this.root_);
130};

133/**
134 * Finds the first text node descendant of root with the given content.
135 * @param {string|RegExp} textOrRegexp The text to find, or a regular
136 * expression to find a match of.
137 * @return {Node} The first text node that matches, or null if none is found.
138 */
139goog.testing.editor.TestHelper.prototype.findTextNode = function(textOrRegexp) {
140 return goog.testing.dom.findTextNode(textOrRegexp, this.root_);
141};

144/**
145 * Select from the given from offset in the given from node to the given
146 * to offset in the optionally given to node. If nodes are passed in, uses them,
147 * otherwise uses findTextNode to find the nodes to select. Selects a caret
148 * if opt_to and opt_toOffset are not given.
149 * @param {Node|string} from Node or text of the node to start the selection at.
150 * @param {number} fromOffset Offset within the above node to start the
151 * selection at.
152 * @param {Node|string=} opt_to Node or text of the node to end the selection
153 * at.
154 * @param {number=} opt_toOffset Offset within the above node to end the
155 * selection at.
156 */
157goog.testing.editor.TestHelper.prototype.select = function(from, fromOffset,
158 opt_to, opt_toOffset) {
159 var end;
160 var start = end = goog.isString(from) ? this.findTextNode(from) : from;
161 var endOffset;
162 var startOffset = endOffset = fromOffset;
163
164 if (opt_to && goog.isNumber(opt_toOffset)) {
165 end = goog.isString(opt_to) ? this.findTextNode(opt_to) : opt_to;
166 endOffset = opt_toOffset;
167 }
168
169 goog.dom.Range.createFromNodes(start, startOffset, end, endOffset).select();
170};

173/** @override */
174goog.testing.editor.TestHelper.prototype.disposeInternal = function() {
175 if (goog.editor.node.isEditableContainer(this.root_)) {
176 this.tearDownEditableElement();
177 }
178 delete this.root_;
179 goog.testing.editor.TestHelper.base(this, 'disposeInternal');
