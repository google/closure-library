60/**
61 * Tests if the given iterator over nodes matches the given Array of node
62 * descriptors. Throws an error if any match fails.
63 * @param {goog.iter.Iterator} it An iterator over nodes.
64 * @param {Array<Node|number|string>} array Array of node descriptors to match
65 * against. Node descriptors can be any of the following:
66 * Node: Test if the two nodes are equal.
67 * number: Test node.nodeType == number.
68 * string starting with '#': Match the node's id with the text
69 * after "#".
70 * other string: Match the text node's contents.
71 */
72goog.testing.dom.assertNodesMatch = function(it, array) {
73 var i = 0;
74 goog.iter.forEach(it, function(node) {
75 if (array.length <= i) {
76 fail('Got more nodes than expected: ' + goog.testing.dom.describeNode_(
77 node));
78 }
79 var expected = array[i];
80
81 if (goog.dom.isNodeLike(expected)) {
82 assertEquals('Nodes should match at position ' + i, expected, node);
83 } else if (goog.isNumber(expected)) {
84 assertEquals('Node types should match at position ' + i, expected,
85 node.nodeType);
86 } else if (expected.charAt(0) == '#') {
87 assertEquals('Expected element at position ' + i,
88 goog.dom.NodeType.ELEMENT, node.nodeType);
89 var expectedId = expected.substr(1);
90 assertEquals('IDs should match at position ' + i,
91 expectedId, node.id);
92
93 } else {
94 assertEquals('Expected text node at position ' + i,
95 goog.dom.NodeType.TEXT, node.nodeType);
96 assertEquals('Node contents should match at position ' + i,
97 expected, node.nodeValue);
98 }
99
100 i++;
101 });
102
103 assertEquals('Used entire match array', array.length, i);
104};

119 * Exposes the nodes of a range wrapper as a string.
120 * @param {goog.dom.AbstractRange} range A range.
121 * @return {string} A string representation of the range.
122 */
123goog.testing.dom.exposeRange = function(range) {
124 // This is deliberately not implemented as
125 // goog.dom.AbstractRange.prototype.toString, because it is non-authoritative.
126 // Two equivalent ranges may have very different exposeRange values, and
127 // two different ranges may have equal exposeRange values.
128 // (The mapping of ranges to DOM nodes/offsets is a many-to-many mapping).
129 if (!range) {
130 return 'null';
131 }
132 return goog.testing.dom.exposeNode(range.getStartNode()) + ':' +

135 range.getEndOffset();

139/**
140 * Determines if the current user agent matches the specified string. Returns
141 * false if the string does specify at least one user agent but does not match
142 * the running agent.
143 * @param {string} userAgents Space delimited string of user agents.
144 * @return {boolean} Whether the user agent was matched. Also true if no user
145 * agent was listed in the expectation string.
146 * @private
147 */
148goog.testing.dom.checkUserAgents_ = function(userAgents) {
149 if (goog.string.startsWith(userAgents, '!')) {
150 if (goog.string.contains(userAgents, ' ')) {
151 throw new Error('Only a single negative user agent may be specified');
152 }
153 return !goog.userAgent[userAgents.substr(1)];
154 }
155
156 var agents = userAgents.split(' ');
157 var hasUserAgent = false;
158 for (var i = 0, len = agents.length; i < len; i++) {
159 var cls = agents[i];
160 if (cls in goog.userAgent) {
161 hasUserAgent = true;
162 if (goog.userAgent[cls]) {
163 return true;
164 }
165 }
166 }
167 // If we got here, there was a user agent listed but we didn't match it.
168 return !hasUserAgent;
169};

185/**
186 * Check if the given node is important. A node is important if it is a
187 * non-empty text node, a non-annotated element, or an element annotated to
188 * match on this user agent.
189 * @param {Node} node The node to test.
190 * @return {boolean} Whether this node should be included for iteration.
191 * @private
192 */
193goog.testing.dom.nodeFilter_ = function(node) {
194 if (node.nodeType == goog.dom.NodeType.TEXT) {
195 // If a node is part of a string of text nodes and it has spaces in it,
196 // we allow it since it's going to affect the merging of nodes done below.
197 if (goog.string.isBreakingWhitespace(node.nodeValue) &&
198 (!node.previousSibling ||
199 node.previousSibling.nodeType != goog.dom.NodeType.TEXT) &&
200 (!node.nextSibling ||
201 node.nextSibling.nodeType != goog.dom.NodeType.TEXT)) {
202 return false;
203 }
204 // Allow optional text to be specified as [[BROWSER1 BROWSER2]]Text
205 var match = node.nodeValue.match(/^\[\[(.+)\]\]/);
206 if (match) {
207 return goog.testing.dom.checkUserAgents_(match[1]);
208 }
209 } else if (node.className && goog.isString(node.className)) {
210 return goog.testing.dom.checkUserAgents_(node.className);
211 }
212 return true;
213};

244/**
245 * Assert that the html in {@code actual} is substantially similar to
246 * htmlPattern. This method tests for the same set of styles, for the same
247 * order of nodes, and the presence of attributes. Breaking whitespace nodes
248 * are ignored. Elements can be
249 * annotated with classnames corresponding to keys in goog.userAgent and will be
250 * expected to show up in that user agent and expected not to show up in
251 * others.
252 * @param {string} htmlPattern The pattern to match.
253 * @param {!Node} actual The element to check: its contents are matched
254 * against the HTML pattern.
255 * @param {boolean=} opt_strictAttributes If false, attributes that appear in
256 * htmlPattern must be in actual, but actual can have attributes not
257 * present in htmlPattern. If true, htmlPattern and actual must have the
258 * same set of attributes. Default is false.
259 */
260goog.testing.dom.assertHtmlContentsMatch = function(htmlPattern, actual,
261 opt_strictAttributes) {
262 var div = goog.dom.createDom(goog.dom.TagName.DIV);
263 div.innerHTML = htmlPattern;
264
265 var errorSuffix = '\nExpected\n' + htmlPattern + '\nActual\n' +
266 actual.innerHTML;
267
268 var actualIt = goog.iter.filter(
269 goog.iter.map(new goog.dom.TagIterator(actual),
270 goog.testing.dom.endTagMap_),
271 goog.testing.dom.nodeFilter_);
272
273 var expectedIt = goog.iter.filter(new goog.dom.NodeIterator(div),
274 goog.testing.dom.nodeFilter_);
275
276 var actualNode;
277 var preIterated = false;
278 var advanceActualNode = function() {
279 // If the iterator has already been advanced, don't advance it again.
280 if (!preIterated) {
281 actualNode = /** @type {Node} */ (goog.iter.nextOrValue(actualIt, null));
282 }
283 preIterated = false;
284
285 // Advance the iterator so long as it is return end tags.
286 while (actualNode == goog.testing.dom.END_TAG_MARKER_) {
287 actualNode = /** @type {Node} */ (goog.iter.nextOrValue(actualIt, null));
288 }
289 };
290
291 // HACK(brenneman): IE has unique ideas about whitespace handling when setting
292 // innerHTML. This results in elision of leading whitespace in the expected
293 // nodes where doing so doesn't affect visible rendering. As a workaround, we
294 // remove the leading whitespace in the actual nodes where necessary.
295 //
296 // The collapsible variable tracks whether we should collapse the whitespace
297 // in the next Text node we encounter.
298 var IE_TEXT_COLLAPSE =
299 goog.userAgent.IE && !goog.userAgent.isVersionOrHigher('9');
300
301 var collapsible = true;
302
303 var number = 0;
304 goog.iter.forEach(expectedIt, function(expectedNode) {
305 expectedNode = /** @type {Node} */ (expectedNode);
306
307 advanceActualNode();
308 assertNotNull('Finished actual HTML before finishing expected HTML at ' +
309 'node number ' + number + ': ' +
310 goog.testing.dom.describeNode_(expectedNode) + errorSuffix,
311 actualNode);
312
313 // Do no processing for expectedNode == div.
314 if (expectedNode == div) {
315 return;
316 }
317
318 assertEquals('Should have the same node type, got ' +
319 goog.testing.dom.describeNode_(actualNode) + ' but expected ' +
320 goog.testing.dom.describeNode_(expectedNode) + '.' + errorSuffix,
321 expectedNode.nodeType, actualNode.nodeType);
322
323 if (expectedNode.nodeType == goog.dom.NodeType.ELEMENT) {
324 var expectedElem = goog.asserts.assertElement(expectedNode);
325 var actualElem = goog.asserts.assertElement(actualNode);
326
327 assertEquals('Tag names should match' + errorSuffix,
328 expectedElem.tagName, actualElem.tagName);
329 assertObjectEquals('Should have same styles' + errorSuffix,
330 goog.style.parseStyleAttribute(expectedElem.style.cssText),
331 goog.style.parseStyleAttribute(actualElem.style.cssText));
332 goog.testing.dom.assertAttributesEqual_(errorSuffix, expectedElem,
333 actualElem, !!opt_strictAttributes);
334
335 if (IE_TEXT_COLLAPSE &&
336 goog.style.getCascadedStyle(actualElem, 'display') != 'inline') {
337 // Text may be collapsed after any non-inline element.
338 collapsible = true;
339 }
340 } else {
341 // Concatenate text nodes until we reach a non text node.
342 var actualText = actualNode.nodeValue;
343 preIterated = true;
344 while ((actualNode = /** @type {Node} */
345 (goog.iter.nextOrValue(actualIt, null))) &&
346 actualNode.nodeType == goog.dom.NodeType.TEXT) {
347 actualText += actualNode.nodeValue;
348 }
349
350 if (IE_TEXT_COLLAPSE) {
351 // Collapse the leading whitespace, unless the string consists entirely
352 // of whitespace.
353 if (collapsible && !goog.string.isEmptyOrWhitespace(actualText)) {
354 actualText = goog.string.trimLeft(actualText);
355 }
356 // Prepare to collapse whitespace in the next Text node if this one does
357 // not end in a whitespace character.
358 collapsible = /\s$/.test(actualText);
359 }
360
361 var expectedText = goog.testing.dom.getExpectedText_(expectedNode);
362 if ((actualText && !goog.string.isBreakingWhitespace(actualText)) ||
363 (expectedText && !goog.string.isBreakingWhitespace(expectedText))) {
364 var normalizedActual = actualText.replace(/\s+/g, ' ');
365 var normalizedExpected = expectedText.replace(/\s+/g, ' ');
366
367 assertEquals('Text should match' + errorSuffix, normalizedExpected,
368 normalizedActual);
369 }
370 }
371
372 number++;
373 });
374
375 advanceActualNode();
376 assertNull('Finished expected HTML before finishing actual HTML' +
377 errorSuffix, goog.iter.nextOrValue(actualIt, null));
378};

405/**
406 * Finds the first text node descendant of root with the given content. Note
407 * that this operates on a text node level, so if text nodes get split this
408 * may not match the user visible text. Using normalize() may help here.
409 * @param {string|RegExp} textOrRegexp The text to find, or a regular
410 * expression to find a match of.
411 * @param {Element} root The element to search in.
412 * @return {Node} The first text node that matches, or null if none is found.
413 */
414goog.testing.dom.findTextNode = function(textOrRegexp, root) {
415 var it = new goog.dom.NodeIterator(root);
416 var ret = goog.iter.nextOrValue(goog.iter.filter(it, function(node) {
417 if (node.nodeType == goog.dom.NodeType.TEXT) {
418 if (goog.isString(textOrRegexp)) {
419 return node.nodeValue == textOrRegexp;
420 } else {
421 return !!node.nodeValue.match(textOrRegexp);
422 }
423 } else {
424 return false;
425 }
426 }), null);
427 return /** @type {Node} */ (ret);
428};

454/**
455 * Gets the value of a DOM attribute in deterministic way.
456 * @param {!Node} node A node.
457 * @param {string} name Attribute name.
458 * @return {*} Attribute value.
459 * @private
460 */
461goog.testing.dom.getAttributeValue_ = function(node, name) {
462 // These hacks avoid nondetermistic results in the following cases:
463 // IE7: document.createElement(goog.dom.TagName.INPUT).height returns
464 // a random number.
465 // FF3: getAttribute('disabled') returns different value for <div disabled="">
466 // and <div disabled="disabled">
467 // WebKit: Two radio buttons with the same name can't be checked at the same
468 // time, even if only one of them is in the document.
469 if (goog.userAgent.WEBKIT && node.tagName == goog.dom.TagName.INPUT &&
470 node['type'] == goog.dom.InputType.RADIO && name == 'checked') {
471 return false;
472 }
473 return goog.isDef(node[name]) &&
474 typeof node.getAttribute(name) != typeof node[name] ?
475 node[name] : node.getAttribute(name);
476};

479/**
480 * Assert that the attributes of two Nodes are the same (ignoring any
481 * instances of the style attribute).
482 * @param {string} errorSuffix String to add to end of error messages.
483 * @param {!Element} expectedElem The element whose attributes we are expecting.
484 * @param {!Element} actualElem The element with the actual attributes.
485 * @param {boolean} strictAttributes If false, attributes that appear in
486 * expectedNode must also be in actualNode, but actualNode can have
487 * attributes not present in expectedNode. If true, expectedNode and
488 * actualNode must have the same set of attributes.
489 * @private
490 */
491goog.testing.dom.assertAttributesEqual_ = function(errorSuffix,
492 expectedElem, actualElem, strictAttributes) {
493 if (strictAttributes) {
494 goog.testing.dom.compareClassAttribute_(expectedElem, actualElem);
495 }
496
497 var expectedAttributes = expectedElem.attributes;
498 var actualAttributes = actualElem.attributes;
499
500 for (var i = 0, len = expectedAttributes.length; i < len; i++) {
501 var expectedName = expectedAttributes[i].name;
502 var expectedValue = goog.testing.dom.getAttributeValue_(expectedElem,
503 expectedName);
504
505 var actualAttribute = actualAttributes[expectedName];
506 var actualValue = goog.testing.dom.getAttributeValue_(actualElem,
507 expectedName);
508
509 // IE enumerates attribute names in the expected node that are not present,
510 // causing an undefined actualAttribute.
511 if (!expectedValue && !actualValue) {
512 continue;
513 }
514
515 if (expectedName == 'id' && goog.userAgent.IE) {
516 goog.testing.dom.compareIdAttributeForIe_(
517 /** @type {string} */ (expectedValue), actualAttribute,
518 strictAttributes, errorSuffix);
519 continue;
520 }
521
522 if (goog.testing.dom.ignoreAttribute_(expectedName)) {
523 continue;
524 }
525
526 assertNotUndefined('Expected to find attribute with name ' +
527 expectedName + ', in element ' +
528 goog.testing.dom.describeNode_(actualElem) + errorSuffix,
529 actualAttribute);
530 assertEquals('Expected attribute ' + expectedName +
531 ' has a different value ' + errorSuffix,
532 String(expectedValue),
533 String(goog.testing.dom.getAttributeValue_(
534 actualElem, actualAttribute.name)));
535 }
536
537 if (strictAttributes) {
538 for (i = 0; i < actualAttributes.length; i++) {
539 var actualName = actualAttributes[i].name;
540 var actualAttribute = actualAttributes.getNamedItem(actualName);
541
542 if (!actualAttribute || goog.testing.dom.ignoreAttribute_(actualName)) {
543 continue;
544 }
545
546 assertNotUndefined('Unexpected attribute with name ' +
547 actualName + ' in element ' +
548 goog.testing.dom.describeNode_(actualElem) + errorSuffix,
549 expectedAttributes[actualName]);
550 }
551 }
552};

555/**
556 * Assert the class attribute of actualElem is the same as the one in
557 * expectedElem, ignoring classes that are useragents.
558 * @param {!Element} expectedElem The DOM element whose class we expect.
559 * @param {!Element} actualElem The DOM element with the actual class.
560 * @private
561 */
562goog.testing.dom.compareClassAttribute_ = function(expectedElem,
563 actualElem) {
564 var classes = goog.dom.classlist.get(expectedElem);
565
566 var expectedClasses = [];
567 for (var i = 0, len = classes.length; i < len; i++) {
568 if (!(classes[i] in goog.userAgent)) {
569 expectedClasses.push(classes[i]);
570 }
571 }
572 expectedClasses.sort();
573
574 var actualClasses = goog.array.toArray(goog.dom.classlist.get(actualElem));
575 actualClasses.sort();
576
577 assertArrayEquals(
578 'Expected class was: ' + expectedClasses.join(' ') +
579 ', but actual class was: ' + actualElem.className,
580 expectedClasses, actualClasses);
581};

607/**
608 * Compare id attributes for IE. In IE, if an element lacks an id attribute
609 * in the original HTML, the element object will still have such an attribute,
610 * but its value will be the empty string.
611 * @param {string} expectedValue The expected value of the id attribute.
612 * @param {Attr} actualAttribute The actual id attribute.
613 * @param {boolean} strictAttributes Whether strict attribute checking should be
614 * done.
615 * @param {string} errorSuffix String to append to error messages.
616 * @private
617 */
618goog.testing.dom.compareIdAttributeForIe_ = function(expectedValue,
619 actualAttribute, strictAttributes, errorSuffix) {
620 if (expectedValue === '') {
621 if (strictAttributes) {
622 assertTrue('Unexpected attribute with name id in element ' +
623 errorSuffix, actualAttribute.value == '');
624 }
625 } else {
626 assertNotUndefined('Expected to find attribute with name id, in element ' +
627 errorSuffix, actualAttribute);
628 assertNotEquals('Expected to find attribute with name id, in element ' +
629 errorSuffix, '', actualAttribute.value);
630 assertEquals('Expected attribute has a different value ' + errorSuffix,
631 expectedValue, actualAttribute.value);
632 }
