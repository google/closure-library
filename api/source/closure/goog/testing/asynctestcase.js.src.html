16/**
17 * @fileoverview A class representing a set of test functions that use
18 * asynchronous functions that cannot be meaningfully mocked.
19 *
20 * To create a Google-compatable JsUnit test using this test case, put the
21 * following snippet in your test:
22 *
23 * var asyncTestCase = goog.testing.AsyncTestCase.createAndInstall();
24 *
25 * To make the test runner wait for your asynchronous behaviour, use:
26 *
27 * asyncTestCase.waitForAsync('Waiting for xhr to respond');
28 *
29 * The next test will not start until the following call is made, or a
30 * timeout occurs:
31 *
32 * asyncTestCase.continueTesting();
33 *
34 * There does NOT need to be a 1:1 mapping of waitForAsync calls and
35 * continueTesting calls. The next test will be run after a single call to
36 * continueTesting is made, as long as there is no subsequent call to
37 * waitForAsync in the same thread.
38 *
39 * Example:
40 * // Returning here would cause the next test to be run.
41 * asyncTestCase.waitForAsync('description 1');
42 * // Returning here would *not* cause the next test to be run.
43 * // Only effect of additional waitForAsync() calls is an updated
44 * // description in the case of a timeout.
45 * asyncTestCase.waitForAsync('updated description');
46 * asyncTestCase.continueTesting();
47 * // Returning here would cause the next test to be run.
48 * asyncTestCase.waitForAsync('just kidding, still running.');
49 * // Returning here would *not* cause the next test to be run.
50 *
51 * The test runner can also be made to wait for more than one asynchronous
52 * event with:
53 *
54 * asyncTestCase.waitForSignals(n);
55 *
56 * The next test will not start until asyncTestCase.signal() is called n times,
57 * or the test step timeout is exceeded.
58 *
59 * This class supports asynchronous behaviour in all test functions except for
60 * tearDownPage. If such support is needed, it can be added.
61 *
62 * Example Usage:
63 *
64 * var asyncTestCase = goog.testing.AsyncTestCase.createAndInstall();
65 * // Optionally, set a longer-than-normal step timeout.
66 * asyncTestCase.stepTimeout = 30 * 1000;
67 *
68 * function testSetTimeout() {
69 * var step = 0;
70 * function stepCallback() {
71 * step++;
72 * switch (step) {
73 * case 1:
74 * var startTime = goog.now();
75 * asyncTestCase.waitForAsync('step 1');
76 * window.setTimeout(stepCallback, 100);
77 * break;
78 * case 2:
79 * assertTrue('Timeout fired too soon',
80 * goog.now() - startTime >= 100);
81 * asyncTestCase.waitForAsync('step 2');
82 * window.setTimeout(stepCallback, 100);
83 * break;
84 * case 3:
85 * assertTrue('Timeout fired too soon',
86 * goog.now() - startTime >= 200);
87 * asyncTestCase.continueTesting();
88 * break;
89 * default:
90 * fail('Unexpected call to stepCallback');
91 * }
92 * }
93 * stepCallback();
94 * }
95 *
96 * Known Issues:
97 * IE7 Exceptions:
98 * As the failingtest.html will show, it appears as though ie7 does not
99 * propagate an exception past a function called using the func.call()
100 * syntax. This causes case 3 of the failing tests (exceptions) to show up
101 * as timeouts in IE.
102 * window.onerror:
103 * This seems to catch errors only in ff2/ff3. It does not work in Safari or
104 * IE7. The consequence of this is that exceptions that would have been
105 * caught by window.onerror show up as timeouts.
106 *
107 * @author agrieve@google.com (Andrew Grieve)
108 */

118/**
119 * A test case that is capable of running tests the contain asynchronous logic.
120 * @param {string=} opt_name A descriptive name for the test case.
121 * @extends {goog.testing.TestCase}
122 * @constructor

363/**
364 * Informs the testcase not to continue to the next step in the test cycle
365 * until continueTesting is called.
366 * @param {string=} opt_name A description of what we are waiting for.
367 */
368goog.testing.AsyncTestCase.prototype.waitForAsync = function(opt_name) {
369 this.isReady_ = false;
370 this.curStepName_ = opt_name || this.curStepName_;
371
372 // Reset the timer that tracks if the async test takes too long.
373 this.stopTimeoutTimer_();
374 this.startTimeoutTimer_();
375};

378/**
379 * Continue with the next step in the test cycle.
380 */
381goog.testing.AsyncTestCase.prototype.continueTesting = function() {
382 if (this.receivedSignalCount_ < this.expectedSignalCount_) {
383 var remaining = this.expectedSignalCount_ - this.receivedSignalCount_;
384 throw Error('Still waiting for ' + remaining + ' signals.');
385 }
386 this.endCurrentStep_();
387};

390/**
391 * Ends the current test step and queues the next test step to run.
392 * @private
393 */
394goog.testing.AsyncTestCase.prototype.endCurrentStep_ = function() {
395 if (!this.isReady_) {
396 // We are a potential entry point, so we pump.
397 this.isReady_ = true;
398 this.stopTimeoutTimer_();
399 // Run this in a setTimeout so that the caller has a chance to call
400 // waitForAsync() again before we continue.
401 this.timeout(goog.bind(this.pump_, this, null), 0);
402 }
403};

406/**
407 * Informs the testcase not to continue to the next step in the test cycle
408 * until signal is called the specified number of times. Within a test, this
409 * function behaves additively if called multiple times; the number of signals
410 * to wait for will be the sum of all expected number of signals this function
411 * was called with.
412 * @param {number} times The number of signals to receive before
413 * continuing testing.
414 * @param {string=} opt_name A description of what we are waiting for.
415 */
416goog.testing.AsyncTestCase.prototype.waitForSignals =
417 function(times, opt_name) {
418 this.expectedSignalCount_ += times;
419 if (this.receivedSignalCount_ < this.expectedSignalCount_) {
420 this.waitForAsync(opt_name);
421 }
422};

425/**
426 * Signals once to continue with the test. If this is the last signal that the
427 * test was waiting on, call continueTesting.
428 */
429goog.testing.AsyncTestCase.prototype.signal = function() {
430 if (++this.receivedSignalCount_ === this.expectedSignalCount_ &&
431 this.expectedSignalCount_ > 0) {
432 this.endCurrentStep_();
433 }

437/**
438 * Handles an exception thrown by a test.
439 * @param {*=} opt_e The exception object associated with the failure
440 * or a string.
441 * @throws Always throws a ControlBreakingException.
442 */
443goog.testing.AsyncTestCase.prototype.doAsyncError = function(opt_e) {
444 // If we've caught an exception that we threw, then just pass it along. This
445 // can happen if doAsyncError() was called from a call to assert and then
446 // again by pump_().
447 if (opt_e && opt_e.isControlBreakingException) {
448 throw opt_e;
449 }
450
451 // Prevent another timeout error from triggering for this test step.
452 this.stopTimeoutTimer_();
453
454 // doError() uses test.name. Here, we create a dummy test and give it a more
455 // helpful name based on the step we're currently on.
456 var fakeTestObj = new goog.testing.TestCase.Test(this.curStepName_,
457 goog.nullFunction);
458 if (this.activeTest) {
459 fakeTestObj.name = this.activeTest.name + ' [' + fakeTestObj.name + ']';
460 }
461
462 if (this.activeTest) {
463 // Note: if the test has an error, and then tearDown has an error, they will
464 // both be reported.
465 this.doError(fakeTestObj, opt_e);
466 } else {
467 this.exceptionBeforeTest = opt_e;
468 }
469
470 // This is a potential entry point, so we pump. We also add in a bit of a
471 // delay to try and prevent any async behavior from the failed test from
472 // causing the next test to fail.
473 this.timeout(goog.bind(this.pump_, this, this.doAsyncErrorTearDown_),
474 this.timeToSleepAfterFailure);
475
476 // We just caught an exception, so we do not want the code above us on the
477 // stack to continue executing. If pump_ is in our call-stack, then it will
478 // batch together multiple errors, so we only increment the count if pump_ is
479 // not in the stack and let pump_ increment the count when it batches them.
480 if (!this.returnWillPump_) {
481 this.numControlExceptionsExpected_ += 1;
482 this.dbgLog_('doAsynError: numControlExceptionsExpected_ = ' +
483 this.numControlExceptionsExpected_ + ' and throwing exception.');
484 }
485
486 // Copy the error message to ControlBreakingException.
487 var message = '';
488 if (typeof opt_e == 'string') {
489 message = opt_e;
490 } else if (opt_e && opt_e.message) {
491 message = opt_e.message;
492 }
493 throw new goog.testing.AsyncTestCase.ControlBreakingException(message);
494};

555/**
556 * Wraps doAsyncError() for when we are sure that the test runner has no user
557 * code above it in the stack.
558 * @param {string|Error=} opt_e The exception object associated with the
559 * failure or a string.
560 * @private
561 */
562goog.testing.AsyncTestCase.prototype.doTopOfStackAsyncError_ =
563 function(opt_e) {
564 /** @preserveTry */
565 try {
566 this.doAsyncError(opt_e);
567 } catch (e) {
568 // We know that we are on the top of the stack, so there is no need to
569 // throw this exception in this case.
570 if (e.isControlBreakingException) {
571 this.numControlExceptionsExpected_ -= 1;
572 this.dbgLog_('doTopOfStackAsyncError_: numControlExceptionsExpected_ = ' +
573 this.numControlExceptionsExpected_ + ' and catching exception.');
574 } else {
575 throw e;
576 }
577 }
578};

581/**
582 * Calls the tearDown function, catching any errors, and then moves on to
583 * the next step in the testing cycle.
584 * @private
585 */
586goog.testing.AsyncTestCase.prototype.doAsyncErrorTearDown_ = function() {
587 if (this.inException_) {
588 // We get here if tearDown is throwing the error.
589 // Upon calling continueTesting, the inline function 'doAsyncError' (set
590 // below) is run.
591 this.endCurrentStep_();
592 } else {
593 this.inException_ = true;
594 this.isReady_ = true;
595
596 // The continue point is different depending on if the error happened in
597 // setUpPage() or in setUp()/test*()/tearDown().
598 var stepFuncAfterError = this.nextStepFunc_;
599 var stepNameAfterError = 'TestCase.execute (after error)';
600 if (this.activeTest) {
601 stepFuncAfterError = this.doIteration_;
602 stepNameAfterError = 'doIteration (after error)';
603 }
604
605 // We must set the next step before calling tearDown.
606 this.setNextStep_(function() {
607 this.inException_ = false;
608 // This is null when an error happens in setUpPage.
609 this.setNextStep_(stepFuncAfterError, stepNameAfterError);
610 }, 'doAsyncError');
611
612 // Call the test's tearDown().
613 if (!this.cleanedUp_) {
614 this.cleanedUp_ = true;
615 this.tearDown();
616 }
617 }
618};

621/**
622 * Replaces the asserts.js assert_() and fail() functions with a wrappers to
623 * catch the exceptions.
624 * @private
625 */
626goog.testing.AsyncTestCase.prototype.hookAssert_ = function() {
627 if (!this.origAssert_) {
628 this.origAssert_ = _assert;
629 this.origFail_ = fail;
630 var self = this;
631 _assert = function() {
632 /** @preserveTry */
633 try {
634 self.origAssert_.apply(this, arguments);
635 } catch (e) {
636 self.dbgLog_('Wrapping failed assert()');
637 self.doAsyncError(e);
638 }
639 };
640 fail = function() {
641 /** @preserveTry */
642 try {
643 self.origFail_.apply(this, arguments);
644 } catch (e) {
645 self.dbgLog_('Wrapping fail()');
646 self.doAsyncError(e);
647 }
648 };
649 }
650};

653/**
654 * Sets a window.onerror handler for catching exceptions that happen in async
655 * callbacks. Note that as of Safari 3.1, Safari does not support this.
656 * @private
657 */
658goog.testing.AsyncTestCase.prototype.hookOnError_ = function() {
659 if (!this.origOnError_) {
660 this.origOnError_ = window.onerror;
661 var self = this;
662 window.onerror = function(error, url, line) {
663 // Ignore exceptions that we threw on purpose.
664 var cbe =
665 goog.testing.AsyncTestCase.ControlBreakingException.TO_STRING;
666 if (String(error).indexOf(cbe) != -1 &&
667 self.numControlExceptionsExpected_) {
668 self.numControlExceptionsExpected_ -= 1;
669 self.dbgLog_('window.onerror: numControlExceptionsExpected_ = ' +
670 self.numControlExceptionsExpected_ + ' and ignoring exception. ' +
671 error);
672 // Tell the browser not to compain about the error.
673 return true;
674 } else {
675 self.dbgLog_('window.onerror caught exception.');
676 var message = error + '\nURL: ' + url + '\nLine: ' + line;
677 self.doTopOfStackAsyncError_(message);
678 // Tell the browser to complain about the error.
679 return false;
680 }
681 };
682 }
683};

690goog.testing.AsyncTestCase.prototype.unhookAll_ = function() {

702/**
703 * Enables the timeout timer. This timer fires unless continueTesting is
704 * called.
705 * @private
706 */
707goog.testing.AsyncTestCase.prototype.startTimeoutTimer_ = function() {
708 if (!this.timeoutHandle_ && this.stepTimeout > 0) {
709 this.timeoutHandle_ = this.timeout(goog.bind(function() {
710 this.dbgLog_('Timeout timer fired with id ' + this.timeoutHandle_);
711 this.timeoutHandle_ = 0;
712
713 this.doTopOfStackAsyncError_('Timed out while waiting for ' +
714 'continueTesting() to be called.');
715 }, this, null), this.stepTimeout);
716 this.dbgLog_('Started timeout timer with id ' + this.timeoutHandle_);
717 }
718};

746/**
747 * Calls the given function, redirecting any exceptions to doAsyncError.
748 * @param {Function} func The function to call.
749 * @return {!goog.testing.AsyncTestCase.TopStackFuncResult_} Returns a
750 * TopStackFuncResult_.
751 * @private
752 */
753goog.testing.AsyncTestCase.prototype.callTopOfStackFunc_ = function(func) {
754 /** @preserveTry */
755 try {
756 func.call(this);
757 return {controlBreakingExceptionThrown: false, message: ''};
758 } catch (e) {
759 this.dbgLog_('Caught exception in callTopOfStackFunc_');
760 /** @preserveTry */
761 try {
762 this.doAsyncError(e);
763 return {controlBreakingExceptionThrown: false, message: ''};
764 } catch (e2) {
765 if (!e2.isControlBreakingException) {
766 throw e2;
767 }
768 return {controlBreakingExceptionThrown: true, message: e2.message};
769 }
770 }

774/**
775 * Calls the next callback when the isReady_ flag is true.
776 * @param {Function=} opt_doFirst A function to call before pumping.
777 * @private
778 * @throws Throws a ControlBreakingException if there were any failing steps.
779 */
780goog.testing.AsyncTestCase.prototype.pump_ = function(opt_doFirst) {
781 // If this function is already above us in the call-stack, then we should
782 // return rather than pumping in order to minimize call-stack depth.
783 if (!this.returnWillPump_) {
784 this.setBatchTime(this.now());
785 this.returnWillPump_ = true;
786 var topFuncResult = {};
787
788 if (opt_doFirst) {
789 topFuncResult = this.callTopOfStackFunc_(opt_doFirst);
790 }
791 // Note: we don't check for this.running here because it is not set to true
792 // while executing setUpPage and tearDownPage.
793 // Also, if isReady_ is false, then one of two things will happen:
794 // 1. Our timeout callback will be called.
795 // 2. The tests will call continueTesting(), which will call pump_() again.
796 while (this.isReady_ && this.nextStepFunc_ &&
797 !topFuncResult.controlBreakingExceptionThrown) {
798 this.curStepFunc_ = this.nextStepFunc_;
799 this.curStepName_ = this.nextStepName_;
800 this.nextStepFunc_ = null;
801 this.nextStepName_ = '';
802
803 this.dbgLog_('Performing step: ' + this.curStepName_);
804 topFuncResult =
805 this.callTopOfStackFunc_(/** @type {Function} */(this.curStepFunc_));
806
807 // If the max run time is exceeded call this function again async so as
808 // not to block the browser.
809 var delta = this.now() - this.getBatchTime();
810 if (delta > goog.testing.TestCase.maxRunTime &&
811 !topFuncResult.controlBreakingExceptionThrown) {
812 this.saveMessage('Breaking async');
813 var self = this;
814 this.timeout(function() { self.pump_(); }, 100);
815 break;
816 }
817 }
818 this.returnWillPump_ = false;
819 } else if (opt_doFirst) {
820 opt_doFirst.call(this);
821 }
