16 * @fileoverview Utility for running multiple test files that utilize the same
17 * interface as goog.testing.TestRunner. Each test is run in series and their
18 * results aggregated. The main usecase for the MultiTestRunner is to allow
19 * the testing of all tests in a project locally.
20 *
21 */

635/**
636 * Starts executing the tests.
637 */
638goog.testing.MultiTestRunner.prototype.start = function() {
639 this.startButtonEl_.disabled = true;
640 this.stopButtonEl_.disabled = false;
641 this.stopped_ = false;
642 this.active_ = true;
643 this.finished_ = {};
644 this.activeTests_ = this.getTestsToRun();
645 this.startedCount_ = 0;
646 this.resultCount_ = 0;
647 this.passes_ = 0;
648 this.stats_ = [];
649 this.startTime_ = goog.now();
650
651 this.resetProgressDom_();
652 goog.dom.removeChildren(this.logEl_);
653
654 this.resetReport_();
655 this.clearStats_();
656 this.showTab_(0);
657
658 // Ensure the pool isn't too big.
659 while (this.getChildCount() > this.poolSize_) {
660 this.removeChildAt(0, true).dispose();
661 }
662
663 // Start a test in each runner.
664 for (var i = 0; i < this.poolSize_; i++) {
665 if (i >= this.getChildCount()) {
666 var testFrame = new goog.testing.MultiTestRunner.TestFrame(
667 this.basePath_, this.timeoutMs_, this.verbosePasses_, this.dom_);
668 this.addChild(testFrame, true);
669 }
670 this.runNextTest_(
671 /** @type {goog.testing.MultiTestRunner.TestFrame} */
672 (this.getChildAt(i)));
673 }
674};

677/**
678 * Logs a message to the log window.
679 * @param {string} msg A message to log.
680 */
681goog.testing.MultiTestRunner.prototype.log = function(msg) {
682 if (msg != '.') {
683 msg = this.getTimeStamp_() + ' : ' + msg;
684 }
685
686 this.logEl_.appendChild(this.dom_.createDom(goog.dom.TagName.DIV, null, msg));

689 var top = this.logEl_.scrollTop;

691 if (top == 0 || top > height - 50) {
692 this.logEl_.scrollTop = height;
693 }

697/**
698 * Processes a result returned from a TestFrame. If there are tests remaining
699 * it will trigger the next one to be run, otherwise if there are no tests and
700 * all results have been recieved then it will call finish.
701 * @param {goog.testing.MultiTestRunner.TestFrame} frame The frame that just
702 * finished.
703 */
704goog.testing.MultiTestRunner.prototype.processResult = function(frame) {
705 var success = frame.isSuccess();
706 var report = frame.getReport();
707 var test = frame.getTestFile();
708
709 this.stats_.push(frame.getStats());
710 this.finished_[test] = true;
711
712 var prefix = success ? '' : '*** FAILURE *** ';
713 this.log(prefix +
714 this.trimFileName_(test) + ' : ' + (success ? 'Passed' : 'Failed'));
715
716 this.resultCount_++;
717
718 if (success) {
719 this.passes_++;
720 }
721
722 this.drawProgressSegment_(test, success);
723 this.writeCurrentSummary_();
724 if (!(success && this.hidePasses_)) {
725 this.drawTestResult_(test, success, report);
726 }
727
728 if (!this.stopped_ && this.startedCount_ < this.activeTests_.length) {
729 this.runNextTest_(frame);
730 } else if (this.resultCount_ == this.activeTests_.length) {
731 this.finish_();
732 }
733};

736/**
737 * Runs the next available test, if there are any left.
738 * @param {goog.testing.MultiTestRunner.TestFrame} frame Where to run the test.
739 * @private
740 */
741goog.testing.MultiTestRunner.prototype.runNextTest_ = function(frame) {
742 if (this.startedCount_ < this.activeTests_.length) {
743 var nextTest = this.activeTests_[this.startedCount_++];
744 this.log(this.trimFileName_(nextTest) + ' : Loading');
745 frame.runTest(nextTest);
746 }
747};

750/**
751 * Handles the test finishing, processing the results and rendering the report.
752 * @private
753 */
754goog.testing.MultiTestRunner.prototype.finish_ = function() {
755 if (this.stopped_) {
756 this.log('Stopped');
757 } else {
758 this.log('Finished');
759 }
760
761 this.startButtonEl_.disabled = false;
762 this.stopButtonEl_.disabled = true;
763 this.active_ = false;
764
765 this.showTab_(1);
766 this.drawStats_();
767
768 // Remove all the test frames
769 while (this.getChildCount() > 0) {
770 this.removeChildAt(0, true).dispose();
771 }
772
773 // Compute tests that did not finish before the stop button was hit.
774 var unfinished = [];
775 for (var i = 0; i < this.activeTests_.length; i++) {
776 var test = this.activeTests_[i];
777 if (!this.finished_[test]) {
778 unfinished.push(test);
779 }
780 }
781
782 if (unfinished.length) {
783 this.reportEl_.appendChild(goog.dom.createDom(
784 goog.dom.TagName.PRE, undefined,
785 'Theses tests did not finish:\n' + unfinished.join('\n')));
786 }
787};

861 var hist = {}, data = [], xlabels = [], ylabels = [];

864 var num = this.stats_[i][statsField];

886 ylabels.push(i);

1409/**
1410 * Checks the active test for completion, keeping track of the tests' various
1411 * execution stages.
1412 * @private
1413 */
1414goog.testing.MultiTestRunner.TestFrame.prototype.checkForCompletion_ =
1415 function() {
1416 var js = goog.dom.getFrameContentWindow(this.iframeEl_);
1417 switch (this.currentState_) {
1418 case 0:
1419 if (this.iframeLoaded_ && js['G_testRunner']) {
1420 this.lastStateTime_ = goog.now();
1421 this.currentState_++;
1422 }
1423 break;
1424 case 1:
1425 if (js['G_testRunner']['isInitialized']()) {
1426 this.lastStateTime_ = goog.now();
1427 this.currentState_++;
1428 }
1429 break;
1430 case 2:
1431 if (js['G_testRunner']['isFinished']()) {
1432 var tr = js['G_testRunner'];
1433 this.isSuccess_ = tr['isSuccess']();
1434 this.report_ = tr['getReport'](this.verbosePasses_);
1435 this.runTime_ = tr['getRunTime']();
1436 this.numFilesLoaded_ = tr['getNumFilesLoaded']();
1437 this.finish_();
1438 return;
1439 }
1440 }
1441
1442 // Check to see if the test has timed out.
1443 if (goog.now() - this.lastStateTime_ > this.timeoutMs_) {
1444 this.report_ = this.testFile_ + ' timed out ' +
1445 goog.testing.MultiTestRunner.STATES[this.currentState_];
1446 this.isSuccess_ = false;
1447 this.finish_();
1448 return;
1449 }
1450
1451 // Check again in 100ms.
1452 goog.Timer.callOnce(this.checkForCompletion_, 100, this);
