105/**
106 * Simulates the static XhrIo send method.
107 * @param {string} url Uri to make request to.
108 * @param {Function=} opt_callback Callback function for when request is
109 * complete.
110 * @param {string=} opt_method Send method, default: GET.
111 * @param {string=} opt_content Post data.
112 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
113 * request.
114 * @param {number=} opt_timeoutInterval Number of milliseconds after which an
115 * incomplete request will be aborted; 0 means no timeout is set.
116 * @param {boolean=} opt_withCredentials Whether to send credentials with the
117 * request. Default to false. See {@link goog.net.XhrIo#setWithCredentials}.
118 * @return {!goog.testing.net.XhrIo} The mocked sent XhrIo.
119 */
120goog.testing.net.XhrIo.send = function(url, opt_callback, opt_method,
121 opt_content, opt_headers,
122 opt_timeoutInterval,
123 opt_withCredentials) {
124 var x = new goog.testing.net.XhrIo();
125 goog.testing.net.XhrIo.sendInstances_.push(x);
126 if (opt_callback) {
127 goog.events.listen(x, goog.net.EventType.COMPLETE, opt_callback);
128 }
129 goog.events.listen(x,
130 goog.net.EventType.READY,
131 goog.partial(goog.testing.net.XhrIo.cleanupSend_, x));
132 if (opt_timeoutInterval) {
133 x.setTimeoutInterval(opt_timeoutInterval);
134 }
135 x.setWithCredentials(Boolean(opt_withCredentials));
136 x.send(url, opt_method, opt_content, opt_headers);
137
138 return x;
139};

377/**
378 * Simulates the XhrIo send.
379 * @param {string} url Uri to make request too.
380 * @param {string=} opt_method Send method, default: GET.
381 * @param {string=} opt_content Post data.
382 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
383 * request.
384 */
385goog.testing.net.XhrIo.prototype.send = function(url, opt_method, opt_content,
386 opt_headers) {
387 if (this.xhr_) {
388 throw Error('[goog.net.XhrIo] Object is active with another request');
389 }
390
391 this.lastUri_ = url;
392 this.lastMethod_ = opt_method || 'GET';
393 this.lastContent_ = opt_content;
394 this.lastHeaders_ = opt_headers;
395
396 if (this.testQueue_) {
397 this.testQueue_.enqueue(['s', url, opt_method, opt_content, opt_headers]);
398 }
399 this.xhr_ = true;
400 this.active_ = true;
401 this.readyState_ = goog.net.XmlHttp.ReadyState.UNINITIALIZED;
402 this.simulateReadyStateChange(goog.net.XmlHttp.ReadyState.LOADING);
403};

417/**
418 * Simulates changing to the new ready state.
419 * @param {number} readyState Ready state to change to.
420 */
421goog.testing.net.XhrIo.prototype.simulateReadyStateChange =
422 function(readyState) {
423 if (readyState < this.readyState_) {
424 throw Error('Readystate cannot go backwards');
425 }
426
427 // INTERACTIVE can be dispatched repeatedly as more data is reported.
428 if (readyState == goog.net.XmlHttp.ReadyState.INTERACTIVE &&
429 readyState == this.readyState_) {
430 this.dispatchEvent(goog.net.EventType.READY_STATE_CHANGE);
431 return;
432 }
433
434 while (this.readyState_ < readyState) {
435 this.readyState_++;
436 this.dispatchEvent(goog.net.EventType.READY_STATE_CHANGE);
437
438 if (this.readyState_ == goog.net.XmlHttp.ReadyState.COMPLETE) {
439 this.active_ = false;
440 this.dispatchEvent(goog.net.EventType.COMPLETE);
441 }
442 }
443};

461/**
462 * Simulates receiving a response.
463 * @param {number} statusCode Simulated status code.
464 * @param {string|Document|ArrayBuffer|null} response Simulated response.
465 * @param {Object=} opt_headers Simulated response headers.
466 */
467goog.testing.net.XhrIo.prototype.simulateResponse = function(statusCode,
468 response, opt_headers) {
469 this.statusCode_ = statusCode;
470 this.response_ = response || '';
471 this.responseHeaders_ = opt_headers || {};
472
473 try {
474 if (this.isSuccess()) {
475 this.simulateReadyStateChange(goog.net.XmlHttp.ReadyState.COMPLETE);
476 this.dispatchEvent(goog.net.EventType.SUCCESS);
477 } else {
478 this.lastErrorCode_ = goog.net.ErrorCode.HTTP_ERROR;
479 this.lastError_ = this.getStatusText() + ' [' + this.getStatus() + ']';
480 this.simulateReadyStateChange(goog.net.XmlHttp.ReadyState.COMPLETE);
481 this.dispatchEvent(goog.net.EventType.ERROR);
482 }
483 } finally {
484 this.simulateReady();
485 }
486};

721/**
722 * Returns all response headers as a key-value map.
723 * Multiple values for the same header key can be combined into one,
724 * separated by a comma and a space.
725 * Note that the native getResponseHeader method for retrieving a single header
726 * does a case insensitive match on the header name. This method does not
727 * include any case normalization logic, it will just return a key-value
728 * representation of the headers.
729 * See: http://www.w3.org/TR/XMLHttpRequest/#the-getresponseheader()-method
730 * @return {!Object<string, string>} An object with the header keys as keys
731 * and header values as values.
732 */
733goog.testing.net.XhrIo.prototype.getResponseHeaders = function() {
734 var headersObject = {};
735 goog.object.forEach(this.responseHeaders_, function(value, key) {
736 if (headersObject[key]) {
737 headersObject[key] += ', ' + value;
738 } else {
739 headersObject[key] = value;
740 }
741 });
742 return headersObject;
