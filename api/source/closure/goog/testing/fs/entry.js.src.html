197/**
198 * Return a deferred that will call its errback if this entry has been deleted.
199 * In addition, the deferred will only run after a timeout of 0, and all its
200 * callbacks will run with the entry as "this".
201 *
202 * @param {string} action The name of the action being performed. For error
203 * reporting.
204 * @return {!goog.async.Deferred} The deferred that will be called after a
205 * timeout of 0.
206 * @protected
207 */
208goog.testing.fs.Entry.prototype.checkNotDeleted = function(action) {
209 var d = new goog.async.Deferred(undefined, this);
210 goog.Timer.callOnce(function() {
211 if (this.deleted) {
212 var err = new goog.fs.Error(
213 /** @type {!FileError} */ ({'name': 'NotFoundError'}),
214 action);
215 d.errback(err);
216 } else {
217 d.callback();
218 }
219 }, 0, this);
220 return d;
221};

425/**
426 * Get a file or directory entry from a path. This handles parsing the path for
427 * subdirectories and throwing appropriate errors should something go wrong.
428 *
429 * @param {string} path The path to the entry, relative to this directory.
430 * @param {goog.fs.DirectoryEntry.Behavior} behavior The behavior for loading
431 * the entry.
432 * @param {boolean} isFile Whether a file or directory is being loaded.
433 * @param {function(!goog.testing.fs.DirectoryEntry, string) :
434 * !goog.testing.fs.Entry} createFn
435 * The function for creating the entry if it doesn't yet exist. This is
436 * passed the parent entry and the name of the new entry.
437 * @return {!goog.testing.fs.Entry} The loaded entry.
438 * @private
439 */
440goog.testing.fs.DirectoryEntry.prototype.getEntry_ = function(
441 path, behavior, isFile, createFn) {
442 // Filter out leading, trailing, and duplicate slashes.
443 var components = goog.array.filter(path.split('/'), goog.functions.identity);
444
445 var basename = /** @type {string} */ (goog.array.peek(components)) || '';
446 var dir = goog.string.startsWith(path, '/') ?
447 this.getFileSystem().getRoot() : this;
448
449 goog.array.forEach(components.slice(0, -1), function(p) {
450 var subdir = dir.children[p];
451 if (!subdir) {
452 throw new goog.fs.Error(
453 /** @type {!FileError} */ ({'name': 'NotFoundError'}),
454 'loading ' + path + ' from ' + this.getFullPath() + ' (directory ' +
455 dir.getFullPath() + '/' + p + ')');
456 }
457 dir = subdir;
458 }, this);
459
460 // If there is no basename, the path must resolve to the root directory.
461 var entry = basename ? dir.children[basename] : dir;
462
463 if (!entry) {
464 if (behavior == goog.fs.DirectoryEntry.Behavior.DEFAULT) {
465 throw new goog.fs.Error(
466 /** @type {!FileError} */ ({'name': 'NotFoundError'}),
467 'loading ' + path + ' from ' + this.getFullPath());
468 } else {
469 goog.asserts.assert(
470 behavior == goog.fs.DirectoryEntry.Behavior.CREATE ||
471 behavior == goog.fs.DirectoryEntry.Behavior.CREATE_EXCLUSIVE);
472 entry = createFn(dir, basename);
473 dir.children[basename] = entry;
474 this.lastModifiedTimestamp_ = goog.now();
475 return entry;
476 }
477 } else if (behavior == goog.fs.DirectoryEntry.Behavior.CREATE_EXCLUSIVE) {
478 throw new goog.fs.Error(
479 /** @type {!FileError} */ ({'name': 'InvalidModificationError'}),
480 'loading ' + path + ' from ' + this.getFullPath());
481 } else if (entry.isFile() != isFile) {
482 throw new goog.fs.Error(
483 /** @type {!FileError} */ ({'name': 'TypeMismatchError'}),
484 'loading ' + path + ' from ' + this.getFullPath());
485 } else {
486 if (behavior == goog.fs.DirectoryEntry.Behavior.CREATE) {
487 this.lastModifiedTimestamp_ = goog.now();
488 }
489 return entry;
490 }
491};
