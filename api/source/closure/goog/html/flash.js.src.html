27/**
28 * Attributes and param tag name attributes not allowed to be overriden
29 * when calling createObject() and createObjectForOldIe().
30 *
31 * While values that should be specified as params are probably not
32 * recognized as attributes, we block them anyway just to be sure.
33 * @const {!Array<string>}
34 * @private
35 */
36goog.html.flash.FORBIDDEN_ATTRS_AND_PARAMS_ON_FLASH_ = [
37 'classid', // Used on old IE.
38 'data', // Used in <object> to specify a URL.
39 'movie', // Used on old IE.
40 'type', // Used in <object> on for non-IE/modern IE.
41 'typemustmatch' // Always set to a fixed value.
42];

45goog.html.flash.createEmbed = function(src, opt_attributes) {
46 var fixedAttributes = {
47 'src': src,
48 'type': 'application/x-shockwave-flash',
49 'pluginspage': 'https://www.macromedia.com/go/getflashplayer'
50 };
51 var defaultAttributes = {
52 'allownetworking': 'none',
53 'allowscriptaccess': 'never'
54 };
55 var attributes = goog.html.SafeHtml.combineAttributes(
56 fixedAttributes, defaultAttributes, opt_attributes);
57 return goog.html.SafeHtml.
58 createSafeHtmlTagSecurityPrivateDoNotAccessOrElse('embed', attributes);
59};

62goog.html.flash.createObject = function(
63 data, opt_params, opt_attributes) {
64 goog.html.flash.verifyKeysNotInMaps(
65 goog.html.flash.FORBIDDEN_ATTRS_AND_PARAMS_ON_FLASH_,
66 opt_attributes,
67 opt_params);
68
69 var paramTags = goog.html.flash.combineParams(
70 {
71 'allownetworking': 'none',
72 'allowscriptaccess': 'never'
73 },
74 opt_params);
75 var fixedAttributes = {
76 'data': data,
77 'type': 'application/x-shockwave-flash',
78 'typemustmatch': ''
79 };
80 var attributes = goog.html.SafeHtml.combineAttributes(
81 fixedAttributes, {}, opt_attributes);
82
83 return goog.html.SafeHtml.createSafeHtmlTagSecurityPrivateDoNotAccessOrElse(
84 'object', attributes, paramTags);
85};

112/**
113 * @param {!Object<string, string|!goog.string.TypedString>} defaultParams
114 * @param {!Object<string, string>=}
115 * opt_params Optional params passed to create*().
116 * @return {!Array<!goog.html.SafeHtml>} Combined params.
117 * @throws {Error} If opt_attributes contains an attribute with the same name
118 * as an attribute in fixedAttributes.
119 * @package
120 */
121goog.html.flash.combineParams = function(defaultParams, opt_params) {
122 var combinedParams = {};
123 var name;
124
125 for (name in defaultParams) {
126 goog.asserts.assert(name.toLowerCase() == name, 'Must be lower case');
127 combinedParams[name] = defaultParams[name];
128 }
129 for (name in opt_params) {
130 var nameLower = name.toLowerCase();
131 if (nameLower in defaultParams) {
132 delete combinedParams[nameLower];
133 }
134 combinedParams[name] = opt_params[name];
135 }
136
137 var paramTags = [];
138 for (name in combinedParams) {
139 paramTags.push(
140 goog.html.SafeHtml.createSafeHtmlTagSecurityPrivateDoNotAccessOrElse(
141 'param', {'name': name, 'value': combinedParams[name]}));
142
143 }
144 return paramTags;
145};

148/**
149 * Checks that keys are not present as keys in maps.
150 * @param {!Array<string>} keys Keys that must not be present, lower-case.
151 * @param {!Object<string, goog.html.SafeHtml.AttributeValue_>=}
152 * opt_attributes Optional attributes passed to create*().
153 * @param {!Object<string, string>=} opt_params Optional params passed to
154 * createObject*().
155 * @throws {Error} If any of keys exist as a key, ignoring case, in
156 * opt_attributes or opt_params.
157 * @package
158 */
159goog.html.flash.verifyKeysNotInMaps = function(
160 keys, opt_attributes, opt_params) {
161 var verifyNotInMap = function(keys, map, type) {
162 for (var keyMap in map) {
163 var keyMapLower = keyMap.toLowerCase();
164 for (var i = 0; i < keys.length; i++) {
165 var keyToCheck = keys[i];
166 goog.asserts.assert(keyToCheck.toLowerCase() == keyToCheck);
167 if (keyMapLower == keyToCheck) {
168 throw Error('Cannot override "' + keyToCheck + '" ' + type +
169 ', got "' + keyMap + '" with value "' + map[keyMap] + '"');
170 }
171 }
172 }
173 };
174
175 verifyNotInMap(keys, opt_attributes, 'attribute');
176 verifyNotInMap(keys, opt_params, 'param');
177};
