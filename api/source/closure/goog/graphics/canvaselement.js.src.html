408/**
409 * Draw the path. Should be treated as package scope.
410 * @param {CanvasRenderingContext2D} ctx The context to draw the element in.
411 * @suppress {deprecated} goog.graphics is deprecated.
412 */
413goog.graphics.CanvasPathElement.prototype.draw = function(ctx) {
414 this.drawn_ = true;
415
416 ctx.beginPath();
417 this.path_.forEachSegment(function(segment, args) {
418 switch (segment) {
419 case goog.graphics.Path.Segment.MOVETO:
420 ctx.moveTo(args[0], args[1]);
421 break;
422 case goog.graphics.Path.Segment.LINETO:
423 for (var i = 0; i < args.length; i += 2) {
424 ctx.lineTo(args[i], args[i + 1]);
425 }
426 break;
427 case goog.graphics.Path.Segment.CURVETO:
428 for (var i = 0; i < args.length; i += 6) {
429 ctx.bezierCurveTo(args[i], args[i + 1], args[i + 2],
430 args[i + 3], args[i + 4], args[i + 5]);
431 }
432 break;
433 case goog.graphics.Path.Segment.ARCTO:
434 throw Error('Canvas paths cannot contain arcs');
435 case goog.graphics.Path.Segment.CLOSE:
436 ctx.closePath();
437 break;
438 }
439 });
440};

583/**
584 * Update the styles of the DIVs.
585 * @private
586 */
587goog.graphics.CanvasTextElement.prototype.updateStyle_ = function() {
588 var x1 = this.x1_;
589 var x2 = this.x2_;
590 var y1 = this.y1_;
591 var y2 = this.y2_;
592 var align = this.align_;
593 var font = this.font_;
594 var style = this.getElement().style;
595 var scaleX = this.getGraphics().getPixelScaleX();
596 var scaleY = this.getGraphics().getPixelScaleY();
597
598 if (x1 == x2) {
599 // Special case vertical text
600 style.lineHeight = '90%';
601
602 this.innerElement_.style.verticalAlign = align == 'center' ? 'middle' :
603 align == 'left' ? (y1 < y2 ? 'top' : 'bottom') :
604 y1 < y2 ? 'bottom' : 'top';
605 style.textAlign = 'center';
606
607 var w = font.size * scaleX;
608 style.top = Math.round(Math.min(y1, y2) * scaleY) + 'px';
609 style.left = Math.round((x1 - w / 2) * scaleX) + 'px';
610 style.width = Math.round(w) + 'px';
611 style.height = Math.abs(y1 - y2) * scaleY + 'px';
612
613 style.fontSize = font.size * 0.6 * scaleY + 'pt';
614 } else {
615 style.lineHeight = '100%';
616 this.innerElement_.style.verticalAlign = 'top';
617 style.textAlign = align;
618
619 style.top = Math.round(((y1 + y2) / 2 - font.size * 2 / 3) * scaleY) + 'px';
620 style.left = Math.round(x1 * scaleX) + 'px';
621 style.width = Math.round(Math.abs(x2 - x1) * scaleX) + 'px';
622 style.height = 'auto';
623
624 style.fontSize = font.size * scaleY + 'pt';
625 }
626
627 style.fontWeight = font.bold ? 'bold' : 'normal';
628 style.fontStyle = font.italic ? 'italic' : 'normal';
629 style.fontFamily = font.family;
630
631 var fill = this.getFill();
632 style.color = fill.getColor() || fill.getColor1();
633};

636/**
637 * Update the text content.
638 * @private
639 */
640goog.graphics.CanvasTextElement.prototype.updateText_ = function() {
641 if (this.x1_ == this.x2_) {
642 // Special case vertical text
643 var html =
644 goog.array.map(
645 this.text_.split(''),
646 function(entry) { return goog.string.htmlEscape(entry); })
647 .join('<br>');
648 // Creating a SafeHtml for each character would be quite expensive, and it's
649 // obvious that this is safe, so an unchecked conversion is appropriate.
650 var safeHtml = goog.html.uncheckedconversions
651 .safeHtmlFromStringKnownToSatisfyTypeContract(
652 goog.string.Const.from('Concatenate escaped chars and <br>'),
653 html);
654 goog.dom.safe.setInnerHtml(
655 /** @type {!Element} */ (this.innerElement_), safeHtml);
656 } else {
657 goog.dom.safe.setInnerHtml(
658 /** @type {!Element} */ (this.innerElement_),
659 goog.html.SafeHtml.htmlEscape(this.text_));
660 }
661};

782 * @param {CanvasRenderingContext2D} ctx The context to draw the element in.

784goog.graphics.CanvasImageElement.prototype.draw = function(ctx) {
785 if (this.img_) {
786 if (this.w_ && this.h_) {
787 // If the image is already loaded, draw it.
788 ctx.drawImage(this.img_, this.x_, this.y_, this.w_, this.h_);
789 }
790 this.drawn_ = true;
791
792 } else {
793 // Otherwise, load it.
794 var img = new Image();
795 img.onload = goog.bind(this.handleImageLoad_, this, img);
796 // TODO(robbyw): Handle image load errors.
797 img.src = this.src_;
798 }
