40/**
41 * A Graphics implementation for drawing using canvas.
42 * @param {string|number} width The (non-zero) width in pixels. Strings
43 * expressing percentages of parent with (e.g. '80%') are also accepted.
44 * @param {string|number} height The (non-zero) height in pixels. Strings
45 * expressing percentages of parent with (e.g. '80%') are also accepted.
46 * @param {?number=} opt_coordWidth The coordinate width - if
47 * omitted or null, defaults to same as width.
48 * @param {?number=} opt_coordHeight The coordinate height - if
49 * omitted or null, defaults to same as height.
50 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper object for the
51 * document we want to render in.
52 * @constructor
53 * @extends {goog.graphics.AbstractGraphics}
54 * @deprecated goog.graphics is deprecated. It existed to abstract over browser
55 * differences before the canvas tag was widely supported. See
56 * http://en.wikipedia.org/wiki/Canvas_element for details.
57 */
58goog.graphics.CanvasGraphics = function(width, height,
59 opt_coordWidth, opt_coordHeight,
60 opt_domHelper) {
61 goog.graphics.AbstractGraphics.call(this, width, height,
62 opt_coordWidth, opt_coordHeight,
63 opt_domHelper);
64};
65goog.inherits(goog.graphics.CanvasGraphics, goog.graphics.AbstractGraphics);

129/**
130 * Push an element transform on to the transform stack.
131 * @param {goog.graphics.Element} element The transformed element.
132 */
133goog.graphics.CanvasGraphics.prototype.pushElementTransform = function(
134 element) {
135 var ctx = this.getContext();
136 ctx.save();
137
138 var transform = element.getTransform();
139
140 // TODO(robbyw): Test for unsupported transforms i.e. skews.
141 var tx = transform.getTranslateX();
142 var ty = transform.getTranslateY();
143 if (tx || ty) {
144 ctx.translate(tx, ty);
145 }
146
147 var sinTheta = transform.getShearY();
148 if (sinTheta) {
149 ctx.rotate(Math.asin(sinTheta));
150 }
151};

255/** @override */
256goog.graphics.CanvasGraphics.prototype.getPixelSize = function() {
257 // goog.style.getSize does not work for Canvas elements. We
258 // have to compute the size manually if it is percentage based.
259 var width = this.width;
260 var height = this.height;
261 var computeWidth = goog.isString(width) && width.indexOf('%') != -1;
262 var computeHeight = goog.isString(height) && height.indexOf('%') != -1;
263
264 if (!this.isInDocument() && (computeWidth || computeHeight)) {
265 return null;
266 }
267
268 var parent;
269 var parentSize;
270
271 if (computeWidth) {
272 parent = /** @type {Element} */ (this.getElement().parentNode);
273 parentSize = goog.style.getSize(parent);
274 width = parseFloat(/** @type {string} */ (width)) * parentSize.width / 100;
275 }
276
277 if (computeHeight) {
278 parent = parent || /** @type {Element} */ (this.getElement().parentNode);
279 parentSize = parentSize || goog.style.getSize(parent);
280 height = parseFloat(/** @type {string} */ (height)) * parentSize.height /
281 100;
282 }
283
284 return new goog.math.Size(/** @type {number} */ (width),
285 /** @type {number} */ (height));
286};

337/**
338 * Redraw the entire canvas.
339 */
340goog.graphics.CanvasGraphics.prototype.redraw = function() {
341 if (this.preventRedraw_) {
342 this.needsRedraw_ = true;
343 return;
344 }
345
346 if (this.isInDocument()) {
347 this.reset();
348
349 if (this.coordWidth) {
350 var pixels = this.getPixelSize();
351 this.getContext().scale(pixels.width / this.coordWidth,
352 pixels.height / this.coordHeight);
353 }
354 if (this.coordLeft || this.coordTop) {
355 this.getContext().translate(-this.coordLeft, -this.coordTop);
356 }
357 this.pushElementTransform(this.canvasElement);
358 this.canvasElement.draw(this.context_);
359 this.popElementTransform();
360 }
361};

364/**
365 * Draw an element, including any stroke or fill.
366 * @param {goog.graphics.Element} element The element to draw.
367 */
368goog.graphics.CanvasGraphics.prototype.drawElement = function(element) {
369 if (element instanceof goog.graphics.CanvasTextElement) {
370 // Don't draw text since that is not implemented using canvas.
371 return;
372 }
373
374 var ctx = this.getContext();
375 this.pushElementTransform(element);
376
377 if (!element.getFill || !element.getStroke) {
378 // Draw without stroke or fill (e.g. the element is an image or group).
379 element.draw(ctx);
380 this.popElementTransform();
381 return;
382 }
383
384 var fill = element.getFill();
385 if (fill) {
386 if (fill instanceof goog.graphics.SolidFill) {
387 if (fill.getOpacity() != 0) {
388 ctx.globalAlpha = fill.getOpacity();
389 ctx.fillStyle = fill.getColor();
390 element.draw(ctx);
391 ctx.fill();
392 ctx.globalAlpha = 1;
393 }
394 } else { // (fill instanceof goog.graphics.LinearGradient)
395 var linearGradient = ctx.createLinearGradient(fill.getX1(), fill.getY1(),
396 fill.getX2(), fill.getY2());
397 linearGradient.addColorStop(0.0, fill.getColor1());
398 linearGradient.addColorStop(1.0, fill.getColor2());
399
400 ctx.fillStyle = linearGradient;
401 element.draw(ctx);
402 ctx.fill();
403 }
404 }
405
406 var stroke = element.getStroke();
407 if (stroke) {
408 element.draw(ctx);
409 ctx.strokeStyle = stroke.getColor();
410
411 var width = stroke.getWidth();
412 if (goog.isString(width) && width.indexOf('px') != -1) {
413 width = parseFloat(width) / this.getPixelScaleX();
414 }
415 ctx.lineWidth = width;
416
417 ctx.stroke();
418 }
419
420 this.popElementTransform();
421};
