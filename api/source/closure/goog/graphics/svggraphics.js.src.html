45/**
46 * A Graphics implementation for drawing using SVG.
47 * @param {string|number} width The width in pixels. Strings
48 * expressing percentages of parent with (e.g. '80%') are also accepted.
49 * @param {string|number} height The height in pixels. Strings
50 * expressing percentages of parent with (e.g. '80%') are also accepted.
51 * @param {?number=} opt_coordWidth The coordinate width - if
52 * omitted or null, defaults to same as width.
53 * @param {?number=} opt_coordHeight The coordinate height - if
54 * omitted or null, defaults to same as height.
55 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper object for the
56 * document we want to render in.
57 * @constructor
58 * @extends {goog.graphics.AbstractGraphics}
59 * @deprecated goog.graphics is deprecated. It existed to abstract over browser
60 * differences before the canvas tag was widely supported. See
61 * http://en.wikipedia.org/wiki/Canvas_element for details.
62 * @final
63 */
64goog.graphics.SvgGraphics = function(width, height,
65 opt_coordWidth, opt_coordHeight,
66 opt_domHelper) {
67 goog.graphics.AbstractGraphics.call(this, width, height,
68 opt_coordWidth, opt_coordHeight,
69 opt_domHelper);
70
71 /**
72 * Map from def key to id of def root element.
73 * Defs are global "defines" of svg that are used to share common attributes,
74 * for example gradients.
75 * @type {Object}
76 * @private
77 */
78 this.defs_ = {};
79
80 /**
81 * Whether to manually implement viewBox by using a coordinate transform.
82 * As of 1/11/08 this is necessary for Safari 3 but not for the nightly
83 * WebKit build. Apply to webkit versions < 526. 525 is the
84 * last version used by Safari 3.1.
85 * @type {boolean}
86 * @private
87 */
88 this.useManualViewbox_ = goog.userAgent.WEBKIT &&
89 !goog.userAgent.isVersionOrHigher(526);
90
91 /**
92 * Event handler.
93 * @type {goog.events.EventHandler<!goog.graphics.SvgGraphics>}
94 * @private
95 */
96 this.handler_ = new goog.events.EventHandler(this);
97};

183/**
184 * Sets the fill of the given element.
185 * @param {goog.graphics.StrokeAndFillElement} element The element wrapper.
186 * @param {goog.graphics.Fill?} fill The fill object.
187 * @override
188 */
189goog.graphics.SvgGraphics.prototype.setElementFill = function(element, fill) {
190 var svgElement = element.getElement();
191 if (fill instanceof goog.graphics.SolidFill) {
192 svgElement.setAttribute('fill', fill.getColor());
193 svgElement.setAttribute('fill-opacity', fill.getOpacity());
194 } else if (fill instanceof goog.graphics.LinearGradient) {
195 // create a def key which is just a concat of all the relevant fields
196 var defKey = 'lg-' +
197 fill.getX1() + '-' + fill.getY1() + '-' +
198 fill.getX2() + '-' + fill.getY2() + '-' +
199 fill.getColor1() + '-' + fill.getColor2();
200 // It seems that the SVG version accepts opacity where the VML does not
201
202 var id = this.getDef(defKey);
203
204 if (!id) { // No def for this yet, create it
205 // Create the gradient def entry (only linear gradient are supported)
206 var gradient = this.createSvgElement_('linearGradient', {
207 'x1': fill.getX1(),
208 'y1': fill.getY1(),
209 'x2': fill.getX2(),
210 'y2': fill.getY2(),
211 'gradientUnits': 'userSpaceOnUse'
212 });
213
214 var gstyle = 'stop-color:' + fill.getColor1();
215 if (goog.isNumber(fill.getOpacity1())) {
216 gstyle += ';stop-opacity:' + fill.getOpacity1();
217 }
218 var stop1 = this.createSvgElement_(
219 'stop', {'offset': '0%', 'style': gstyle});
220 gradient.appendChild(stop1);
221
222 // LinearGradients don't have opacity in VML so implement that before
223 // enabling the following code.
224 // if (fill.getOpacity() != null) {
225 // gstyles += 'opacity:' + fill.getOpacity() + ';'
226 // }
227 gstyle = 'stop-color:' + fill.getColor2();
228 if (goog.isNumber(fill.getOpacity2())) {
229 gstyle += ';stop-opacity:' + fill.getOpacity2();
230 }
231 var stop2 = this.createSvgElement_(
232 'stop', {'offset': '100%', 'style': gstyle});
233 gradient.appendChild(stop2);
234
235 // LinearGradients don't have opacity in VML so implement that before
236 // enabling the following code.
237 // if (fill.getOpacity() != null) {
238 // gstyles += 'opacity:' + fill.getOpacity() + ';'
239 // }
240
241 id = this.addDef(defKey, gradient);
242 }
243
244 // Link element to linearGradient definition
245 svgElement.setAttribute('fill', 'url(#' + id + ')');
246 } else {
247 svgElement.setAttribute('fill', 'none');
248 }
249};

252/**
253 * Sets the stroke of the given element.
254 * @param {goog.graphics.StrokeAndFillElement} element The element wrapper.
255 * @param {goog.graphics.Stroke?} stroke The stroke object.
256 * @override
257 */
258goog.graphics.SvgGraphics.prototype.setElementStroke = function(element,
259 stroke) {
260 var svgElement = element.getElement();
261 if (stroke) {
262 svgElement.setAttribute('stroke', stroke.getColor());
263 svgElement.setAttribute('stroke-opacity', stroke.getOpacity());
264
265 var width = stroke.getWidth();
266 if (goog.isString(width) && width.indexOf('px') != -1) {
267 svgElement.setAttribute('stroke-width',
268 parseFloat(width) / this.getPixelScaleX());
269 } else {
270 svgElement.setAttribute('stroke-width', width);
271 }
272 } else {
273 svgElement.setAttribute('stroke', 'none');
274 }

278/**
279 * Set the translation and rotation of an element.
280 *
281 * If a more general affine transform is needed than this provides
282 * (e.g. skew and scale) then use setElementAffineTransform.
283 * @param {goog.graphics.Element} element The element wrapper.
284 * @param {number} x The x coordinate of the translation transform.
285 * @param {number} y The y coordinate of the translation transform.
286 * @param {number} angle The angle of the rotation transform.
287 * @param {number} centerX The horizontal center of the rotation transform.
288 * @param {number} centerY The vertical center of the rotation transform.
289 * @override
290 */
291goog.graphics.SvgGraphics.prototype.setElementTransform = function(element, x,
292 y, angle, centerX, centerY) {
293 element.getElement().setAttribute('transform', 'translate(' + x + ',' + y +
294 ') rotate(' + angle + ' ' + centerX + ' ' + centerY + ')');
295};

298/**
299 * Set the transformation of an element.
300 * @param {goog.graphics.Element} element The element wrapper.
301 * @param {!goog.graphics.AffineTransform} affineTransform The
302 * transformation applied to this element.
303 * @override
304 */
305goog.graphics.SvgGraphics.prototype.setElementAffineTransform = function(
306 element, affineTransform) {
307 var t = affineTransform;
308 var substr = [t.getScaleX(), t.getShearY(), t.getShearX(), t.getScaleY(),
309 t.getTranslateX(), t.getTranslateY()].join(',');
310 element.getElement().setAttribute('transform', 'matrix(' + substr + ')');

382/**
383 * Sets up the view box.
384 * @private
385 */
386goog.graphics.SvgGraphics.prototype.setViewBox_ = function() {
387 if (this.coordWidth || this.coordLeft || this.coordTop) {
388 this.getElement().setAttribute('preserveAspectRatio', 'none');
389 if (this.useManualViewbox_) {
390 this.updateManualViewBox_();
391 } else {
392 this.getElement().setAttribute('viewBox', this.getViewBox_());
393 }

398/**
399 * Updates the transform of the root element to fake a viewBox. Should only
400 * be called when useManualViewbox_ is set.
401 * @private
402 */
403goog.graphics.SvgGraphics.prototype.updateManualViewBox_ = function() {
404 if (!this.isInDocument() ||
405 !(this.coordWidth || this.coordLeft || !this.coordTop)) {
406 return;
407 }
408
409 var size = this.getPixelSize();
410 if (size.width == 0) {
411 // In Safari, invisible SVG is sometimes shown. Explicitly hide it.
412 this.getElement().style.visibility = 'hidden';
413 return;
414 }
415
416 this.getElement().style.visibility = '';
417
418 var offsetX = - this.coordLeft;
419 var offsetY = - this.coordTop;
420 var scaleX = size.width / this.coordWidth;
421 var scaleY = size.height / this.coordHeight;
422
423 this.canvasElement.getElement().setAttribute('transform',
424 'scale(' + scaleX + ' ' + scaleY + ') ' +
425 'translate(' + offsetX + ' ' + offsetY + ')');
426};

441/** @override */
442goog.graphics.SvgGraphics.prototype.getPixelSize = function() {
443 if (!goog.userAgent.GECKO) {
444 return this.isInDocument() ?
445 goog.style.getSize(this.getElement()) :
446 goog.graphics.SvgGraphics.base(this, 'getPixelSize');
447 }
448
449 // In Gecko, goog.style.getSize does not work for SVG elements. We have to
450 // compute the size manually if it is percentage based.
451 var width = this.width;
452 var height = this.height;
453 var computeWidth = goog.isString(width) && width.indexOf('%') != -1;
454 var computeHeight = goog.isString(height) && height.indexOf('%') != -1;
455
456 if (!this.isInDocument() && (computeWidth || computeHeight)) {
457 return null;
458 }
459
460 var parent;
461 var parentSize;
462
463 if (computeWidth) {
464 parent = /** @type {Element} */ (this.getElement().parentNode);
465 parentSize = goog.style.getSize(parent);
466 width = parseFloat(/** @type {string} */ (width)) * parentSize.width / 100;
467 }
468
469 if (computeHeight) {
470 parent = parent || /** @type {Element} */ (this.getElement().parentNode);
471 parentSize = parentSize || goog.style.getSize(parent);
472 height = parseFloat(/** @type {string} */ (height)) * parentSize.height /
473 100;
474 }
475
476 return new goog.math.Size(/** @type {number} */ (width),
477 /** @type {number} */ (height));
478};

576/**
577 * Draw a text string vertically centered on a given line.
578 *
579 * @param {string} text The text to draw.
580 * @param {number} x1 X coordinate of start of line.
581 * @param {number} y1 Y coordinate of start of line.
582 * @param {number} x2 X coordinate of end of line.
583 * @param {number} y2 Y coordinate of end of line.
584 * @param {string} align Horizontal alignment: left (default), center, right.
585 * @param {goog.graphics.Font} font Font describing the font properties.
586 * @param {goog.graphics.Stroke?} stroke Stroke object describing the
587 * stroke.
588 * @param {goog.graphics.Fill?} fill Fill object describing the fill.
589 * @param {goog.graphics.GroupElement=} opt_group The group wrapper element
590 * to append to. If not specified, appends to the main canvas.
591 *
592 * @return {!goog.graphics.TextElement} The newly created element.
593 * @override
594 */
595goog.graphics.SvgGraphics.prototype.drawTextOnLine = function(
596 text, x1, y1, x2, y2, align, font, stroke, fill, opt_group) {
597 var angle = Math.round(goog.math.angle(x1, y1, x2, y2));
598 var dx = x2 - x1;
599 var dy = y2 - y1;
600 var lineLength = Math.round(Math.sqrt(dx * dx + dy * dy)); // Length of line
601
602 // SVG baseline is on the glyph's base line. We estimate it as 85% of the
603 // font height. This is just a rough estimate, but do not have a better way.
604 var fontSize = font.size;
605 var attributes = {'font-family': font.family, 'font-size': fontSize};
606 var baseline = Math.round(fontSize * 0.85);
607 var textY = Math.round(y1 - (fontSize / 2) + baseline);
608 var textX = x1;
609 if (align == 'center') {
610 textX += Math.round(lineLength / 2);
611 attributes['text-anchor'] = 'middle';
612 } else if (align == 'right') {
613 textX += lineLength;
614 attributes['text-anchor'] = 'end';
615 }
616 attributes['x'] = textX;
617 attributes['y'] = textY;
618 if (font.bold) {
619 attributes['font-weight'] = 'bold';
620 }
621 if (font.italic) {
622 attributes['font-style'] = 'italic';
623 }
624 if (angle != 0) {
625 attributes['transform'] = 'rotate(' + angle + ' ' + x1 + ' ' + y1 + ')';
626 }
627
628 var element = this.createSvgElement_('text', attributes);
629 element.appendChild(this.dom_.getDocument().createTextNode(text));
630
631 // Bypass a Firefox-Mac bug where text fill is ignored. If text has no stroke,
632 // set a stroke, otherwise the text will not be visible.
633 if (stroke == null && goog.userAgent.GECKO && goog.userAgent.MAC) {
634 var color = 'black';
635 // For solid fills, use the fill color
636 if (fill instanceof goog.graphics.SolidFill) {
637 color = fill.getColor();
638 }
639 stroke = new goog.graphics.Stroke(1, color);
640 }
641
642 var wrapper = new goog.graphics.SvgTextElement(element, this, stroke, fill);
643 this.append_(wrapper, opt_group);
644 return wrapper;
645};

672/**
673 * Returns a string representation of a logical path suitable for use in
674 * an SVG element.
675 *
676 * @param {goog.graphics.Path} path The logical path.
677 * @return {string} The SVG path representation.
678 * @suppress {deprecated} goog.graphics is deprecated.
679 */
680goog.graphics.SvgGraphics.getSvgPath = function(path) {
681 var list = [];
682 path.forEachSegment(function(segment, args) {
683 switch (segment) {
684 case goog.graphics.Path.Segment.MOVETO:
685 list.push('M');
686 Array.prototype.push.apply(list, args);
687 break;
688 case goog.graphics.Path.Segment.LINETO:
689 list.push('L');
690 Array.prototype.push.apply(list, args);
691 break;
692 case goog.graphics.Path.Segment.CURVETO:
693 list.push('C');
694 Array.prototype.push.apply(list, args);
695 break;
696 case goog.graphics.Path.Segment.ARCTO:
697 var extent = args[3];
698 list.push('A', args[0], args[1],
699 0, Math.abs(extent) > 180 ? 1 : 0, extent > 0 ? 1 : 0,
700 args[4], args[5]);
701 break;
702 case goog.graphics.Path.Segment.CLOSE:
703 list.push('Z');
704 break;
705 }
706 });
707 return list.join(' ');
708};
