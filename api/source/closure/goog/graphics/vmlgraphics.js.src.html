50/**
51 * A Graphics implementation for drawing using VML.
52 * @param {string|number} width The (non-zero) width in pixels. Strings
53 * expressing percentages of parent with (e.g. '80%') are also accepted.
54 * @param {string|number} height The (non-zero) height in pixels. Strings
55 * expressing percentages of parent with (e.g. '80%') are also accepted.
56 * @param {?number=} opt_coordWidth The coordinate width - if
57 * omitted or null, defaults to same as width.
58 * @param {?number=} opt_coordHeight The coordinate height - if
59 * omitted or null, defaults to same as height.
60 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper object for the
61 * document we want to render in.
62 * @constructor
63 * @extends {goog.graphics.AbstractGraphics}
64 * @deprecated goog.graphics is deprecated. It existed to abstract over browser
65 * differences before the canvas tag was widely supported. See
66 * http://en.wikipedia.org/wiki/Canvas_element for details.
67 * @final
68 */
69goog.graphics.VmlGraphics = function(width, height,
70 opt_coordWidth, opt_coordHeight,
71 opt_domHelper) {
72 goog.graphics.AbstractGraphics.call(this, width, height,
73 opt_coordWidth, opt_coordHeight,
74 opt_domHelper);
75 this.handler_ = new goog.events.EventHandler(this);
76 this.registerDisposable(this.handler_);
77};
78goog.inherits(goog.graphics.VmlGraphics, goog.graphics.AbstractGraphics);

138 * This function is internal for the VML supporting classes, and

247/**
248 * Resets the graphics so they will display properly on IE8. Noop in older
249 * versions.
250 * @private
251 */
252goog.graphics.VmlGraphics.prototype.updateGraphics_ = function() {
253 if (goog.graphics.VmlGraphics.IE8_MODE_ && this.isInDocument()) {
254 // There's a risk of mXSS here, as the browser is not guaranteed to
255 // return the HTML that was originally written, when innerHTML is read.
256 // However, given that this a deprecated API and affects only IE, it seems
257 // an acceptable risk.
258 var html = goog.html.uncheckedconversions
259 .safeHtmlFromStringKnownToSatisfyTypeContract(
260 goog.string.Const.from('Assign innerHTML to itself'),
261 this.getElement().innerHTML);
262 goog.dom.safe.setInnerHtml(
263 /** @type {!Element} */ (this.getElement()), html);
264 }
265};

283/**
284 * Sets the fill for the given element.
285 * @param {goog.graphics.StrokeAndFillElement} element The element wrapper.
286 * @param {goog.graphics.Fill?} fill The fill object.
287 * @override
288 */
289goog.graphics.VmlGraphics.prototype.setElementFill = function(element, fill) {
290 var vmlElement = element.getElement();
291 goog.graphics.VmlGraphics.removeFill_(vmlElement);
292 if (fill instanceof goog.graphics.SolidFill) {
293 // NOTE(arv): VML does not understand 'transparent' so hard code support
294 // for it.
295 if (fill.getColor() == 'transparent') {
296 vmlElement.filled = false;
297 } else if (fill.getOpacity() != 1) {
298 vmlElement.filled = true;
299 // Set opacity (number 0-1 is translated to percent)
300 var fillNode = this.createVmlElement('fill');
301 fillNode.opacity = Math.round(fill.getOpacity() * 100) + '%';
302 fillNode.color = fill.getColor();
303 vmlElement.appendChild(fillNode);
304 } else {
305 vmlElement.filled = true;
306 vmlElement.fillcolor = fill.getColor();
307 }
308 } else if (fill instanceof goog.graphics.LinearGradient) {
309 vmlElement.filled = true;
310 // Add a 'fill' element
311 var gradient = this.createVmlElement('fill');
312 gradient.color = fill.getColor1();
313 gradient.color2 = fill.getColor2();
314 if (goog.isNumber(fill.getOpacity1())) {
315 gradient.opacity = fill.getOpacity1();
316 }
317 if (goog.isNumber(fill.getOpacity2())) {
318 gradient.opacity2 = fill.getOpacity2();
319 }
320 var angle = goog.math.angle(fill.getX1(), fill.getY1(),
321 fill.getX2(), fill.getY2());
322 // Our angles start from 0 to the right, and grow clockwise.
323 // MSIE starts from 0 to top, and grows anti-clockwise.
324 angle = Math.round(goog.math.standardAngle(270 - angle));
325 gradient.angle = angle;
326 gradient.type = 'gradient';
327 vmlElement.appendChild(gradient);
328 } else {
329 vmlElement.filled = false;
330 }
331 this.updateGraphics_();
332};

335/**
336 * Sets the stroke for the given element.
337 * @param {goog.graphics.StrokeAndFillElement} element The element wrapper.
338 * @param {goog.graphics.Stroke?} stroke The stroke object.
339 * @override
340 */
341goog.graphics.VmlGraphics.prototype.setElementStroke = function(element,
342 stroke) {
343 var vmlElement = element.getElement();
344 if (stroke) {
345 vmlElement.stroked = true;
346
347 var width = stroke.getWidth();
348 if (goog.isString(width) && width.indexOf('px') == -1) {
349 width = parseFloat(width);
350 } else {
351 width = width * this.getPixelScaleX();
352 }
353
354 var strokeElement = vmlElement.getElementsByTagName('stroke')[0];
355 if (!strokeElement) {
356 strokeElement = strokeElement || this.createVmlElement('stroke');
357 vmlElement.appendChild(strokeElement);
358 }
359 strokeElement.opacity = stroke.getOpacity();
360 strokeElement.weight = width + 'px';
361 strokeElement.color = stroke.getColor();
362 } else {
363 vmlElement.stroked = false;
364 }
365 this.updateGraphics_();
366};

369/**
370 * Set the translation and rotation of an element.
371 *
372 * If a more general affine transform is needed than this provides
373 * (e.g. skew and scale) then use setElementAffineTransform.
374 * @param {number} x The x coordinate of the translation transform.
375 * @param {number} y The y coordinate of the translation transform.
376 * @param {number} angle The angle of the rotation transform.
377 * @param {number} centerX The horizontal center of the rotation transform.
378 * @param {number} centerY The vertical center of the rotation transform.
379 * @override
380 */
381goog.graphics.VmlGraphics.prototype.setElementTransform = function(element, x,
382 y, angle, centerX, centerY) {
383 var el = element.getElement();
384
385 el.style.left = goog.graphics.VmlGraphics.toPosPx(x);
386 el.style.top = goog.graphics.VmlGraphics.toPosPx(y);
387 if (angle || el.rotation) {
388 el.rotation = angle;
389 el.coordsize = goog.graphics.VmlGraphics.toSizeCoord(centerX * 2) + ' ' +
390 goog.graphics.VmlGraphics.toSizeCoord(centerY * 2);
391 }
392};

395/**
396 * Set the transformation of an element.
397 * @param {!goog.graphics.Element} element The element wrapper.
398 * @param {!goog.graphics.AffineTransform} affineTransform The
399 * transformation applied to this element.
400 * @override
401 */
402goog.graphics.VmlGraphics.prototype.setElementAffineTransform = function(
403 element, affineTransform) {
404 var t = affineTransform;
405 var vmlElement = element.getElement();
406 goog.graphics.VmlGraphics.removeSkew_(vmlElement);
407 var skewNode = this.createVmlElement('skew');
408 skewNode.on = 'true';
409 // Move the transform origin to 0px,0px of the graphics.
410 // In VML, 0,0 means the center of the element, -0.5,-0.5 left top conner of
411 // it.
412 skewNode.origin =
413 (-vmlElement.style.pixelLeft / vmlElement.style.pixelWidth - 0.5) + ',' +
414 (-vmlElement.style.pixelTop / vmlElement.style.pixelHeight - 0.5);
415 skewNode.offset = t.getTranslateX().toFixed(1) + 'px,' +
416 t.getTranslateY().toFixed(1) + 'px';
417 skewNode.matrix = [t.getScaleX().toFixed(6), t.getShearX().toFixed(6),
418 t.getShearY().toFixed(6), t.getScaleY().toFixed(6),
419 0, 0].join(',');
420 vmlElement.appendChild(skewNode);
421 this.updateGraphics_();
422};

511/**
512 * Creates the DOM representation of the graphics area.
513 * @override
514 */
515goog.graphics.VmlGraphics.prototype.createDom = function() {
516 var doc = this.dom_.getDocument();
517
518 // Add the namespace.
519 if (!doc.namespaces[goog.graphics.VmlGraphics.VML_PREFIX_]) {
520 if (goog.graphics.VmlGraphics.IE8_MODE_) {
521 doc.namespaces.add(goog.graphics.VmlGraphics.VML_PREFIX_,
522 goog.graphics.VmlGraphics.VML_NS_,
523 goog.graphics.VmlGraphics.VML_IMPORT_);
524 } else {
525 doc.namespaces.add(goog.graphics.VmlGraphics.VML_PREFIX_,
526 goog.graphics.VmlGraphics.VML_NS_);
527 }
528
529 // We assume that we only need to add the CSS if the namespace was not
530 // present
531 var ss = doc.createStyleSheet();
532 ss.cssText = goog.graphics.VmlGraphics.VML_PREFIX_ + '\\:*' +
533 '{behavior:url(#default#VML)}';
534 }
535
536 // Outer a DIV with overflow hidden for clipping.
537 // All inner elements are absolutly positioned on-top of this div.
538 var pixelWidth = this.width;
539 var pixelHeight = this.height;
540 var divElement = this.dom_.createDom(goog.dom.TagName.DIV, {
541 'style': 'overflow:hidden;position:relative;width:' +
542 goog.graphics.VmlGraphics.toCssSize(pixelWidth) + ';height:' +
543 goog.graphics.VmlGraphics.toCssSize(pixelHeight)
544 });
545
546 this.setElementInternal(divElement);
547
548 var group = this.createVmlElement('group');
549 var style = group.style;
550
551 style.position = 'absolute';
552 style.left = style.top = 0;
553 style.width = this.width;
554 style.height = this.height;
555 if (this.coordWidth) {
556 group.coordsize =
557 goog.graphics.VmlGraphics.toSizeCoord(this.coordWidth) + ' ' +
558 goog.graphics.VmlGraphics.toSizeCoord(
559 /** @type {number} */ (this.coordHeight));
560 } else {
561 group.coordsize = goog.graphics.VmlGraphics.toSizeCoord(pixelWidth) + ' ' +
562 goog.graphics.VmlGraphics.toSizeCoord(pixelHeight);
563 }
564
565 if (goog.isDef(this.coordLeft)) {
566 group.coordorigin = goog.graphics.VmlGraphics.toSizeCoord(this.coordLeft) +
567 ' ' + goog.graphics.VmlGraphics.toSizeCoord(this.coordTop);
568 } else {
569 group.coordorigin = '0 0';
570 }
571 divElement.appendChild(group);
572
573 this.canvasElement = new goog.graphics.VmlGroupElement(group, this);
574
575 goog.events.listen(divElement, goog.events.EventType.RESIZE, goog.bind(
576 this.handleContainerResize_, this));
577};

580/**
581 * Changes the canvas element size to match the container element size.
582 * @private
583 */
584goog.graphics.VmlGraphics.prototype.handleContainerResize_ = function() {
585 var size = goog.style.getSize(this.getElement());
586 var style = this.canvasElement.getElement().style;
587
588 if (size.width) {
589 style.width = size.width + 'px';
590 style.height = size.height + 'px';
591 } else {
592 var current = this.getElement();
593 while (current && current.currentStyle &&
594 current.currentStyle.display != 'none') {
595 current = current.parentNode;
596 }
597 if (current && current.currentStyle) {
598 this.handler_.listen(current, 'propertychange',
599 this.handleContainerResize_);
600 }
601 }
602
603 this.dispatchEvent(goog.events.EventType.RESIZE);
604};

607/**
608 * Handle property changes on hidden ancestors.
609 * @param {goog.events.BrowserEvent} e The browser event.
610 * @private

612goog.graphics.VmlGraphics.prototype.handlePropertyChange_ = function(e) {

614 if (prop == 'display' || prop == 'className') {

616 'propertychange', this.handlePropertyChange_);

844/**
845 * Returns a string representation of a logical path suitable for use in
846 * a VML element.
847 *
848 * @param {goog.graphics.Path} path The logical path.
849 * @return {string} The VML path representation.
850 * @suppress {deprecated} goog.graphics is deprecated.
851 */
852goog.graphics.VmlGraphics.getVmlPath = function(path) {
853 var list = [];
854 path.forEachSegment(function(segment, args) {
855 switch (segment) {
856 case goog.graphics.Path.Segment.MOVETO:
857 list.push('m');
858 Array.prototype.push.apply(list, goog.array.map(args,
859 goog.graphics.VmlGraphics.toSizeCoord));
860 break;
861 case goog.graphics.Path.Segment.LINETO:
862 list.push('l');
863 Array.prototype.push.apply(list, goog.array.map(args,
864 goog.graphics.VmlGraphics.toSizeCoord));
865 break;
866 case goog.graphics.Path.Segment.CURVETO:
867 list.push('c');
868 Array.prototype.push.apply(list, goog.array.map(args,
869 goog.graphics.VmlGraphics.toSizeCoord));
870 break;
871 case goog.graphics.Path.Segment.CLOSE:
872 list.push('x');
873 break;
874 case goog.graphics.Path.Segment.ARCTO:
875 var toAngle = args[2] + args[3];
876 var cx = goog.graphics.VmlGraphics.toSizeCoord(
877 args[4] - goog.math.angleDx(toAngle, args[0]));
878 var cy = goog.graphics.VmlGraphics.toSizeCoord(
879 args[5] - goog.math.angleDy(toAngle, args[1]));
880 var rx = goog.graphics.VmlGraphics.toSizeCoord(args[0]);
881 var ry = goog.graphics.VmlGraphics.toSizeCoord(args[1]);
882 // VML angles are in fd units (see http://www.w3.org/TR/NOTE-VML) and
883 // are positive counter-clockwise.
884 var fromAngle = Math.round(args[2] * -65536);
885 var extent = Math.round(args[3] * -65536);
886 list.push('ae', cx, cy, rx, ry, fromAngle, extent);
887 break;
888 }
889 });
890 return list.join(' ');
891};
