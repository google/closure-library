15/**
16 * @fileoverview Implementation of sprintf-like, python-%-operator-like,
17 * .NET-String.Format-like functionality. Uses JS string's replace method to
18 * extract format specifiers and sends those specifiers to a handler function,
19 * which then, based on conversion type part of the specifier, calls the
20 * appropriate function to handle the specific conversion.
21 * For specific functionality implemented, look at formatRe below, or look
22 * at the tests.
23 */
24
25goog.provide('goog.string.format');
26
27goog.require('goog.string');

30/**
31 * Performs sprintf-like conversion, ie. puts the values in a template.
32 * DO NOT use it instead of built-in conversions in simple cases such as
33 * 'Cost: %.2f' as it would introduce unneccessary latency oposed to
34 * 'Cost: ' + cost.toFixed(2).
35 * @param {string} formatString Template string containing % specifiers.
36 * @param {...string|number} var_args Values formatString is to be filled with.
37 * @return {string} Formatted string.
38 */
39goog.string.format = function(formatString, var_args) {
40
41 // Convert the arguments to an array (MDC recommended way).
42 var args = Array.prototype.slice.call(arguments);
43
44 // Try to get the template.
45 var template = args.shift();
46 if (typeof template == 'undefined') {
47 throw Error('[goog.string.format] Template required');
48 }
49
50 // This re is used for matching, it also defines what is supported.
51 var formatRe = /%([0\-\ \+]*)(\d+)?(\.(\d+))?([%sfdiu])/g;
52
53 /**
54 * Chooses which conversion function to call based on type conversion
55 * specifier.
56 * @param {string} match Contains the re matched string.
57 * @param {string} flags Formatting flags.
58 * @param {string} width Replacement string minimum width.
59 * @param {string} dotp Matched precision including a dot.
60 * @param {string} precision Specifies floating point precision.
61 * @param {string} type Type conversion specifier.
62 * @param {string} offset Matching location in the original string.
63 * @param {string} wholeString Has the actualString being searched.
64 * @return {string} Formatted parameter.
65 */
66 function replacerDemuxer(match,
67 flags,
68 width,
69 dotp,
70 precision,
71 type,
72 offset,
73 wholeString) {
74
75 // The % is too simple and doesn't take an argument.
76 if (type == '%') {
77 return '%';
78 }
79
80 // Try to get the actual value from parent function.
81 var value = args.shift();
82
83 // If we didn't get any arguments, fail.
84 if (typeof value == 'undefined') {
85 throw Error('[goog.string.format] Not enough arguments');
86 }
87
88 // Patch the value argument to the beginning of our type specific call.
89 arguments[0] = value;
90
91 return goog.string.format.demuxes_[type].apply(null, arguments);
92
93 }
94
95 return template.replace(formatRe, replacerDemuxer);
96};

100 * Contains various conversion functions (to be filled in later on).

102 * @private

104goog.string.format.demuxes_ = {};

107/**
108 * Processes %s conversion specifier.
109 * @param {string} value Contains the formatRe matched string.
110 * @param {string} flags Formatting flags.
111 * @param {string} width Replacement string minimum width.
112 * @param {string} dotp Matched precision including a dot.
113 * @param {string} precision Specifies floating point precision.
114 * @param {string} type Type conversion specifier.
115 * @param {string} offset Matching location in the original string.
116 * @param {string} wholeString Has the actualString being searched.
117 * @return {string} Replacement string.
118 */
119goog.string.format.demuxes_['s'] = function(value,
120 flags,
121 width,
122 dotp,
123 precision,
124 type,
125 offset,
126 wholeString) {
127 var replacement = value;
128 // If no padding is necessary we're done.
129 // The check for '' is necessary because Firefox incorrectly provides the
130 // empty string instead of undefined for non-participating capture groups,
131 // and isNaN('') == false.
132 if (isNaN(width) || width == '' || replacement.length >= width) {
133 return replacement;
134 }
135
136 // Otherwise we should find out where to put spaces.
137 if (flags.indexOf('-', 0) > -1) {
138 replacement =
139 replacement + goog.string.repeat(' ', width - replacement.length);
140 } else {
141 replacement =
142 goog.string.repeat(' ', width - replacement.length) + replacement;
143 }
144 return replacement;
145};

148/**
149 * Processes %f conversion specifier.
150 * @param {string} value Contains the formatRe matched string.
151 * @param {string} flags Formatting flags.
152 * @param {string} width Replacement string minimum width.
153 * @param {string} dotp Matched precision including a dot.
154 * @param {string} precision Specifies floating point precision.
155 * @param {string} type Type conversion specifier.
156 * @param {string} offset Matching location in the original string.
157 * @param {string} wholeString Has the actualString being searched.
158 * @return {string} Replacement string.
159 */
160goog.string.format.demuxes_['f'] = function(value,
161 flags,
162 width,
163 dotp,
164 precision,
165 type,
166 offset,
167 wholeString) {
168
169 var replacement = value.toString();
170
171 // The check for '' is necessary because Firefox incorrectly provides the
172 // empty string instead of undefined for non-participating capture groups,
173 // and isNaN('') == false.
174 if (!(isNaN(precision) || precision == '')) {
175 replacement = parseFloat(value).toFixed(precision);
176 }
177
178 // Generates sign string that will be attached to the replacement.
179 var sign;
180 if (value < 0) {
181 sign = '-';
182 } else if (flags.indexOf('+') >= 0) {
183 sign = '+';
184 } else if (flags.indexOf(' ') >= 0) {
185 sign = ' ';
186 } else {
187 sign = '';
188 }
189
190 if (value >= 0) {
191 replacement = sign + replacement;
192 }
193
194 // If no padding is neccessary we're done.
195 if (isNaN(width) || replacement.length >= width) {
196 return replacement;
197 }
198
199 // We need a clean signless replacement to start with
200 replacement = isNaN(precision) ?
201 Math.abs(value).toString() :
202 Math.abs(value).toFixed(precision);
203
204 var padCount = width - replacement.length - sign.length;
205
206 // Find out which side to pad, and if it's left side, then which character to
207 // pad, and set the sign on the left and padding in the middle.
208 if (flags.indexOf('-', 0) >= 0) {
209 replacement = sign + replacement + goog.string.repeat(' ', padCount);
210 } else {
211 // Decides which character to pad.
212 var paddingChar = (flags.indexOf('0', 0) >= 0) ? '0' : ' ';
213 replacement =
214 sign + goog.string.repeat(paddingChar, padCount) + replacement;
215 }
216
217 return replacement;
218};

221/**
222 * Processes %d conversion specifier.
223 * @param {string} value Contains the formatRe matched string.
224 * @param {string} flags Formatting flags.
225 * @param {string} width Replacement string minimum width.
226 * @param {string} dotp Matched precision including a dot.
227 * @param {string} precision Specifies floating point precision.
228 * @param {string} type Type conversion specifier.
229 * @param {string} offset Matching location in the original string.
230 * @param {string} wholeString Has the actualString being searched.
231 * @return {string} Replacement string.
232 */
233goog.string.format.demuxes_['d'] = function(value,
234 flags,
235 width,
236 dotp,
237 precision,
238 type,
239 offset,
240 wholeString) {
241 return goog.string.format.demuxes_['f'](
242 parseInt(value, 10) /* value */,
243 flags, width, dotp, 0 /* precision */,
244 type, offset, wholeString);
245};
