15/**
16 * @fileoverview Definitions for all tweak entries.
17 * The class hierarchy is as follows (abstract entries are denoted with a *):
18 * BaseEntry(id, description) *
19 * -> ButtonAction(buttons in the UI)
20 * -> BaseSetting(query parameter) *
21 * -> BooleanGroup(child booleans)
22 * -> BasePrimitiveSetting(value, defaultValue) *
23 * -> BooleanSetting
24 * -> StringSetting
25 * -> NumericSetting
26 * -> BooleanInGroupSetting(token)
27 * Most clients should not use these classes directly, but instead use the API
28 * defined in tweak.js. One possible use case for directly using them is to
29 * register tweaks that are not known at compile time.
30 *
31 * @author agrieve@google.com (Andrew Grieve)
32 */
33

51/**
52 * Base class for all Registry entries.
53 * @param {string} id The ID for the entry. Must contain only letters,
54 * numbers, underscores and periods.
55 * @param {string} description A description of what the entry does.
56 * @constructor
57 */
58goog.tweak.BaseEntry = function(id, description) {
59 /**
60 * An ID to uniquely identify the entry.
61 * @type {string}
62 * @private
63 */
64 this.id_ = id;
65
66 /**
67 * A descriptive label for the entry.
68 * @type {string}
69 */
70 this.label = id;
71
72 /**
73 * A description of what this entry does.
74 * @type {string}
75 */
76 this.description = description;
77
78 /**
79 * Functions to be called whenever a setting is changed or a button is
80 * clicked.
81 * @type {!Array<!Function>}
82 * @private
83 */
84 this.callbacks_ = [];
85};

164 * Base class for all tweak entries that are settings. Settings are entries
165 * that are associated with a query parameter.

168 * @constructor
169 * @extends {goog.tweak.BaseEntry}

173 // Apply this restriction for settings since they turn in to query
174 // parameters. For buttons, it's not really important.

176 'Tweak id contains illegal characters: ', id);

179 * The value of this setting's query parameter.

181 * @protected
182 */
183 this.initialQueryParamValue;
184
185 /**
186 * The query parameter that controls this setting.
187 * @type {?string}
188 * @private

190 this.paramName_ = this.getId().toLowerCase();

195/**
196 * States of initialization. Entries are initialized lazily in order to allow
197 * their initialization to happen in multiple statements.
198 * @enum {number}
199 * @private
200 */
201goog.tweak.BaseSetting.InitializeState_ = {
202 // The start state for all settings.
203 NOT_INITIALIZED: 0,
204 // This is used to allow concrete classes to call assertNotInitialized()
205 // during their initialize() function.
206 INITIALIZING: 1,
207 // One a setting is initialized, it may no longer change its configuration
208 // settings (associated query parameter, token, etc).
209 INITIALIZED: 2
210};

306/**
307 * Applies the default value or query param value if this is the first time
308 * that the function has been called.
309 * @protected
310 */
311goog.tweak.BaseSetting.prototype.ensureInitialized = function() {
312 if (this.initializeState_ ==
313 goog.tweak.BaseSetting.InitializeState_.NOT_INITIALIZED) {
314 // Instead of having only initialized / not initialized, there is a
315 // separate in-between state so that functions that call
316 // assertNotInitialized() will not fail when called inside of the
317 // initialize().
318 this.initializeState_ =
319 goog.tweak.BaseSetting.InitializeState_.INITIALIZING;
320 var value = this.initialQueryParamValue == undefined ? null :
321 this.initialQueryParamValue;
322 this.initialize(value);
323 this.initializeState_ =
324 goog.tweak.BaseSetting.InitializeState_.INITIALIZED;
325 }

395 * Returns the value of the setting to use once "Apply Tweaks" is clicked.

398goog.tweak.BasePrimitiveSetting.prototype.getNewValue = function() {
399 this.ensureInitialized();
400 return this.newValue_;

404/**
405 * Sets the value of the setting. If the setting has the restartRequired
406 * option, then the value will not be changed until the "Apply Tweaks" button
407 * is clicked. If it does not have the option, the value will be update
408 * immediately and all registered callbacks will be called.
409 * @param {*} value The value.
410 */
411goog.tweak.BasePrimitiveSetting.prototype.setValue = function(value) {
412 this.ensureInitialized();
413 var changed = this.newValue_ != value;
414 this.newValue_ = value;
415 // Don't fire callbacks if we are currently in the initialize() method.
416 if (this.isInitializing()) {
417 this.value_ = value;
418 } else {
419 if (!this.isRestartRequired()) {
420 // Update the current value only if the tweak has been marked as not
421 // needing a restart.
422 this.value_ = value;
423 }
424 if (changed) {
425 this.fireCallbacks();
426 }
427 }
428};

535 * @param {Array<string>|undefined} values Valid values.
536 */
537goog.tweak.StringSetting.prototype.setValidValues = function(values) {
538 this.assertNotInitialized('setValidValues');
539 this.validValues_ = values;
540 // Set the default value to the first value in the list if the current
541 // default value is not within it.
542 if (values && !goog.array.contains(values, this.getDefaultValue())) {
543 this.setDefaultValue(values[0]);

548/**
549 * Returns the valid values for the setting.
550 * @return {Array<string>|undefined} Valid values.
551 */
552goog.tweak.StringSetting.prototype.getValidValues = function() {
553 return this.validValues_;
554};

556
557/**
558 * @override
559 */
560goog.tweak.StringSetting.prototype.initialize = function(value) {
561 if (value == null) {
562 this.setValue(this.getDefaultValue());
563 } else {
564 var validValues = this.validValues_;
565 if (validValues) {
566 // Make the query parameter values case-insensitive since users might
567 // type them by hand. Make the capitalization that is actual used come
568 // from the list of valid values.
569 value = value.toLowerCase();
570 for (var i = 0, il = validValues.length; i < il; ++i) {
571 if (value == validValues[i].toLowerCase()) {
572 this.setValue(validValues[i]);
573 return;
574 }
575 }
576 // Warn if the value is not in the list of allowed values.
577 goog.log.warning(this.logger, 'Tweak ' + this.getId() +
578 ' has value outside of expected range:' + value);
579 }
580 this.setValue(value);
581 }
582};

683/**
684 * @override
685 */
686goog.tweak.NumericSetting.prototype.initialize = function(value) {
687 if (value == null) {
688 this.setValue(this.getDefaultValue());
689 } else {
690 var coercedValue = +value;
691 // Warn if the value is not in the list of allowed values.
692 if (this.validValues_ &&
693 !goog.array.contains(this.validValues_, coercedValue)) {
694 goog.log.warning(this.logger, 'Tweak ' + this.getId() +
695 ' has value outside of expected range: ' + value);
696 }
697
698 if (isNaN(coercedValue)) {
699 goog.log.warning(this.logger, 'Tweak ' + this.getId() +
700 ' has value of NaN, resetting to ' + this.getDefaultValue());
701 this.setValue(this.getDefaultValue());
702 } else {
703 this.setValue(coercedValue);
704 }
705 }
