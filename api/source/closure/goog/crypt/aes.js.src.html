29/**
30 * Implementation of AES in JavaScript.
31 * See http://en.wikipedia.org/wiki/Advanced_Encryption_Standard
32 *
33 * WARNING: This is ECB mode only. If you are encrypting something
34 * longer than 16 bytes, or encrypting more than one value with the same key
35 * (so basically, always) you need to use this with a block cipher mode of
36 * operation. See goog.crypt.Cbc.
37 *
38 * See http://en.wikipedia.org/wiki/Block_cipher_modes_of_operation for more
39 * information.
40 *
41 * @constructor
42 * @implements {goog.crypt.BlockCipher}
43 * @param {!Array<number>} key The key as an array of integers in {0, 255}.
44 * The key must have lengths of 16, 24, or 32 integers for 128-,
45 * 192-, or 256-bit encryption, respectively.
46 * @final
47 * @struct
48 */
49goog.crypt.Aes = function(key) {
50 goog.crypt.Aes.assertKeyArray_(key);
51
52 /**
53 * The AES key.
54 * @type {!Array<number>}
55 * @private
56 */
57 this.key_ = key;
58
59 /**
60 * Key length, in words.
61 * @type {number}
62 * @private
63 */
64 this.keyLength_ = this.key_.length / 4;
65
66 /**
67 * Number of rounds. Based on key length per AES spec.
68 * @type {number}
69 * @private
70 */
71 this.numberOfRounds_ = this.keyLength_ + 6;
72
73 /**
74 * 4x4 byte array containing the current state.
75 * @type {!Array<!Array<number>>}
76 * @private
77 */
78 this.state_ = [[], [], [], []];
79
80 /**
81 * Scratch temporary array for calculation.
82 * @type {!Array<!Array<number>>}
83 * @private
84 */
85 this.temp_ = [[], [], [], []];
86
87 /**
88 * The key schedule.
89 * @type {!Array<!Array<number>>}
90 * @private
91 */
92 this.keySchedule_;
93
94 this.keyExpansion_();
95};
96

98/**
99 * @define {boolean} Whether to call test method stubs. This can be enabled
100 * for unit testing.
101 */
102goog.define('goog.crypt.Aes.ENABLE_TEST_MODE', false);
103
104
105/**
106 * @override
107 */
108goog.crypt.Aes.prototype.encrypt = function(input) {
109
110 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
111 this.testKeySchedule_(0, this.keySchedule_, 0);
112 }
113
114 this.copyInput_(input);
115 this.addRoundKey_(0);
116
117 for (var round = 1; round < this.numberOfRounds_; ++round) {
118 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
119 this.testKeySchedule_(round, this.keySchedule_, round);
120 this.testStartRound_(round, this.state_);
121 }
122
123 this.subBytes_(goog.crypt.Aes.SBOX_);
124 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
125 this.testAfterSubBytes_(round, this.state_);
126 }
127
128 this.shiftRows_();
129 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
130 this.testAfterShiftRows_(round, this.state_);
131 }
132
133 this.mixColumns_();
134 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
135 this.testAfterMixColumns_(round, this.state_);
136 }
137
138 this.addRoundKey_(round);
139 }
140
141 this.subBytes_(goog.crypt.Aes.SBOX_);
142 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
143 this.testAfterSubBytes_(round, this.state_);
144 }
145
146 this.shiftRows_();
147 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
148 this.testAfterShiftRows_(round, this.state_);
149 }
150
151 this.addRoundKey_(this.numberOfRounds_);
152
153 return this.generateOutput_();
154};
155
156
157/**
158 * @override
159 */
160goog.crypt.Aes.prototype.decrypt = function(input) {
161
162 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
163 this.testKeySchedule_(0, this.keySchedule_, this.numberOfRounds_);
164 }
165
166 this.copyInput_(input);
167 this.addRoundKey_(this.numberOfRounds_);
168
169 for (var round = 1; round < this.numberOfRounds_; ++round) {
170 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
171 this.testKeySchedule_(round, this.keySchedule_,
172 this.numberOfRounds_ - round);
173 this.testStartRound_(round, this.state_);
174 }
175
176 this.invShiftRows_();
177 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
178 this.testAfterShiftRows_(round, this.state_);
179 }
180
181 this.subBytes_(goog.crypt.Aes.INV_SBOX_);
182 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
183 this.testAfterSubBytes_(round, this.state_);
184 }
185
186 this.addRoundKey_(this.numberOfRounds_ - round);
187 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
188 this.testAfterAddRoundKey_(round, this.state_);
189 }
190
191 this.invMixColumns_();
192 }
193
194 this.invShiftRows_();
195 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
196 this.testAfterShiftRows_(round, this.state_);
197 }
198
199 this.subBytes_(goog.crypt.Aes.INV_SBOX_);
200 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
201 this.testAfterSubBytes_(this.numberOfRounds_, this.state_);
202 }
203
204 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
205 this.testKeySchedule_(this.numberOfRounds_, this.keySchedule_, 0);
206 }
207
208 this.addRoundKey_(0);
209
210 return this.generateOutput_();
211};

222/**
223 * Asserts that the key's array of integers is in the correct format.
224 * @param {!Array<number>} arr AES key as array of integers.
225 * @private
226 */
227goog.crypt.Aes.assertKeyArray_ = function(arr) {
228 if (goog.asserts.ENABLE_ASSERTS) {
229 goog.asserts.assert(arr.length == 16 || arr.length == 24 ||
230 arr.length == 32,
231 'Key must have length 16, 24, or 32.');
232 for (var i = 0; i < arr.length; i++) {
233 goog.asserts.assertNumber(arr[i]);
234 goog.asserts.assert(arr[i] >= 0 && arr[i] <= 255);
235 }
236 }

307/**
308 * Helper to copy input into the AES state matrix.
309 * @param {!Array<number>} input Byte array to copy into the state matrix.
310 * @private
311 */
312goog.crypt.Aes.prototype.copyInput_ = function(input) {
313 var v, p;
314
315 goog.asserts.assert(input.length == goog.crypt.Aes.BLOCK_SIZE_ * 4,
316 'Expecting input of 4 times block size.');
317
318 for (var r = 0; r < goog.crypt.Aes.BLOCK_SIZE_; r++) {
319 for (var c = 0; c < 4; c++) {
320 p = c * 4 + r;
321 v = input[p];
322
323 goog.asserts.assert(
324 v <= 255 && v >= 0,
325 'Invalid input. Value %s at position %s is not a byte.', v, p);
326
327 this.state_[r][c] = v;
328 }
329 }
330};
331
332
333/**
334 * Helper to copy the state matrix into an output array.
335 * @return {!Array<number>} Output byte array.
336 * @private
337 */
338goog.crypt.Aes.prototype.generateOutput_ = function() {
339 var output = [];
340 for (var r = 0; r < goog.crypt.Aes.BLOCK_SIZE_; r++) {
341 for (var c = 0; c < 4; c++) {
342 output[c * 4 + r] = this.state_[r][c];
343 }
344 }
345 return output;
346};

349/**
350 * AES's AddRoundKey procedure. Add the current round key to the state.
351 * @param {number} round The current round.
352 * @private
353 */
354goog.crypt.Aes.prototype.addRoundKey_ = function(round) {
355 for (var r = 0; r < 4; r++) {
356 for (var c = 0; c < 4; c++) {
357 this.state_[r][c] ^= this.keySchedule_[round * 4 + c][r];
358 }
359 }
360};

363/**
364 * AES's SubBytes procedure. Substitute bytes from the precomputed SBox lookup
365 * into the state.
366 * @param {!Array<number>} box The SBox or invSBox.
367 * @private
368 */
369goog.crypt.Aes.prototype.subBytes_ = function(box) {
370 for (var r = 0; r < 4; r++) {
371 for (var c = 0; c < 4; c++) {
372 this.state_[r][c] = box[this.state_[r][c]];
373 }
374 }
375};

378/**
379 * AES's ShiftRows procedure. Shift the values in each row to the right. Each
380 * row is shifted one more slot than the one above it.
381 * @private
382 */
383goog.crypt.Aes.prototype.shiftRows_ = function() {
384 for (var r = 1; r < 4; r++) {
385 for (var c = 0; c < 4; c++) {
386 this.temp_[r][c] = this.state_[r][c];
387 }
388 }
389
390 for (var r = 1; r < 4; r++) {
391 for (var c = 0; c < 4; c++) {
392 this.state_[r][c] = this.temp_[r][(c + r) %
393 goog.crypt.Aes.BLOCK_SIZE_];
394 }
395 }
396};
397
398
399/**
400 * AES's InvShiftRows procedure. Shift the values in each row to the right.
401 * @private
402 */
403goog.crypt.Aes.prototype.invShiftRows_ = function() {
404 for (var r = 1; r < 4; r++) {
405 for (var c = 0; c < 4; c++) {
406 this.temp_[r][(c + r) % goog.crypt.Aes.BLOCK_SIZE_] =
407 this.state_[r][c];
408 }
409 }
410
411 for (var r = 1; r < 4; r++) {
412 for (var c = 0; c < 4; c++) {
413 this.state_[r][c] = this.temp_[r][c];
414 }
415 }
416};

419/**
420 * AES's MixColumns procedure. Mix the columns of the state using magic.
421 * @private
422 */
423goog.crypt.Aes.prototype.mixColumns_ = function() {
424 var s = this.state_;
425 var t = this.temp_[0];
426
427 for (var c = 0; c < 4; c++) {
428 t[0] = s[0][c];
429 t[1] = s[1][c];
430 t[2] = s[2][c];
431 t[3] = s[3][c];
432
433 s[0][c] = (goog.crypt.Aes.MULT_2_[t[0]] ^
434 goog.crypt.Aes.MULT_3_[t[1]] ^ t[2] ^ t[3]);
435 s[1][c] = (t[0] ^ goog.crypt.Aes.MULT_2_[t[1]] ^
436 goog.crypt.Aes.MULT_3_[t[2]] ^ t[3]);
437 s[2][c] = (t[0] ^ t[1] ^ goog.crypt.Aes.MULT_2_[t[2]] ^
438 goog.crypt.Aes.MULT_3_[t[3]]);
439 s[3][c] = (goog.crypt.Aes.MULT_3_[t[0]] ^ t[1] ^ t[2] ^
440 goog.crypt.Aes.MULT_2_[t[3]]);
441 }
442};

445/**
446 * AES's InvMixColumns procedure.
447 * @private
448 */
449goog.crypt.Aes.prototype.invMixColumns_ = function() {
450 var s = this.state_;
451 var t = this.temp_[0];
452
453 for (var c = 0; c < 4; c++) {
454 t[0] = s[0][c];
455 t[1] = s[1][c];
456 t[2] = s[2][c];
457 t[3] = s[3][c];
458
459 s[0][c] = (
460 goog.crypt.Aes.MULT_E_[t[0]] ^ goog.crypt.Aes.MULT_B_[t[1]] ^
461 goog.crypt.Aes.MULT_D_[t[2]] ^ goog.crypt.Aes.MULT_9_[t[3]]);
462
463 s[1][c] = (
464 goog.crypt.Aes.MULT_9_[t[0]] ^ goog.crypt.Aes.MULT_E_[t[1]] ^
465 goog.crypt.Aes.MULT_B_[t[2]] ^ goog.crypt.Aes.MULT_D_[t[3]]);
466
467 s[2][c] = (
468 goog.crypt.Aes.MULT_D_[t[0]] ^ goog.crypt.Aes.MULT_9_[t[1]] ^
469 goog.crypt.Aes.MULT_E_[t[2]] ^ goog.crypt.Aes.MULT_B_[t[3]]);
470
471 s[3][c] = (
472 goog.crypt.Aes.MULT_B_[t[0]] ^ goog.crypt.Aes.MULT_D_[t[1]] ^
473 goog.crypt.Aes.MULT_9_[t[2]] ^ goog.crypt.Aes.MULT_E_[t[3]]);
474 }
475};

478/**
479 * AES's KeyExpansion procedure. Create the key schedule from the initial key.
480 * @private
481 */
482goog.crypt.Aes.prototype.keyExpansion_ = function() {
483 this.keySchedule_ = new Array(goog.crypt.Aes.BLOCK_SIZE_ * (
484 this.numberOfRounds_ + 1));
485
486 for (var rowNum = 0; rowNum < this.keyLength_; rowNum++) {
487 this.keySchedule_[rowNum] = [
488 this.key_[4 * rowNum],
489 this.key_[4 * rowNum + 1],
490 this.key_[4 * rowNum + 2],
491 this.key_[4 * rowNum + 3]
492 ];
493 }
494
495 var temp = new Array(4);
496
497 for (var rowNum = this.keyLength_;
498 rowNum < (goog.crypt.Aes.BLOCK_SIZE_ * (this.numberOfRounds_ + 1));
499 rowNum++) {
500 temp[0] = this.keySchedule_[rowNum - 1][0];
501 temp[1] = this.keySchedule_[rowNum - 1][1];
502 temp[2] = this.keySchedule_[rowNum - 1][2];
503 temp[3] = this.keySchedule_[rowNum - 1][3];
504
505 if (rowNum % this.keyLength_ == 0) {
506 this.rotWord_(temp);
507 this.subWord_(temp);
508
509 temp[0] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][0];
510 temp[1] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][1];
511 temp[2] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][2];
512 temp[3] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][3];
513 } else if (this.keyLength_ > 6 && rowNum % this.keyLength_ == 4) {
514 this.subWord_(temp);
515 }
516
517 this.keySchedule_[rowNum] = new Array(4);
518 this.keySchedule_[rowNum][0] =
519 this.keySchedule_[rowNum - this.keyLength_][0] ^ temp[0];
520 this.keySchedule_[rowNum][1] =
521 this.keySchedule_[rowNum - this.keyLength_][1] ^ temp[1];
522 this.keySchedule_[rowNum][2] =
523 this.keySchedule_[rowNum - this.keyLength_][2] ^ temp[2];
524 this.keySchedule_[rowNum][3] =
525 this.keySchedule_[rowNum - this.keyLength_][3] ^ temp[3];
526 }
527};

530/**
531 * AES's SubWord procedure.
532 * @param {!Array<number>} w Bytes to find the SBox substitution for.
533 * @return {!Array<number>} The substituted bytes.
534 * @private
535 */
536goog.crypt.Aes.prototype.subWord_ = function(w) {
537 w[0] = goog.crypt.Aes.SBOX_[w[0]];
538 w[1] = goog.crypt.Aes.SBOX_[w[1]];
539 w[2] = goog.crypt.Aes.SBOX_[w[2]];
540 w[3] = goog.crypt.Aes.SBOX_[w[3]];
541
542 return w;
543};

546/**
547 * AES's RotWord procedure.
548 * @param {!Array<number>} w Array of bytes to rotate.
549 * @return {!Array<number>} The rotated bytes.
550 * @private
551 */
552goog.crypt.Aes.prototype.rotWord_ = function(w) {
553 var temp = w[0];
554
555 w[0] = w[1];
556 w[1] = w[2];
557 w[2] = w[3];
558 w[3] = temp;
559
560 return w;
561};
