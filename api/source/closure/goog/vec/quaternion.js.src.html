228 * Computes the conjugate of the quaternion in quat storing the result into

231 * @param {goog.vec.Quaternion.AnyType} quat The source quaternion.

237goog.vec.Quaternion.conjugate = function(quat, resultQuat) {
238 resultQuat[0] = -quat[0];
239 resultQuat[1] = -quat[1];
240 resultQuat[2] = -quat[2];
241 resultQuat[3] = quat[3];

246/**
247 * Concatenates the two quaternions storing the result into resultQuat.
248 *
249 * @param {goog.vec.Quaternion.AnyType} quat0 The first quaternion.
250 * @param {goog.vec.Quaternion.AnyType} quat1 The second quaternion.
251 * @param {goog.vec.Quaternion.AnyType} resultQuat The quaternion to
252 * receive the result.
253 * @return {!goog.vec.Quaternion.AnyType} Return q so that
254 * operations can be chained together.
255 */
256goog.vec.Quaternion.concat = function(quat0, quat1, resultQuat) {
257 var x0 = quat0[0], y0 = quat0[1], z0 = quat0[2], w0 = quat0[3];
258 var x1 = quat1[0], y1 = quat1[1], z1 = quat1[2], w1 = quat1[3];
259 resultQuat[0] = w0 * x1 + x0 * w1 + y0 * z1 - z0 * y1;
260 resultQuat[1] = w0 * y1 - x0 * z1 + y0 * w1 + z0 * x1;
261 resultQuat[2] = w0 * z1 + x0 * y1 - y0 * x1 + z0 * w1;
262 resultQuat[3] = w0 * w1 - x0 * x1 - y0 * y1 - z0 * z1;
263 return resultQuat;
264};

267/**
268 * Generates a unit quaternion from the given angle-axis rotation pair.
269 * The rotation axis is not required to be a unit vector, but should
270 * have non-zero length. The angle should be specified in radians.
271 *
272 * @param {number} angle The angle (in radians) to rotate about the axis.
273 * @param {goog.vec.Quaternion.AnyType} axis Unit vector specifying the
274 * axis of rotation.
275 * @param {goog.vec.Quaternion.AnyType} quat Unit quaternion to store the
276 * result.
277 * @return {goog.vec.Quaternion.AnyType} Return q so that
278 * operations can be chained together.
279 */
280goog.vec.Quaternion.fromAngleAxis = function(angle, axis, quat) {
281 // Normalize the axis of rotation.
282 goog.vec.Vec3.normalize(axis, axis);
283
284 var halfAngle = 0.5 * angle;
285 var sin = Math.sin(halfAngle);
286 goog.vec.Quaternion.setFromValues(
287 quat, sin * axis[0], sin * axis[1], sin * axis[2], Math.cos(halfAngle));
288
289 // Normalize the resulting quaternion.
290 goog.vec.Quaternion.normalize(quat, quat);
291 return quat;
292};

295/**
296 * Generates an angle-axis rotation pair from a unit quaternion.
297 * The quaternion is assumed to be of unit length. The calculated
298 * values are returned via the passed 'axis' object and the 'angle'
299 * number returned by the function itself. The returned rotation axis
300 * is a non-zero length unit vector, and the returned angle is in
301 * radians in the range of [-PI, +PI].
302 *
303 * @param {goog.vec.Quaternion.AnyType} quat Unit quaternion to convert.
304 * @param {goog.vec.Quaternion.AnyType} axis Vector to store the returned
305 * rotation axis.
306 * @return {number} angle Angle (in radians) to rotate about 'axis'.
307 * The range of the returned angle is [-PI, +PI].
308 */
309goog.vec.Quaternion.toAngleAxis = function(quat, axis) {
310 var angle = 2 * Math.acos(quat[3]);
311 var magnitude = Math.min(Math.max(1 - quat[3] * quat[3], 0), 1);
312 if (magnitude < goog.vec.EPSILON) {
313 // This is nearly an identity rotation, so just use a fixed +X axis.
314 goog.vec.Vec3.setFromValues(axis, 1, 0, 0);
315 } else {
316 // Compute the proper rotation axis.
317 goog.vec.Vec3.setFromValues(axis, quat[0], quat[1], quat[2]);
318 // Make sure the rotation axis is of unit length.
319 goog.vec.Vec3.normalize(axis, axis);
320 }
321 // Adjust the range of the returned angle to [-PI, +PI].
322 if (angle > Math.PI) {
323 angle -= 2 * Math.PI;
324 }
325 return angle;
326};

329/**
330 * Generates the quaternion from the given rotation matrix.
331 *
332 * @param {goog.vec.Quaternion.AnyType} matrix The source matrix.
333 * @param {goog.vec.Quaternion.AnyType} quat The resulting quaternion.
334 * @return {!goog.vec.Quaternion.AnyType} Return q so that
335 * operations can be chained together.
336 */
337goog.vec.Quaternion.fromRotationMatrix4 = function(matrix, quat) {
338 var sx = matrix[0], sy = matrix[5], sz = matrix[10];
339 quat[3] = Math.sqrt(Math.max(0, 1 + sx + sy + sz)) / 2;
340 quat[0] = Math.sqrt(Math.max(0, 1 + sx - sy - sz)) / 2;
341 quat[1] = Math.sqrt(Math.max(0, 1 - sx + sy - sz)) / 2;
342 quat[2] = Math.sqrt(Math.max(0, 1 - sx - sy + sz)) / 2;
343
344 quat[0] = (matrix[6] - matrix[9] < 0) != (quat[0] < 0) ? -quat[0] : quat[0];
345 quat[1] = (matrix[8] - matrix[2] < 0) != (quat[1] < 0) ? -quat[1] : quat[1];
346 quat[2] = (matrix[1] - matrix[4] < 0) != (quat[2] < 0) ? -quat[2] : quat[2];
347 return quat;
348};

351/**
352 * Generates the rotation matrix from the given quaternion.
353 *
354 * @param {goog.vec.Quaternion.AnyType} quat The source quaternion.
355 * @param {goog.vec.AnyType} matrix The resulting matrix.
356 * @return {!goog.vec.AnyType} Return resulting matrix so that
357 * operations can be chained together.
358 */
359goog.vec.Quaternion.toRotationMatrix4 = function(quat, matrix) {
360 var x = quat[0], y = quat[1], z = quat[2], w = quat[3];
361 var x2 = 2 * x, y2 = 2 * y, z2 = 2 * z;
362 var wx = x2 * w;
363 var wy = y2 * w;
364 var wz = z2 * w;
365 var xx = x2 * x;
366 var xy = y2 * x;
367 var xz = z2 * x;
368 var yy = y2 * y;
369 var yz = z2 * y;
370 var zz = z2 * z;
371
372 matrix[0] = 1 - (yy + zz);
373 matrix[1] = xy + wz;
374 matrix[2] = xz - wy;
375 matrix[3] = 0;
376 matrix[4] = xy - wz;
377 matrix[5] = 1 - (xx + zz);
378 matrix[6] = yz + wx;
379 matrix[7] = 0;
380 matrix[8] = xz + wy;
381 matrix[9] = yz - wx;
382 matrix[10] = 1 - (xx + yy);
383 matrix[11] = 0;
384 matrix[12] = 0;
385 matrix[13] = 0;
386 matrix[14] = 0;
387 matrix[15] = 1;
388 return matrix;
389};

392/**
393 * Computes the spherical linear interpolated value from the given quaternions
394 * q0 and q1 according to the coefficient t. The resulting quaternion is stored
395 * in resultQuat.
396 *
397 * @param {goog.vec.Quaternion.AnyType} q0 The first quaternion.
398 * @param {goog.vec.Quaternion.AnyType} q1 The second quaternion.
399 * @param {number} t The interpolating coefficient.
400 * @param {goog.vec.Quaternion.AnyType} resultQuat The quaternion to
401 * receive the result.
402 * @return {goog.vec.Quaternion.AnyType} Return q so that
403 * operations can be chained together.
404 */
405goog.vec.Quaternion.slerp = function(q0, q1, t, resultQuat) {
406 // Compute the dot product between q0 and q1 (cos of the angle between q0 and
407 // q1). If it's outside the interval [-1,1], then the arccos is not defined.
408 // The usual reason for this is that q0 and q1 are colinear. In this case
409 // the angle between the two is zero, so just return q1.
410 var cosVal = goog.vec.Quaternion.dot(q0, q1);
411 if (cosVal > 1 || cosVal < -1) {
412 goog.vec.Vec4.setFromArray(resultQuat, q1);
413 return resultQuat;
414 }
415
416 // Quaternions are a double cover on the space of rotations. That is, q and -q
417 // represent the same rotation. Thus we have two possibilities when
418 // interpolating between q0 and q1: going the short way or the long way. We
419 // prefer the short way since that is the likely expectation from users.
420 var factor = 1;
421 if (cosVal < 0) {
422 factor = -1;
423 cosVal = -cosVal;
424 }
425
426 // Compute the angle between q0 and q1. If it's very small, then just return
427 // q1 to avoid a very large denominator below.
428 var angle = Math.acos(cosVal);
429 if (angle <= goog.vec.EPSILON) {
430 goog.vec.Vec4.setFromArray(resultQuat, q1);
431 return resultQuat;
432 }
433
434 // Compute the coefficients and interpolate.
435 var invSinVal = 1 / Math.sin(angle);
436 var c0 = Math.sin((1 - t) * angle) * invSinVal;
437 var c1 = factor * Math.sin(t * angle) * invSinVal;
438
439 resultQuat[0] = q0[0] * c0 + q1[0] * c1;
440 resultQuat[1] = q0[1] * c0 + q1[1] * c1;
441 resultQuat[2] = q0[2] * c0 + q1[2] * c1;
442 resultQuat[3] = q0[3] * c0 + q1[3] * c1;
443 return resultQuat;
444};
