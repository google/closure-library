15/**
16 * @fileoverview Implements 3x3 matrices and their related functions which are
17 * compatible with WebGL. The API is structured to avoid unnecessary memory
18 * allocations. The last parameter will typically be the output vector and
19 * an object can be both an input and output parameter to all methods except
20 * where noted. Matrix operations follow the mathematical form when multiplying
21 * vectors as follows: resultVec = matrix * vec.
22 *
23 * The matrices are stored in column-major order.
24 *
25 */
26goog.provide('goog.vec.Mat3');
27
28goog.require('goog.vec');

647/**
648 * Performs a per-component addition of the matrices mat0 and mat1, storing
649 * the result into resultMat.
650 *
651 * @param {goog.vec.Mat3.AnyType} mat0 The first addend.
652 * @param {goog.vec.Mat3.AnyType} mat1 The second addend.
653 * @param {goog.vec.Mat3.AnyType} resultMat The matrix to
654 * receive the results (may be either mat0 or mat1).
655 * @return {goog.vec.Mat3.AnyType} return resultMat so that operations can be
656 * chained together.
657 */
658goog.vec.Mat3.addMat = function(mat0, mat1, resultMat) {
659 resultMat[0] = mat0[0] + mat1[0];
660 resultMat[1] = mat0[1] + mat1[1];
661 resultMat[2] = mat0[2] + mat1[2];
662 resultMat[3] = mat0[3] + mat1[3];

792 * @param {goog.vec.Mat3.AnyType} mat0 The matrix to invert.
793 * @param {goog.vec.Mat3.AnyType} resultMat The matrix to receive

1166/**
1167 * Decomposes a rotation matrix into Euler angles using the ZXZ convention so
1168 * that rotation = rotation_z(theta1) * rotation_x(theta2) * rotation_z(theta3),
1169 * with theta1 in [0, 2 * pi], theta2 in [0, pi] and theta3 in [0, 2 * pi].
1170 * rotation_x(theta) means rotation around the X axis of theta radians.
1171 *
1172 * @param {goog.vec.Mat3.AnyType} mat The matrix.
1173 * @param {goog.vec.Vec3.AnyType} euler The ZXZ Euler angles in
1174 * radians as [theta1, theta2, theta3].
1175 * @param {boolean=} opt_theta2IsNegative Whether theta2 is in [-pi, 0] instead
1176 * of the default [0, pi].
1177 * @return {goog.vec.Vec3.AnyType} return euler so that operations can be
1178 * chained together.
1179 */
1180goog.vec.Mat3.toEulerZXZ = function(mat, euler, opt_theta2IsNegative) {
1181 // There is an ambiguity in the sign of sinTheta2 because of the sqrt.
1182 var sinTheta2 = Math.sqrt(mat[2] * mat[2] + mat[5] * mat[5]);
1183
1184 // By default we explicitely constrain theta2 to be in [0, pi],
1185 // so sinTheta2 is always positive. We can change the behavior and specify
1186 // theta2 to be negative in [-pi, 0] with opt_Theta2IsNegative.
1187 var signTheta2 = opt_theta2IsNegative ? -1 : 1;
1188
1189 if (sinTheta2 > goog.vec.EPSILON) {
1190 euler[2] = Math.atan2(mat[2] * signTheta2, mat[5] * signTheta2);
1191 euler[1] = Math.atan2(sinTheta2 * signTheta2, mat[8]);
1192 euler[0] = Math.atan2(mat[6] * signTheta2, -mat[7] * signTheta2);
1193 } else {
1194 // There is also an arbitrary choice for theta1 = 0 or theta2 = 0 here.
1195 // We assume theta1 = 0 as some applications do not allow the camera to roll
1196 // (i.e. have theta1 != 0).
1197 euler[0] = 0;
1198 euler[1] = Math.atan2(sinTheta2 * signTheta2, mat[8]);
1199 euler[2] = Math.atan2(mat[1], mat[0]);
1200 }
1201
1202 // Atan2 outputs angles in [-pi, pi] so we bring them back to [0, 2 * pi].
1203 euler[0] = (euler[0] + Math.PI * 2) % (Math.PI * 2);
1204 euler[2] = (euler[2] + Math.PI * 2) % (Math.PI * 2);
1205 // For theta2 we want the angle to be in [0, pi] or [-pi, 0] depending on
1206 // signTheta2.
1207 euler[1] = ((euler[1] * signTheta2 + Math.PI * 2) % (Math.PI * 2)) *
1208 signTheta2;
1209
1210 return euler;
1211};
