16/**
17 * @fileoverview A utility to get better currency format pattern.
18 *
19 * This module implements a new currency format representation model. It
20 * provides 3 currency representation forms: global, portable and local. Local
21 * format is the most popular format people use to represent currency in its
22 * circulating country without worrying about how it should be distinguished
23 * from other currencies. Global format is a formal representation in context
24 * of multiple currencies in same page, it is ISO 4217 currency code. Portable
25 * format is a compromise between global and local. It looks similar to how
26 * people would like to see how their currency is being represented in other
27 * media. While at the same time, it should be distinguishable to world's
28 * popular currencies (like USD, EUR) and currencies somewhat relevant in the
29 * area (like CNY in HK, though native currency is HKD). There is no guarantee
30 * of uniqueness.
31 *
32 */

68/**
69 * This function will add tier2 currency support. Be default, only tier1
70 * (most popular currencies) are supported. If an application really needs
71 * to support some of the rarely used currencies, it should call this function
72 * before any other functions in this namespace.
73 */
74goog.i18n.currency.addTier2Support = function() {
75 // Protection from executing this these again and again.
76 if (!goog.i18n.currency.tier2Enabled_) {
77 for (var key in goog.i18n.currency.CurrencyInfoTier2) {
78 goog.i18n.currency.CurrencyInfo[key] =
79 goog.i18n.currency.CurrencyInfoTier2[key];
80 }
81 goog.i18n.currency.tier2Enabled_ = true;
82 }
83};

86/**
87 * Global currency pattern always uses ISO-4217 currency code as prefix. Local
88 * currency sign is added if it is different from currency code. Each currency
89 * is unique in this form. The negative side is that ISO code looks weird in
90 * some countries as people normally do not use it. Local currency sign
91 * alleviates the problem, but also makes it a little verbose.
92 *
93 * @param {string} currencyCode ISO-4217 3-letter currency code.
94 * @return {string} Global currency pattern string for given currency.
95 */
96goog.i18n.currency.getGlobalCurrencyPattern = function(currencyCode) {
97 var info = goog.i18n.currency.CurrencyInfo[currencyCode];
98 var patternNum = info[0];
99 if (currencyCode == info[1]) {
100 return goog.i18n.currency.getCurrencyPattern_(patternNum, info[1]);
101 }
102 return currencyCode + ' ' +
103 goog.i18n.currency.getCurrencyPattern_(patternNum, info[1]);
104};

147/**
148 * Portable currency pattern is a compromise between local and global. It is
149 * not a mere blend or mid-way between the two. Currency sign is chosen so that
150 * it looks familiar to native users. It also has enough information to
151 * distinguish itself from other popular currencies in its native region.
152 * In this pattern, currency sign symbols that has availability problem in
153 * popular fonts are also avoided.

155 * @param {string} currencyCode ISO-4217 3-letter currency code.
156 * @return {string} Portable currency pattern string for given currency.
157 */
158goog.i18n.currency.getPortableCurrencyPattern = function(currencyCode) {
159 var info = goog.i18n.currency.CurrencyInfo[currencyCode];
160 return goog.i18n.currency.getCurrencyPattern_(info[0], info[2]);

176/**
177 * This function returns the default currency sign position. Some applications
178 * may want to handle currency sign and currency amount separately. This
179 * function can be used in such situations to correctly position the currency
180 * sign relative to the amount.
181 *
182 * To match the behavior of ICU, position is not determined by display locale.
183 *
184 * @param {string} currencyCode ISO-4217 3-letter currency code.
185 * @return {boolean} true if currency should be positioned before amount field.
186 */
187goog.i18n.currency.isPrefixSignPosition = function(currencyCode) {
188 return (goog.i18n.currency.CurrencyInfo[currencyCode][0] &
189 goog.i18n.currency.POSITION_FLAG_) == 0;
190};

193/**
194 * This function constructs the currency pattern. Currency sign is provided. The
195 * pattern information is encoded in patternNum.
196 *
197 * @param {number} patternNum Encoded pattern number that has
198 * currency pattern information.
199 * @param {string} sign The currency sign that will be used in pattern.
200 * @return {string} currency pattern string.
201 * @private
202 */
203goog.i18n.currency.getCurrencyPattern_ = function(patternNum, sign) {
204 var strParts = ['#,##0'];
205 var precision = patternNum & goog.i18n.currency.PRECISION_MASK_;
206 if (precision > 0) {
207 strParts.push('.');
208 for (var i = 0; i < precision; i++) {
209 strParts.push('0');
210 }
211 }
212 if ((patternNum & goog.i18n.currency.POSITION_FLAG_) == 0) {
213 strParts.unshift((patternNum & goog.i18n.currency.SPACE_FLAG_) ?
214 "' " : "'");
215 strParts.unshift(sign);
216 strParts.unshift("'");
217 } else {
218 strParts.push((patternNum & goog.i18n.currency.SPACE_FLAG_) ? " '" : "'",
219 sign, "'");
220 }
221 return strParts.join('');
222};

225/**
226 * Modify currency pattern string by adjusting precision for given currency.
227 * Standard currency pattern will have 2 digit after decimal point.
228 * Examples:
229 * $#,##0.00 -> $#,##0 (precision == 0)
230 * $#,##0.00 -> $#,##0.0 (precision == 1)
231 * $#,##0.00 -> $#,##0.000 (precision == 3)
232 *
233 * @param {string} pattern currency pattern string.
234 * @param {string} currencyCode 3-letter currency code.
235 * @return {string} modified currency pattern string.
236 */
237goog.i18n.currency.adjustPrecision = function(pattern, currencyCode) {
238 var strParts = ['0'];
239 var info = goog.i18n.currency.CurrencyInfo[currencyCode];
240 var precision = info[0] & goog.i18n.currency.PRECISION_MASK_;
241 if (precision > 0) {
242 strParts.push('.');
243 for (var i = 0; i < precision; i++) {
244 strParts.push('0');
245 }
246 }
247 return pattern.replace(/0.00/g, strParts.join(''));
248};
