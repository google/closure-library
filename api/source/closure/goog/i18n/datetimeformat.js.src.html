33/**
34 * Datetime formatting functions following the pattern specification as defined
35 * in JDK, ICU and CLDR, with minor modification for typical usage in JS.
36 * Pattern specification: (Refer to JDK/ICU/CLDR)
37 * <pre>
38 * Symbol Meaning Presentation Example
39 * ------ ------- ------------ -------
40 * G era designator (Text) AD
41 * y# year (Number) 1996
42 * Y* year (week of year) (Number) 1997
43 * u* extended year (Number) 4601
44 * M month in year (Text & Number) July & 07
45 * d day in month (Number) 10
46 * h hour in am/pm (1~12) (Number) 12
47 * H hour in day (0~23) (Number) 0
48 * m minute in hour (Number) 30
49 * s second in minute (Number) 55
50 * S fractional second (Number) 978
51 * E day of week (Text) Tuesday
52 * e* day of week (local 1~7) (Number) 2
53 * D* day in year (Number) 189
54 * F* day of week in month (Number) 2 (2nd Wed in July)
55 * w week in year (Number) 27
56 * W* week in month (Number) 2
57 * a am/pm marker (Text) PM
58 * k hour in day (1~24) (Number) 24
59 * K hour in am/pm (0~11) (Number) 0
60 * z time zone (Text) Pacific Standard Time
61 * Z time zone (RFC 822) (Number) -0800
62 * v time zone (generic) (Text) Pacific Time
63 * g* Julian day (Number) 2451334
64 * A* milliseconds in day (Number) 69540000
65 * ' escape for text (Delimiter) 'Date='
66 * '' single quote (Literal) 'o''clock'
67 *
68 * Item marked with '*' are not supported yet.
69 * Item marked with '#' works different than java
70 *
71 * The count of pattern letters determine the format.
72 * (Text): 4 or more, use full form, <4, use short or abbreviated form if it
73 * exists. (e.g., "EEEE" produces "Monday", "EEE" produces "Mon")
74 *
75 * (Number): the minimum number of digits. Shorter numbers are zero-padded to
76 * this amount (e.g. if "m" produces "6", "mm" produces "06"). Year is handled
77 * specially; that is, if the count of 'y' is 2, the Year will be truncated to
78 * 2 digits. (e.g., if "yyyy" produces "1997", "yy" produces "97".) Unlike other
79 * fields, fractional seconds are padded on the right with zero.
80 *
81 * (Text & Number): 3 or over, use text, otherwise use number. (e.g., "M"
82 * produces "1", "MM" produces "01", "MMM" produces "Jan", and "MMMM" produces
83 * "January".)
84 *
85 * Any characters in the pattern that are not in the ranges of ['a'..'z'] and
86 * ['A'..'Z'] will be treated as quoted text. For instance, characters like ':',
87 * '.', ' ', '#' and '@' will appear in the resulting time text even they are
88 * not embraced within single quotes.
89 * </pre>
90 */

94/**
95 * Construct a DateTimeFormat object based on current locale.
96 * @constructor
97 * @param {string|number} pattern pattern specification or pattern type.
98 * @param {!Object=} opt_dateTimeSymbols Optional symbols to use use for this
99 * instance rather than the global symbols.
100 * @final
101 */
102goog.i18n.DateTimeFormat = function(pattern, opt_dateTimeSymbols) {
103 goog.asserts.assert(goog.isDef(pattern), 'Pattern must be defined');
104 goog.asserts.assert(
105 goog.isDef(opt_dateTimeSymbols) || goog.isDef(goog.i18n.DateTimeSymbols),
106 'goog.i18n.DateTimeSymbols or explicit symbols must be defined');
107
108 this.patternParts_ = [];
109
110 /**
111 * Data structure that with all the locale info needed for date formatting.
112 * (day/month names, most common patterns, rules for week-end, etc.)
113 * @type {!Object}
114 * @private
115 */
116 this.dateTimeSymbols_ = opt_dateTimeSymbols || goog.i18n.DateTimeSymbols;
117 if (typeof pattern == 'number') {
118 this.applyStandardPattern_(pattern);
119 } else {
120 this.applyPattern_(pattern);
121 }
122};

145/**
146 * regular expression pattern for parsing pattern string
147 * @type {Array<RegExp>}
148 * @private
149 */
150goog.i18n.DateTimeFormat.TOKENS_ = [
151 //quote string
152 /^\'(?:[^\']|\'\')*\'/,
153 // pattern chars
154 /^(?:G+|y+|M+|k+|S+|E+|a+|h+|K+|H+|c+|L+|Q+|d+|m+|s+|v+|w+|z+|Z+)/,
155 // and all the other chars
156 /^[^\'GyMkSEahKHcLQdmsvwzZ]+/ // and all the other chars
157];
158
159
160/**
161 * These are token types, corresponding to above token definitions.
162 * @enum {number}
163 * @private
164 */
165goog.i18n.DateTimeFormat.PartTypes_ = {
166 QUOTED_STRING: 0,
167 FIELD: 1,
168 LITERAL: 2
169};

172/**
173 * Apply specified pattern to this formatter object.
174 * @param {string} pattern String specifying how the date should be formatted.
175 * @private
176 */
177goog.i18n.DateTimeFormat.prototype.applyPattern_ = function(pattern) {
178 // lex the pattern, once for all uses
179 while (pattern) {
180 for (var i = 0; i < goog.i18n.DateTimeFormat.TOKENS_.length; ++i) {
181 var m = pattern.match(goog.i18n.DateTimeFormat.TOKENS_[i]);
182 if (m) {
183 var part = m[0];
184 pattern = pattern.substring(part.length);
185 if (i == goog.i18n.DateTimeFormat.PartTypes_.QUOTED_STRING) {
186 if (part == "''") {
187 part = "'"; // '' -> '
188 } else {
189 part = part.substring(1, part.length - 1); // strip quotes
190 part = part.replace(/\'\'/, "'");
191 }
192 }
193 this.patternParts_.push({ text: part, type: i });
194 break;
195 }
196 }
197 }
198};

201/**
202 * Format the given date object according to preset pattern and current locale.
203 * @param {goog.date.DateLike} date The Date object that is being formatted.
204 * @param {goog.i18n.TimeZone=} opt_timeZone optional, if specified, time
205 * related fields will be formatted based on its setting. When this field
206 * is not specified, "undefined" will be pass around and those function
207 * that really need time zone service will create a default one.
208 * @return {string} Formatted string for the given date.
209 * Throws an error if the date is null or if one tries to format a date-only
210 * object (for instance goog.date.Date) using a pattern with time fields.
211 */
212goog.i18n.DateTimeFormat.prototype.format = function(date, opt_timeZone) {
213 if (!date)
214 throw Error('The date to format must be non-null.');
215
216 // We don't want to write code to calculate each date field because we
217 // want to maximize performance and minimize code size.
218 // JavaScript only provide API to render local time.
219 // Suppose target date is: 16:00 GMT-0400
220 // OS local time is: 12:00 GMT-0800
221 // We want to create a Local Date Object : 16:00 GMT-0800, and fix the
222 // time zone display ourselves.
223 // Thing get a little bit tricky when daylight time transition happens. For
224 // example, suppose OS timeZone is America/Los_Angeles, it is impossible to
225 // represent "2006/4/2 02:30" even for those timeZone that has no transition
226 // at this time. Because 2:00 to 3:00 on that day does not exist in
227 // America/Los_Angeles time zone. To avoid calculating date field through
228 // our own code, we uses 3 Date object instead, one for "Year, month, day",
229 // one for time within that day, and one for timeZone object since it need
230 // the real time to figure out actual time zone offset.
231 var diff = opt_timeZone ?
232 (date.getTimezoneOffset() - opt_timeZone.getOffset(date)) * 60000 : 0;
233 var dateForDate = diff ? new Date(date.getTime() + diff) : date;
234 var dateForTime = dateForDate;
235 // When the time manipulation applied above spans the DST on/off hour, this
236 // could alter the time incorrectly by adding or subtracting an additional
237 // hour.
238 // We can mitigate this by:
239 // - Adding the difference in timezone offset to the date. This ensures that
240 // the dateForDate is still within the right day if the extra DST hour
241 // affected the date.
242 // - Move the time one day forward if we applied a timezone offset backwards,
243 // or vice versa. This trick ensures that the time is in the same offset
244 // as the original date, so we remove the additional hour added or
245 // subtracted by the DST switch.
246 if (opt_timeZone &&
247 dateForDate.getTimezoneOffset() != date.getTimezoneOffset()) {
248 var dstDiff = (dateForDate.getTimezoneOffset() - date.getTimezoneOffset()) *
249 60000;
250 dateForDate = new Date(dateForDate.getTime() + dstDiff);
251
252 diff += diff > 0 ? -goog.date.MS_PER_DAY : goog.date.MS_PER_DAY;
253 dateForTime = new Date(date.getTime() + diff);
254 }
255
256 var out = [];
257 for (var i = 0; i < this.patternParts_.length; ++i) {
258 var text = this.patternParts_[i].text;
259 if (goog.i18n.DateTimeFormat.PartTypes_.FIELD ==
260 this.patternParts_[i].type) {
261 out.push(this.formatField_(text, date, dateForDate, dateForTime,
262 opt_timeZone));
263 } else {
264 out.push(text);
265 }
266 }
267 return out.join('');
268};

271/**
272 * Apply a predefined pattern as identified by formatType, which is stored in
273 * locale specific repository.
274 * @param {number} formatType A number that identified the predefined pattern.
275 * @private
276 */
277goog.i18n.DateTimeFormat.prototype.applyStandardPattern_ =
278 function(formatType) {
279 var pattern;
280 if (formatType < 4) {
281 pattern = this.dateTimeSymbols_.DATEFORMATS[formatType];
282 } else if (formatType < 8) {
283 pattern = this.dateTimeSymbols_.TIMEFORMATS[formatType - 4];
284 } else if (formatType < 12) {
285 pattern = this.dateTimeSymbols_.DATETIMEFORMATS[formatType - 8];
286 pattern = pattern.replace('{1}',
287 this.dateTimeSymbols_.DATEFORMATS[formatType - 8]);
288 pattern = pattern.replace('{0}',
289 this.dateTimeSymbols_.TIMEFORMATS[formatType - 8]);
290 } else {
291 this.applyStandardPattern_(goog.i18n.DateTimeFormat.Format.MEDIUM_DATETIME);
292 return;
293 }
294 this.applyPattern_(pattern);
295};

310/**
311 * Localizes a string potentially containing numbers, replacing ASCII digits
312 * with native digits if specified so by the locale. Leaves other characters.
313 * @param {number|string} input the string to be localized, using ASCII digits.
314 * @param {!Object=} opt_dateTimeSymbols Optional symbols to use use rather than
315 * the global symbols.
316 * @return {string} localized string, potentially using native digits.
317 */
318goog.i18n.DateTimeFormat.localizeNumbers =
319 function(input, opt_dateTimeSymbols) {
320 input = String(input);
321 var dateTimeSymbols = opt_dateTimeSymbols || goog.i18n.DateTimeSymbols;
322 if (dateTimeSymbols.ZERODIGIT === undefined) {
323 return input;
324 }
325
326 var parts = [];
327 for (var i = 0; i < input.length; i++) {
328 var c = input.charCodeAt(i);
329 parts.push((0x30 <= c && c <= 0x39) ? // '0' <= c <= '9'
330 String.fromCharCode(dateTimeSymbols.ZERODIGIT + c - 0x30) :
331 input.charAt(i));
332 }
333 return parts.join('');
334};

353/**
354 * Formats Year field according to pattern specified
355 * Javascript Date object seems incapable handling 1BC and
356 * year before. It can show you year 0 which does not exists.
357 * following we just keep consistent with javascript's
358 * toString method. But keep in mind those things should be
359 * unsupported.
360 * @param {number} count Number of time pattern char repeats, it controls
361 * how a field should be formatted.
362 * @param {!goog.date.DateLike} date It holds the date object to be formatted.
363 * @return {string} Formatted string that represent this field.
364 * @private
365 */
366goog.i18n.DateTimeFormat.prototype.formatYear_ = function(count, date) {
367 var value = date.getFullYear();
368 if (value < 0) {
369 value = -value;
370 }
371 if (count == 2) {
372 // See comment about special casing 'yy' at the start of the file, this
373 // matches ICU and CLDR behaviour. See also:
374 // http://icu-project.org/apiref/icu4j/com/ibm/icu/text/SimpleDateFormat.html
375 // http://www.unicode.org/reports/tr35/tr35-dates.html
376 value = value % 100;
377 }
378 return this.localizeNumbers_(goog.string.padNumber(value, count));
379};

733 * Formatting one date field.
734 * @param {string} patternStr The pattern string for the field being formatted.
735 * @param {!goog.date.DateLike} date represents the real date to be formatted.
736 * @param {!goog.date.DateLike} dateForDate used to resolve date fields
737 * for formatting.
738 * @param {!goog.date.DateLike} dateForTime used to resolve time fields
739 * for formatting.
740 * @param {goog.i18n.TimeZone=} opt_timeZone This holds current time zone info.
741 * @return {string} string representation for the given field.
742 * @private
743 */
744goog.i18n.DateTimeFormat.prototype.formatField_ =
745 function(patternStr, date, dateForDate, dateForTime, opt_timeZone) {
746 var count = patternStr.length;
747 switch (patternStr.charAt(0)) {
748 case 'G': return this.formatEra_(count, dateForDate);
749 case 'y': return this.formatYear_(count, dateForDate);
750 case 'M': return this.formatMonth_(count, dateForDate);
751 case 'k': return this.format24Hours_(count, dateForTime);
752 case 'S': return this.formatFractionalSeconds_(count, dateForTime);
753 case 'E': return this.formatDayOfWeek_(count, dateForDate);
754 case 'a': return this.formatAmPm_(count, dateForTime);
755 case 'h': return this.format1To12Hours_(count, dateForTime);
756 case 'K': return this.format0To11Hours_(count, dateForTime);
757 case 'H': return this.format0To23Hours_(count, dateForTime);
758 case 'c': return this.formatStandaloneDay_(count, dateForDate);
759 case 'L': return this.formatStandaloneMonth_(count, dateForDate);
760 case 'Q': return this.formatQuarter_(count, dateForDate);
761 case 'd': return this.formatDate_(count, dateForDate);
762 case 'm': return this.formatMinutes_(count, dateForTime);
763 case 's': return this.formatSeconds_(count, dateForTime);
764 case 'v': return this.formatTimeZoneId_(date, opt_timeZone);
765 case 'w': return this.formatWeekOfYear_(count, dateForTime);
766 case 'z': return this.formatTimeZone_(count, date, opt_timeZone);
767 case 'Z': return this.formatTimeZoneRFC_(count, date, opt_timeZone);
768 default: return '';
