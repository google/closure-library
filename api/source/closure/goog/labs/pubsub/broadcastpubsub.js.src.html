35/**
36 * Topic-based publish/subscribe messaging implementation that provides
37 * communication between browsing contexts that share the same origin.
38 *
39 * Wrapper around PubSub that utilizes localStorage to broadcast publications to
40 * all browser windows with the same origin as the publishing context. This
41 * allows for topic-based publish/subscribe implementation of strings shared by
42 * all browser contexts that share the same origin.
43 *
44 * Delivery is guaranteed on all browsers except IE8 where topics expire after a
45 * timeout. Publishing of a topic within a callback function provides no
46 * guarantee on ordering in that there is a possiblilty that separate origin
47 * contexts may see topics in a different order.
48 *
49 * This class is not secure and in certain cases (e.g., a browser crash) data
50 * that is published can persist in localStorage indefinitely. Do not use this
51 * class to communicate private or confidential information.
52 *
53 * On IE8, localStorage is shared by the http and https origins. An attacker
54 * could possibly leverage this to publish to the secure origin.
55 *
56 * goog.labs.pubsub.BroadcastPubSub wraps an instance of PubSub rather than
57 * subclassing because the base PubSub class allows publishing of arbitrary
58 * objects.
59 *
60 * Special handling is done for the IE8 browsers. See the IE8_EVENTS_KEY_
61 * constant and the {@code publish} function for more information.
62 *
63 *
64 * @constructor @struct @extends {goog.Disposable}
65 * @suppress {checkStructDictInheritance}
66 */
67goog.labs.pubsub.BroadcastPubSub = function() {
68 goog.labs.pubsub.BroadcastPubSub.base(this, 'constructor');
69 goog.labs.pubsub.BroadcastPubSub.instances_.push(this);
70
71 /** @private @const */
72 this.pubSub_ = new goog.pubsub.PubSub();
73 this.registerDisposable(this.pubSub_);
74
75 /** @private @const */
76 this.handler_ = new goog.events.EventHandler(this);
77 this.registerDisposable(this.handler_);
78
79 /** @private @const */
80 this.logger_ = goog.log.getLogger('goog.labs.pubsub.BroadcastPubSub');
81
82 /** @private @const */
83 this.mechanism_ = new goog.storage.mechanism.HTML5LocalStorage();
84
85 /** @private {goog.storage.Storage} */
86 this.storage_ = null;
87
88 /** @private {Object<string, number>} */
89 this.ie8LastEventTimes_ = null;
90
91 /** @private {number} */
92 this.ie8StartupTimestamp_ = goog.now() - 1;
93
94 if (this.mechanism_.isAvailable()) {
95 this.storage_ = new goog.storage.Storage(this.mechanism_);
96
97 var target = window;
98 if (goog.labs.pubsub.BroadcastPubSub.IS_IE8_) {
99 this.ie8LastEventTimes_ = {};
100
101 target = document;
102 }
103 this.handler_.listen(target,
104 goog.events.EventType.STORAGE,
105 this.handleStorageEvent_);
106 }
107};
108goog.inherits(goog.labs.pubsub.BroadcastPubSub, goog.Disposable);

122/**
123 * Handle the storage event and possibly dispatch topics.
124 * @param {!goog.events.Event} e Event object.
125 * @private
126 */
127goog.labs.pubsub.BroadcastPubSub.prototype.handleStorageEvent_ =
128 function(e) {
129 if (goog.labs.pubsub.BroadcastPubSub.IS_IE8_) {
130 // Even though we have the event, IE8 doesn't update our localStorage until
131 // after we handle the actual event.
132 goog.async.run(this.handleIe8StorageEvent_, this);
133 return;
134 }
135
136 var browserEvent = e.getBrowserEvent();
137 if (browserEvent.key !=
138 goog.labs.pubsub.BroadcastPubSub.STORAGE_KEY_) {
139 return;
140 }
141
142 var data = goog.json.parse(browserEvent.newValue);
143 var args = goog.isObject(data) && data['args'];
144 if (goog.isArray(args) && goog.array.every(args, goog.isString)) {
145 this.dispatch_(args);
146 } else {
147 goog.log.warning(this.logger_, 'storage event contained invalid arguments');
148 }
149};

162/**
163 * Publishes a message to a topic. Remote subscriptions in other tabs/windows
164 * are dispatched via local storage events. Local subscriptions are called
165 * asynchronously via Timer event in order to simulate remote behavior locally.
166 * @param {string} topic Topic to publish to.
167 * @param {...string} var_args String arguments that are applied to each
168 * subscription function.
169 */
170goog.labs.pubsub.BroadcastPubSub.prototype.publish =
171 function(topic, var_args) {
172 var args = goog.array.toArray(arguments);
173
174 // Dispatch to localStorage.
175 if (this.storage_) {
176 // Update topics to use the optional prefix.
177 var now = goog.now();
178 var data = {
179 'args': args,
180 'timestamp': now
181 };
182
183 if (!goog.labs.pubsub.BroadcastPubSub.IS_IE8_) {
184 // Generated events will contain all the data in modern browsers.
185 this.storage_.set(goog.labs.pubsub.BroadcastPubSub.STORAGE_KEY_, data);
186 this.storage_.remove(goog.labs.pubsub.BroadcastPubSub.STORAGE_KEY_);
187 } else {
188 // With IE8 we need to manage our own events queue.
189 var events = null;
190 /** @preserveTry */
191 try {
192 events = this.storage_.get(
193 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
194 } catch (ex) {
195 goog.log.error(this.logger_,
196 'publish encountered invalid event queue at ' +
197 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
198 }
199 if (!goog.isArray(events)) {
200 events = [];
201 }
202 // Avoid a race condition where we're publishing in the same
203 // millisecond that another event that may be getting
204 // processed. In short, we try go guarantee that whatever event
205 // we put on the event queue has a timestamp that is older than
206 // any other timestamp in the queue.
207 var lastEvent = events[events.length - 1];
208 var lastTimestamp = lastEvent && lastEvent['timestamp'] ||
209 this.ie8StartupTimestamp_;
210 if (lastTimestamp >= now) {
211 now = lastTimestamp +
212 goog.labs.pubsub.BroadcastPubSub.IE8_TIMESTAMP_UNIQUE_OFFSET_MS_;
213 data['timestamp'] = now;
214 }
215 events.push(data);
216 this.storage_.set(
217 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_, events);
218
219 // Cleanup this event in IE8_EVENT_LIFETIME_MS_ milliseconds.
220 goog.Timer.callOnce(goog.bind(this.cleanupIe8StorageEvents_, this, now),
221 goog.labs.pubsub.BroadcastPubSub.IE8_EVENT_LIFETIME_MS_);
222 }
223 }
224
225 // W3C spec is to not dispatch the storage event to the same window that
226 // modified localStorage. For conforming browsers we have to manually dispatch
227 // the publish event to subscriptions on instances of BroadcastPubSub in the
228 // current window.
229 if (!goog.userAgent.IE) {
230 // Dispatch the publish event to local instances asynchronously to fix some
231 // quirks with timings. The result is that all subscriptions are dispatched
232 // before any future publishes are processed. The effect is that
233 // subscriptions in the same window are dispatched as if they are the result
234 // of a publish from another tab.
235 goog.array.forEach(goog.labs.pubsub.BroadcastPubSub.instances_,
236 function(instance) {
237 goog.async.run(goog.bind(instance.dispatch_, instance, args));
238 });
239 }
240};

372/**
373 * Name for this window/tab's storage key that stores its IE8 event queue.
374 *
375 * The browsers storage events are supposed to track the key which was changed,
376 * the previous value for that key, and the new value of that key. Our
377 * implementation is dependent on this information but IE8 doesn't provide it.
378 * We implement our own event queue using local storage to track this
379 * information in IE8. Since all instances share the same localStorage context
380 * in a particular tab, we share the events queue.
381 *
382 * This key is a static member shared by all instances of BroadcastPubSub in the
383 * same Window context. To avoid read-update-write contention, this key is only
384 * written in a single context in the cleanupIe8StorageEvents_ function. Since
385 * instances in other contexts will read this key there is code in the {@code
386 * publish} function to make sure timestamps are unique even within the same
387 * millisecond.
388 *
389 * @private @const
390 */
391goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_ =
392 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_PREFIX_ +
393 goog.math.randomInt(1e9);

459/**
460 * Processes the events array for key if all elements are valid IE8 events.
461 * @param {string} key The key in localStorage where the event queue is stored.
462 * @param {!Array<!Object>} events Array of possible events stored at key.
463 * @return {boolean} Return true if all elements in the array are valid
464 * events, false otherwise.
465 * @private
466 */
467goog.labs.pubsub.BroadcastPubSub.prototype.maybeProcessIe8Events_ =
468 function(key, events) {
469 if (!events.length) {
470 return false;
471 }
472
473 var validEvents =
474 goog.labs.pubsub.BroadcastPubSub.filterValidIe8Events_(events);
475 if (validEvents.length == events.length) {
476 var lastTimestamp = goog.array.peek(validEvents)['timestamp'];
477 var previousTime =
478 this.ie8LastEventTimes_[key] || this.ie8StartupTimestamp_;
479 if (lastTimestamp > previousTime -
480 goog.labs.pubsub.BroadcastPubSub.IE8_QUEUE_LIFETIME_MS_) {
481 this.ie8LastEventTimes_[key] = lastTimestamp;
482 validEvents = goog.labs.pubsub.BroadcastPubSub.filterNewIe8Events_(
483 previousTime, validEvents);
484 for (var i = 0, event; event = validEvents[i]; i++) {
485 this.dispatch_(event['args']);
486 }
487 return true;
488 }
489 } else {
490 goog.log.warning(this.logger_, 'invalid events found in queue ' + key);
491 }
492
493 return false;
494};

497/**
498 * Handle the storage event and possibly dispatch events. Looks through all keys
499 * in localStorage for valid keys.
500 * @private
501 */
502goog.labs.pubsub.BroadcastPubSub.prototype.handleIe8StorageEvent_ = function() {
503 var numKeys = this.mechanism_.getCount();
504 for (var idx = 0; idx < numKeys; idx++) {
505 var key = this.mechanism_.key(idx);
506 // Don't process events we generated. The W3C standard says that storage
507 // events should be queued by the browser for each window whose document's
508 // storage object is affected by a change in localStorage. Chrome, Firefox,
509 // and modern IE don't dispatch the event to the window which made the
510 // change. This code simulates that behavior in IE8.
511 if (!(goog.isString(key) && goog.string.startsWith(
512 key, goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_PREFIX_))) {
513 continue;
514 }
515
516 var events = null;
517 /** @preserveTry */
518 try {
519 events = this.storage_.get(key);
520 } catch (ex) {
521 goog.log.warning(this.logger_, 'invalid remote event queue ' + key);
522 }
523
524 if (!(goog.isArray(events) && this.maybeProcessIe8Events_(key, events))) {
525 // Events is not an array, empty, contains invalid events, or expired.
526 this.storage_.remove(key);
527 }
528 }

532/**
533 * Cleanup our IE8 event queue by removing any events that come at or before the
534 * given timestamp.
535 * @param {number} timestamp Maximum timestamp to remove from the queue.
536 * @private
537 */
538goog.labs.pubsub.BroadcastPubSub.prototype.cleanupIe8StorageEvents_ =
539 function(timestamp) {
540 var events = null;
541 /** @preserveTry */
542 try {
543 events = this.storage_.get(
544 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
545 } catch (ex) {
546 goog.log.error(this.logger_,
547 'cleanup encountered invalid event queue key ' +
548 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
549 }
550 if (!goog.isArray(events)) {
551 this.storage_.remove(goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
552 return;
553 }
554
555 events = goog.labs.pubsub.BroadcastPubSub.filterNewIe8Events_(
556 timestamp, goog.labs.pubsub.BroadcastPubSub.filterValidIe8Events_(
557 events));
558
559 if (events.length > 0) {
560 this.storage_.set(goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_, events);
561 } else {
562 this.storage_.remove(goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
563 }
