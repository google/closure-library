32/**
33 * Replaces tags not on the white-list with empty text nodes, dropping all
34 * attributes, and drops other non-text nodes such as comments.
35 *
36 * @param {!Object<string, boolean>} tagWhitelist a set of lower-case tag names
37 * following the convention established by {@link goog.object.createSet}.
38 * @param {!Object<string, Object<string, goog.labs.html.AttributeRewriter>>}
39 * attrWhitelist
40 * maps lower-case tag names and the special string {@code "*"} to functions
41 * from decoded attribute values to sanitized values or {@code null} to
42 * indicate that the attribute is not allowed with that value.
43 *
44 * For example, if {@code attrWhitelist['a']['href']} is defined then it
45 * is used to sanitize the value of the link's URL.
46 *
47 * If {@code attrWhitelist['*']['id']} is defined, and
48 * {@code attrWhitelist['div']['id']} is not, then the former is used to
49 * sanitize any {@code id} attribute on a {@code <div>} element.
50 * @param {string} html a string of HTML
51 * @return {string} the input but with potentially dangerous tokens removed.
52 */
53goog.labs.html.scrubber.scrub = function(tagWhitelist, attrWhitelist, html) {
54 return goog.labs.html.scrubber.render_(
55 goog.labs.html.scrubber.balance_(
56 goog.labs.html.scrubber.filter_(
57 tagWhitelist,
58 attrWhitelist,
59 goog.labs.html.scrubber.lex_(html))));
60};

63/**
64 * Balances tags in trusted HTML.
65 * @param {string} html a string of HTML
66 * @return {string} the input but with an end-tag for each non-void start tag
67 * and only for non-void start tags, and with start and end tags nesting
68 * properly.
69 */
70goog.labs.html.scrubber.balance = function(html) {
71 return goog.labs.html.scrubber.render_(
72 goog.labs.html.scrubber.balance_(
73 goog.labs.html.scrubber.lex_(html)));
74};

150/**
151 * Matches the open tag and body of a special element :
152 * one whose body cannot contain nested elements so uses special parsing rules.
153 * It does not include the end tag.
154 * @private
155 */
156goog.labs.html.scrubber.SPECIAL_ELEMENT_ = '<(?:' +
157 // Special tag name.
158 '(iframe|script|style|textarea|title|xmp)' +
159 // End of tag name
160 goog.labs.html.scrubber.BREAK_ +
161 // Attributes
162 goog.labs.html.scrubber.ATTRS_ + '>' +
163 // Element content includes non '<' characters, and
164 // '<' that don't start a matching end tag.
165 // This uses a back-reference to the tag name to determine whether
166 // the tag names match.
167 // Since matching is case-insensitive, this can only be used in
168 // a case-insensitive regular expression.
169 // JavaScript does not treat Turkish dotted I's as equivalent to their
170 // ASCII equivalents.
171 '(?:[^<]|<(?!/\\1' + goog.labs.html.scrubber.BREAK_ + '))*' +
172 ')';

248/**
249 * Replaces tags not on the white-list with empty text nodes, dropping all
250 * attributes, and drops other non-text nodes such as comments.
251 *
252 * @param {!Object<string, boolean>} tagWhitelist a set of lower-case tag names
253 * following the convention established by {@link goog.object.createSet}.
254 * @param {!Object<string, Object<string, goog.labs.html.AttributeRewriter>>
255 * } attrWhitelist
256 * maps lower-case tag names and the special string {@code "*"} to functions
257 * from decoded attribute values to sanitized values or {@code null} to
258 * indicate that the attribute is not allowed with that value.
259 *
260 * For example, if {@code attrWhitelist['a']['href']} is defined then it is
261 * used to sanitize the value of the link's URL.
262 *
263 * If {@code attrWhitelist['*']['id']} is defined, and
264 * {@code attrWhitelist['div']['id']} is not, then the former is used to
265 * sanitize any {@code id} attribute on a {@code <div>} element.
266 * @param {!Array<string>} htmlTokens an array of HTML tokens as returned by
267 * {@link goog.labs.html.scrubber.lex_}.
268 * @return {!Array<string>} the input array modified in place to have some
269 * tokens removed.
270 * @private
271 */
272goog.labs.html.scrubber.filter_ = function(
273 tagWhitelist, attrWhitelist, htmlTokens) {
274 var genericAttrWhitelist = attrWhitelist['*'];
275 for (var i = 0, n = htmlTokens.length; i < n; ++i) {
276 var htmlToken = htmlTokens[i];
277 if (htmlToken.charCodeAt(0) !== goog.labs.html.scrubber.CC_LT_) {
278 // Definitely not a tag
279 continue;
280 }
281
282 var tag = htmlToken.match(goog.labs.html.scrubber.TAG_RE_);
283 if (tag) {
284 var lowerCaseTagName = tag[1].toLowerCase();
285 var isCloseTag =
286 htmlToken.charCodeAt(1) === goog.labs.html.scrubber.CC_SLASH_;
287 var attrs = '';
288 if (!isCloseTag && tag[2]) {
289 var tagSpecificAttrWhitelist =
290 /** @type {Object<string, goog.labs.html.AttributeRewriter>} */ (
291 goog.labs.html.scrubber.readOwnProperty_(
292 attrWhitelist, lowerCaseTagName));
293 if (genericAttrWhitelist || tagSpecificAttrWhitelist) {
294 attrs = goog.labs.html.scrubber.filterAttrs_(
295 tag[2], genericAttrWhitelist, tagSpecificAttrWhitelist);
296 }
297 }
298 var specialContent = htmlToken.substring(tag[0].length);
299 htmlTokens[i] =
300 (tagWhitelist[lowerCaseTagName] === true) ?
301 (
302 (isCloseTag ? '</' : '<') + lowerCaseTagName + attrs + '>' +
303 specialContent
304 ) :
305 '';
306 } else if (htmlToken.length > 1) {
307 switch (htmlToken.charCodeAt(1)) {
308 case goog.labs.html.scrubber.CC_BANG_:
309 case goog.labs.html.scrubber.CC_SLASH_:
310 case goog.labs.html.scrubber.CC_QMARK_:
311 htmlTokens[i] = ''; // Elide comments.
312 break;
313 default:
314 // Otherwise, token is just a text node that starts with '<'.
315 // Speed up later passes by normalizing the text node.
316 htmlTokens[i] = htmlTokens[i].replace(/</g, '&lt;');
317 }
318 }
319 }
320 return htmlTokens;
321};

324/**
325 * Parses attribute names and values out of a tag body and applies the attribute
326 * white-list to produce a tag body containing only safe attributes.
327 *
328 * @param {string} attrsText the text of a tag between the end of the tag name
329 * and the beginning of the tag end marker, so {@code " foo bar='baz'"} for
330 * the tag {@code <tag foo bar='baz'/>}.
331 * @param {Object<string, goog.labs.html.AttributeRewriter>}
332 * genericAttrWhitelist
333 * a whitelist of attribute transformations for attributes that are allowed
334 * on any element.
335 * @param {Object<string, goog.labs.html.AttributeRewriter>}
336 * tagSpecificAttrWhitelist
337 * a whitelist of attribute transformations for attributes that are allowed
338 * on the element started by the tag whose body is {@code tagBody}.
339 * @return {string} a tag-body that consists only of safe attributes.
340 * @private
341 */
342goog.labs.html.scrubber.filterAttrs_ =
343 function(attrsText, genericAttrWhitelist, tagSpecificAttrWhitelist) {
344 var attrs = attrsText.match(goog.labs.html.scrubber.ATTRS_RE_);
345 var nAttrs = attrs ? attrs.length : 0;
346 var safeAttrs = '';
347 for (var i = 0; i < nAttrs; ++i) {
348 var attr = attrs[i];
349 var eq = attr.indexOf('=');
350 var name, value;
351 if (eq >= 0) {
352 name = goog.string.trim(attr.substr(0, eq));
353 value = goog.string.stripQuotes(
354 goog.string.trim(attr.substr(eq + 1)), '"\'');
355 } else {
356 name = value = attr;
357 }
358 name = name.toLowerCase();
359 var rewriter = /** @type {?goog.labs.html.AttributeRewriter} */ (
360 tagSpecificAttrWhitelist &&
361 goog.labs.html.scrubber.readOwnProperty_(
362 tagSpecificAttrWhitelist, name) ||
363 genericAttrWhitelist &&
364 goog.labs.html.scrubber.readOwnProperty_(genericAttrWhitelist, name));
365 if (rewriter) {
366 var safeValue = rewriter(goog.string.unescapeEntities(value));
367 if (safeValue != null) {
368 if (safeValue.implementsGoogStringTypedString) {
369 safeValue = /** @type {goog.string.TypedString} */
370 (safeValue).getTypedStringValue();
371 }
372 safeValue = String(safeValue);
373 if (safeValue.indexOf('`') >= 0) {
374 safeValue += ' ';
375 }
376 safeAttrs +=
377 ' ' + name + '="' + goog.string.htmlEscape(safeValue, false) +
378 '"';
379 }
380 }
381 }
382 return safeAttrs;
383};

397/**
398 * We limit the nesting limit of balanced HTML to a large but manageable number
399 * so that built-in browser limits aren't likely to kick in and undo all our
400 * matching of start and end tags.
401 * <br>
402 * This mitigates the HTML parsing equivalent of stack smashing attacks.
403 * <br>
404 * Otherwise, crafted inputs like
405 * {@code <p><p><p><p>...Ad nauseam..</p></p></p></p>} could exploit
406 * browser bugs, and/or undocumented nesting limit recovery code to misnest
407 * tags.
408 * @private
409 * @const
410 */
411goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_ = 256;
412
413
414/**
415 * Ensures that there are end-tags for all and only for non-void start tags.
416 * @param {Array<string>} htmlTokens an array of HTML tokens as returned by
417 * {@link goog.labs.html.scrubber.lex}.
418 * @return {!Array<string>} the input array modified in place to have some
419 * tokens removed.
420 * @private
421 */
422goog.labs.html.scrubber.balance_ = function(htmlTokens) {
423 var openElementStack = [];
424 for (var i = 0, n = htmlTokens.length; i < n; ++i) {
425 var htmlToken = htmlTokens[i];
426 if (htmlToken.charCodeAt(0) !== goog.labs.html.scrubber.CC_LT_) {
427 // Definitely not a tag
428 continue;
429 }
430 var tag = htmlToken.match(goog.labs.html.scrubber.TAG_RE_);
431 if (tag) {
432 var lowerCaseTagName = tag[1].toLowerCase();
433 var isCloseTag =
434 htmlToken.charCodeAt(1) === goog.labs.html.scrubber.CC_SLASH_;
435 // Special case: HTML5 mandates that </br> be treated as <br>.
436 if (isCloseTag && lowerCaseTagName == 'br') {
437 isCloseTag = false;
438 htmlToken = '<br>';
439 }
440 var isVoidTag = goog.dom.tags.isVoidTag(lowerCaseTagName);
441 if (isVoidTag && isCloseTag) {
442 htmlTokens[i] = '';
443 continue;
444 }
445
446 var prefix = '';
447
448 // Insert implied open tags.
449 var nOpenElements = openElementStack.length;
450 if (nOpenElements && !isCloseTag) {
451 var top = openElementStack[nOpenElements - 1];
452 var groups = goog.labs.html.scrubber.ELEMENT_GROUPS_[lowerCaseTagName];
453 if (groups === undefined) {
454 groups = goog.labs.html.scrubber.Group_.INLINE_;
455 }
456 var canContain = goog.labs.html.scrubber.ELEMENT_CONTENTS_[top];
457 if (!(groups & canContain)) {
458 var blockContainer = goog.labs.html.scrubber.BLOCK_CONTAINERS_[top];
459 if ('string' === typeof blockContainer) {
460 var containerCanContain =
461 goog.labs.html.scrubber.ELEMENT_CONTENTS_[blockContainer];
462 if (containerCanContain & groups) {
463 if (nOpenElements <
464 goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_) {
465 prefix = '<' + blockContainer + '>';
466 }
467 openElementStack[nOpenElements] = blockContainer;
468 ++nOpenElements;
469 }
470 }
471 }
472 }
473
474 // Insert any missing close tags we need.
475 var newStackLen = goog.labs.html.scrubber.pickElementsToClose_(
476 lowerCaseTagName, isCloseTag, openElementStack);
477
478 var nClosed = nOpenElements - newStackLen;
479 if (nClosed) { // ["p", "a", "b"] -> "</b></a></p>"
480 // First, dump anything past the nesting limit.
481 if (nOpenElements > goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_) {
482 nClosed -= nOpenElements -
483 goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_;
484 nOpenElements = goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_;
485 }
486 // Truncate to the new limit, and produce end tags.
487 var closeTags = openElementStack.splice(newStackLen, nClosed);
488 if (closeTags.length) {
489 closeTags.reverse();
490 prefix += '</' + closeTags.join('></') + '>';
491 }
492 }
493
494 // We could do resumption here to handle misnested tags like
495 // <b><i class="c">Foo</b>Bar</i>
496 // which is equivalent to
497 // <b><i class="c">Foo</i></b><i class="c">Bar</i>
498 // but that requires storing attributes on the open element stack
499 // which complicates all the code using it for marginal added value.
500
501 if (isCloseTag) {
502 // If the close tag matched an open tag, then the closed section
503 // included that tag name.
504 htmlTokens[i] = prefix;
505 } else {
506 if (!isVoidTag) {
507 openElementStack[openElementStack.length] = lowerCaseTagName;
508 }
509 if (openElementStack.length >
510 goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_) {
511 htmlToken = '';
512 }
513 htmlTokens[i] = prefix + htmlToken;
514 }
515 }
516 }
517 if (openElementStack.length) {
518 if (openElementStack.length >
519 goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_) {
520 openElementStack.length = goog.labs.html.scrubber.BALANCE_NESTING_LIMIT_;
521 }
522 if (openElementStack.length) {
523 openElementStack.reverse();
524 htmlTokens[htmlTokens.length] = '</' + openElementStack.join('></') + '>';
525 }
526 }
527 return htmlTokens;
528};

531/**
532 * Normalizes HTML tokens and concatenates them into a string.
533 * @param {Array<string>} htmlTokens an array of HTML tokens as returned by
534 * {@link goog.labs.html.scrubber.lex}.
535 * @return {string} a string of HTML.
536 * @private
537 */
538goog.labs.html.scrubber.render_ = function(htmlTokens) {
539 for (var i = 0, n = htmlTokens.length; i < n; ++i) {
540 var htmlToken = htmlTokens[i];
541 if (htmlToken.charCodeAt(0) === goog.labs.html.scrubber.CC_LT_ &&
542 goog.labs.html.scrubber.TAG_RE_.test(htmlToken)) {
543 // The well-formedness and quotedness of attributes must be ensured by
544 // earlier passes. filter does this.
545 } else {
546 if (htmlToken.indexOf('<') >= 0) {
547 htmlToken = htmlToken.replace(/</g, '&lt;');
548 }
549 if (htmlToken.indexOf('>') >= 0) {
550 htmlToken = htmlToken.replace(/>/g, '&gt;');
551 }
552 htmlTokens[i] = htmlToken;
553 }
554 }
555 return htmlTokens.join('');
556};

614/**
615 * Picks which open HTML elements to close.
616 *
617 * @param {string} lowerCaseTagName The name of the tag.
618 * @param {boolean} isCloseTag True for a {@code </tagname>} tag.
619 * @param {Array<string>} openElementStack The names of elements that have been
620 * opened and not subsequently closed.
621 * @return {number} the length of openElementStack after closing any tags that
622 * need to be closed.
623 * @private
624 */
625goog.labs.html.scrubber.pickElementsToClose_ =
626 function(lowerCaseTagName, isCloseTag, openElementStack) {
627 var nOpenElements = openElementStack.length;
628 if (isCloseTag) {
629 // Look for a matching close tag inside blocking scopes.
630 var topMost;
631 if (/^h[1-6]$/.test(lowerCaseTagName)) {
632 // </h1> will close any header.
633 topMost = -1;
634 for (var i = nOpenElements; --i >= 0;) {
635 if (/^h[1-6]$/.test(openElementStack[i])) {
636 topMost = i;
637 }
638 }
639 } else {
640 topMost = goog.array.lastIndexOf(openElementStack, lowerCaseTagName);
641 }
642 if (topMost >= 0) {
643 var blockers = goog.labs.html.scrubber.ALL_SCOPES_ &
644 ~(goog.labs.html.scrubber.ELEMENT_SCOPES_[lowerCaseTagName] | 0);
645 for (var i = nOpenElements; --i > topMost;) {
646 var blocks =
647 goog.labs.html.scrubber.ELEMENT_SCOPES_[openElementStack[i]] | 0;

650 }

652 return topMost;
653 }
654 return nOpenElements;
655 } else {
656 // Close anything that cannot contain the tag name.
657 var groups = goog.labs.html.scrubber.ELEMENT_GROUPS_[lowerCaseTagName];
658 if (groups === undefined) {
659 groups = goog.labs.html.scrubber.Group_.INLINE_;
660 }
661 for (var i = nOpenElements; --i >= 0;) {
662 var canContain =

664 if (canContain === undefined) {
665 canContain = goog.labs.html.scrubber.Group_.INLINE_;
666 }
667 if (groups & canContain) {
668 return i + 1;
669 }
670 }
671 return 0;
672 }
