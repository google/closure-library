30/**
31 * A TestChannel is used during the first part of channel negotiation
32 * with the server to create the channel. It helps us determine whether we're

150/**
151 * Starts the test channel. This initiates connections to the server.
152 *
153 * @param {string} path The relative uri for the test connection.
154 */
155BaseTestChannel.prototype.connect = function(path) {
156 this.path_ = path;
157 var sendDataUri = this.channel_.getForwardChannelUri(this.path_);
158
159 requestStats.notifyStatEvent(requestStats.Stat.TEST_STAGE_ONE_START);
160
161 // If the channel already has the result of the handshake, then skip it.
162 var handshakeResult = this.channel_.getConnectionState().handshakeResult;
163 if (goog.isDefAndNotNull(handshakeResult)) {
164 this.hostPrefix_ = this.channel_.correctHostPrefix(handshakeResult[0]);
165 this.state_ = BaseTestChannel.State_.CONNECTION_TESTING;
166 this.checkBufferingProxy_();
167 return;
168 }
169
170 // the first request returns server specific parameters
171 sendDataUri.setParameterValues('MODE', 'init');
172 this.request_ = ChannelRequest.createChannelRequest(this, this.channelDebug_);
173 this.request_.setExtraHeaders(this.extraHeaders_);
174 this.request_.xmlHttpGet(sendDataUri, false /* decodeChunks */,
175 null /* hostPrefix */, true /* opt_noClose */);
176 this.state_ = BaseTestChannel.State_.INIT;
177};
178
179
180/**
181 * Begins the second stage of the test channel where we test to see if we're
182 * behind a buffering proxy. The server sends back a multi-chunked response
183 * with the first chunk containing the content '1' and then two seconds later
184 * sending the second chunk containing the content '2'. Depending on how we
185 * receive the content, we can tell if we're behind a buffering proxy.
186 * @private
187 */
188BaseTestChannel.prototype.checkBufferingProxy_ = function() {
189 this.channelDebug_.debug('TestConnection: starting stage 2');
190
191 // If the test result is already available, skip its execution.
192 var bufferingProxyResult =
193 this.channel_.getConnectionState().bufferingProxyResult;
194 if (goog.isDefAndNotNull(bufferingProxyResult)) {
195 this.channelDebug_.debug(
196 'TestConnection: skipping stage 2, precomputed result is ' +
197 bufferingProxyResult ? 'Buffered' : 'Unbuffered');
198 requestStats.notifyStatEvent(requestStats.Stat.TEST_STAGE_TWO_START);
199 if (bufferingProxyResult) { // Buffered/Proxy connection
200 requestStats.notifyStatEvent(requestStats.Stat.PROXY);
201 this.channel_.testConnectionFinished(this, false);
202 } else { // Unbuffered/NoProxy connection
203 requestStats.notifyStatEvent(requestStats.Stat.NOPROXY);
204 this.channel_.testConnectionFinished(this, true);
205 }
206 return; // Skip the test
207 }
208 this.request_ = ChannelRequest.createChannelRequest(this, this.channelDebug_);
209 this.request_.setExtraHeaders(this.extraHeaders_);
210 var recvDataUri = this.channel_.getBackChannelUri(this.hostPrefix_,
211 /** @type {string} */ (this.path_));
212
213 requestStats.notifyStatEvent(requestStats.Stat.TEST_STAGE_TWO_START);
214 recvDataUri.setParameterValues('TYPE', 'xmlhttp');
215 this.request_.xmlHttpGet(recvDataUri, false /** decodeChunks */,
216 this.hostPrefix_, false /** opt_noClose */);
217};

252/**
253 * Callback from ChannelRequest for when new data is received
254 *
255 * @param {ChannelRequest} req The request object.
256 * @param {string} responseText The text of the response.
257 * @override
258 */
259BaseTestChannel.prototype.onRequestData = function(req, responseText) {
260 this.lastStatusCode_ = req.getLastStatusCode();
261 if (this.state_ == BaseTestChannel.State_.INIT) {
262 this.channelDebug_.debug('TestConnection: Got data for stage 1');
263 if (!responseText) {
264 this.channelDebug_.debug('TestConnection: Null responseText');
265 // The server should always send text; something is wrong here
266 this.channel_.testConnectionFailure(this, ChannelRequest.Error.BAD_DATA);
267 return;
268 }
269 /** @preserveTry */
270 try {
271 var respArray = this.channel_.getWireCodec().decodeMessage(responseText);
272 } catch (e) {
273 this.channelDebug_.dumpException(e);
274 this.channel_.testConnectionFailure(this, ChannelRequest.Error.BAD_DATA);
275 return;
276 }
277 this.hostPrefix_ = this.channel_.correctHostPrefix(respArray[0]);
278 } else if (this.state_ == BaseTestChannel.State_.CONNECTION_TESTING) {
279 if (this.receivedIntermediateResult_) {
280 requestStats.notifyStatEvent(requestStats.Stat.TEST_STAGE_TWO_DATA_TWO);
281 } else {
282 // '11111' is used instead of '1' to prevent a small amount of buffering
283 // by Safari.
284 if (responseText == '11111') {
285 requestStats.notifyStatEvent(requestStats.Stat.TEST_STAGE_TWO_DATA_ONE);
286 this.receivedIntermediateResult_ = true;
287 if (this.checkForEarlyNonBuffered_()) {
288 // If early chunk detection is on, and we passed the tests,
289 // assume HTTP_OK, cancel the test and turn on noproxy mode.
290 this.lastStatusCode_ = 200;
291 this.request_.cancel();
292 this.channelDebug_.debug(
293 'Test connection succeeded; using streaming connection');
294 requestStats.notifyStatEvent(requestStats.Stat.NOPROXY);
295 this.channel_.testConnectionFinished(this, true);
296 }
297 } else {
298 requestStats.notifyStatEvent(
299 requestStats.Stat.TEST_STAGE_TWO_DATA_BOTH);
300 this.receivedIntermediateResult_ = false;
301 }
302 }
303 }
304};

307/**
308 * Callback from ChannelRequest that indicates a request has completed.
309 *
310 * @param {!ChannelRequest} req The request object.
311 * @override
312 */
313BaseTestChannel.prototype.onRequestComplete = function(req) {
314 this.lastStatusCode_ = this.request_.getLastStatusCode();
315 if (!this.request_.getSuccess()) {
316 this.channelDebug_.debug(
317 'TestConnection: request failed, in state ' + this.state_);
318 if (this.state_ == BaseTestChannel.State_.INIT) {
319 requestStats.notifyStatEvent(requestStats.Stat.TEST_STAGE_ONE_FAILED);
320 } else if (this.state_ == BaseTestChannel.State_.CONNECTION_TESTING) {
321 requestStats.notifyStatEvent(requestStats.Stat.TEST_STAGE_TWO_FAILED);
322 }
323 this.channel_.testConnectionFailure(this,
324 /** @type {ChannelRequest.Error} */
325 (this.request_.getLastError()));
326 return;
327 }
328
329 if (this.state_ == BaseTestChannel.State_.INIT) {
330 this.recordClientProtocol_(req);
331 this.state_ = BaseTestChannel.State_.CONNECTION_TESTING;
332
333 this.channelDebug_.debug(
334 'TestConnection: request complete for initial check');
335
336 this.checkBufferingProxy_();
337 } else if (this.state_ == BaseTestChannel.State_.CONNECTION_TESTING) {
338 this.channelDebug_.debug('TestConnection: request complete for stage 2');
339
340 var goodConn = this.receivedIntermediateResult_;
341 if (goodConn) {
342 this.channelDebug_.debug(
343 'Test connection succeeded; using streaming connection');
344 requestStats.notifyStatEvent(requestStats.Stat.NOPROXY);
345 this.channel_.testConnectionFinished(this, true);
346 } else {
347 this.channelDebug_.debug(
348 'Test connection failed; not using streaming');
349 requestStats.notifyStatEvent(requestStats.Stat.PROXY);
350 this.channel_.testConnectionFinished(this, false);
351 }
352 }

356/**
357 * Record the client protocol header from the initial handshake response.
358 *
359 * @param {!ChannelRequest} req The request object.

362BaseTestChannel.prototype.recordClientProtocol_ = function(req) {
363 var xmlHttp = req.getXhr();

366 this.clientProtocol_ = protocolHeader ? protocolHeader : null;
