15/**
16 * @fileoverview The module loader for loading modules across the network.
17 *
18 * Browsers do not guarantee that scripts appended to the document
19 * are executed in the order they are added. For production mode, we use
20 * XHRs to load scripts, because they do not have this problem and they
21 * have superior mechanisms for handling failure. However, XHR-evaled
22 * scripts are harder to debug.
23 *
24 * In debugging mode, we use normal script tags. In order to make this work,
25 * we load the scripts in serial: we do not execute script B to the document
26 * until we are certain that script A is finished loading.
27 *
28 */

136/**
137 * When enabled, we will add a sourceURL comment to the end of all scripts
138 * to mark their origin.
139 *
140 * On WebKit, stack traces will refect the sourceURL comment, so this is
141 * useful for debugging webkit stack traces in production.
142 *
143 * Notice that in debug mode, we will use source url injection + eval rather
144 * then appending script nodes to the DOM, because the scripts will load far
145 * faster. (Appending script nodes is very slow, because we can't parallelize
146 * the downloading and evaling of the script).
147 *
148 * The cost of appending sourceURL information is negligible when compared to
149 * the cost of evaling the script. Almost all clients will want this on.
150 *
151 * TODO(nicksantos): Turn this on by default. We may want to turn this off
152 * for clients that inject their own sourceURL.
153 *
154 * @param {boolean} enabled Whether source url injection is enabled.
155 */
156goog.module.ModuleLoader.prototype.setSourceUrlInjection = function(enabled) {
157 this.sourceUrlInjection_ = enabled;
158};

172/** @override */
173goog.module.ModuleLoader.prototype.loadModules = function(
174 ids, moduleInfoMap, opt_successFn, opt_errorFn, opt_timeoutFn,
175 opt_forceReload) {
176 var loadStatus = this.loadingModulesStatus_[ids] ||
177 new goog.module.ModuleLoader.LoadStatus();
178 loadStatus.loadRequested = true;
179 loadStatus.successFn = opt_successFn || null;
180 loadStatus.errorFn = opt_errorFn || null;
181
182 if (!this.loadingModulesStatus_[ids]) {
183 // Modules were not prefetched.
184 this.loadingModulesStatus_[ids] = loadStatus;
185 this.downloadModules_(ids, moduleInfoMap);
186 // TODO(user): Need to handle timeouts in the module loading code.
187 } else if (goog.isDefAndNotNull(loadStatus.responseTexts)) {
188 // Modules prefetch is complete.
189 this.evaluateCode_(ids);
190 }
191 // Otherwise modules prefetch is in progress, and these modules will be
192 // executed after the prefetch is complete.
193};

196/**
197 * Evaluate the JS code.
198 * @param {Array<string>} moduleIds The module ids.
199 * @private
200 */
201goog.module.ModuleLoader.prototype.evaluateCode_ = function(moduleIds) {
202 this.dispatchEvent(new goog.module.ModuleLoader.Event(
203 goog.module.ModuleLoader.EventType.REQUEST_SUCCESS, moduleIds));
204
205 goog.log.info(this.logger, 'evaluateCode ids:' + moduleIds);
206 var success = true;
207 var loadStatus = this.loadingModulesStatus_[moduleIds];
208 var uris = loadStatus.requestUris;
209 var texts = loadStatus.responseTexts;
210 try {
211 if (this.usingSourceUrlInjection_()) {
212 for (var i = 0; i < uris.length; i++) {
213 var uri = uris[i];
214 goog.globalEval(texts[i] + ' //@ sourceURL=' + uri);
215 }
216 } else {
217 goog.globalEval(texts.join('\n'));
218 }
219 } catch (e) {
220 success = false;
221 // TODO(user): Consider throwing an exception here.
222 goog.log.warning(this.logger, 'Loaded incomplete code for module(s): ' +
223 moduleIds, e);
224 }
225
226 this.dispatchEvent(
227 new goog.module.ModuleLoader.Event(
228 goog.module.ModuleLoader.EventType.EVALUATE_CODE, moduleIds));
229
230 if (!success) {
231 this.handleErrorHelper_(moduleIds, loadStatus.errorFn, null /* status */);
232 } else if (loadStatus.successFn) {
233 loadStatus.successFn();
234 }
235 delete this.loadingModulesStatus_[moduleIds];
236};

239/**
240 * Handles a successful response to a request for prefetch or load one or more
241 * modules.
242 *
243 * @param {goog.net.BulkLoader} bulkLoader The bulk loader.
244 * @param {Array<string>} moduleIds The ids of the modules requested.
245 * @private
246 */
247goog.module.ModuleLoader.prototype.handleSuccess_ = function(
248 bulkLoader, moduleIds) {
249 goog.log.info(this.logger, 'Code loaded for module(s): ' + moduleIds);
250
251 var loadStatus = this.loadingModulesStatus_[moduleIds];
252 loadStatus.responseTexts = bulkLoader.getResponseTexts();
253
254 if (loadStatus.loadRequested) {
255 this.evaluateCode_(moduleIds);
256 }
257
258 // NOTE: A bulk loader instance is used for loading a set of module ids.
259 // Once these modules have been loaded successfully or in error the bulk
260 // loader should be disposed as it is not needed anymore. A new bulk loader
261 // is instantiated for any new modules to be loaded. The dispose is called
262 // on a timer so that the bulkloader has a chance to release its
263 // objects.
264 goog.Timer.callOnce(bulkLoader.dispose, 5, bulkLoader);
265};

268/** @override */
269goog.module.ModuleLoader.prototype.prefetchModule = function(
270 id, moduleInfo) {
271 // Do not prefetch in debug mode.
272 if (this.getDebugMode()) {
273 return;
274 }
275 var loadStatus = this.loadingModulesStatus_[[id]];
276 if (loadStatus) {
277 return;
278 }
279
280 var moduleInfoMap = {};
281 moduleInfoMap[id] = moduleInfo;
282 this.loadingModulesStatus_[[id]] = new goog.module.ModuleLoader.LoadStatus();
283 this.downloadModules_([id], moduleInfoMap);
284};

287/**
288 * Downloads a list of JavaScript modules.
289 *
290 * @param {Array<string>} ids The module ids in dependency order.
291 * @param {Object} moduleInfoMap A mapping from module id to ModuleInfo object.
292 * @private
293 */
294goog.module.ModuleLoader.prototype.downloadModules_ = function(
295 ids, moduleInfoMap) {
296 var uris = [];
297 for (var i = 0; i < ids.length; i++) {
298 goog.array.extend(uris, moduleInfoMap[ids[i]].getUris());
299 }
300 goog.log.info(this.logger, 'downloadModules ids:' + ids + ' uris:' + uris);
301
302 if (this.getDebugMode() &&
303 !this.usingSourceUrlInjection_()) {
304 // In debug mode use <script> tags rather than XHRs to load the files.
305 // This makes it possible to debug and inspect stack traces more easily.
306 // It's also possible to use it to load JavaScript files that are hosted on
307 // another domain.
308 // The scripts need to load serially, so this is much slower than parallel
309 // script loads with source url injection.
310 goog.net.jsloader.loadMany(uris);
311 } else {
312 var loadStatus = this.loadingModulesStatus_[ids];
313 loadStatus.requestUris = uris;
314
315 var bulkLoader = new goog.net.BulkLoader(uris);
316
317 var eventHandler = this.eventHandler_;
318 eventHandler.listen(
319 bulkLoader,
320 goog.net.EventType.SUCCESS,
321 goog.bind(this.handleSuccess_, this, bulkLoader, ids));
322 eventHandler.listen(
323 bulkLoader,
324 goog.net.EventType.ERROR,
325 goog.bind(this.handleError_, this, bulkLoader, ids));
326 bulkLoader.load();
327 }
328};

331/**
332 * Handles an error during a request for one or more modules.
333 * @param {goog.net.BulkLoader} bulkLoader The bulk loader.
334 * @param {Array<string>} moduleIds The ids of the modules requested.
335 * @param {number} status The response status.
336 * @private
337 */
338goog.module.ModuleLoader.prototype.handleError_ = function(
339 bulkLoader, moduleIds, status) {
340 var loadStatus = this.loadingModulesStatus_[moduleIds];
341 // The bulk loader doesn't cancel other requests when a request fails. We will
342 // delete the loadStatus in the first failure, so it will be undefined in
343 // subsequent errors.
344 if (loadStatus) {
345 delete this.loadingModulesStatus_[moduleIds];
346 this.handleErrorHelper_(moduleIds, loadStatus.errorFn, status);
347 }
348
349 // NOTE: A bulk loader instance is used for loading a set of module ids. Once
350 // these modules have been loaded successfully or in error the bulk loader
351 // should be disposed as it is not needed anymore. A new bulk loader is
352 // instantiated for any new modules to be loaded. The dispose is called
353 // on another thread so that the bulkloader has a chance to release its
354 // objects.
355 goog.Timer.callOnce(bulkLoader.dispose, 5, bulkLoader);
356};

359/**
360 * Handles an error during a request for one or more modules.
361 * @param {Array<string>} moduleIds The ids of the modules requested.
362 * @param {?function(?number)} errorFn The function to call on failure.
363 * @param {?number} status The response status.
364 * @private
365 */
366goog.module.ModuleLoader.prototype.handleErrorHelper_ = function(
367 moduleIds, errorFn, status) {
368 this.dispatchEvent(
369 new goog.module.ModuleLoader.Event(
370 goog.module.ModuleLoader.EventType.REQUEST_ERROR, moduleIds));
371
372 goog.log.warning(this.logger, 'Request failed for module(s): ' + moduleIds);
373
374 if (errorFn) {
375 errorFn(status);
376 }
377};
