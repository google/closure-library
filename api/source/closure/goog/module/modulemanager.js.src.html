40/**
41 * The ModuleManager keeps track of all modules in the environment.
42 * Since modules may not have their code loaded, we must keep track of them.

297/**
298 * Sets the module info for all modules. Should only be called once. Also
299 * marks modules that are currently being loaded.
300 *
301 * @param {string=} opt_info A string representation of the module dependency
302 * graph, in the form: module1:dep1,dep2/module2:dep1,dep2 etc.
303 * Where depX is the base-36 encoded position of the dep in the module list.
304 * @param {Array<string>=} opt_loadingModuleIds A list of moduleIds that
305 * are currently being loaded.
306 */
307goog.module.ModuleManager.prototype.setAllModuleInfoString = function(
308 opt_info, opt_loadingModuleIds) {
309 if (!goog.isString(opt_info)) {
310 // The call to this method is generated in two steps, the argument is added
311 // after some of the compilation passes. This means that the initial code
312 // doesn't have any arguments and causes compiler errors. We make it
313 // optional to satisfy this constraint.
314 return;
315 }
316
317 var modules = opt_info.split('/');
318 var moduleIds = [];
319
320 // Split the string into the infoMap of id->deps
321 for (var i = 0; i < modules.length; i++) {
322 var parts = modules[i].split(':');
323 var id = parts[0];
324 var deps;
325 if (parts[1]) {
326 deps = parts[1].split(',');
327 for (var j = 0; j < deps.length; j++) {
328 var index = parseInt(deps[j], 36);
329 goog.asserts.assert(
330 moduleIds[index], 'No module @ %s, dep of %s @ %s', index, id, i);
331 deps[j] = moduleIds[index];
332 }
333 } else {
334 deps = [];
335 }
336 moduleIds.push(id);
337 this.moduleInfoMap_[id] = new goog.module.ModuleInfo(deps, id);
338 }
339 if (opt_loadingModuleIds && opt_loadingModuleIds.length) {
340 goog.array.extend(this.loadingModuleIds_, opt_loadingModuleIds);
341 // The last module in the list of initial modules. When it has loaded all
342 // initial modules have loaded.
343 this.lastInitialModuleId_ = /** @type {?string} */ (
344 goog.array.peek(opt_loadingModuleIds));
345 } else {
346 if (!this.initialModulesLoaded_.hasFired()) {
347 this.initialModulesLoaded_.callback();
348 }
349 }
350 this.maybeFinishBaseLoad_();
351};

435/**
436 * Dispatches an ACTIVE or IDLE event if necessary.
437 * @private
438 */
439goog.module.ModuleManager.prototype.dispatchActiveIdleChangeIfNeeded_ =
440 function() {
441 var lastActive = this.lastActive_;
442 var active = this.isActive();
443 if (active != lastActive) {
444 this.executeCallbacks_(active ?
445 goog.module.ModuleManager.CallbackType.ACTIVE :
446 goog.module.ModuleManager.CallbackType.IDLE);
447
448 // Flip the last active value.
449 this.lastActive_ = active;
450 }
451
452 // Check if the module manager is user active i.e., there are user initiated
453 // modules being loaded or queued up to be loaded.
454 var userLastActive = this.userLastActive_;
455 var userActive = this.isUserActive();
456 if (userActive != userLastActive) {
457 this.executeCallbacks_(userActive ?
458 goog.module.ModuleManager.CallbackType.USER_ACTIVE :
459 goog.module.ModuleManager.CallbackType.USER_IDLE);
460
461 // Flip the last user active value.
462 this.userLastActive_ = userActive;
463 }

469 *
470 * @param {string} id The id of the module to preload.
471 * @param {number=} opt_timeout The number of ms to wait before adding the
472 * module id to the loading queue (defaults to 0 ms). Note that the module
473 * will be loaded asynchronously regardless of the value of this parameter.

476goog.module.ModuleManager.prototype.preloadModule = function(
477 id, opt_timeout) {

480 goog.bind(this.addLoadModule_, this, id, d),
481 opt_timeout || 0);
482 return d;

486/**
487 * Prefetches a JavaScript module and its dependencies, which means that the
488 * module will be downloaded, but not evaluated. To complete the module load,
489 * the caller should also call load or execOnLoad after prefetching the module.
490 *
491 * @param {string} id The id of the module to prefetch.
492 */
493goog.module.ModuleManager.prototype.prefetchModule = function(id) {
494 var moduleInfo = this.getModuleInfo(id);
495 if (moduleInfo.isLoaded() || this.isModuleLoading(id)) {
496 throw Error('Module load already requested: ' + id);
497 } else if (this.batchModeEnabled_) {
498 throw Error('Modules prefetching is not supported in batch mode');
499 } else {
500 var idWithDeps = this.getNotYetLoadedTransitiveDepIds_(id);
501 for (var i = 0; i < idWithDeps.length; i++) {
502 this.loader_.prefetchModule(idWithDeps[i],
503 this.moduleInfoMap_[idWithDeps[i]]);
504 }
505 }
506};

509/**
510 * Loads a single module for use with a given deferred.
511 *
512 * @param {string} id The id of the module to load.
513 * @param {goog.async.Deferred} d A deferred object.
514 * @private
515 */
516goog.module.ModuleManager.prototype.addLoadModule_ = function(id, d) {
517 var moduleInfo = this.getModuleInfo(id);
518 if (moduleInfo.isLoaded()) {
519 d.callback(this.moduleContext_);
520 return;
521 }
522
523 this.registerModuleLoadCallbacks_(id, moduleInfo, false, d);
524 if (!this.isModuleLoading(id)) {
525 this.loadModulesOrEnqueue_([id]);
526 }
527};

530/**
531 * Loads a list of modules or, if some other module is currently being loaded,
532 * appends the ids to the queue of requested module ids. Registers callbacks a
533 * module that is currently loading and returns a fired deferred for a module
534 * that is already loaded.
535 *
536 * @param {Array<string>} ids The id of the module to load.
537 * @param {boolean=} opt_userInitiated If the load is a result of a user action.
538 * @return {!Object<string, !goog.async.Deferred>} A mapping from id (String)
539 * to deferred objects that will callback or errback when the load for that
540 * id is finished.
541 * @private
542 */
543goog.module.ModuleManager.prototype.loadModulesOrEnqueueIfNotLoadedOrLoading_ =
544 function(ids, opt_userInitiated) {
545 var uniqueIds = [];
546 goog.array.removeDuplicates(ids, uniqueIds);
547 var idsToLoad = [];
548 var deferredMap = {};
549 for (var i = 0; i < uniqueIds.length; i++) {
550 var id = uniqueIds[i];
551 var moduleInfo = this.getModuleInfo(id);
552 if (!moduleInfo) {
553 throw new Error('Unknown module: ' + id);
554 }
555 var d = new goog.async.Deferred();
556 deferredMap[id] = d;
557 if (moduleInfo.isLoaded()) {
558 d.callback(this.moduleContext_);
559 } else {
560 this.registerModuleLoadCallbacks_(id, moduleInfo, !!opt_userInitiated, d);
561 if (!this.isModuleLoading(id)) {
562 idsToLoad.push(id);
563 }
564 }
565 }
566
567 // If there are ids to load, load them, otherwise, they are all loading or
568 // loaded.
569 if (idsToLoad.length > 0) {
570 this.loadModulesOrEnqueue_(idsToLoad);
571 }
572 return deferredMap;
573};

576/**
577 * Registers the callbacks and handles logic if it is a user initiated module
578 * load.
579 *
580 * @param {string} id The id of the module to possibly load.
581 * @param {!goog.module.ModuleInfo} moduleInfo The module identifier for the
582 * given id.
583 * @param {boolean} userInitiated If the load was user initiated.
584 * @param {goog.async.Deferred} d A deferred object.
585 * @private
586 */
587goog.module.ModuleManager.prototype.registerModuleLoadCallbacks_ =
588 function(id, moduleInfo, userInitiated, d) {
589 moduleInfo.registerCallback(d.callback, d);
590 moduleInfo.registerErrback(function(err) { d.errback(Error(err)); });
591 // If it's already loading, we don't have to do anything besides handle
592 // if it was user initiated
593 if (this.isModuleLoading(id)) {
594 if (userInitiated) {
595 goog.log.info(this.logger_,
596 'User initiated module already loading: ' + id);
597 this.addUserInitiatedLoadingModule_(id);
598 this.dispatchActiveIdleChangeIfNeeded_();
599 }
600 } else {
601 if (userInitiated) {
602 goog.log.info(this.logger_, 'User initiated module load: ' + id);
603 this.addUserInitiatedLoadingModule_(id);
604 } else {
605 goog.log.info(this.logger_, 'Initiating module load: ' + id);
606 }
607 }
608};

611/**
612 * Initiates loading of a list of modules or, if a module is currently being
613 * loaded, appends the modules to the queue of requested module ids.
614 *
615 * The caller should verify that the requested modules are not already loaded or
616 * loading. {@link #loadModulesOrEnqueueIfNotLoadedOrLoading_} is a more lenient
617 * alternative to this method.
618 *
619 * @param {Array<string>} ids The ids of the modules to load.
620 * @private
621 */
622goog.module.ModuleManager.prototype.loadModulesOrEnqueue_ = function(ids) {
623 // With concurrent loading we always just send off the request.
624 if (this.concurrentLoadingEnabled_) {
625 // For now we wait for initial modules to have downloaded as this puts the
626 // loader in a good state for calculating the needed deps of additional
627 // loads.
628 // TODO(user): Make this wait unnecessary.
629 this.initialModulesLoaded_.addCallback(
630 goog.bind(this.loadModules_, this, ids));
631 } else {
632 if (goog.array.isEmpty(this.loadingModuleIds_)) {
633 this.loadModules_(ids);
634 } else {
635 this.requestedModuleIdsQueue_.push(ids);
636 this.dispatchActiveIdleChangeIfNeeded_();
637 }
638 }
639};

642/**
643 * Gets the amount of delay to wait before sending a request for more modules.
644 * If a certain module request fails, we backoff a little bit and try again.
645 * @return {number} Delay, in ms.
646 * @private
647 */
648goog.module.ModuleManager.prototype.getBackOff_ = function() {
649 // 5 seconds after one error, 20 seconds after 2.
650 return Math.pow(this.consecutiveFailures_, 2) * 5000;
651};

654/**
655 * Loads a list of modules and any of their not-yet-loaded prerequisites.
656 * If batch mode is enabled, the prerequisites will be loaded together with the
657 * requested modules and all requested modules will be loaded at the same time.
658 *
659 * The caller should verify that the requested modules are not already loaded
660 * and that no modules are currently loading before calling this method.
661 *
662 * @param {Array<string>} ids The ids of the modules to load.
663 * @param {boolean=} opt_isRetry If the load is a retry of a previous load
664 * attempt.
665 * @param {boolean=} opt_forceReload Whether to bypass cache while loading the
666 * module.
667 * @private
668 */
669goog.module.ModuleManager.prototype.loadModules_ = function(
670 ids, opt_isRetry, opt_forceReload) {
671 if (!opt_isRetry) {
672 this.consecutiveFailures_ = 0;
673 }
674
675 // Not all modules may be loaded immediately if batch mode is not enabled.
676 var idsToLoadImmediately = this.processModulesForLoad_(ids);
677
678 goog.log.info(this.logger_, 'Loading module(s): ' + idsToLoadImmediately);
679 this.loadingModuleIds_ = idsToLoadImmediately;
680
681 if (this.batchModeEnabled_) {
682 this.requestedLoadingModuleIds_ = ids;
683 } else {
684 // If batch mode is disabled, we treat each dependency load as a separate
685 // load.
686 this.requestedLoadingModuleIds_ = goog.array.clone(idsToLoadImmediately);
687 }
688
689 // Dispatch an active/idle change if needed.
690 this.dispatchActiveIdleChangeIfNeeded_();
691
692 if (goog.array.isEmpty(idsToLoadImmediately)) {
693 // All requested modules and deps have been either loaded already or have
694 // already been requested.
695 return;
696 }
697
698 this.requestedModuleIds_.push.apply(this.requestedModuleIds_,
699 idsToLoadImmediately);
700
701 var loadFn = goog.bind(this.loader_.loadModules, this.loader_,
702 goog.array.clone(idsToLoadImmediately),
703 this.moduleInfoMap_,
704 null,
705 goog.bind(this.handleLoadError_, this, this.requestedLoadingModuleIds_,
706 idsToLoadImmediately),
707 goog.bind(this.handleLoadTimeout_, this),
708 !!opt_forceReload);
709
710 var delay = this.getBackOff_();
711 if (delay) {
712 window.setTimeout(loadFn, delay);
713 } else {
714 loadFn();
715 }
716};

719/**
720 * Processes a list of module ids for loading. Checks if any of the modules are
721 * already loaded and then gets transitive deps. Queues any necessary modules
722 * if batch mode is not enabled. Returns the list of ids that should be loaded.
723 *
724 * @param {Array<string>} ids The ids that need to be loaded.
725 * @return {!Array<string>} The ids to load, including dependencies.
726 * @throws {Error} If the module is already loaded.
727 * @private
728 */
729goog.module.ModuleManager.prototype.processModulesForLoad_ = function(ids) {
730 for (var i = 0; i < ids.length; i++) {
731 var moduleInfo = this.moduleInfoMap_[ids[i]];
732 if (moduleInfo.isLoaded()) {
733 throw Error('Module already loaded: ' + ids[i]);
734 }
735 }
736
737 // Build a list of the ids of this module and any of its not-yet-loaded
738 // prerequisite modules in dependency order.
739 var idsWithDeps = [];
740 for (var i = 0; i < ids.length; i++) {
741 idsWithDeps = idsWithDeps.concat(
742 this.getNotYetLoadedTransitiveDepIds_(ids[i]));
743 }
744 goog.array.removeDuplicates(idsWithDeps);
745
746 if (!this.batchModeEnabled_ && idsWithDeps.length > 1) {
747 var idToLoad = idsWithDeps.shift();
748 goog.log.info(this.logger_,
749 'Must load ' + idToLoad + ' module before ' + ids);
750
751 // Insert the requested module id and any other not-yet-loaded prereqs
752 // that it has at the front of the queue.
753 var queuedModules = goog.array.map(idsWithDeps, function(id) {
754 return [id];
755 });

765/**
766 * Builds a list of the ids of the not-yet-loaded modules that a particular
767 * module transitively depends on, including itself.
768 *
769 * @param {string} id The id of a not-yet-loaded module.
770 * @return {!Array<string>} An array of module ids in dependency order that's
771 * guaranteed to end with the provided module id.
772 * @private
773 */
774goog.module.ModuleManager.prototype.getNotYetLoadedTransitiveDepIds_ =
775 function(id) {
776 // NOTE(user): We want the earliest occurrance of a module, not the first
777 // dependency we find. Therefore we strip duplicates at the end rather than
778 // during. See the tests for concrete examples.
779 var ids = [];
780 if (!goog.array.contains(this.requestedModuleIds_, id)) {
781 ids.push(id);
782 }
783 var depIds = goog.array.clone(this.getModuleInfo(id).getDependencies());
784 while (depIds.length) {
785 var depId = depIds.pop();
786 if (!this.getModuleInfo(depId).isLoaded() &&
787 !goog.array.contains(this.requestedModuleIds_, depId)) {
788 ids.unshift(depId);
789 // We need to process direct dependencies first.
790 Array.prototype.unshift.apply(depIds,
791 this.getModuleInfo(depId).getDependencies());
792 }
793 }
794 goog.array.removeDuplicates(ids);
795 return ids;

883/**
884 * Requests that a function be called once a particular module is loaded.
885 * Client code can use this method to safely call into modules that may not yet
886 * be loaded. For consistency, this method always calls the function
887 * asynchronously -- even if the module is already loaded. Initiates loading of
888 * the module if necessary, unless opt_noLoad is true.
889 *
890 * @param {string} moduleId A module id.
891 * @param {Function} fn Function to execute when the module has loaded.
892 * @param {Object=} opt_handler Optional handler under whose scope to execute
893 * the callback.
894 * @param {boolean=} opt_noLoad TRUE iff not to initiate loading of the module.
895 * @param {boolean=} opt_userInitiated TRUE iff the loading of the module was
896 * user initiated.
897 * @param {boolean=} opt_preferSynchronous TRUE iff the function should be
898 * executed synchronously if the module has already been loaded.
899 * @return {!goog.module.ModuleLoadCallback} A callback wrapper that exposes
900 * an abort and execute method.
901 */
902goog.module.ModuleManager.prototype.execOnLoad = function(
903 moduleId, fn, opt_handler, opt_noLoad,
904 opt_userInitiated, opt_preferSynchronous) {
905 var moduleInfo = this.moduleInfoMap_[moduleId];
906 var callbackWrapper;
907
908 if (moduleInfo.isLoaded()) {
909 goog.log.info(this.logger_, moduleId + ' module already loaded');
910 // Call async so that code paths don't change between loaded and unloaded
911 // cases.
912 callbackWrapper = new goog.module.ModuleLoadCallback(fn, opt_handler);
913 if (opt_preferSynchronous) {
914 callbackWrapper.execute(this.moduleContext_);
915 } else {
916 window.setTimeout(
917 goog.bind(callbackWrapper.execute, callbackWrapper), 0);
918 }
919 } else if (this.isModuleLoading(moduleId)) {
920 goog.log.info(this.logger_, moduleId + ' module already loading');
921 callbackWrapper = moduleInfo.registerCallback(fn, opt_handler);
922 if (opt_userInitiated) {
923 goog.log.info(this.logger_,
924 'User initiated module already loading: ' + moduleId);
925 this.addUserInitiatedLoadingModule_(moduleId);
926 this.dispatchActiveIdleChangeIfNeeded_();
927 }
928 } else {
929 goog.log.info(this.logger_,
930 'Registering callback for module: ' + moduleId);
931 callbackWrapper = moduleInfo.registerCallback(fn, opt_handler);
932 if (!opt_noLoad) {
933 if (opt_userInitiated) {
934 goog.log.info(this.logger_, 'User initiated module load: ' + moduleId);
935 this.addUserInitiatedLoadingModule_(moduleId);
936 }
937 goog.log.info(this.logger_, 'Initiating module load: ' + moduleId);
938 this.loadModulesOrEnqueue_([moduleId]);
939 }
940 }
941 return callbackWrapper;

1114/**
1115 * Handles a module load failure.
1116 *
1117 * @param {!Array<string>} requestedLoadingModuleIds Modules ids that were
1118 * requested in failed request. Does not included calculated dependencies.
1119 * @param {!Array<string>} requestedModuleIdsWithDeps All module ids requested
1120 * in the failed request including all dependencies.
1121 * @param {?number} status The error status.
1122 * @private
1123 */
1124goog.module.ModuleManager.prototype.handleLoadError_ = function(
1125 requestedLoadingModuleIds, requestedModuleIdsWithDeps, status) {
1126 this.consecutiveFailures_++;
1127 // Module manager was not designed to be reentrant. Reinstate the instance
1128 // var with actual value when request failed (Other requests may have
1129 // started already.)
1130 this.requestedLoadingModuleIds_ = requestedLoadingModuleIds;
1131 // Pretend we never requested the failed modules.
1132 goog.array.forEach(requestedModuleIdsWithDeps,
1133 goog.partial(goog.array.remove, this.requestedModuleIds_), this);
1134
1135 if (status == 401) {
1136 // The user is not logged in. They've cleared their cookies or logged out
1137 // from another window.
1138 goog.log.info(this.logger_, 'Module loading unauthorized');
1139 this.dispatchModuleLoadFailed_(
1140 goog.module.ModuleManager.FailureType.UNAUTHORIZED);
1141 // Drop any additional module requests.
1142 this.requestedModuleIdsQueue_.length = 0;
1143 } else if (status == 410) {
1144 // The requested module js is old and not available.
1145 this.requeueBatchOrDispatchFailure_(
1146 goog.module.ModuleManager.FailureType.OLD_CODE_GONE);
1147 this.loadNextModules_();
1148 } else if (this.consecutiveFailures_ >= 3) {
1149 goog.log.info(this.logger_, 'Aborting after failure to load: ' +
1150 this.loadingModuleIds_);
1151 this.requeueBatchOrDispatchFailure_(
1152 goog.module.ModuleManager.FailureType.CONSECUTIVE_FAILURES);
1153 this.loadNextModules_();
1154 } else {
1155 goog.log.info(this.logger_, 'Retrying after failure to load: ' +
1156 this.loadingModuleIds_);
1157 var forceReload =
1158 status == goog.module.ModuleManager.CORRUPT_RESPONSE_STATUS_CODE;
1159 this.loadModules_(this.requestedLoadingModuleIds_, true, forceReload);
1160 }
1161};

1204/**
1205 * Handles when a module load failed.
1206 * @param {goog.module.ModuleManager.FailureType} cause The reason for the
1207 * failure.
1208 * @private
1209 */
1210goog.module.ModuleManager.prototype.dispatchModuleLoadFailed_ = function(
1211 cause) {
1212 var failedIds = this.requestedLoadingModuleIds_;
1213 this.loadingModuleIds_.length = 0;
1214 // If any pending modules depend on the id that failed,
1215 // they need to be removed from the queue.
1216 var idsToCancel = [];
1217 for (var i = 0; i < this.requestedModuleIdsQueue_.length; i++) {
1218 var dependentModules = goog.array.filter(
1219 this.requestedModuleIdsQueue_[i],
1220 /**
1221 * Returns true if the requestedId has dependencies on the modules that
1222 * just failed to load.
1223 * @param {string} requestedId The module to check for dependencies.
1224 * @return {boolean} True if the module depends on failed modules.
1225 */
1226 function(requestedId) {
1227 var requestedDeps = this.getNotYetLoadedTransitiveDepIds_(
1228 requestedId);
1229 return goog.array.some(failedIds, function(id) {
1230 return goog.array.contains(requestedDeps, id);
1231 });
1232 }, this);
1233 goog.array.extend(idsToCancel, dependentModules);
1234 }
1235
1236 // Also insert the ids that failed to load as ids to cancel.
1237 for (var i = 0; i < failedIds.length; i++) {
1238 goog.array.insert(idsToCancel, failedIds[i]);
1239 }
1240
1241 // Remove ids to cancel from the queues.
1242 for (var i = 0; i < idsToCancel.length; i++) {
1243 for (var j = 0; j < this.requestedModuleIdsQueue_.length; j++) {
1244 goog.array.remove(this.requestedModuleIdsQueue_[j], idsToCancel[i]);
1245 }
1246 goog.array.remove(this.userInitiatedLoadingModuleIds_, idsToCancel[i]);
1247 }
1248
1249 // Call the functions for error notification.
1250 var errorCallbacks = this.callbackMap_[
1251 goog.module.ModuleManager.CallbackType.ERROR];
1252 if (errorCallbacks) {
1253 for (var i = 0; i < errorCallbacks.length; i++) {
1254 var callback = errorCallbacks[i];
1255 for (var j = 0; j < idsToCancel.length; j++) {
1256 callback(goog.module.ModuleManager.CallbackType.ERROR, idsToCancel[j],
1257 cause);
1258 }
1259 }
1260 }
1261
1262 // Call the errbacks on the module info.
1263 for (var i = 0; i < failedIds.length; i++) {
1264 if (this.moduleInfoMap_[failedIds[i]]) {
1265 this.moduleInfoMap_[failedIds[i]].onError(cause);
1266 }
1267 }
1268
1269 // Clear the requested loading module ids.
1270 this.requestedLoadingModuleIds_.length = 0;
1271
1272 this.dispatchActiveIdleChangeIfNeeded_();
1273};
