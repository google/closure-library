30/**
31 * A SimpleResult object is a basic implementation of the
32 * goog.result.Result interface. This could be subclassed(e.g. XHRResult)
33 * or instantiated and returned by another class as a form of result. The caller
34 * receiving the result could then attach handlers to be called when the result
35 * is resolved(success or error).
36 *
37 * @constructor
38 * @implements {goog.result.Result}
39 * @deprecated Use {@link goog.Promise} instead - http://go/promisemigration
40 */
41goog.result.SimpleResult = function() {
42 /**
43 * The current state of this Result.
44 * @type {goog.result.Result.State}
45 * @private
46 */
47 this.state_ = goog.result.Result.State.PENDING;
48
49 /**
50 * The list of handlers to call when this Result is resolved.
51 * @type {!Array<!goog.result.SimpleResult.HandlerEntry_>}
52 * @private
53 */
54 this.handlers_ = [];
55
56 // The value_ and error_ properties are initialized in the constructor to
57 // ensure that all SimpleResult instances share the same hidden class in
58 // modern JavaScript engines.
59
60 /**
61 * The 'value' of this Result.
62 * @type {*}
63 * @private
64 */
65 this.value_ = undefined;
66
67 /**
68 * The error slug for this Result.
69 * @type {*}
70 * @private
71 */
72 this.error_ = undefined;
73};
74goog.Thenable.addImplementation(goog.result.SimpleResult);

123/**
124 * Attaches handlers to be called when the value of this Result is available.
125 *
126 * @param {!function(this:T, !goog.result.SimpleResult)} handler The function
127 * called when the value is available. The function is passed the Result
128 * object as the only argument.
129 * @param {T=} opt_scope Optional scope for the handler.
130 * @template T
131 * @override
132 */
133goog.result.SimpleResult.prototype.wait = function(handler, opt_scope) {
134 if (this.isPending_()) {
135 this.handlers_.push({
136 callback: handler,
137 scope: opt_scope || null
138 });
139 } else {
140 handler.call(opt_scope, this);
141 }
142};

145/**
146 * Sets the value of this Result, changing the state.
147 *
148 * @param {*} value The value to set for this Result.
149 */
150goog.result.SimpleResult.prototype.setValue = function(value) {
151 if (this.isPending_()) {
152 this.value_ = value;
153 this.state_ = goog.result.Result.State.SUCCESS;
154 this.callHandlers_();
155 } else if (!this.isCanceled()) {
156 // setValue is a no-op if this Result has been canceled.
157 throw new goog.result.SimpleResult.StateError();
158 }
159};

162/**
163 * Sets the Result to be an error Result.
164 *
165 * @param {*=} opt_error Optional error slug to set for this Result.
166 */
167goog.result.SimpleResult.prototype.setError = function(opt_error) {
168 if (this.isPending_()) {
169 this.error_ = opt_error;
170 this.state_ = goog.result.Result.State.ERROR;
171 this.callHandlers_();
172 } else if (!this.isCanceled()) {
173 // setError is a no-op if this Result has been canceled.
174 throw new goog.result.SimpleResult.StateError();
175 }
176};

179/**
180 * Calls the handlers registered for this Result.
181 *
182 * @private
183 */
184goog.result.SimpleResult.prototype.callHandlers_ = function() {
185 var handlers = this.handlers_;
186 this.handlers_ = [];
187 for (var n = 0; n < handlers.length; n++) {
188 var handlerEntry = handlers[n];
189 handlerEntry.callback.call(handlerEntry.scope, this);
190 }
191};

194/**

198goog.result.SimpleResult.prototype.isPending_ = function() {
199 return this.state_ == goog.result.Result.State.PENDING;

203/**
204 * Cancels the Result.
205 *
206 * @return {boolean} Whether the result was canceled. It will not be canceled if
207 * the result was already canceled or has already resolved.
208 * @override
209 */
210goog.result.SimpleResult.prototype.cancel = function() {
211 // cancel is a no-op if the result has been resolved.
212 if (this.isPending_()) {
213 this.setError(new goog.result.Result.CancelError());
214 return true;
215 }
216 return false;
217};

220/** @override */
221goog.result.SimpleResult.prototype.isCanceled = function() {
222 return this.state_ == goog.result.Result.State.ERROR &&
223 this.error_ instanceof goog.result.Result.CancelError;
224};

227/** @override */
228goog.result.SimpleResult.prototype.then = function(
229 opt_onFulfilled, opt_onRejected, opt_context) {
230 var resolve, reject;
231 // Copy the resolvers to outer scope, so that they are available
232 // when the callback to wait() fires (which may be synchronous).
233 var promise = new goog.Promise(function(res, rej) {
234 resolve = res;
235 reject = rej;
236 });
237 this.wait(function(result) {
238 if (result.isCanceled()) {
239 promise.cancel();
240 } else if (result.getState() == goog.result.Result.State.SUCCESS) {
241 resolve(result.getValue());
242 } else if (result.getState() == goog.result.Result.State.ERROR) {
243 reject(result.getError());
244 }
245 });
246 return promise.then(opt_onFulfilled, opt_onRejected, opt_context);
247};

250/**
251 * Creates a SimpleResult that fires when the given promise resolves.
252 * Use only during migration to Promises.
253 * @param {!goog.Promise<?>} promise
254 * @return {!goog.result.Result}
255 */
256goog.result.SimpleResult.fromPromise = function(promise) {
257 var result = new goog.result.SimpleResult();
258 promise.then(result.setValue, result.setError, result);
259 return result;
