27/**
28 * Returns the normalized scrollLeft position for a scrolled element.
29 * @param {Element} element The scrolled element.
30 * @return {number} The number of pixels the element is scrolled. 0 indicates
31 * that the element is not scrolled at all (which, in general, is the
32 * left-most position in ltr and the right-most position in rtl).
33 */
34goog.style.bidi.getScrollLeft = function(element) {
35 var isRtl = goog.style.isRightToLeft(element);
36 if (isRtl && goog.userAgent.GECKO) {
37 // ScrollLeft starts at 0 and then goes negative as the element is scrolled
38 // towards the left.
39 return -element.scrollLeft;
40 } else if (isRtl &&
41 !(goog.userAgent.IE && goog.userAgent.isVersionOrHigher('8'))) {
42 // ScrollLeft starts at the maximum positive value and decreases towards
43 // 0 as the element is scrolled towards the left. However, for overflow
44 // visible, there is no scrollLeft and the value always stays correctly at 0
45 var overflowX = goog.style.getComputedOverflowX(element);
46 if (overflowX == 'visible') {
47 return element.scrollLeft;
48 } else {
49 return element.scrollWidth - element.clientWidth - element.scrollLeft;
50 }
51 }
52 // ScrollLeft behavior is identical in rtl and ltr, it starts at 0 and
53 // increases as the element is scrolled away from the start.
54 return element.scrollLeft;
55};

58/**
59 * Returns the "offsetStart" of an element, analagous to offsetLeft but
60 * normalized for right-to-left environments and various browser
61 * inconsistencies. This value returned can always be passed to setScrollOffset
62 * to scroll to an element's left edge in a left-to-right offsetParent or
63 * right edge in a right-to-left offsetParent.
64 *
65 * For example, here offsetStart is 10px in an LTR environment and 5px in RTL:
66 *
67 * <pre>
68 * | xxxxxxxxxx |
69 * ^^^^^^^^^^ ^^^^ ^^^^^
70 * 10px elem 5px
71 * </pre>
72 *
73 * If an element is positioned before the start of its offsetParent, the
74 * startOffset may be negative. This can be used with setScrollOffset to
75 * reliably scroll to an element:
76 *
77 * <pre>
78 * var scrollOffset = goog.style.bidi.getOffsetStart(element);
79 * goog.style.bidi.setScrollOffset(element.offsetParent, scrollOffset);
80 * </pre>
81 *
82 * @see setScrollOffset
83 *
84 * @param {Element} element The element for which we need to determine the
85 * offsetStart position.
86 * @return {number} The offsetStart for that element.
87 */
88goog.style.bidi.getOffsetStart = function(element) {
89 var offsetLeftForReal = element.offsetLeft;
90
91 // The element might not have an offsetParent.
92 // For example, the node might not be attached to the DOM tree,
93 // and position:fixed children do not have an offset parent.
94 // Just try to do the best we can with what we have.
95 var bestParent = element.offsetParent;
96
97 if (!bestParent && goog.style.getComputedPosition(element) == 'fixed') {
98 bestParent = goog.dom.getOwnerDocument(element).documentElement;
99 }
100
101 // Just give up in this case.
102 if (!bestParent) {
103 return offsetLeftForReal;
104 }
105
106 if (goog.userAgent.GECKO) {
107 // When calculating an element's offsetLeft, Firefox erroneously subtracts
108 // the border width from the actual distance. So we need to add it back.
109 var borderWidths = goog.style.getBorderBox(bestParent);
110 offsetLeftForReal += borderWidths.left;
111 } else if (goog.userAgent.isDocumentModeOrHigher(8) &&
112 !goog.userAgent.isDocumentModeOrHigher(9)) {
113 // When calculating an element's offsetLeft, IE8/9-Standards Mode
114 // erroneously adds the border width to the actual distance. So we need to
115 // subtract it.
116 var borderWidths = goog.style.getBorderBox(bestParent);
117 offsetLeftForReal -= borderWidths.left;
118 }
119
120 if (goog.style.isRightToLeft(bestParent)) {
121 // Right edge of the element relative to the left edge of its parent.
122 var elementRightOffset = offsetLeftForReal + element.offsetWidth;
123
124 // Distance from the parent's right edge to the element's right edge.
125 return bestParent.clientWidth - elementRightOffset;
126 }
127
128 return offsetLeftForReal;
129};

132/**
133 * Sets the element's scrollLeft attribute so it is correctly scrolled by
134 * offsetStart pixels. This takes into account whether the element is RTL and
135 * the nuances of different browsers. To scroll to the "beginning" of an
136 * element use getOffsetStart to obtain the element's offsetStart value and then
137 * pass the value to setScrollOffset.
138 * @see getOffsetStart
139 * @param {Element} element The element to set scrollLeft on.
140 * @param {number} offsetStart The number of pixels to scroll the element.
141 * If this value is < 0, 0 is used.
142 */
143goog.style.bidi.setScrollOffset = function(element, offsetStart) {
144 offsetStart = Math.max(offsetStart, 0);
145 // In LTR and in "mirrored" browser RTL (such as IE), we set scrollLeft to
146 // the number of pixels to scroll.
147 // Otherwise, in RTL, we need to account for different browser behavior.
148 if (!goog.style.isRightToLeft(element)) {
149 element.scrollLeft = offsetStart;
150 } else if (goog.userAgent.GECKO) {
151 // Negative scroll-left positions in RTL.
152 element.scrollLeft = -offsetStart;
153 } else if (!(goog.userAgent.IE && goog.userAgent.isVersionOrHigher('8'))) {
154 // Take the current scrollLeft value and move to the right by the
155 // offsetStart to get to the left edge of the element, and then by
156 // the clientWidth of the element to get to the right edge.
157 element.scrollLeft =
158 element.scrollWidth - offsetStart - element.clientWidth;
159 } else {
160 element.scrollLeft = offsetStart;
161 }
162};

165/**
166 * Sets the element's left style attribute in LTR or right style attribute in
167 * RTL. Also clears the left attribute in RTL and the right attribute in LTR.
168 * @param {Element} elem The element to position.
169 * @param {number} left The left position in LTR; will be set as right in RTL.
170 * @param {?number} top The top position. If null only the left/right is set.
171 * @param {boolean} isRtl Whether we are in RTL mode.
172 */
173goog.style.bidi.setPosition = function(elem, left, top, isRtl) {
174 if (!goog.isNull(top)) {
175 elem.style.top = top + 'px';
176 }
177 if (isRtl) {
178 elem.style.right = left + 'px';
179 elem.style.left = '';
180 } else {
181 elem.style.left = left + 'px';
182 elem.style.right = '';
183 }
