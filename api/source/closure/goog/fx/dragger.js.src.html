44/**
45 * A class that allows mouse or touch-based dragging (moving) of an element
46 *
47 * @param {Element} target The element that will be dragged.
48 * @param {Element=} opt_handle An optional handle to control the drag, if null
49 * the target is used.
50 * @param {goog.math.Rect=} opt_limits Object containing left, top, width,
51 * and height.
52 *
53 * @extends {goog.events.EventTarget}
54 * @constructor
55 */
56goog.fx.Dragger = function(target, opt_handle, opt_limits) {
57 goog.events.EventTarget.call(this);
58 this.target = target;
59 this.handle = opt_handle || target;
60 this.limits = opt_limits || new goog.math.Rect(NaN, NaN, NaN, NaN);
61
62 this.document_ = goog.dom.getOwnerDocument(target);
63 this.eventHandler_ = new goog.events.EventHandler(this);
64 this.registerDisposable(this.eventHandler_);
65
66 // Add listener. Do not use the event handler here since the event handler is
67 // used for listeners added and removed during the drag operation.
68 goog.events.listen(this.handle,
69 [goog.events.EventType.TOUCHSTART, goog.events.EventType.MOUSEDOWN],
70 this.startDrag, false, this);
71};
72goog.inherits(goog.fx.Dragger, goog.events.EventTarget);
73// Dragger is meant to be extended, but defines most properties on its
74// prototype, thus making it unsuitable for sealing.
75goog.tagUnsealableClass(goog.fx.Dragger);

90/**
91 * Creates copy of node being dragged. This is a utility function to be used
92 * wherever it is inappropriate for the original source to follow the mouse
93 * cursor itself.
94 *
95 * @param {Element} sourceEl Element to copy.
96 * @return {!Element} The clone of {@code sourceEl}.
97 */
98goog.fx.Dragger.cloneNode = function(sourceEl) {
99 var clonedEl = /** @type {Element} */ (sourceEl.cloneNode(true)),
100 origTexts = sourceEl.getElementsByTagName(goog.dom.TagName.TEXTAREA),
101 dragTexts = clonedEl.getElementsByTagName(goog.dom.TagName.TEXTAREA);
102 // Cloning does not copy the current value of textarea elements, so correct
103 // this manually.
104 for (var i = 0; i < origTexts.length; i++) {
105 dragTexts[i].value = origTexts[i].value;
106 }
107 switch (sourceEl.tagName) {
108 case goog.dom.TagName.TR:
109 return goog.dom.createDom(goog.dom.TagName.TABLE, null,
110 goog.dom.createDom(goog.dom.TagName.TBODY,
111 null, clonedEl));
112 case goog.dom.TagName.TD:
113 case goog.dom.TagName.TH:
114 return goog.dom.createDom(
115 goog.dom.TagName.TABLE, null, goog.dom.createDom(
116 goog.dom.TagName.TBODY, null, goog.dom.createDom(
117 goog.dom.TagName.TR, null, clonedEl)));
118 case goog.dom.TagName.TEXTAREA:
119 clonedEl.value = sourceEl.value;
120 default:
121 return clonedEl;
122 }
123};

433/**
434 * Event handler that is used to start the drag
435 * @param {goog.events.BrowserEvent} e Event object.
436 */
437goog.fx.Dragger.prototype.startDrag = function(e) {
438 var isMouseDown = e.type == goog.events.EventType.MOUSEDOWN;
439
440 // Dragger.startDrag() can be called by AbstractDragDrop with a mousemove
441 // event and IE does not report pressed mouse buttons on mousemove. Also,
442 // it does not make sense to check for the button if the user is already
443 // dragging.
444
445 if (this.enabled_ && !this.dragging_ &&
446 (!isMouseDown || e.isMouseActionButton())) {
447 this.maybeReinitTouchEvent_(e);
448 if (this.hysteresisDistanceSquared_ == 0) {
449 if (this.fireDragStart_(e)) {
450 this.dragging_ = true;
451 e.preventDefault();
452 } else {
453 // If the start drag is cancelled, don't setup for a drag.
454 return;
455 }
456 } else {
457 // Need to preventDefault for hysteresis to prevent page getting selected.
458 e.preventDefault();
459 }
460 this.setupDragHandlers();
461
462 this.clientX = this.startX = e.clientX;
463 this.clientY = this.startY = e.clientY;
464 this.screenX = e.screenX;
465 this.screenY = e.screenY;
466 this.computeInitialPosition();
467 this.pageScroll = goog.dom.getDomHelper(this.document_).getDocumentScroll();
468
469 this.mouseDownTime_ = goog.now();
470 } else {
471 this.dispatchEvent(goog.fx.Dragger.EventType.EARLY_CANCEL);
472 }
473};

476/**
477 * Sets up event handlers when dragging starts.
478 * @protected
479 */
480goog.fx.Dragger.prototype.setupDragHandlers = function() {
481 var doc = this.document_;
482 var docEl = doc.documentElement;
483 // Use bubbling when we have setCapture since we got reports that IE has
484 // problems with the capturing events in combination with setCapture.
485 var useCapture = !goog.fx.Dragger.HAS_SET_CAPTURE_;
486
487 this.eventHandler_.listen(doc,
488 [goog.events.EventType.TOUCHMOVE, goog.events.EventType.MOUSEMOVE],
489 this.handleMove_, useCapture);
490 this.eventHandler_.listen(doc,
491 [goog.events.EventType.TOUCHEND, goog.events.EventType.MOUSEUP],
492 this.endDrag, useCapture);
493
494 if (goog.fx.Dragger.HAS_SET_CAPTURE_) {
495 docEl.setCapture(false);
496 this.eventHandler_.listen(docEl,
497 goog.events.EventType.LOSECAPTURE,
498 this.endDrag);
499 } else {
500 // Make sure we stop the dragging if the window loses focus.
501 // Don't use capture in this listener because we only want to end the drag
502 // if the actual window loses focus. Since blur events do not bubble we use
503 // a bubbling listener on the window.
504 this.eventHandler_.listen(goog.dom.getWindow(doc),
505 goog.events.EventType.BLUR,
506 this.endDrag);
507 }
508
509 if (goog.userAgent.IE && this.ieDragStartCancellingOn_) {
510 // Cancel IE's 'ondragstart' event.
511 this.eventHandler_.listen(doc, goog.events.EventType.DRAGSTART,
512 goog.events.Event.preventDefault);
513 }
514
515 if (this.scrollTarget_) {
516 this.eventHandler_.listen(this.scrollTarget_, goog.events.EventType.SCROLL,
517 this.onScroll_, useCapture);
518 }
519};

547/**
548 * Event handler that is used to end the drag.
549 * @param {goog.events.BrowserEvent} e Event object.
550 * @param {boolean=} opt_dragCanceled Whether the drag has been canceled.
551 */
552goog.fx.Dragger.prototype.endDrag = function(e, opt_dragCanceled) {
553 this.cleanUpAfterDragging_();
554
555 if (this.dragging_) {
556 this.maybeReinitTouchEvent_(e);
557 this.dragging_ = false;
558
559 var x = this.limitX(this.deltaX);
560 var y = this.limitY(this.deltaY);
561 var dragCanceled = opt_dragCanceled ||
562 e.type == goog.events.EventType.TOUCHCANCEL;
563 this.dispatchEvent(new goog.fx.DragEvent(
564 goog.fx.Dragger.EventType.END, this, e.clientX, e.clientY, e, x, y,
565 dragCanceled));
566 } else {
567 this.dispatchEvent(goog.fx.Dragger.EventType.EARLY_CANCEL);
568 }
569};

581/**
582 * Re-initializes the event with the first target touch event or, in the case
583 * of a stop event, the last changed touch.
584 * @param {goog.events.BrowserEvent} e A TOUCH... event.
585 * @private
586 */
587goog.fx.Dragger.prototype.maybeReinitTouchEvent_ = function(e) {
588 var type = e.type;
589
590 if (type == goog.events.EventType.TOUCHSTART ||
591 type == goog.events.EventType.TOUCHMOVE) {
592 e.init(e.getBrowserEvent().targetTouches[0], e.currentTarget);
593 } else if (type == goog.events.EventType.TOUCHEND ||
594 type == goog.events.EventType.TOUCHCANCEL) {
595 e.init(e.getBrowserEvent().changedTouches[0], e.currentTarget);
596 }
597};

600/**
601 * Event handler that is used on mouse / touch move to update the drag
602 * @param {goog.events.BrowserEvent} e Event object.
603 * @private
604 */
605goog.fx.Dragger.prototype.handleMove_ = function(e) {
606 if (this.enabled_) {
607 this.maybeReinitTouchEvent_(e);
608 // dx in right-to-left cases is relative to the right.
609 var sign = this.useRightPositioningForRtl_ &&
610 this.isRightToLeft_() ? -1 : 1;
611 var dx = sign * (e.clientX - this.clientX);
612 var dy = e.clientY - this.clientY;
613 this.clientX = e.clientX;
614 this.clientY = e.clientY;
615 this.screenX = e.screenX;
616 this.screenY = e.screenY;
617
618 if (!this.dragging_) {
619 var diffX = this.startX - this.clientX;
620 var diffY = this.startY - this.clientY;
621 var distance = diffX * diffX + diffY * diffY;
622 if (distance > this.hysteresisDistanceSquared_) {
623 if (this.fireDragStart_(e)) {
624 this.dragging_ = true;
625 } else {
626 // DragListGroup disposes of the dragger if BEFOREDRAGSTART is
627 // canceled.
628 if (!this.isDisposed()) {
629 this.endDrag(e);
630 }
631 return;
632 }
633 }
634 }
635
636 var pos = this.calculatePosition_(dx, dy);
637 var x = pos.x;
638 var y = pos.y;
639
640 if (this.dragging_) {
641
642 var rv = this.dispatchEvent(new goog.fx.DragEvent(
643 goog.fx.Dragger.EventType.BEFOREDRAG, this, e.clientX, e.clientY,
644 e, x, y));
645
646 // Only do the defaultAction and dispatch drag event if predrag didn't
647 // prevent default
648 if (rv) {
649 this.doDrag(e, x, y, false);
650 e.preventDefault();
651 }
652 }
653 }
654};

657/**
658 * Calculates the drag position.
659 *
660 * @param {number} dx The horizontal movement delta.
661 * @param {number} dy The vertical movement delta.
662 * @return {!goog.math.Coordinate} The newly calculated drag element position.
663 * @private
664 */
665goog.fx.Dragger.prototype.calculatePosition_ = function(dx, dy) {
666 // Update the position for any change in body scrolling
667 var pageScroll = goog.dom.getDomHelper(this.document_).getDocumentScroll();
668 dx += pageScroll.x - this.pageScroll.x;
669 dy += pageScroll.y - this.pageScroll.y;
670 this.pageScroll = pageScroll;
671
672 this.deltaX += dx;
673 this.deltaY += dy;
674
675 var x = this.limitX(this.deltaX);
676 var y = this.limitY(this.deltaY);
677 return new goog.math.Coordinate(x, y);
678};

710/**
711 * Returns the 'real' x after limits are applied (allows for some
712 * limits to be undefined).
713 * @param {number} x X-coordinate to limit.
714 * @return {number} The 'real' X-coordinate after limits are applied.
715 */
716goog.fx.Dragger.prototype.limitX = function(x) {
717 var rect = this.limits;
718 var left = !isNaN(rect.left) ? rect.left : null;
719 var width = !isNaN(rect.width) ? rect.width : 0;
720 var maxX = left != null ? left + width : Infinity;
721 var minX = left != null ? left : -Infinity;
722 return Math.min(maxX, Math.max(minX, x));
723};
724
725
726/**
727 * Returns the 'real' y after limits are applied (allows for some
728 * limits to be undefined).
729 * @param {number} y Y-coordinate to limit.
730 * @return {number} The 'real' Y-coordinate after limits are applied.
731 */
732goog.fx.Dragger.prototype.limitY = function(y) {
733 var rect = this.limits;
734 var top = !isNaN(rect.top) ? rect.top : null;
735 var height = !isNaN(rect.height) ? rect.height : 0;
736 var maxY = top != null ? top + height : Infinity;
737 var minY = top != null ? top : -Infinity;
738 return Math.min(maxY, Math.max(minY, y));
739};

742/**
743 * Overridable function for computing the initial position of the target
744 * before dragging begins.
745 * @protected
746 */
747goog.fx.Dragger.prototype.computeInitialPosition = function() {
748 this.deltaX = this.useRightPositioningForRtl_ ?
749 goog.style.bidi.getOffsetStart(this.target) : this.target.offsetLeft;
750 this.deltaY = this.target.offsetTop;

789 * representing the browser event that caused this drag event.

791 * @param {number=} opt_actY Optional actual y for drag if it has been limited.
792 * @param {boolean=} opt_dragCanceled Whether the drag has been canceled.
793 * @constructor
794 * @extends {goog.events.Event}
795 */
796goog.fx.DragEvent = function(type, dragobj, clientX, clientY, browserEvent,
797 opt_actX, opt_actY, opt_dragCanceled) {
798 goog.events.Event.call(this, type);
799
800 /**
801 * X-coordinate relative to the viewport
802 * @type {number}
803 */
804 this.clientX = clientX;
805
806 /**
807 * Y-coordinate relative to the viewport
808 * @type {number}
809 */
810 this.clientY = clientY;
811
812 /**
813 * The closure object representing the browser event that caused this drag
814 * event.
815 * @type {goog.events.BrowserEvent}
816 */
817 this.browserEvent = browserEvent;
818

820 * The real x-position of the drag if it has been limited
821 * @type {number}
822 */
823 this.left = goog.isDef(opt_actX) ? opt_actX : dragobj.deltaX;
824
825 /**
826 * The real y-position of the drag if it has been limited
827 * @type {number}
828 */

830
831 /**

833 * @type {goog.fx.Dragger}
834 */
835 this.dragger = dragobj;
836

839 * a legitimate drag END that can result in an action and a drag END which is
840 * a result of a drag cancelation. For now it can happen 1) with drag END

844 * released, 3) when TOUCHCANCEL is raised instead of TOUCHEND (on touch
