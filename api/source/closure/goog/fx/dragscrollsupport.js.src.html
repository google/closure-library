33/**
34 * A scroll support class. Currently this class will automatically scroll
35 * a scrollable container node and scroll it by a fixed amount at a timed
36 * interval when the mouse is moved above or below the container or in vertical
37 * margin areas. Intended for use in drag and drop. This could potentially be
38 * made more general and could support horizontal scrolling.
39 *
40 * @param {Element} containerNode A container that can be scrolled.
41 * @param {number=} opt_margin Optional margin to use while scrolling.
42 * @param {boolean=} opt_externalMouseMoveTracking Whether mouse move events
43 * are tracked externally by the client object which calls the mouse move
44 * event handler, useful when events are generated for more than one source
45 * element and/or are not real mousemove events.
46 * @constructor
47 * @extends {goog.Disposable}
48 * @see ../demos/dragscrollsupport.html
49 */
50goog.fx.DragScrollSupport = function(containerNode, opt_margin,
51 opt_externalMouseMoveTracking) {
52 goog.Disposable.call(this);
53
54 /**
55 * The container to be scrolled.
56 * @type {Element}
57 * @private
58 */
59 this.containerNode_ = containerNode;
60
61 /**
62 * Scroll timer that will scroll the container until it is stopped.
63 * It will scroll when the mouse is outside the scrolling area of the
64 * container.
65 *
66 * @type {goog.Timer}
67 * @private
68 */
69 this.scrollTimer_ = new goog.Timer(goog.fx.DragScrollSupport.TIMER_STEP_);
70
71 /**
72 * EventHandler used to set up and tear down listeners.
73 * @type {goog.events.EventHandler<!goog.fx.DragScrollSupport>}
74 * @private
75 */
76 this.eventHandler_ = new goog.events.EventHandler(this);
77
78 /**
79 * The current scroll delta.
80 * @type {goog.math.Coordinate}
81 * @private
82 */
83 this.scrollDelta_ = new goog.math.Coordinate();
84
85 /**
86 * The container bounds.
87 * @type {goog.math.Rect}
88 * @private
89 */
90 this.containerBounds_ = goog.style.getBounds(containerNode);
91
92 /**
93 * The margin for triggering a scroll.
94 * @type {number}
95 * @private
96 */
97 this.margin_ = opt_margin || 0;
98
99 /**
100 * The bounding rectangle which if left triggers scrolling.
101 * @type {goog.math.Rect}
102 * @private
103 */
104 this.scrollBounds_ = opt_margin ?
105 this.constrainBounds_(this.containerBounds_.clone()) :
106 this.containerBounds_;
107
108 this.setupListeners_(!!opt_externalMouseMoveTracking);
109};
110goog.inherits(goog.fx.DragScrollSupport, goog.Disposable);

153/**
154 * Sets whether scrolling should be constrained to happen only when the cursor
155 * is inside the container node.
156 * NOTE: If a margin is not set, then it does not make sense to
157 * contain the scroll, because in that case scroll will never be triggered.
158 * @param {boolean} constrain Whether scrolling should be constrained to happen
159 * only when the cursor is inside the container node.
160 */
161goog.fx.DragScrollSupport.prototype.setConstrainScroll = function(constrain) {
162 this.constrainScroll_ = !!this.margin_ && constrain;
163};

176/**
177 * Constrains the container bounds with respect to the margin.
178 *
179 * @param {goog.math.Rect} bounds The container element.
180 * @return {goog.math.Rect} The bounding rectangle used to calculate scrolling
181 * direction.
182 * @private
183 */
184goog.fx.DragScrollSupport.prototype.constrainBounds_ = function(bounds) {
185 var margin = this.margin_;
186 if (margin) {
187 var quarterHeight = bounds.height * 0.25;
188 var yMargin = Math.min(margin, quarterHeight);
189 bounds.top += yMargin;
190 bounds.height -= 2 * yMargin;
191
192 var quarterWidth = bounds.width * 0.25;
193 var xMargin = Math.min(margin, quarterWidth);
194 bounds.top += xMargin;
195 bounds.height -= 2 * xMargin;
196 }
197 return bounds;
198};

201/**
202 * Attaches listeners and activates automatic scrolling.
203 * @param {boolean} externalMouseMoveTracking Whether to enable internal
204 * mouse move event handling.
205 * @private
206 */
207goog.fx.DragScrollSupport.prototype.setupListeners_ = function(
208 externalMouseMoveTracking) {
209 if (!externalMouseMoveTracking) {
210 // Track mouse pointer position to determine scroll direction.
211 this.eventHandler_.listen(goog.dom.getOwnerDocument(this.containerNode_),
212 goog.events.EventType.MOUSEMOVE, this.onMouseMove);
213 }
214
215 // Scroll with a constant speed.
216 this.eventHandler_.listen(this.scrollTimer_, goog.Timer.TICK, this.onTick_);
217};

219
220/**
221 * Handler for timer tick event, scrolls the container by one scroll step if
222 * needed.
223 * @param {goog.events.Event} event Timer tick event.
224 * @private
225 */
226goog.fx.DragScrollSupport.prototype.onTick_ = function(event) {
227 this.containerNode_.scrollTop += this.scrollDelta_.y;
228 this.containerNode_.scrollLeft += this.scrollDelta_.x;
229};

232/**
233 * Handler for mouse moves events.
234 * @param {goog.events.Event} event Mouse move event.
235 */
236goog.fx.DragScrollSupport.prototype.onMouseMove = function(event) {
237 var deltaX = this.horizontalScrolling_ ? this.calculateScrollDelta(
238 event.clientX, this.scrollBounds_.left, this.scrollBounds_.width) : 0;
239 var deltaY = this.calculateScrollDelta(event.clientY,
240 this.scrollBounds_.top, this.scrollBounds_.height);
241 this.scrollDelta_.x = deltaX;
242 this.scrollDelta_.y = deltaY;
243
244 // If the scroll data is 0 or the event fired outside of the
245 // bounds of the container node.
246 if ((!deltaX && !deltaY) ||
247 (this.constrainScroll_ &&
248 !this.isInContainerBounds_(event.clientX, event.clientY))) {
249 this.scrollTimer_.stop();
250 } else if (!this.scrollTimer_.enabled) {
251 this.scrollTimer_.start();
252 }
253};

256/**
257 * Gets whether the input coordinate is in the container bounds.
258 * @param {number} x The x coordinate.
259 * @param {number} y The y coordinate.
260 * @return {boolean} Whether the input coordinate is in the container bounds.
261 * @private
262 */
263goog.fx.DragScrollSupport.prototype.isInContainerBounds_ = function(x, y) {
264 var containerBounds = this.containerBounds_;
265 return containerBounds.left <= x &&
266 containerBounds.left + containerBounds.width >= x &&
267 containerBounds.top <= y &&
268 containerBounds.top + containerBounds.height >= y;
269};

272/**
273 * Calculates scroll delta.
274 *
275 * @param {number} coordinate Current mouse pointer coordinate.
276 * @param {number} min The coordinate value below which scrolling up should be
277 * started.
278 * @param {number} rangeLength The length of the range in which scrolling should
279 * be disabled and above which scrolling down should be started.
280 * @return {number} The calculated scroll delta.
281 * @protected
282 */
283goog.fx.DragScrollSupport.prototype.calculateScrollDelta = function(
284 coordinate, min, rangeLength) {
285 var delta = 0;
286 if (coordinate < min) {
287 delta = -goog.fx.DragScrollSupport.SCROLL_STEP_;
288 } else if (coordinate > min + rangeLength) {
289 delta = goog.fx.DragScrollSupport.SCROLL_STEP_;
290 }
291 return delta;
292};
