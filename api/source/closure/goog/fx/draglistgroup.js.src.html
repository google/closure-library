44/**
45 * A class representing a group of one or more "drag lists" with items that can
46 * be dragged within them and between them.
47 *
48 * Example usage:
49 * var dragListGroup = new goog.fx.DragListGroup();
50 * dragListGroup.setDragItemHandleHoverClass(className1, className2);
51 * dragListGroup.setDraggerElClass(className3);
52 * dragListGroup.addDragList(vertList, goog.fx.DragListDirection.DOWN);
53 * dragListGroup.addDragList(horizList, goog.fx.DragListDirection.RIGHT);
54 * dragListGroup.init();
55 *
56 * @extends {goog.events.EventTarget}
57 * @constructor
58 */
59goog.fx.DragListGroup = function() {
60 goog.events.EventTarget.call(this);
61
62 /**
63 * The drag lists.
64 * @type {Array<Element>}
65 * @private
66 */
67 this.dragLists_ = [];
68
69 /**
70 * All the drag items. Set by init().
71 * @type {Array<Element>}
72 * @private
73 */
74 this.dragItems_ = [];
75
76 /**
77 * Which drag item corresponds to a given handle. Set by init().
78 * Specifically, this maps from the unique ID (as given by goog.getUid)
79 * of the handle to the drag item.
80 * @type {Object}
81 * @private
82 */
83 this.dragItemForHandle_ = {};
84
85 /**
86 * The event handler for this instance.
87 * @type {goog.events.EventHandler<!goog.fx.DragListGroup>}
88 * @private
89 */
90 this.eventHandler_ = new goog.events.EventHandler(this);
91
92 /**
93 * Whether the setup has been done to make all items in all lists draggable.
94 * @type {boolean}
95 * @private
96 */
97 this.isInitialized_ = false;
98
99 /**
100 * Whether the currDragItem is always displayed. By default the list
101 * collapses, the currDragItem's display is set to none, when we do not
102 * hover over a draglist.
103 * @type {boolean}
104 * @private
105 */
106 this.isCurrDragItemAlwaysDisplayed_ = false;
107
108 /**
109 * Whether to update the position of the currDragItem as we drag, i.e.,
110 * insert the currDragItem each time to the position where it would land if
111 * we were to end the drag at that point. Defaults to true.
112 * @type {boolean}
113 * @private
114 */
115 this.updateWhileDragging_ = true;
116};
117goog.inherits(goog.fx.DragListGroup, goog.events.EventTarget);

403/**
404 * Performs the initial setup to make all items in all lists draggable.
405 */
406goog.fx.DragListGroup.prototype.init = function() {
407 if (this.isInitialized_) {
408 return;
409 }
410
411 for (var i = 0, numLists = this.dragLists_.length; i < numLists; i++) {
412 var dragList = this.dragLists_[i];
413
414 var dragItems = goog.dom.getChildren(dragList);
415 for (var j = 0, numItems = dragItems.length; j < numItems; ++j) {
416 this.listenForDragEvents(dragItems[j]);
417 }
418 }
419
420 this.isInitialized_ = true;
421};

424/**
425 * Adds a single item to the given drag list and sets up the drag listeners for
426 * it.
427 * If opt_index is specified the item is inserted at this index, otherwise the
428 * item is added as the last child of the list.
429 *
430 * @param {!Element} list The drag list where to add item to.
431 * @param {!Element} item The new element to add.
432 * @param {number=} opt_index Index where to insert the item in the list. If not
433 * specified item is inserted as the last child of list.
434 */
435goog.fx.DragListGroup.prototype.addItemToDragList = function(list, item,
436 opt_index) {
437 if (goog.isDef(opt_index)) {
438 goog.dom.insertChildAt(list, item, opt_index);
439 } else {
440 goog.dom.appendChild(list, item);
441 }
442 this.listenForDragEvents(item);
443};

446/** @override */
447goog.fx.DragListGroup.prototype.disposeInternal = function() {
448 this.eventHandler_.dispose();
449
450 for (var i = 0, n = this.dragLists_.length; i < n; i++) {
451 var dragList = this.dragLists_[i];
452 // Note: IE doesn't allow 'delete' for fields on HTML elements (because
453 // they're not real JS objects in IE), so we just set them to undefined.
454 dragList.dlgGrowthDirection_ = undefined;
455 dragList.dlgDragHoverClass_ = undefined;
456 }
457
458 this.dragLists_.length = 0;
459 this.dragItems_.length = 0;
460 this.dragItemForHandle_ = null;
461
462 // In the case where a drag event is currently in-progress and dispose is
463 // called, this cleans up the extra state.
464 this.cleanupDragDom_();
465
466 goog.fx.DragListGroup.superClass_.disposeInternal.call(this);
467};

470/**
471 * Caches the heights of each drag list and drag item, except for the current
472 * drag item.
473 *
474 * @param {Element} currDragItem The item currently being dragged.
475 * @private
476 */
477goog.fx.DragListGroup.prototype.recacheListAndItemBounds_ = function(
478 currDragItem) {
479 for (var i = 0, n = this.dragLists_.length; i < n; i++) {
480 var dragList = this.dragLists_[i];
481 dragList.dlgBounds_ = goog.style.getBounds(dragList);
482 }
483
484 for (var i = 0, n = this.dragItems_.length; i < n; i++) {
485 var dragItem = this.dragItems_[i];
486 if (dragItem != currDragItem) {
487 dragItem.dlgBounds_ = goog.style.getBounds(dragItem);
488 }
489 }
490};

493/**
494 * Listens for drag events on the given drag item. This method is currently used
495 * to initialize drag items.
496 *
497 * @param {Element} dragItem the element to initialize. This element has to be
498 * in one of the drag lists.
499 * @protected
500 */
501goog.fx.DragListGroup.prototype.listenForDragEvents = function(dragItem) {
502 var dragItemHandle = this.getHandleForDragItem_(dragItem);
503 var uid = goog.getUid(dragItemHandle);
504 this.dragItemForHandle_[uid] = dragItem;
505
506 if (this.dragItemHoverClasses_) {
507 this.eventHandler_.listen(
508 dragItem, goog.events.EventType.MOUSEOVER,
509 this.handleDragItemMouseover_);
510 this.eventHandler_.listen(
511 dragItem, goog.events.EventType.MOUSEOUT,
512 this.handleDragItemMouseout_);
513 }
514 if (this.dragItemHandleHoverClasses_) {
515 this.eventHandler_.listen(
516 dragItemHandle, goog.events.EventType.MOUSEOVER,
517 this.handleDragItemHandleMouseover_);
518 this.eventHandler_.listen(
519 dragItemHandle, goog.events.EventType.MOUSEOUT,
520 this.handleDragItemHandleMouseout_);
521 }
522
523 this.dragItems_.push(dragItem);
524 this.eventHandler_.listen(dragItemHandle,
525 [goog.events.EventType.MOUSEDOWN, goog.events.EventType.TOUCHSTART],
526 this.handlePotentialDragStart_);
527};

530/**
531 * Handles mouse and touch events which may start a drag action.
532 * @param {!goog.events.BrowserEvent} e MOUSEDOWN or TOUCHSTART event.
533 * @private
534 */
535goog.fx.DragListGroup.prototype.handlePotentialDragStart_ = function(e) {
536 var uid = goog.getUid(/** @type {Node} */ (e.currentTarget));
537 this.currDragItem_ = /** @type {Element} */ (this.dragItemForHandle_[uid]);
538
539 this.draggerEl_ = this.createDragElementInternal(this.currDragItem_);
540 if (this.draggerElClasses_) {
541 // Add CSS class for the clone, if any.
542 goog.dom.classlist.addAll(
543 goog.asserts.assert(this.draggerEl_), this.draggerElClasses_ || []);
544 }
545
546 // Place the clone (i.e. draggerEl) at the same position as the actual
547 // current drag item. This is a bit tricky since
548 // goog.style.getPageOffset() gets the left-top pos of the border, but
549 // goog.style.setPageOffset() sets the left-top pos of the margin.
550 // It's difficult to adjust for the margins of the clone because it's
551 // difficult to read it: goog.style.getComputedStyle() doesn't work for IE.
552 // Instead, our workaround is simply to set the clone's margins to 0px.
553 this.draggerEl_.style.margin = '0';
554 this.draggerEl_.style.position = 'absolute';
555 this.draggerEl_.style.visibility = 'hidden';
556 var doc = goog.dom.getOwnerDocument(this.currDragItem_);
557 doc.body.appendChild(this.draggerEl_);
558
559 // Important: goog.style.setPageOffset() only works correctly for IE when the
560 // element is already in the document.
561 var currDragItemPos = goog.style.getPageOffset(this.currDragItem_);
562 goog.style.setPageOffset(this.draggerEl_, currDragItemPos);
563
564 this.dragger_ = new goog.fx.Dragger(this.draggerEl_);
565 this.dragger_.setHysteresis(this.hysteresisDistance_);
566
567 // Listen to events on the dragger. These handlers will be unregistered at
568 // DRAGEND, when the dragger is disposed of. We can't use eventHandler_,
569 // because it creates new references to the handler functions at each
570 // dragging action, and keeps them until DragListGroup is disposed of.
571 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.START,
572 this.handleDragStart_, false, this);
573 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.END,
574 this.handleDragEnd_, false, this);
575 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.EARLY_CANCEL,
576 this.cleanup_, false, this);
577 this.dragger_.startDrag(e);
578};

611/**
612 * Handles the start of a drag action.
613 * @param {!goog.fx.DragEvent} e goog.fx.Dragger.EventType.START event.
614 * @private
615 */
616goog.fx.DragListGroup.prototype.handleDragStart_ = function(e) {
617 if (!this.dispatchEvent(new goog.fx.DragListGroupEvent(
618 goog.fx.DragListGroup.EventType.BEFOREDRAGSTART, this, e.browserEvent,
619 this.currDragItem_, null, null))) {
620 e.preventDefault();
621 this.cleanup_();
622 return;
623 }
624
625 // Record the original location of the current drag item.
626 // Note: this.origNextItem_ may be null.
627 this.origList_ = /** @type {Element} */ (this.currDragItem_.parentNode);
628 this.origNextItem_ = goog.dom.getNextElementSibling(this.currDragItem_);
629 this.currHoverItem_ = this.origNextItem_;
630 this.currHoverList_ = this.origList_;
631
632 // If there's a CSS class specified for the current drag item, add it.
633 // Otherwise, make the actual current drag item hidden (takes up space).
634 if (this.currDragItemClasses_) {
635 goog.dom.classlist.addAll(
636 goog.asserts.assert(this.currDragItem_),
637 this.currDragItemClasses_ || []);
638 } else {
639 this.currDragItem_.style.visibility = 'hidden';
640 }
641
642 // Precompute distances from top-left corner to center for efficiency.
643 var draggerElSize = goog.style.getSize(this.draggerEl_);
644 this.draggerEl_.halfWidth = draggerElSize.width / 2;
645 this.draggerEl_.halfHeight = draggerElSize.height / 2;
646
647 this.draggerEl_.style.visibility = '';
648
649 // Record the bounds of all the drag lists and all the other drag items. This
650 // caching is for efficiency, so that we don't have to recompute the bounds on
651 // each drag move. Do this in the state where the current drag item is not in
652 // any of the lists, except when update while dragging is disabled, as in this
653 // case the current drag item does not get removed until drag ends.
654 if (this.updateWhileDragging_) {
655 this.currDragItem_.style.display = 'none';
656 }
657 this.recacheListAndItemBounds_(this.currDragItem_);
658 this.currDragItem_.style.display = '';
659
660 // Listen to events on the dragger.
661 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.DRAG,
662 this.handleDragMove_, false, this);
663
664 this.dispatchEvent(
665 new goog.fx.DragListGroupEvent(
666 goog.fx.DragListGroup.EventType.DRAGSTART, this, e.browserEvent,
667 this.currDragItem_, this.draggerEl_, this.dragger_));
668};

671/**
672 * Handles a drag movement (i.e. DRAG event fired by the dragger).
673 *
674 * @param {goog.fx.DragEvent} dragEvent Event object fired by the dragger.
675 * @return {boolean} The return value for the event.
676 * @private
677 */
678goog.fx.DragListGroup.prototype.handleDragMove_ = function(dragEvent) {
679
680 // Compute the center of the dragger element (i.e. the cloned drag item).
681 var draggerElPos = goog.style.getPageOffset(this.draggerEl_);
682 var draggerElCenter = new goog.math.Coordinate(
683 draggerElPos.x + this.draggerEl_.halfWidth,
684 draggerElPos.y + this.draggerEl_.halfHeight);
685
686 // Check whether the center is hovering over one of the drag lists.
687 var hoverList = this.getHoverDragList_(draggerElCenter);
688
689 // If hovering over a list, find the next item (if drag were to end now).
690 var hoverNextItem =
691 hoverList ? this.getHoverNextItem_(hoverList, draggerElCenter) : null;
692
693 var rv = this.dispatchEvent(
694 new goog.fx.DragListGroupEvent(
695 goog.fx.DragListGroup.EventType.BEFOREDRAGMOVE, this, dragEvent,
696 this.currDragItem_, this.draggerEl_, this.dragger_,
697 draggerElCenter, hoverList, hoverNextItem));
698 if (!rv) {
699 return false;
700 }
701
702 if (hoverList) {
703 if (this.updateWhileDragging_) {
704 this.insertCurrDragItem_(hoverList, hoverNextItem);
705 } else {
706 // If update while dragging is disabled do not insert
707 // the dragged item, but update the hovered item instead.

714 goog.asserts.assert(hoverList), hoverList.dlgDragHoverClass_);

920/**
921 * Helper for handleDragMove_().
922 * Given the position of the center of the dragger element, figures out whether
923 * it's currently hovering over any of the drag lists.
924 *
925 * @param {goog.math.Coordinate} draggerElCenter The center position of the
926 * dragger element.
927 * @return {Element} If currently hovering over a drag list, returns the drag
928 * list element. Else returns null.
929 * @private
930 */
931goog.fx.DragListGroup.prototype.getHoverDragList_ = function(draggerElCenter) {
932
933 // If the current drag item was in a list last time we did this, then check
934 // that same list first.
935 var prevHoverList = null;
936 if (this.currDragItem_.style.display != 'none') {
937 prevHoverList = /** @type {Element} */ (this.currDragItem_.parentNode);
938 // Important: We can't use the cached bounds for this list because the
939 // cached bounds are based on the case where the current drag item is not
940 // in the list. Since the current drag item is known to be in this list, we
941 // must recompute the list's bounds.
942 var prevHoverListBounds = goog.style.getBounds(prevHoverList);
943 if (this.isInRect_(draggerElCenter, prevHoverListBounds)) {
944 return prevHoverList;
945 }
946 }
947
948 for (var i = 0, n = this.dragLists_.length; i < n; i++) {
949 var dragList = this.dragLists_[i];
950 if (dragList == prevHoverList) {
951 continue;
952 }
953 if (this.isInRect_(draggerElCenter, dragList.dlgBounds_)) {
954 return dragList;
955 }
956 }
957
958 return null;

1011/**
1012 * Helper for handleDragMove_().
1013 * Given the position of the center of the dragger element, plus the drag list
1014 * that it's currently hovering over, figures out the next drag item in the
1015 * list that follows the current position of the dragger element. (I.e. if
1016 * the drag action ends right now, it would become the item after the current
1017 * drag item.)
1018 *
1019 * @param {Element} hoverList The drag list that we're hovering over.
1020 * @param {goog.math.Coordinate} draggerElCenter The center position of the
1021 * dragger element.
1022 * @return {Element} Returns the earliest item in the hover list that belongs
1023 * after the current position of the dragger element. If all items in the
1024 * list should come before the current drag item, then returns null.
1025 * @private
1026 */
1027goog.fx.DragListGroup.prototype.getHoverNextItem_ = function(
1028 hoverList, draggerElCenter) {
1029 if (hoverList == null) {
1030 throw Error('getHoverNextItem_ called with null hoverList.');
1031 }
1032
1033 // The definition of what it means for the draggerEl to be "before" a given
1034 // item in the hover drag list is not always the same. It changes based on
1035 // the growth direction of the hover drag list in question.
1036 /** @type {number} */
1037 var relevantCoord;
1038 var getRelevantBoundFn;
1039 var isBeforeFn;
1040 var pickClosestRow = false;
1041 var distanceToClosestRow = undefined;
1042 switch (hoverList.dlgGrowthDirection_) {
1043 case goog.fx.DragListDirection.DOWN:
1044 // "Before" means draggerElCenter.y is less than item's bottom y-value.
1045 relevantCoord = draggerElCenter.y;
1046 getRelevantBoundFn = goog.fx.DragListGroup.getBottomBound_;
1047 isBeforeFn = goog.fx.DragListGroup.isLessThan_;
1048 break;
1049 case goog.fx.DragListDirection.RIGHT_2D:
1050 pickClosestRow = true;
1051 case goog.fx.DragListDirection.RIGHT:
1052 // "Before" means draggerElCenter.x is less than item's right x-value.
1053 relevantCoord = draggerElCenter.x;
1054 getRelevantBoundFn = goog.fx.DragListGroup.getRightBound_;
1055 isBeforeFn = goog.fx.DragListGroup.isLessThan_;
1056 break;
1057 case goog.fx.DragListDirection.LEFT_2D:
1058 pickClosestRow = true;
1059 case goog.fx.DragListDirection.LEFT:
1060 // "Before" means draggerElCenter.x is greater than item's left x-value.
1061 relevantCoord = draggerElCenter.x;
1062 getRelevantBoundFn = goog.fx.DragListGroup.getLeftBound_;
1063 isBeforeFn = goog.fx.DragListGroup.isGreaterThan_;
1064 break;
1065 }
1066
1067 // This holds the earliest drag item found so far that should come after
1068 // this.currDragItem_ in the hover drag list (based on draggerElCenter).
1069 var earliestAfterItem = null;
1070 // This is the position of the relevant bound for the earliestAfterItem,
1071 // where "relevant" is determined by the growth direction of hoverList.
1072 var earliestAfterItemRelevantBound;
1073
1074 var hoverListItems = goog.dom.getChildren(hoverList);
1075 for (var i = 0, n = hoverListItems.length; i < n; i++) {
1076 var item = hoverListItems[i];
1077 if (item == this.currDragItem_) {
1078 continue;
1079 }
1080
1081 var relevantBound = getRelevantBoundFn(item.dlgBounds_);
1082 // When the hoverlist is broken into multiple rows (i.e., in the case of
1083 // LEFT_2D and RIGHT_2D) it is no longer enough to only look at the
1084 // x-coordinate alone in order to find the {@earliestAfterItem} in the
1085 // hoverlist. Make sure it is chosen from the row closest to the
1086 // {@code draggerElCenter}.
1087 if (pickClosestRow) {
1088 var distanceToRow = goog.fx.DragListGroup.verticalDistanceFromItem_(item,
1089 draggerElCenter);
1090 // Initialize the distance to the closest row to the current value if
1091 // undefined.
1092 if (!goog.isDef(distanceToClosestRow)) {
1093 distanceToClosestRow = distanceToRow;
1094 }
1095 if (isBeforeFn(relevantCoord, relevantBound) &&
1096 (earliestAfterItemRelevantBound == undefined ||
1097 (distanceToRow < distanceToClosestRow) ||
1098 ((distanceToRow == distanceToClosestRow) &&
1099 (isBeforeFn(relevantBound, earliestAfterItemRelevantBound) ||
1100 relevantBound == earliestAfterItemRelevantBound)))) {
1101 earliestAfterItem = item;
1102 earliestAfterItemRelevantBound = relevantBound;
1103 }
1104 // Update distance to closest row.
1105 if (distanceToRow < distanceToClosestRow) {
1106 distanceToClosestRow = distanceToRow;
1107 }
1108 } else if (isBeforeFn(relevantCoord, relevantBound) &&
1109 (earliestAfterItemRelevantBound == undefined ||
1110 isBeforeFn(relevantBound, earliestAfterItemRelevantBound))) {
1111 earliestAfterItem = item;
1112 earliestAfterItemRelevantBound = relevantBound;
1113 }
1114 }
1115 // If we ended up picking an element that is not in the closest row it can
1116 // only happen if we should have picked the last one in which case there is
1117 // no consecutive element.
1118 if (!goog.isNull(earliestAfterItem) &&
1119 goog.fx.DragListGroup.verticalDistanceFromItem_(
1120 earliestAfterItem, draggerElCenter) > distanceToClosestRow) {
1121 return null;
1122 } else {
1123 return earliestAfterItem;
1124 }
1125};
